96 private static Task HandleExceptionAsync(HttpContext context, Exception exception)
97 {
98 context.Response.ContentType = "application/json";
99 var (statusCode, problemDetails) = exception switch
100 {
101 ValidationException ve => (
102 StatusCodes.Status400BadRequest,
103 (ProblemDetails)new ValidationProblemDetails(ve.Errors)
104 {
105 Title = "Validation failed",
106 Status = StatusCodes.Status400BadRequest
107 }),
108 NotFoundException ne => (
109 StatusCodes.Status404NotFound,
110 new ProblemDetails
111 {
112 Title = "Resource not found",
113 Detail = ne.Message,
114 Status = StatusCodes.Status404NotFound
115 }),
116 DomainException de => (
117 StatusCodes.Status422UnprocessableEntity,
118 new ProblemDetails
119 {
120 Title = "Business rule violation",
121 Detail = de.Message,
122 Status = StatusCodes.Status422UnprocessableEntity
123 }),
124 _ => (
125 StatusCodes.Status500InternalServerError,
126 new ProblemDetails
127 {
128 Title = "An unexpected error occurred",
129 Status = StatusCodes.Status500InternalServerError,
130 Detail = "Please try again later or contact support."
131 })
132 };
133 context.Response.StatusCode = statusCode;
134 return context.Response.WriteAsJsonAsync(problemDetails);
135 }
