1// Este archivo es la raíz de composición y configuración de inicio de la aplicación.
2//
3// Rol en Clean Architecture:
4// - Parte de la capa de presentación (Capa de API)
5// - Raíz de Composición: Conecta todas las dependencias a través de Inyección de Dependencias
6// - Configuración: Configura pipeline de middleware, autenticación y servicios
7// - Punto de entrada: ASP.NET Core ejecuta este archivo en el inicio de la aplicación
8//
9// Responsabilidades:
10// - Registrar servicios de todas las capas en el contenedor de DI
11// - Configurar autenticación y autorización
12// - Agregar servicios de acceso a datos y persistencia
13// - Configurar pipeline de middleware (¡el orden importa!)
14// - Inicializar base de datos si es necesario
15//
16// Patrón de Inyección de Dependencias:
17// - Registra interfaces de capas de Aplicación y Dominio
18// - Implementaciones de capa de Infraestructura
19// - Los controladores usan servicios registrados (inyección de constructor)
20// - Los manejadores de MediatR usan servicios registrados
21//
22// Registro de Capa de Arquitectura:
23// - Capa de Dominio: IUnitOfWork
24// - Capa de Aplicación: ITaskRepository, IDomainEventDispatcher, MediatR, FluentValidation, ValidationBehavior
25// - Capa de Infraestructura: DbContext, Repositorios, Distribuidor de Eventos
26// - Capa de UI: Controladores, Middleware
27//
28// Directrices de Raíz de Composición:
29// - La capa de UI puede hacer referencia a Infraestructura para registro de DI
30// - Infraestructura implementa interfaces de otras capas
31// - Las capas de Dominio y Aplicación permanecen independientes
32// - Este archivo es el único lugar donde se hace referencia directa a Infraestructura
33//
34// pipeline de Middleware:
35// - El orden es crítico: el middleware anterior procesa solicitudes primero
36// - ExceptionHandlingMiddleware: Usualmente primero para capturar todas las excepciones
37// - Autenticación, Autorización: Middleware relacionado con seguridad
38// - MapControllers: Enruta solicitudes a controladores

74var bearerSection = builder.Configuration.GetSection("Authentication:Schemes:Bearer");
75var validIssuer = bearerSection["ValidIssuer"];
76var validAudiences = bearerSection
77 .GetSection("ValidAudiences")
78 .Get<string[]>() ?? [];
79var signingKeys = bearerSection
80 .GetSection("SigningKeys")
81 .GetChildren()
82 .Select(x => x["Value"])
83 .Where(x => !string.IsNullOrWhiteSpace(x))
84 .Select(x => new SymmetricSecurityKey(Convert.FromBase64String(x!)))
85 .Cast<SecurityKey>()
86 .ToArray();
87
88builder.Services
89 .AddAuthentication(JwtBearerDefaults.AuthenticationScheme)
90 .AddJwtBearer(options =>
91 {
92 options.TokenValidationParameters = new TokenValidationParameters
93 {
94 ValidateIssuer = !string.IsNullOrWhiteSpace(validIssuer),
95 ValidIssuer = validIssuer,
96 ValidateAudience = validAudiences.Length > 0,
97 ValidAudiences = validAudiences,
98 ValidateIssuerSigningKey = true,
99 IssuerSigningKeys = signingKeys,
100 ValidateLifetime = true,
101 TryAllIssuerSigningKeys = true,
102 ClockSkew = TimeSpan.FromMinutes(1)
103 };
104 });
105builder.Services.AddAuthorization();
106
107// Agregar servicios al contenedor
108builder.Services
109 .AddMediatR(cfg => cfg.RegisterServicesFromAssembly(typeof(TaskManagement.Application.Commands.CreateTask.CreateTaskCommand).Assembly))
110 .AddValidatorsFromAssembly(typeof(TaskManagement.Application.Commands.CreateTask.CreateTaskCommand).Assembly)
111 .AddTransient(typeof(IPipelineBehavior<,>), typeof(ValidationBehavior<,>))
112 .AddDbContext<TaskDbContext>(options =>
113 options.UseSqlite(builder.Configuration.GetConnectionString("DefaultConnection")))
114 .AddScoped<IUnitOfWork>(sp => sp.GetRequiredService<TaskDbContext>())
115 .AddScoped<ITaskRepository, TaskRepository>()
116 .AddScoped<ITaskReadRepository, TaskReadRepository>()
117 .AddScoped<IDomainEventDispatcher, MediatRDomainEventDispatcher>()
118 .AddControllers()
119 .AddJsonOptions(options =>
120 {
121 options.JsonSerializerOptions.Converters.Add(new JsonStringEnumConverter());
122 });
123var app = builder.Build();

125using (var scope = app.Services.CreateScope())
