109def create(req, sock, client, server, cfg):
110 resp = Response(req, sock, cfg)
111
112 # set initial environ
113 environ = default_environ(req, sock, cfg)
114
115 # default variables
116 host = None
117 script_name = os.environ.get("SCRIPT_NAME", "")
118
119 # add the headers to the environ
120 for hdr_name, hdr_value in req.headers:
121 if hdr_name == "EXPECT":
122 # handle expect
123 if hdr_value.lower() == "100-continue":
124 sock.send(b"HTTP/1.1 100 Continue\r\n\r\n")
125 elif hdr_name == 'HOST':
126 host = hdr_value
127 elif hdr_name == "SCRIPT_NAME":
128 script_name = hdr_value
129 elif hdr_name == "CONTENT-TYPE":
130 environ['CONTENT_TYPE'] = hdr_value
131 continue
132 elif hdr_name == "CONTENT-LENGTH":
133 environ['CONTENT_LENGTH'] = hdr_value
134 continue
135
136 key = 'HTTP_' + hdr_name.replace('-', '_')
137 if key in environ:
138 hdr_value = "%s,%s" % (environ[key], hdr_value)
139 environ[key] = hdr_value
140
141 # set the url scheme
142 environ['wsgi.url_scheme'] = req.scheme
143
144 # set the REMOTE_* keys in environ
145 # authors should be aware that REMOTE_HOST and REMOTE_ADDR
146 # may not qualify the remote addr:
147 # http://www.ietf.org/rfc/rfc3875
148 if isinstance(client, str):
149 environ['REMOTE_ADDR'] = client
150 elif isinstance(client, bytes):
151 environ['REMOTE_ADDR'] = client.decode()
152 else:
153 environ['REMOTE_ADDR'] = client[0]
154 environ['REMOTE_PORT'] = str(client[1])
155
156 # handle the SERVER_*
157 # Normally only the application should use the Host header but since the
158 # WSGI spec doesn't support unix sockets, we are using it to create
159 # viable SERVER_* if possible.
160 if isinstance(server, str):
161 server = server.split(":")
162 if len(server) == 1:
163 # unix socket
164 if host:
165 server = host.split(':')
166 if len(server) == 1:
167 if req.scheme == "http":
168 server.append(80)
169 elif req.scheme == "https":
170 server.append(443)
171 else:
172 server.append('')
173 else:
174 # no host header given which means that we are not behind a
175 # proxy, so append an empty port.
176 server.append('')
177 environ['SERVER_NAME'] = server[0]
178 environ['SERVER_PORT'] = str(server[1])
179
180 # set the path and script name
181 path_info = req.path
182 if script_name:
183 path_info = path_info.split(script_name, 1)[1]
184 environ['PATH_INFO'] = util.unquote_to_wsgi_str(path_info)
185 environ['SCRIPT_NAME'] = script_name
186
187 # override the environ with the correct remote and server address if
188 # we are behind a proxy using the proxy protocol.
189 environ.update(proxy_environ(req))
190 return resp, environ

223 def start_response(self, status, headers, exc_info=None):
224 if exc_info:
225 try:
226 if self.status and self.headers_sent:

229 exc_info = None

247 def process_headers(self, headers):
248 for name, value in headers:
249 if not isinstance(name, str):
250 raise TypeError('%r is not a string' % name)
251
252 if HEADER_RE.search(name):
253 raise InvalidHeaderName('%r' % name)
254
255 if not isinstance(value, str):
256 raise TypeError('%r is not a string' % value)
257
258 if HEADER_VALUE_RE.search(value):
259 raise InvalidHeader('%r' % value)
260
261 value = value.strip()
262 lname = name.lower().strip()
263 if lname == "content-length":
264 self.response_length = int(value)
265 elif util.is_hoppish(name):
266 if lname == "connection":
267 # handle websocket
268 if value.lower().strip() == "upgrade":
269 self.upgrade = True
270 elif lname == "upgrade":
271 if value.lower().strip() == "websocket":
272 self.headers.append((name.strip(), value))
273
274 # ignore hopbyhop headers
275 continue
276 self.headers.append((name.strip(), value))

278 def is_chunked(self):
279 # Only use chunked responses when the client is
280 # speaking HTTP/1.1 or newer and there was
281 # no Content-Length header set.
282 if self.response_length is not None:
283 return False
284 elif self.req.version <= (1, 0):
285 return False
286 elif self.req.method == 'HEAD':
287 # Responses to a HEAD request MUST NOT contain a response body.
288 return False
289 elif self.status_code in (204, 304):
290 # Do not use chunked responses when the response is guaranteed to
291 # not have a response body.
292 return False
293 return True

295 def default_headers(self):
296 # set the connection header
297 if self.upgrade:
298 connection = "upgrade"
299 elif self.should_close():
300 connection = "close"
301 else:
302 connection = "keep-alive"
303
304 headers = [
305 "HTTP/%s.%s %s\r\n" % (self.req.version[0],
306 self.req.version[1], self.status),
307 "Server: %s\r\n" % self.version,
308 "Date: %s\r\n" % util.http_date(),
309 "Connection: %s\r\n" % connection
310 ]
311 if self.chunked:
312 headers.append("Transfer-Encoding: chunked\r\n")
313 return headers

315 def send_headers(self):
316 if self.headers_sent:
317 return
318 tosend = self.default_headers()
319 tosend.extend(["%s: %s\r\n" % (k, v) for k, v in self.headers])
320
321 header_str = "%s\r\n" % "".join(tosend)
322 util.write(self.sock, util.to_bytestring(header_str, "latin-1"))
323 self.headers_sent = True
324
325 def write(self, arg):
326 self.send_headers()
327 if not isinstance(arg, bytes):
328 raise TypeError('%r is not a byte' % arg)
329 arglen = len(arg)
330 tosend = arglen
331 if self.response_length is not None:
332 if self.sent >= self.response_length:
333 # Never write more than self.response_length bytes
334 return
335
336 tosend = min(self.response_length - self.sent, tosend)
337 if tosend < arglen:
338 arg = arg[:tosend]
339
340 # Sending an empty chunk signals the end of the
341 # response and prematurely closes the response
342 if self.chunked and tosend == 0:
343 return
344
345 self.sent += tosend
346 util.write(self.sock, arg, self.chunked)

351 def sendfile(self, respiter):
352 if self.cfg.is_ssl or not self.can_sendfile():
353 return False
354
355 if not util.has_fileno(respiter.filelike):
356 return False
357
358 fileno = respiter.filelike.fileno()
359 try:
360 offset = os.lseek(fileno, 0, os.SEEK_CUR)
361 if self.response_length is None:
362 filesize = os.fstat(fileno).st_size
363 nbytes = filesize - offset
364 else:
365 nbytes = self.response_length
366 except (OSError, io.UnsupportedOperation):
367 return False
368
369 self.send_headers()
370
371 if self.is_chunked():
372 chunk_size = "%X\r\n" % nbytes
373 self.sock.sendall(chunk_size.encode('utf-8'))
374
375 self.sock.sendfile(respiter.filelike, count=nbytes)
376
377 if self.is_chunked():
378 self.sock.sendall(b"\r\n")
379
380 os.lseek(fileno, offset, os.SEEK_SET)
381
382 return True
