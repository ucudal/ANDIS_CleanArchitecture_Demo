25class PromptTemplate(StringPromptTemplate):
26 """Prompt template for a language model.
27
28 A prompt template consists of a string template. It accepts a set of parameters
29 from the user that can be used to generate a prompt for a language model.
30
31 The template can be formatted using either f-strings (default), jinja2,
32 or mustache syntax.
33
34 *Security warning*:
35 Prefer using `template_format="f-string"` instead of
36 `template_format="jinja2"`, or make sure to NEVER accept jinja2 templates
37 from untrusted sources as they may lead to arbitrary Python code execution.
38
39 As of LangChain 0.0.329, Jinja2 templates will be rendered using
40 Jinja2's SandboxedEnvironment by default. This sand-boxing should
41 be treated as a best-effort approach rather than a guarantee of security,
42 as it is an opt-out rather than opt-in approach.
43
44 Despite the sand-boxing, we recommend to never use jinja2 templates
45 from untrusted sources.
46
47 Example:
48
49 .. code-block:: python
50
51 from langchain_core.prompts import PromptTemplate
52
53 # Instantiation using from_template (recommended)
54 prompt = PromptTemplate.from_template("Say {foo}")
55 prompt.format(foo="bar")
56
57 # Instantiation using initializer
58 prompt = PromptTemplate(template="Say {foo}")
59 """

136 def __add__(self, other: Any) -> PromptTemplate:
137 """Override the + operator to allow for combining prompt templates."""
138 # Allow for easy combining
139 if isinstance(other, PromptTemplate):
140 if self.template_format != "f-string":
141 msg = "Adding prompt templates only supported for f-strings."
142 raise ValueError(msg)
143 if other.template_format != "f-string":
144 msg = "Adding prompt templates only supported for f-strings."
145 raise ValueError(msg)
146 input_variables = list(
147 set(self.input_variables) | set(other.input_variables)
148 )
149 template = self.template + other.template
150 # If any do not want to validate, then don't
151 validate_template = self.validate_template and other.validate_template
152 partial_variables = dict(self.partial_variables.items())
153 for k, v in other.partial_variables.items():
154 if k in partial_variables:
155 msg = "Cannot have same variable partialed twice."
156 raise ValueError(msg)
157 partial_variables[k] = v
158 return PromptTemplate(
159 template=template,
160 input_variables=input_variables,
161 partial_variables=partial_variables,
162 template_format="f-string",
163 validate_template=validate_template,
164 )
165 if isinstance(other, str):
166 prompt = PromptTemplate.from_template(other)
167 return self + prompt
168 msg = f"Unsupported operand type for +: {type(other)}"
169 raise NotImplementedError(msg)

273

293 input_variables = [
294 var for var in input_variables if var not in _partial_variables
295 ]
296
