49StreamMode = Literal["values", "updates", "debug", "messages", "custom"]
50"""How the stream method should emit outputs.
51
52- `"values"`: Emit all values in the state after each step.
53 When used with functional API, values are emitted once at the end of the workflow.
54- `"updates"`: Emit only the node or task names and updates returned by the nodes or tasks after each step.
55 If multiple updates are made in the same step (e.g. multiple nodes are run) then those updates are emitted separately.
56- `"custom"`: Emit custom data using from inside nodes or tasks using `StreamWriter`.
57- `"messages"`: Emit LLM messages token-by-token together with metadata for any LLM invocations inside nodes or tasks.
58- `"debug"`: Emit debug events with as much information as possible for each step.

181class Send:
182 """A message or packet to send to a specific node in the graph.
183
184 The `Send` class is used within a `StateGraph`'s conditional edges to
185 dynamically invoke a node with a custom state at the next step.
186
187 Importantly, the sent state can differ from the core graph's state,
188 allowing for flexible and dynamic workflow management.
189
190 One such example is a "map-reduce" workflow where your graph invokes
191 the same node multiple times in parallel with different states,
192 before aggregating the results back into the main graph's state.
193
194 Attributes:
195 node (str): The name of the target node to send the message to.
196 arg (Any): The state or message to send to the target node.
197
198 Examples:
199 >>> from typing import Annotated
200 >>> import operator
201 >>> class OverallState(TypedDict):
202 ... subjects: list[str]
203 ... jokes: Annotated[list[str], operator.add]
204 ...
205 >>> from langgraph.types import Send
206 >>> from langgraph.graph import END, START
207 >>> def continue_to_jokes(state: OverallState):
208 ... return [Send("generate_joke", {"subject": s}) for s in state['subjects']]
209 ...
210 >>> from langgraph.graph import StateGraph
211 >>> builder = StateGraph(OverallState)
212 >>> builder.add_node("generate_joke", lambda state: {"jokes": [f"Joke about {state['subject']}"]})
213 >>> builder.add_conditional_edges(START, continue_to_jokes)
214 >>> builder.add_edge("generate_joke", END)
215 >>> graph = builder.compile()
216 >>>
217 >>> # Invoking with two subjects results in a generated joke for each
218 >>> graph.invoke({"subjects": ["cats", "dogs"]})
219 {'subjects': ['cats', 'dogs'], 'jokes': ['Joke about cats', 'Joke about dogs']}

369def interrupt(value: Any) -> Any:
370 """Interrupt the graph with a resumable exception from within a node.
371
372 The `interrupt` function enables human-in-the-loop workflows by pausing graph
373 execution and surfacing a value to the client. This value can communicate context
374 or request input required to resume execution.
375
376 In a given node, the first invocation of this function raises a `GraphInterrupt`
377 exception, halting execution. The provided `value` is included with the exception
378 and sent to the client executing the graph.
379
380 A client resuming the graph must use the [`Command`][langgraph.types.Command]
381 primitive to specify a value for the interrupt and continue execution.
382 The graph resumes from the start of the node, **re-executing** all logic.
383
384 If a node contains multiple `interrupt` calls, LangGraph matches resume values
385 to interrupts based on their order in the node. This list of resume values
386 is scoped to the specific task executing the node and is not shared across tasks.
387
388 To use an `interrupt`, you must enable a checkpointer, as the feature relies
389 on persisting the graph state.
390
391 Example:
392 ```python
393 import uuid
394 from typing import Optional
395 from typing_extensions import TypedDict
396
397 from langgraph.checkpoint.memory import MemorySaver
398 from langgraph.constants import START
399 from langgraph.graph import StateGraph
400 from langgraph.types import interrupt
401
402
403 class State(TypedDict):
404 \"\"\"The graph state.\"\"\"
405
406 foo: str
407 human_value: Optional[str]
408 \"\"\"Human value will be updated using an interrupt.\"\"\"
409
410
411 def node(state: State):
412 answer = interrupt(
413 # This value will be sent to the client
414 # as part of the interrupt information.
415 \"what is your age?\"
416 )
417 print(f\"> Received an input from the interrupt: {answer}\")
418 return {\"human_value\": answer}
419
420
421 builder = StateGraph(State)
422 builder.add_node(\"node\", node)
423 builder.add_edge(START, \"node\")
424
425 # A checkpointer must be enabled for interrupts to work!
426 checkpointer = MemorySaver()
427 graph = builder.compile(checkpointer=checkpointer)
428
429 config = {
430 \"configurable\": {
431 \"thread_id\": uuid.uuid4(),
432 }
433 }
434
435 for chunk in graph.stream({\"foo\": \"abc\"}, config):
436 print(chunk)
437 ```
438
439 ```pycon
440 {'__interrupt__': (Interrupt(value='what is your age?', resumable=True, ns=['node:62e598fa-8653-9d6d-2046-a70203020e37'], when='during'),)}
441 ```
442
443 ```python
444 command = Command(resume=\"some input from a human!!!\")
445
446 for chunk in graph.stream(Command(resume=\"some input from a human!!!\"), config):
447 print(chunk)
448 ```
449
450 ```pycon
451 Received an input from the interrupt: some input from a human!!!
452 {'node': {'human_value': 'some input from a human!!!'}}
453 ```
454
455 Args:
456 value: The value to surface to the client when the graph is interrupted.
457
458 Returns:
459 Any: On subsequent invocations within the same node (same task to be precise), returns the value provided during the first invocation
460
461 Raises:
462 GraphInterrupt: On the first invocation within the node, halts execution and surfaces the provided value to the client.
463 """
464 from langgraph.constants import (
465 CONFIG_KEY_CHECKPOINT_NS,
466 CONFIG_KEY_SCRATCHPAD,
467 CONFIG_KEY_SEND,
468 NS_SEP,
469 RESUME,
470 )
471 from langgraph.errors import GraphInterrupt
472 from langgraph.utils.config import get_config
473
474 conf = get_config()["configurable"]
475 # track interrupt index
476 scratchpad: PregelScratchpad = conf[CONFIG_KEY_SCRATCHPAD]
477 idx = scratchpad.interrupt_counter()
478 # find previous resume values
479 if scratchpad.resume:
480 if idx < len(scratchpad.resume):
481 return scratchpad.resume[idx]
482 # find current resume value
483 if scratchpad.null_resume is not None:
484 assert len(scratchpad.resume) == idx, (scratchpad.resume, idx)
485 v = scratchpad.consume_null_resume()
486 scratchpad.resume.append(v)
487 conf[CONFIG_KEY_SEND]([(RESUME, scratchpad.resume)])
488 return v
489 # no resume value found
490 raise GraphInterrupt(
491 (
492 Interrupt(
493 value=value,
494 resumable=True,
495 ns=cast(str, conf[CONFIG_KEY_CHECKPOINT_NS]).split(NS_SEP),
496 ),
497 )
