219# Parse the args from the command line, then from the environment.
220# Args from the environment are only used if they are not already set from the command line.
221def parse_args():
222
223 # keep track of the switches we've seen so far
224 seen = set()

261def parse_args_helper(
262 args: list,
263 seenFromCommandLine: set,
264 seenFromEnvironment: set = set(),
265 isFromEnvironment=False,
266):
267 iterator = consume_args(args)
268
269 while True:
270 try:
271 arg = next(iterator)
272 except StopIteration:
273 break
274
275 switch = arg
276 if not switch.startswith("-"):
277 switch = ""
278 for pattern, placeholder, action in switches:
279 if re.match("^(" + pattern + ")$", switch):
280 break
281 else:
282 raise ValueError("unrecognized switch " + switch)
283
284 # if we're parsing from the command line, and we've already seen the switch on the command line, this is an error
285 if not isFromEnvironment and switch in seenFromCommandLine:
286 raise ValueError("duplicate switch on command line: " + switch)
287 # if we're parsing from the environment, and we've already seen the switch in the environment, this is an error
288 elif isFromEnvironment and switch in seenFromEnvironment:
289 raise ValueError("duplicate switch from environment: " + switch)
290 # if we're parsing from the environment, and we've already seen the switch on the command line, skip it, since command line takes precedence
291 elif isFromEnvironment and switch in seenFromCommandLine:
292 continue
293 # otherwise, the switch is new, so add it to the appropriate set
294 else:
295 if isFromEnvironment:
296 seenFromEnvironment.add(switch)
297 else:
298 seenFromCommandLine.add(switch)
299
300 # process the switch, running the corresponding action
301 try:
302 action(arg, iterator)
303 except StopIteration:
304 assert placeholder is not None
305 raise ValueError("{0}: missing {1}".format(switch, placeholder))
306 except Exception as exc:
307 raise ValueError("invalid {0} {1}: {2}".format(switch, placeholder, exc))
308
309 # If we're parsing the command line, we're done after we've processed the target
310 # Otherwise, we need to keep parsing until all args are consumed, since the target may be set from the command line
311 # already, but there might be additional args in the environment that we want to process.
312 if not isFromEnvironment and options.target is not None:
313 break

318 # because they use it to report the "process" event. Thus, we can't rely on
319 # run_path() and run_module() doing that, even though they will eventually.

327 if options.mode == "listen" and options.address is not None:

340def run_file():
341 target = options.target
342 start_debugging(target)

344 # run_path has one difference with invoking Python from command-line:
345 # if the target is a file (rather than a directory), it does not add its
346 # parent directory to sys.path. Thus, importing other modules from the
347 # same directory is broken unless sys.path is patched here.
348
349 if target is not None and os.path.isfile(target):
350 dir = os.path.dirname(target)
351 sys.path.insert(0, dir)
352 else:
353 log.debug("Not a file: {0!r}", target)
354
355 log.describe_environment("Pre-launch environment:")
356
357 log.info("Running file {0!r}", target)
358 runpy.run_path(target, run_name="__main__")

361def run_module():
362 # Add current directory to path, like Python itself does for -m. This must
363 # be in place before trying to use find_spec below to resolve submodules.
364 sys.path.insert(0, str(""))
365
366 # We want to do the same thing that run_module() would do here, without
367 # actually invoking it.
368 argv_0 = sys.argv[0]
369 try:
370 spec = None if options.target is None else find_spec(options.target)
371 if spec is not None:
372 argv_0 = spec.origin
373 except Exception:
374 log.swallow_exception("Error determining module path for sys.argv")
375
376 start_debugging(argv_0)
377 log.describe_environment("Pre-launch environment:")
378 log.info("Running module {0!r}", options.target)
379
380 # Docs say that runpy.run_module is equivalent to -m, but it's not actually
381 # the case for packages - -m sets __name__ to "__main__", but run_module sets
382 # it to "pkg.__main__". This breaks everything that uses the standard pattern
383 # __name__ == "__main__" to detect being run as a CLI app. On the other hand,
384 # runpy._run_module_as_main is a private function that actually implements -m.
385 try:
386 run_module_as_main = runpy._run_module_as_main
387 except AttributeError:
388 log.warning("runpy._run_module_as_main is missing, falling back to run_module.")
389 runpy.run_module(options.target, alter_sys=True)
390 else:
391 run_module_as_main(options.target, alter_argv=True)

410def attach_to_pid():
411 pid = options.target
412 log.info("Attaching to process with PID={0}", pid)
413
414 encode = lambda s: list(bytearray(s.encode("utf-8"))) if s is not None else None
415
416 script_dir = os.path.dirname(debugpy.server.__file__)
417 assert os.path.exists(script_dir)
418 script_dir = encode(script_dir)
419
420 setup = {
421 "mode": options.mode,
422 "address": options.address,
423 "wait_for_client": options.wait_for_client,
424 "log_to": options.log_to,
425 "adapter_access_token": options.adapter_access_token,
426 }
427 setup = encode(json.dumps(setup))

429 python_code = """
430import codecs;
431import json;
432import sys;
433
434decode = lambda s: codecs.utf_8_decode(bytearray(s))[0] if s is not None else None;
435
436script_dir = decode({script_dir});
437setup = json.loads(decode({setup}));
438
439sys.path.insert(0, script_dir);
440import attach_pid_injected;
441del sys.path[0];
442
443attach_pid_injected.attach(setup);
444"""
445 python_code = (
446 python_code.replace("\r", "")
447 .replace("\n", "")
448 .format(script_dir=script_dir, setup=setup)
449 )
450 log.info("Code to be injected: \n{0}", python_code.replace(";", ";\n"))
451
452 # pydevd restriction on characters in injected code.
453 assert not (
454 {'"', "'", "\r", "\n"} & set(python_code)

458 os.path.dirname(pydevd.__file__), "pydevd_attach_to_process"

467 log.info("Injecting code into process with PID={0} ...", pid)

473 )
474 except Exception:
