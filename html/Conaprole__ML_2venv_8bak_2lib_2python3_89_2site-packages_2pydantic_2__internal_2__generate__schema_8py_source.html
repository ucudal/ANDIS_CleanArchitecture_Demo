189def check_decorator_fields_exist(decorators: Iterable[AnyFieldDecorator], fields: Iterable[str]) -> None:
190 """Check if the defined fields in decorators exist in `fields` param.
191
192 It ignores the check for a decorator if the decorator has `*` as field or `check_fields=False`.
193
194 Args:
195 decorators: An iterable of decorators.
196 fields: An iterable of fields name.
197
198 Raises:
199 PydanticUserError: If one of the field names does not exist in `fields` param.
200 """
201 fields = set(fields)
202 for dec in decorators:
203 if '*' in dec.info.fields:
204 continue
205 if dec.info.check_fields is False:
206 continue
207 for field in dec.info.fields:
208 if field not in fields:
209 raise PydanticUserError(
210 f'Decorators defined with incorrect fields: {dec.cls_ref}.{dec.cls_var_name}'
211 " (use check_fields=False if you're inheriting from the model and intended this)",
212 code='decorator-missing-field',

222def apply_each_item_validators(
223 schema: core_schema.CoreSchema,
224 each_item_validators: list[Decorator[ValidatorDecoratorInfo]],
225 field_name: str | None,
226) -> core_schema.CoreSchema:
227 # This V1 compatibility shim should eventually be removed
228
229 # fail early if each_item_validators is empty
230 if not each_item_validators:
231 return schema
232
233 # push down any `each_item=True` validators
234 # note that this won't work for any Annotated types that get wrapped by a function validator
235 # but that's okay because that didn't exist in V1
236 if schema['type'] == 'nullable':
237 schema['schema'] = apply_each_item_validators(schema['schema'], each_item_validators, field_name)
238 return schema
239 elif schema['type'] == 'tuple':
240 if (variadic_item_index := schema.get('variadic_item_index')) is not None:
241 schema['items_schema'][variadic_item_index] = apply_validators(
242 schema['items_schema'][variadic_item_index],
243 each_item_validators,
244 field_name,
245 )
246 elif is_list_like_schema_with_items_schema(schema):
247 inner_schema = schema.get('items_schema', core_schema.any_schema())
248 schema['items_schema'] = apply_validators(inner_schema, each_item_validators, field_name)
249 elif schema['type'] == 'dict':
250 inner_schema = schema.get('values_schema', core_schema.any_schema())
251 schema['values_schema'] = apply_validators(inner_schema, each_item_validators, field_name)
252 else:
253 raise TypeError(
254 f"`@validator(..., each_item=True)` cannot be applied to fields with a schema of {schema['type']}"
255 )
256 return schema

275def _add_custom_serialization_from_json_encoders(
276 json_encoders: JsonEncoders | None, tp: Any, schema: CoreSchema
277) -> CoreSchema:
278 """Iterate over the json_encoders and add the first matching encoder to the schema.
279
280 Args:
281 json_encoders: A dictionary of types and their encoder functions.
282 tp: The type to check for a matching encoder.
283 schema: The schema to add the encoder to.
284 """
285 if not json_encoders:
286 return schema
287 if 'serialization' in schema:
288 return schema
289 # Check the class type and its superclasses for a matching encoder
290 # Decimal.__class__.__mro__ (and probably other cases) doesn't include Decimal itself
291 # if the type is a GenericAlias (e.g. from list[int]) we need to use __class__ instead of .__mro__
292 for base in (tp, *getattr(tp, '__mro__', tp.__class__.__mro__)[:-1]):
293 encoder = json_encoders.get(base)
294 if encoder is None:
295 continue
296
297 warnings.warn(
298 f'`json_encoders` is deprecated. See https://docs.pydantic.dev/{version_short()}/concepts/serialization/#custom-serializers for alternatives',
299 PydanticDeprecatedSince20,
300 )
301
302 # TODO: in theory we should check that the schema accepts a serialization key
303 schema['serialization'] = core_schema.plain_serializer_function_ser_schema(encoder, when_used='json')
304 return schema

306 return schema

575 def generate_schema(
576 self,
577 obj: Any,
578 from_dunder_get_core_schema: bool = True,
579 ) -> core_schema.CoreSchema:
580 """Generate core schema.
581
582 Args:
583 obj: The object to generate core schema for.
584 from_dunder_get_core_schema: Whether to generate schema from either the
585 `__get_pydantic_core_schema__` function or `__pydantic_core_schema__` property.
586
587 Returns:
588 The generated core schema.
589
590 Raises:
591 PydanticUndefinedAnnotation:
592 If it is not possible to evaluate forward reference.
593 PydanticSchemaGenerationError:
594 If it is not possible to generate pydantic-core schema.
595 TypeError:
596 - If `alias_generator` returns a disallowed type (must be str, AliasPath or AliasChoices).
597 - If V1 style validator with `each_item=True` applied on a wrong field.
598 PydanticUserError:
599 - If `typing.TypedDict` is used instead of `typing_extensions.TypedDict` on Python < 3.12.
600 - If `__modify_schema__` method is used instead of `__get_pydantic_json_schema__`.
601 """
602 schema: CoreSchema | None = None
603
604 if from_dunder_get_core_schema:
605 from_property = self._generate_schema_from_property(obj, obj)
606 if from_property is not None:
607 schema = from_property
608
609 if schema is None:
610 schema = self._generate_schema_inner(obj)
611
612 metadata_js_function = _extract_get_pydantic_json_schema(obj, schema)
613 if metadata_js_function is not None:
614 metadata_schema = resolve_original_schema(schema, self.defs.definitions)
615 if metadata_schema:
616 self._add_js_function(metadata_schema, metadata_js_function)
617
618 schema = _add_custom_serialization_from_json_encoders(self._config_wrapper.json_encoders, obj, schema)
619
620 return schema

622 def _model_schema(self, cls: type[BaseModel]) -> core_schema.CoreSchema:
623 """Generate schema for a Pydantic model."""
624 with self.defs.get_schema_or_ref(cls) as (model_ref, maybe_schema):
625 if maybe_schema is not None:
626 return maybe_schema
627
628 fields = getattr(cls, '__pydantic_fields__', {})
629 decorators = cls.__pydantic_decorators__
630 computed_fields = decorators.computed_fields

804 def _resolve_forward_ref(self, obj: Any) -> Any:
805 # we assume that types_namespace has the target of forward references in its scope,
806 # but this could fail, for example, if calling Validator on an imported type which contains
807 # forward references to other types only defined in the module from which it was imported
808 # `Validator(SomeImportedTypeAliasWithAForwardReference)`
809 # or the equivalent for BaseModel
810 # class Model(BaseModel):
811 # x: SomeImportedTypeAliasWithAForwardReference
812 try:
813 obj = _typing_extra.eval_type_backport(obj, *self._types_namespace)
814 except NameError as e:
815 raise PydanticUndefinedAnnotation.from_name_error(e) from e
816
817 # if obj is still a ForwardRef, it means we can't evaluate it, raise PydanticUndefinedAnnotation
818 if isinstance(obj, ForwardRef):
819 raise PydanticUndefinedAnnotation(obj.__forward_arg__, f'Unable to evaluate forward reference {obj}')
820
821 if self._typevars_map:
822 obj = replace_types(obj, self._typevars_map)
823
824 return obj

997 def _match_generic_type(self, obj: Any, origin: Any) -> CoreSchema: # noqa: C901
998 # Need to handle generic dataclasses before looking for the schema properties because attribute accesses
999 # on _GenericAlias delegate to the origin type, so lose the information about the concrete parametrization
1000 # As a result, currently, there is no way to cache the schema for generic dataclasses. This may be possible
1001 # to resolve by modifying the value returned by `Generic.__class_getitem__`, but that is a dangerous game.
1002 if dataclasses.is_dataclass(origin):
1003 return self._dataclass_schema(obj, origin) # pyright: ignore[reportArgumentType]
1004 if _typing_extra.is_namedtuple(origin):
1005 return self._namedtuple_schema(obj, origin)
1006
1007 from_property = self._generate_schema_from_property(origin, obj)
1008 if from_property is not None:
1009 return from_property
1010
1011 if _typing_extra.is_type_alias_type(origin):
1012 return self._type_alias_type_schema(obj)

1014 return self._union_schema(obj)

1024 return self._dict_schema(*self._get_first_two_args_or_any(obj))

1029 elif origin in SEQUENCE_TYPES:
1030 return self._sequence_schema(self._get_first_arg_or_any(obj))
