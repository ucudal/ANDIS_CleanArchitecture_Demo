617 def _span_dimensions(self, other_tc):
618 """
619 Return a (top, left, height, width) 4-tuple specifying the extents of
620 the merged cell formed by using this tc and *other_tc* as opposite
621 corner extents.
622 """
623 def raise_on_inverted_L(a, b):
624 if a.top == b.top and a.bottom != b.bottom:
625 raise InvalidSpanError('requested span not rectangular')
626 if a.left == b.left and a.right != b.right:
627 raise InvalidSpanError('requested span not rectangular')
628
629 def raise_on_tee_shaped(a, b):
630 top_most, other = (a, b) if a.top < b.top else (b, a)
631 if top_most.top < other.top and top_most.bottom > other.bottom:
632 raise InvalidSpanError('requested span not rectangular')
633
634 left_most, other = (a, b) if a.left < b.left else (b, a)
635 if left_most.left < other.left and left_most.right > other.right:
636 raise InvalidSpanError('requested span not rectangular')
637
638 raise_on_inverted_L(self, other_tc)
639 raise_on_tee_shaped(self, other_tc)
640
641 top = min(self.top, other_tc.top)
642 left = min(self.left, other_tc.left)
643 bottom = max(self.bottom, other_tc.bottom)
644 right = max(self.right, other_tc.right)
645
646 return top, left, bottom - top, right - left

648 def _span_to_width(self, grid_width, top_tc, vMerge):
649 """
650 Incorporate and then remove `w:tc` elements to the right of this one
651 until this cell spans *grid_width*. Raises |ValueError| if
652 *grid_width* cannot be exactly achieved, such as when a merged cell
653 would drive the span width greater than *grid_width* or if not enough
654 grid columns are available to make this cell that wide. All content
655 from incorporated cells is appended to *top_tc*. The val attribute of
656 the vMerge element on the single remaining cell is set to *vMerge*.
657 If *vMerge* is |None|, the vMerge element is removed if present.
658 """
659 self._move_content_to(top_tc)
660 while self.grid_span < grid_width:
661 self._swallow_next_tc(grid_width, top_tc)
662 self.vMerge = vMerge
663
664 def _swallow_next_tc(self, grid_width, top_tc):
665 """
666 Extend the horizontal span of this `w:tc` element to incorporate the
667 following `w:tc` element in the row and then delete that following
668 `w:tc` element. Any content in the following `w:tc` element is
669 appended to the content of *top_tc*. The width of the following
670 `w:tc` element is added to this one, if present. Raises
671 |InvalidSpanError| if the width of the resulting cell is greater than
672 *grid_width* or if there is no next `<w:tc>` element in the row.
673 """
674 def raise_on_invalid_swallow(next_tc):
675 if next_tc is None:
676 raise InvalidSpanError('not enough grid columns')
677 if self.grid_span + next_tc.grid_span > grid_width:
678 raise InvalidSpanError('span is not rectangular')
679
680 next_tc = self._next_tc
681 raise_on_invalid_swallow(next_tc)
682 next_tc._move_content_to(top_tc)
683 self._add_width_of(next_tc)
684 self.grid_span += next_tc.grid_span
685 next_tc._remove()
