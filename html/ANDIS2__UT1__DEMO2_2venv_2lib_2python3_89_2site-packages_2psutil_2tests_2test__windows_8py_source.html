74def wmic(path, what, converter=int):
75 """Currently not used, but available just in case. Usage:
76
77 >>> wmic("Win32_OperatingSystem", "FreePhysicalMemory")
78 2134124534
79 """
80 out = sh(f"wmic path {path} get {what}").strip()
81 data = "".join(out.splitlines()[1:]).strip() # get rid of the header
82 if converter is not None:
83 if "," in what:
84 return tuple(converter(x) for x in data.split())
85 else:
86 return converter(data)
87 else:
88 return data

642@pytest.mark.skipif(not WINDOWS, reason="WINDOWS only")
643class TestDualProcessImplementation(PsutilTestCase):
644 """Certain APIs on Windows have 2 internal implementations, one
645 based on documented Windows APIs, another one based
646 NtQuerySystemInformation() which gets called as fallback in
