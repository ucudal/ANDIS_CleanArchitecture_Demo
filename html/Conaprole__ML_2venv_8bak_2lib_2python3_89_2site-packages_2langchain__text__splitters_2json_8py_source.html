77 def _json_split(
78 self,
79 data: Dict[str, Any],

81 chunks: Optional[List[Dict]] = None,
82 ) -> List[Dict]:
83 """Split json into maximum size dictionaries while preserving structure."""
84 current_path = current_path or []
85 chunks = chunks if chunks is not None else [{}]
86 if isinstance(data, dict):
87 for key, value in data.items():
88 new_path = current_path + [key]
89 chunk_size = self._json_size(chunks[-1])
90 size = self._json_size({key: value})
91 remaining = self.max_chunk_size - chunk_size
92
93 if size < remaining:
94 # Add item to current chunk
95 self._set_nested_dict(chunks[-1], new_path, value)
96 else:
97 if chunk_size >= self.min_chunk_size:
98 # Chunk is big enough, start a new chunk
99 chunks.append({})
100
101 # Iterate
102 self._json_split(value, new_path, chunks)
103 else:
104 # handle single item
105 self._set_nested_dict(chunks[-1], current_path, data)
106 return chunks
