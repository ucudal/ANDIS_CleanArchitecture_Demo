490def start_client(host, port):
491 """connects to a host/port"""
492 pydev_log.info("Connecting to %s:%s", host, port)
493
494 address_family = AF_INET
495 for res in socket_module.getaddrinfo(host, port, 0, SOCK_STREAM):
496 if res[0] == AF_INET:
497 address_family = res[0]
498 # Prefer IPv4 addresses for backward compat.
499 break

501 # Don't break after this - if the socket is dual-stack prefer IPv4.
502 address_family = res[0]
503

505
506 # Set TCP keepalive on an open socket.
507 # It activates after 1 second (TCP_KEEPIDLE,) of idleness,
508 # then sends a keepalive ping once every 3 seconds (TCP_KEEPINTVL),
509 # and closes the connection after 5 failed ping (TCP_KEEPCNT), or 15 seconds
510 try:
511 s.setsockopt(SOL_SOCKET, socket_module.SO_KEEPALIVE, 1)
512 except (AttributeError, OSError):
513 pass # May not be available everywhere.
514 try:
515 s.setsockopt(socket_module.IPPROTO_TCP, socket_module.TCP_KEEPIDLE, 1)
516 except (AttributeError, OSError):
517 pass # May not be available everywhere.
518 try:
519 s.setsockopt(socket_module.IPPROTO_TCP, socket_module.TCP_KEEPINTVL, 3)
520 except (AttributeError, OSError):
521 pass # May not be available everywhere.
522 try:
523 s.setsockopt(socket_module.IPPROTO_TCP, socket_module.TCP_KEEPCNT, 5)
524 except (AttributeError, OSError):
525 pass # May not be available everywhere.
526
527 try:
528 # 10 seconds default timeout
529 timeout = int(os.environ.get("PYDEVD_CONNECT_TIMEOUT", 10))
530 s.settimeout(timeout)
531 s.connect((host, port))
532 s.settimeout(None) # no timeout after connected
533 pydev_log.info(f"Connected to: {s}.")
534 return s
535 except:
536 pydev_log.exception("Could not connect to %s: %s", host, port)

544class InternalThreadCommand(object):
545 """internal commands are generated/executed by the debugger.
546
547 The reason for their existence is that some commands have to be executed
548 on specific threads. These are the InternalThreadCommands that get
549 get posted to PyDB.

552 def __init__(self, thread_id, method=None, *args, **kwargs):
553 self.thread_id = thread_id
554 self.method = method
555 self.args = args
556 self.kwargs = kwargs

559 """By default, it must be in the same thread to be executed"""
560 return self.thread_id == thread_id or self.thread_id.endswith("|" + thread_id)

562 def do_it(self, dbg):
563 try:
564 if self.method is not None:
565 self.method(dbg, *self.args, **self.kwargs)
566 else:
567 raise NotImplementedError("you have to override do_it")
568 finally:
569 self.args = None
570 self.kwargs = None
571
572 def __str__(self):
573 return "InternalThreadCommands(%s, %s, %s)" % (self.method, self.args, self.kwargs)

1452def build_exception_info_response(dbg, thread_id, thread, request_seq, set_additional_thread_info, iter_visible_frames_info, max_frames):
1453 """
1454 :return ExceptionInfoResponse
1455 """
1456 additional_info = set_additional_thread_info(thread)
1457 topmost_frame = additional_info.get_topmost_frame(thread)
1458
1459 current_paused_frame_name = ""
1460
1461 source_path = "" # This is an extra bit of data used by Visual Studio
1462 stack_str_lst = []
1463 name = None
1464 description = None
1465
1466 if topmost_frame is not None:
1467 try:
1468 try:
1469 frames_list = dbg.suspended_frames_manager.get_frames_list(thread_id)
1470 while frames_list is not None and len(frames_list):
1471 frames = []
1472
1473 frame = None
1474
1475 if not name:
1476 exc_type = frames_list.exc_type
1477 if exc_type is not None:
1478 try:
1479 name = exc_type.__qualname__
1480 except:
1481 try:
1482 name = exc_type.__name__
1483 except:
1484 try:
1485 name = str(exc_type)
1486 except:
1487 pass
1488
1489 if not description:
1490 exc_desc = frames_list.exc_desc
1491 if exc_desc is not None:
1492 try:
1493 description = str(exc_desc)
1494 except:
1495 pass
1496
1497 for (
1498 frame_id,
1499 frame,
1500 method_name,
1501 original_filename,
1502 filename_in_utf8,
1503 lineno,
1504 _applied_mapping,
1505 show_as_current_frame,
1506 line_col_info,
1507 ) in iter_visible_frames_info(dbg, frames_list):
1508 line_text = linecache.getline(original_filename, lineno)
1509
1510 # Never filter out plugin frames!
1511 if not getattr(frame, "IS_PLUGIN_FRAME", False):
1512 if dbg.is_files_filter_enabled and dbg.apply_files_filter(frame, original_filename, False):
1513 continue
1514
1515 if show_as_current_frame:
1516 current_paused_frame_name = method_name
1517 method_name += " (Current frame)"
1518 frames.append((filename_in_utf8, lineno, method_name, line_text, line_col_info))
1519
1520 if not source_path and frames:
1521 source_path = frames[0][0]
1522
1523 if IS_PY311_OR_GREATER:
1524 stack_summary = traceback.StackSummary()
1525 for filename_in_utf8, lineno, method_name, line_text, line_col_info in frames[-max_frames:]:

1528 if line_col_info.end_lineno is not None and lineno < line_col_info.end_lineno:
1529 line_text = "\n".join(linecache.getlines(filename_in_utf8)[lineno : line_col_info.end_lineno + 1])

1531 filename_in_utf8,

1533 method_name,

1536 colno=line_col_info.colno,

1541 stack_summary.append(frame_summary)
1542
1543 stack_str = "".join(stack_summary.format())
1544
1545 else:
1546 # Note: remove col info (just used in 3.11).
1547 stack_str = "".join(traceback.format_list((x[:-1] for x in frames[-max_frames:])))
1548
1549 try:
1550 stype = frames_list.exc_type.__qualname__
1551 smod = frames_list.exc_type.__module__
1552 if smod not in ("__main__", "builtins"):
1553 if not isinstance(smod, str):
1554 smod = "<unknown>"
1555 stype = smod + "." + stype
1556 except Exception:
1557 stype = "<unable to get exception type>"
1558 pydev_log.exception("Error getting exception type.")
1559
1560 stack_str += "%s: %s\n" % (stype, frames_list.exc_desc)
1561 stack_str += frames_list.exc_context_msg
1562 stack_str_lst.append(stack_str)
1563
1564 frames_list = frames_list.chained_frames_list
1565 if frames_list is None or not frames_list:
1566 break
1567
1568 except:
1569 pydev_log.exception("Error on build_exception_info_response.")
1570 finally:
1571 topmost_frame = None
1572 full_stack_str = "".join(reversed(stack_str_lst))
1573
1574 if not name:
1575 name = "exception: type unknown"
1576 if not description:
1577 description = "exception: no description"
1578
1579 if current_paused_frame_name:
1580 name += " (note: full exception trace is shown but execution is paused at: %s)" % (current_paused_frame_name,)
1581
1582 if thread.stop_reason == CMD_STEP_CAUGHT_EXCEPTION:
1583 break_mode = pydevd_schema.ExceptionBreakMode.ALWAYS
1584 else:
1585 break_mode = pydevd_schema.ExceptionBreakMode.UNHANDLED
1586
1587 response = pydevd_schema.ExceptionInfoResponse(
1588 request_seq=request_seq,
1589 success=True,
1590 command="exceptionInfo",
1591 body=pydevd_schema.ExceptionInfoResponseBody(
1592 exceptionId=name,
1593 description=description,
1594 breakMode=break_mode,
1595 details=pydevd_schema.ExceptionDetails(
1596 message=description,
1597 typeName=name,
1598 stackTrace=full_stack_str,
1599 source=source_path,
1600 # Note: ExceptionDetails actually accepts an 'innerException', but
1601 # when passing it, VSCode is not showing the stack trace at all.
1602 ),
1603 ),
1604 )
1605 return response
