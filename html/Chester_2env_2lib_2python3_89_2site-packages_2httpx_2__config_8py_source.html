98 def load_ssl_context_verify(self) -> ssl.SSLContext:
99 """
100 Return an SSL context for verified connections.
101 """
102 if self.trust_env and self.verify is True:
103 ca_bundle = get_ca_bundle_from_env()
104 if ca_bundle is not None:
105 self.verify = ca_bundle # type: ignore
106
107 if isinstance(self.verify, ssl.SSLContext):
108 # Allow passing in our own SSLContext object that's pre-configured.
109 context = self.verify
110 self._load_client_certs(context)
111 return context
112 elif isinstance(self.verify, bool):
113 ca_bundle_path = self.DEFAULT_CA_BUNDLE_PATH
114 elif Path(self.verify).exists():
115 ca_bundle_path = Path(self.verify)
116 else:
117 raise IOError(
118 "Could not find a suitable TLS CA certificate bundle, "
119 "invalid path: {}".format(self.verify)
120 )
121
122 context = self._create_default_ssl_context()
123 context.verify_mode = ssl.CERT_REQUIRED
124 context.check_hostname = True
125
126 # Signal to server support for PHA in TLS 1.3. Raises an
127 # AttributeError if only read-only access is implemented.
128 try:
129 context.post_handshake_auth = True # type: ignore
130 except AttributeError: # pragma: nocover
131 pass
132
133 # Disable using 'commonName' for SSLContext.check_hostname
134 # when the 'subjectAltName' extension isn't available.
135 try:
136 context.hostname_checks_common_name = False # type: ignore
137 except AttributeError: # pragma: nocover
138 pass
139
140 if ca_bundle_path.is_file():
141 logger.trace(f"load_verify_locations cafile={ca_bundle_path!s}")
142 context.load_verify_locations(cafile=str(ca_bundle_path))
143 elif ca_bundle_path.is_dir():
144 logger.trace(f"load_verify_locations capath={ca_bundle_path!s}")
145 context.load_verify_locations(capath=str(ca_bundle_path))
146
147 self._load_client_certs(context)
148
149 return context

151 def _create_default_ssl_context(self) -> ssl.SSLContext:
152 """
153 Creates the default SSLContext object that's used for both verified
154 and unverified connections.
155 """
156 context = ssl.SSLContext(ssl.PROTOCOL_TLS)
157 context.options |= ssl.OP_NO_SSLv2
158 context.options |= ssl.OP_NO_SSLv3
159 context.options |= ssl.OP_NO_TLSv1
160 context.options |= ssl.OP_NO_TLSv1_1
161 context.options |= ssl.OP_NO_COMPRESSION
162 context.set_ciphers(DEFAULT_CIPHERS)
163
164 if ssl.HAS_ALPN:
165 alpn_idents = ["http/1.1", "h2"] if self.http2 else ["http/1.1"]
166 context.set_alpn_protocols(alpn_idents)
167
168 if hasattr(context, "keylog_filename"): # pragma: nocover (Available in 3.8+)
169 keylogfile = os.environ.get("SSLKEYLOGFILE")
170 if keylogfile and self.trust_env:
171 context.keylog_filename = keylogfile # type: ignore
172
173 return context

175 def _load_client_certs(self, ssl_context: ssl.SSLContext) -> None:
176 """
177 Loads client certificates into our SSLContext object
178 """
179 if self.cert is not None:
180 if isinstance(self.cert, str):
181 ssl_context.load_cert_chain(certfile=self.cert)
182 elif isinstance(self.cert, tuple) and len(self.cert) == 2:
183 ssl_context.load_cert_chain(certfile=self.cert[0], keyfile=self.cert[1])
184 elif isinstance(self.cert, tuple) and len(self.cert) == 3:
185 ssl_context.load_cert_chain(
186 certfile=self.cert[0],
187 keyfile=self.cert[1],
188 password=self.cert[2], # type: ignore
189 )

332 def __init__(
333 self, url: URLTypes, *, headers: HeaderTypes = None, mode: str = "DEFAULT",
334 ):
335 url = URL(url)
336 headers = Headers(headers)
337
338 if url.scheme not in ("http", "https"):
339 raise ValueError(f"Unknown scheme for proxy URL {url!r}")
340 if mode not in ("DEFAULT", "FORWARD_ONLY", "TUNNEL_ONLY"):
341 raise ValueError(f"Unknown proxy mode {mode!r}")
342
343 if url.username or url.password:
344 headers.setdefault(
345 "Proxy-Authorization",
346 self.build_auth_header(url.username, url.password),
347 )
348 # Remove userinfo from the URL authority, e.g.:
349 # 'username:password@proxy_host:proxy_port' -> 'proxy_host:proxy_port'
350 url = url.copy_with(username=None, password=None)
351
352 self.url = url
353 self.headers = headers
354 self.mode = mode

356 def build_auth_header(self, username: str, password: str) -> str:
357 userpass = (username.encode("utf-8"), password.encode("utf-8"))
358 token = b64encode(b":".join(userpass)).decode().strip()
359 return f"Basic {token}"
