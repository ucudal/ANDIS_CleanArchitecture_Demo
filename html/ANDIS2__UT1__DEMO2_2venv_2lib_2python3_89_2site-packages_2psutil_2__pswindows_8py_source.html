207def virtual_memory():
208 """System virtual memory as a namedtuple."""
209 mem = cext.virtual_mem()
210 totphys, availphys, _totsys, _availsys = mem
211 total = totphys
212 avail = availphys
213 free = availphys
214 used = total - avail
215 percent = usage_percent((total - avail), total, round_=1)
216 return svmem(total, avail, percent, used, free)
217
218
219def swap_memory():
220 """Swap system memory as a (total, used, free, sin, sout) tuple."""
221 mem = cext.virtual_mem()
222
223 total_phys = mem[0]
224 total_system = mem[2]
225
226 # system memory (commit total/limit) is the sum of physical and swap
227 # thus physical memory values need to be subtracted to get swap values
228 total = total_system - total_phys
229 # commit total is incremented immediately (decrementing free_system)
230 # while the corresponding free physical value is not decremented until
231 # pages are accessed, so we can't use free system memory for swap.
232 # instead, we calculate page file usage based on performance counter
233 if total > 0:
234 percentswap = cext.swap_percent()
235 used = int(0.01 * percentswap * total)
236 else:
237 percentswap = 0.0
238 used = 0
239
240 free = total - used
241 percent = round(percentswap, 1)
242 return _common.sswap(total, used, free, percent, 0, 0)

400def sensors_battery():
401 """Return battery information."""
402 # For constants meaning see:
403 # https://msdn.microsoft.com/en-us/library/windows/desktop/
404 # aa373232(v=vs.85).aspx
405 acline_status, flags, percent, secsleft = cext.sensors_battery()
406 power_plugged = acline_status == 1
407 no_battery = bool(flags & 128)
408 charging = bool(flags & 8)
409
410 if no_battery:
411 return None
412 if power_plugged or charging:
413 secsleft = _common.POWER_TIME_UNLIMITED
414 elif secsleft == -1:
415 secsleft = _common.POWER_TIME_UNKNOWN
416
417 return _common.sbattery(percent, secsleft, power_plugged)

515 @contextlib.contextmanager
516 def _wrap_exceptions(self):
517 """Ctx manager which translates bare OSError and WindowsError
518 exceptions into NoSuchProcess and AccessDenied.
519 """
520 try:
521 yield
522 except OSError as err:
523 name = self._name
524 if is_permission_err(err):
525 msg = (
526 f"service {name!r} is not querable (not enough privileges)"
527 )
528 raise AccessDenied(pid=None, name=name, msg=msg) from err
529 elif err.winerror in {
530 cext.ERROR_INVALID_NAME,
531 cext.ERROR_SERVICE_DOES_NOT_EXIST,
532 }:
533 msg = f"service {name!r} does not exist"
534 raise NoSuchProcess(pid=None, name=name, msg=msg) from err
535 else:
536 raise

648def is_permission_err(exc):
649 """Return True if this is a permission error."""
650 assert isinstance(exc, OSError), exc

657def convert_oserror(exc, pid=None, name=None):
658 """Convert OSError into NoSuchProcess or AccessDenied."""
659 assert isinstance(exc, OSError), exc
660 if is_permission_err(exc):
661 return AccessDenied(pid=pid, name=name)
662 if isinstance(exc, ProcessLookupError):
663 return NoSuchProcess(pid=pid, name=name)
664 raise exc

667def wrap_exceptions(fun):
668 """Decorator which converts OSError into NoSuchProcess or AccessDenied."""
669
670 @functools.wraps(fun)
671 def wrapper(self, *args, **kwargs):
672 try:
673 return fun(self, *args, **kwargs)
674 except OSError as err:
675 raise convert_oserror(err, pid=self.pid, name=self._name) from err
676
677 return wrapper

680def retry_error_partial_copy(fun):
681 """Workaround for https://github.com/giampaolo/psutil/issues/875.
682 See: https://stackoverflow.com/questions/4457745#4457745.
683 """
684
685 @functools.wraps(fun)
686 def wrapper(self, *args, **kwargs):
687 delay = 0.0001
688 times = 33
689 for _ in range(times): # retries for roughly 1 second
690 try:
691 return fun(self, *args, **kwargs)
692 except OSError as _:
693 err = _
694 if err.winerror == ERROR_PARTIAL_COPY:
695 time.sleep(delay)
696 delay = min(delay * 2, 0.04)
697 continue
698 raise
699 msg = (
700 f"{fun} retried {times} times, converted to AccessDenied as it's "
701 f"still returning {err}"
702 )
703 raise AccessDenied(pid=self.pid, name=self._name, msg=msg)
704
705 return wrapper

720 def oneshot_enter(self):

725 self._proc_info.cache_deactivate(self)

728 @memoize_when_activated
729 def _proc_info(self):
730 """Return multiple information about this process as a
731 raw tuple.
732 """
733 ret = cext.proc_info(self.pid)
734 assert len(ret) == len(pinfo_map)
735 return ret
736
737 def name(self):
738 """Return process name, which on Windows is always the final
739 part of the executable.
740 """
741 # This is how PIDs 0 and 4 are always represented in taskmgr
742 # and process-hacker.
743 if self.pid == 0:
744 return "System Idle Process"
745 if self.pid == 4:
746 return "System"
747 return os.path.basename(self.exe())
748
749 @wrap_exceptions
750 @memoize_when_activated
751 def exe(self):
752 if PYPY:
753 try:
754 exe = cext.proc_exe(self.pid)
755 except OSError as err:
756 # 24 = ERROR_TOO_MANY_OPEN_FILES. Not sure why this happens
757 # (perhaps PyPy's JIT delaying garbage collection of files?).
758 if err.errno == 24:
759 debug(f"{err!r} translated into AccessDenied")
760 raise AccessDenied(self.pid, self._name) from err
761 raise

866 @wrap_exceptions
867 def wait(self, timeout=None):
868 if timeout is None:
869 cext_timeout = cext.INFINITE
870 else:
871 # WaitForSingleObject() expects time in milliseconds.
872 cext_timeout = int(timeout * 1000)
873
874 timer = getattr(time, 'monotonic', time.time)
875 stop_at = timer() + timeout if timeout is not None else None
876
877 try:
878 # Exit code is supposed to come from GetExitCodeProcess().
879 # May also be None if OpenProcess() failed with
880 # ERROR_INVALID_PARAMETER, meaning PID is already gone.
881 exit_code = cext.proc_wait(self.pid, cext_timeout)
882 except cext.TimeoutExpired as err:
883 # WaitForSingleObject() returned WAIT_TIMEOUT. Just raise.
884 raise TimeoutExpired(timeout, self.pid, self._name) from err
885 except cext.TimeoutAbandoned:
886 # WaitForSingleObject() returned WAIT_ABANDONED, see:
887 # https://github.com/giampaolo/psutil/issues/1224
888 # We'll just rely on the internal polling and return None
889 # when the PID disappears. Subprocess module does the same
890 # (return None):
891 # https://github.com/python/cpython/blob/
892 # be50a7b627d0aa37e08fa8e2d5568891f19903ce/
893 # Lib/subprocess.py#L1193-L1194
894 exit_code = None
895
896 # At this point WaitForSingleObject() returned WAIT_OBJECT_0,
897 # meaning the process is gone. Stupidly there are cases where
898 # its PID may still stick around so we do a further internal
899 # polling.
900 delay = 0.0001
901 while True:
902 if not pid_exists(self.pid):
903 return exit_code
904 if stop_at and timer() >= stop_at:
905 raise TimeoutExpired(timeout, pid=self.pid, name=self._name)
906 time.sleep(delay)
907 delay = min(delay * 2, 0.04) # incremental delay
