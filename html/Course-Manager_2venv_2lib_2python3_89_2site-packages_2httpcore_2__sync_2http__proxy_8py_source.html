14def merge_headers(
15 default_headers: Headers = None, override_headers: Headers = None
16) -> Headers:
17 """
18 Append default_headers and override_headers, de-duplicating if a key existing in both cases.
19 """
20 default_headers = [] if default_headers is None else default_headers
21 override_headers = [] if override_headers is None else override_headers
22 has_override = set([key.lower() for key, value in override_headers])
23 default_headers = [
24 (key, value)
25 for key, value in default_headers
26 if key.lower() not in has_override
27 ]
28 return default_headers + override_headers

115 self,
116 method: bytes,
117 url: URL,
118 headers: Headers = None,
119 stream: SyncByteStream = None,
120 timeout: TimeoutDict = None,
121 ) -> Tuple[bytes, int, bytes, Headers, SyncByteStream]:
122 """
123 Forwarded proxy requests include the entire URL as the HTTP target,
124 rather than just the path.
125 """
126 origin = self.proxy_origin
127 connection = self._get_connection_from_pool(origin)
128
129 if connection is None:
130 connection = SyncHTTPConnection(
131 origin=origin, http2=self._http2, ssl_context=self._ssl_context,
132 )
133 self._add_to_pool(connection)
134
135 # Issue a forwarded proxy request...
136
137 # GET https://www.example.org/path HTTP/1.1
138 # [proxy headers]
139 # [headers]
140 scheme, host, port, path = url
141 if port is None:
142 target = b"%b://%b%b" % (scheme, host, path)
143 else:
144 target = b"%b://%b:%d%b" % (scheme, host, port, path)

151 )
152 wrapped_stream = ResponseByteStream(

157 def _tunnel_request(
158 self,
159 method: bytes,
160 url: URL,
161 headers: Headers = None,
162 stream: SyncByteStream = None,
163 timeout: TimeoutDict = None,
164 ) -> Tuple[bytes, int, bytes, Headers, SyncByteStream]:
165 """
166 Tunnelled proxy requests require an initial CONNECT request to
167 establish the connection, and then send regular requests.
168 """
169 origin = url_to_origin(url)
170 connection = self._get_connection_from_pool(origin)
171
172 if connection is None:
173 # First, create a connection to the proxy server
174 proxy_connection = SyncHTTPConnection(
175 origin=self.proxy_origin,
176 http2=self._http2,
177 ssl_context=self._ssl_context,
178 )
179
180 # Issue a CONNECT request...
181
182 # CONNECT www.example.org:80 HTTP/1.1
183 # [proxy-headers]
184 if url[2] is None:
185 target = url[1]
186 else:
187 target = b"%b:%d" % (url[1], url[2])
188 connect_url = self.proxy_origin + (target,)
189 connect_headers = [(b"Host", target), (b"Accept", b"*/*")]
190 connect_headers = merge_headers(connect_headers, self.proxy_headers)
191 proxy_response = proxy_connection.request(
192 b"CONNECT", connect_url, headers=connect_headers, timeout=timeout
193 )
194 proxy_status_code = proxy_response[1]
195 proxy_reason_phrase = proxy_response[2]
196 proxy_stream = proxy_response[4]
197 logger.trace(
198 "tunnel_response proxy_status_code=%r proxy_reason=%r ",
199 proxy_status_code,
200 proxy_reason_phrase,
201 )
202 # Read the response data without closing the socket
203 for _ in proxy_stream:
204 pass
205
206 # See if the tunnel was successfully established.
207 if proxy_status_code < 200 or proxy_status_code > 299:
208 msg = "%d %s" % (proxy_status_code, proxy_reason_phrase.decode("ascii"))
209 raise ProxyError(msg)
210
211 # Upgrade to TLS if required
212 # We assume the target speaks TLS on the specified port
213 if url[0] == b"https":
214 proxy_connection.start_tls(url[1], timeout)
215
216 # The CONNECT request is successful, so we have now SWITCHED PROTOCOLS.
217 # This means the proxy connection is now unusable, and we must create
218 # a new one for regular requests, making sure to use the same socket to
219 # retain the tunnel.
220 connection = SyncHTTPConnection(
221 origin=origin,
222 http2=self._http2,
223 ssl_context=self._ssl_context,
224 socket=proxy_connection.socket,
225 )
226 self._add_to_pool(connection)
227
228 # Once the connection has been established we can send requests on
229 # it as normal.
230 response = connection.request(
231 method, url, headers=headers, stream=stream, timeout=timeout,
232 )
233 wrapped_stream = ResponseByteStream(
234 response[4], connection=connection, callback=self._response_closed
235 )
236 return response[0], response[1], response[2], response[3], wrapped_stream
