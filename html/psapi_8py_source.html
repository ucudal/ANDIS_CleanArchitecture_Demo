110 ProcessIds = (DWORD * (size // unit))()
111 cbBytesReturned.value = size
112 _EnumProcesses(byref(ProcessIds), cbBytesReturned, byref(cbBytesReturned))
113 returned = cbBytesReturned.value
114 if returned < size:
115 break
