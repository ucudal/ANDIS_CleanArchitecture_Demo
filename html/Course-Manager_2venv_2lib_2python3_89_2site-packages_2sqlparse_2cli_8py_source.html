10"""Module that contains the command line app.
11
12Why does this file exist, and why not put this in __main__?
13 You might be tempted to import things from __main__ later, but that will
14 cause problems: the code will get executed twice:
15 - When you run `python -m sqlparse` python will execute
16 ``__main__.py`` as a script. That means there won't be any
17 ``sqlparse.__main__`` in ``sys.modules``.
18 - When you import __main__ it will get executed again (as a module) because
19 there's no ``sqlparse.__main__`` in ``sys.modules``.
20 Also see (1) from http://click.pocoo.org/5/setuptools/#setuptools-integration
21"""

160def main(args=None):
161 parser = create_parser()
162 args = parser.parse_args(args)
163
164 if args.filename == '-': # read from stdin
165 if PY2:
166 data = getreader(args.encoding)(sys.stdin).read()
167 else:
168 wrapper = TextIOWrapper(sys.stdin.buffer, encoding=args.encoding)
169 try:
170 data = wrapper.read()
171 finally:
172 wrapper.detach()
173 else:
174 try:
175 with open(args.filename, 'r', args.encoding) as f:
176 data = ''.join(f.readlines())
177 except IOError as e:
178 return _error(
179 u'Failed to read {0}: {1}'.format(args.filename, e))
180
181 close_stream = False
182 if args.outfile:
183 try:
184 stream = open(args.outfile, 'w', args.encoding)
185 close_stream = True
186 except IOError as e:
187 return _error(u'Failed to open {0}: {1}'.format(args.outfile, e))
188 else:
189 stream = sys.stdout
190
191 formatter_opts = vars(args)
192 try:
193 formatter_opts = sqlparse.formatter.validate_options(formatter_opts)
194 except SQLParseError as e:
195 return _error(u'Invalid options: {0}'.format(e))
196
197 s = sqlparse.format(data, **formatter_opts)
198 stream.write(s)
199 stream.flush()
200 if close_stream:
201 stream.close()
202 return 0
