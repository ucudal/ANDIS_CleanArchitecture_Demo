125def stop_on_unhandled_exception(py_db, thread, additional_info, arg):
126 exctype, value, tb = arg
127 break_on_uncaught_exceptions = py_db.break_on_uncaught_exceptions
128 if break_on_uncaught_exceptions:
129 exception_breakpoint = py_db.get_exception_breakpoint(exctype, break_on_uncaught_exceptions)
130 else:
131 exception_breakpoint = None
132
133 if not exception_breakpoint:
134 return
135
136 if tb is None: # sometimes it can be None, e.g. with GTK
137 return
138
139 if exctype is KeyboardInterrupt:
140 return
141
142 if exctype is SystemExit and py_db.ignore_system_exit_code(value):
143 return
144
145 frames = []
146 user_frame = None
147
148 while tb is not None:
149 if not py_db.exclude_exception_by_filter(exception_breakpoint, tb):
150 user_frame = tb.tb_frame
151 frames.append(tb.tb_frame)
152 tb = tb.tb_next
153
154 if user_frame is None:
155 return
156
157 frames_byid = dict([(id(frame), frame) for frame in frames])
158 add_exception_to_frame(user_frame, arg)
159 if exception_breakpoint.condition is not None:
160 eval_result = py_db.handle_breakpoint_condition(additional_info, exception_breakpoint, user_frame)
161 if not eval_result:
162 return
163
164 if exception_breakpoint.expression is not None:
165 py_db.handle_breakpoint_expression(exception_breakpoint, additional_info, user_frame)
166
167 try:
168 additional_info.pydev_message = exception_breakpoint.qname
169 except:
170 additional_info.pydev_message = exception_breakpoint.qname.encode("utf-8")
171
172 pydev_log.debug("Handling post-mortem stop on exception breakpoint %s" % (exception_breakpoint.qname,))
173
174 py_db.do_stop_on_unhandled_exception(thread, user_frame, frames_byid, arg)
