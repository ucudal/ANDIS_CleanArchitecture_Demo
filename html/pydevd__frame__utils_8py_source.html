114 def map_columns_to_line(self, original_line: str):
115 """
116 The columns internally are actually based on bytes.
117
118 Also, the position isn't always the ideal one as the start may not be
119 what we want (if the user has many subscripts in the line the start
120 will always be the same and only the end would change).
121 For more details see:
122 https://github.com/microsoft/debugpy/issues/1099#issuecomment-1303403995
123
124 So, this function maps the start/end columns to the position to be shown in the editor.
125 """
126 colno = _utf8_byte_offset_to_character_offset(original_line, self.colno)
127 end_colno = _utf8_byte_offset_to_character_offset(original_line, self.end_colno)
128
129 if self.lineno == self.end_lineno:
130 try:
131 ret = _extract_caret_anchors_in_bytes_from_line_segment(original_line[colno:end_colno])
132 if ret is not None:
133 return (
134 _utf8_byte_offset_to_character_offset(original_line, ret[0] + self.colno),
135 _utf8_byte_offset_to_character_offset(original_line, ret[1] + self.colno),
136 )
137 except Exception:
138 pass # Suppress exception
139
140 return colno, end_colno

148def _utf8_byte_offset_to_character_offset(s: str, offset: int):
149 byte_offset = 0
150 char_offset = 0
151 offset = offset or 0
152
153 for char_offset, character in enumerate(s):
154 byte_offset += 1
155
156 codepoint = ord(character)
157
158 if codepoint >= _utf8_with_4_bytes:
159 byte_offset += 3
160
161 elif codepoint >= _utf8_with_3_bytes:
162 byte_offset += 2
163
164 elif codepoint >= _utf8_with_2_bytes:
165 byte_offset += 1
166
167 if byte_offset > offset:
168 break
169 else:
170 char_offset += 1
171
172 return char_offset

175# Based on traceback._extract_caret_anchors_in_bytes_from_line_segment (Python 3.11.0)
176def _extract_caret_anchors_in_bytes_from_line_segment(segment: str):
177 import ast
178
179 try:
180 segment = segment.encode("utf-8")
181 except UnicodeEncodeError:
182 return None
183 try:
184 tree = ast.parse(segment)
185 except SyntaxError:
186 return None
187
188 if len(tree.body) != 1:
189 return None
190
191 statement = tree.body[0]
192 if isinstance(statement, ast.Expr):
193 expr = statement.value
194 if isinstance(expr, ast.BinOp):
195 operator_str = segment[expr.left.end_col_offset : expr.right.col_offset]
196 operator_offset = len(operator_str) - len(operator_str.lstrip())
197
198 left_anchor = expr.left.end_col_offset + operator_offset
199 right_anchor = left_anchor + 1
200 if operator_offset + 1 < len(operator_str) and not operator_str[operator_offset + 1] == ord(b" "):
201 right_anchor += 1
202 return left_anchor, right_anchor
203 if isinstance(expr, ast.Subscript):
204 return expr.value.end_col_offset, expr.slice.end_col_offset + 1
205
206 return None

372def create_frames_list_from_traceback(trace_obj, frame, exc_type, exc_desc, exception_type=None):
373 """
374 :param trace_obj:
375 This is the traceback from which the list should be created.
376
377 :param frame:
378 This is the first frame to be considered (i.e.: topmost frame). If None is passed, all
379 the frames from the traceback are shown (so, None should be passed for unhandled exceptions).
380
381 :param exception_type:
382 If this is an unhandled exception or user unhandled exception, we'll not trim the stack to create from the passed
383 frame, rather, we'll just mark the frame in the frames list.
384 """
385 lst = []
386
387 tb = trace_obj
388 if tb is not None and tb.tb_frame is not None:
389 f = tb.tb_frame.f_back
390 while f is not None:
391 lst.insert(0, (f, f.f_lineno, None))
392 f = f.f_back
393
394 while tb is not None:
395 lst.append((tb.tb_frame, tb.tb_lineno, _get_line_col_info_from_tb(tb)))
396 tb = tb.tb_next
397
398 frames_list = None
399
400 for tb_frame, tb_lineno, line_col_info in reversed(lst):
401 if frames_list is None and ((frame is tb_frame) or (frame is None) or (exception_type == EXCEPTION_TYPE_USER_UNHANDLED)):
402 frames_list = FramesList()
403
404 if frames_list is not None:
405 frames_list.append(tb_frame)
406 frames_list.frame_id_to_lineno[id(tb_frame)] = tb_lineno
407 frames_list.frame_id_to_line_col_info[id(tb_frame)] = line_col_info
408
409 if frames_list is None and frame is not None:
410 # Fallback (shouldn't happen in practice).
411 pydev_log.info("create_frames_list_from_traceback did not find topmost frame in list.")
412 frames_list = create_frames_list_from_frame(frame)
413
414 frames_list.exc_type = exc_type
415 frames_list.exc_desc = exc_desc
416 frames_list.trace_obj = trace_obj
417
418 if exception_type == EXCEPTION_TYPE_USER_UNHANDLED:
419 frames_list.current_frame = frame
420 elif exception_type == EXCEPTION_TYPE_UNHANDLED:
421 if len(frames_list) > 0:
422 frames_list.current_frame = frames_list.last_frame()
423
424 curr = frames_list
425 memo = set()
426 memo.add(id(exc_desc))
427
428 while True:
429 chained = create_frames_list_from_exception_cause(None, None, None, curr.exc_desc, memo)
430 if chained is None:
431 break
432 else:
433 curr.chained_frames_list = chained
434 curr = chained
435
436 return frames_list
