67class StdIn(BaseStdIn):
68 """
69 Object to be added to stdin (to emulate it as non-blocking while the next line arrives)

72 def __init__(self, interpreter, host, client_port, original_stdin=sys.stdin):
73 BaseStdIn.__init__(self, original_stdin)

78 def readline(self, *args, **kwargs):
79 # Ok, callback into the client to get the new input
80 try:
81 server = xmlrpclib.Server("http://%s:%s" % (self.host, self.client_port))
82 requested_input = server.RequestInput()
83 if not requested_input:
84 return "\n" # Yes, a readline must return something (otherwise we can get an EOFError on the input() call).
85 else:
86 # readline should end with '\n' (not doing so makes IPython 5 remove the last *valid* character).
87 requested_input += "\n"
88 return requested_input
89 except KeyboardInterrupt:
90 raise # Let KeyboardInterrupt go through -- #PyDev-816: Interrupting infinite loop in the Interactive Console
91 except:
92 return "\n"

190 def need_more_for_code(self, source):
191 # PyDev-502: PyDev 3.9 F2 doesn't support backslash continuations
192
193 # Strangely even the IPython console is_complete said it was complete
194 # even with a continuation char at the end.
195 if source.endswith("\\"):
196 return True
197
198 if hasattr(self.interpreter, "is_complete"):
199 return not self.interpreter.is_complete(source)

201 # At this point, it should always be single.
202 # If we don't do this, things as:
203 #
204 # for i in range(10): print(i)

206 # (in a single line) don't work.
207 # Note that it won't give an error and code will be None (so, it'll
208 # use execMultipleLines in the next call in this case).
209 symbol = "single"
210 code = self.interpreter.compile(source, "<input>", symbol)
211 except (OverflowError, SyntaxError, ValueError):
212 # Case 1

215 # Case 2
