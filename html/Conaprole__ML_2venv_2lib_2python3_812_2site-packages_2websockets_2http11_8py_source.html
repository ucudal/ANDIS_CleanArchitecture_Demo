102 @classmethod
103 def parse(
104 cls,
105 read_line: Callable[[int], Generator[None, None, bytes]],
106 ) -> Generator[None, None, Request]:
107 """
108 Parse a WebSocket handshake request.
109
110 This is a generator-based coroutine.
111
112 The request path isn't URL-decoded or validated in any way.
113
114 The request path and headers are expected to contain only ASCII
115 characters. Other characters are represented with surrogate escapes.
116
117 :meth:`parse` doesn't attempt to read the request body because
118 WebSocket handshake requests don't have one. If the request contains a
119 body, it may be read from the data stream after :meth:`parse` returns.
120
121 Args:
122 read_line: Generator-based coroutine that reads a LF-terminated
123 line or raises an exception if there isn't enough data
124
125 Raises:
126 EOFError: If the connection is closed without a full HTTP request.
127 SecurityError: If the request exceeds a security limit.
128 ValueError: If the request isn't well formatted.
129
130 """
131 # https://datatracker.ietf.org/doc/html/rfc7230#section-3.1.1
132
133 # Parsing is simple because fixed values are expected for method and
134 # version and because path isn't checked. Since WebSocket software tends
135 # to implement HTTP/1.1 strictly, there's little need for lenient parsing.
136
137 try:
138 request_line = yield from parse_line(read_line)
139 except EOFError as exc:
140 raise EOFError("connection closed while reading HTTP request line") from exc
141
142 try:
143 method, raw_path, protocol = request_line.split(b" ", 2)
144 except ValueError: # not enough values to unpack (expected 3, got 1-2)
145 raise ValueError(f"invalid HTTP request line: {d(request_line)}") from None
146 if protocol != b"HTTP/1.1":
147 raise ValueError(
148 f"unsupported protocol; expected HTTP/1.1: {d(request_line)}"
149 )
150 if method != b"GET":
151 raise ValueError(f"unsupported HTTP method; expected GET; got {d(method)}")
152 path = raw_path.decode("ascii", "surrogateescape")
153
154 headers = yield from parse_headers(read_line)
155
156 # https://datatracker.ietf.org/doc/html/rfc7230#section-3.3.3
157
158 if "Transfer-Encoding" in headers:
159 raise NotImplementedError("transfer codings aren't supported")
160
161 if "Content-Length" in headers:
162 raise ValueError("unsupported request body")
163
164 return cls(path, headers)

215 Parse a WebSocket handshake response.

222 Args:

229

231 EOFError: If the connection is closed without a full HTTP response.
232 SecurityError: If the response exceeds a security limit.
233 LookupError: If the response isn't well formatted.
234 ValueError: If the response isn't well formatted.
235

237 # https://datatracker.ietf.org/doc/html/rfc7230#section-3.1.2
238

240 status_line = yield from parse_line(read_line)
241 except EOFError as exc:
242 raise EOFError("connection closed while reading HTTP status line") from exc
243
244 try:
245 protocol, raw_status_code, raw_reason = status_line.split(b" ", 2)
246 except ValueError: # not enough values to unpack (expected 3, got 1-2)
247 raise ValueError(f"invalid HTTP status line: {d(status_line)}") from None
248 if protocol != b"HTTP/1.1":
249 raise ValueError(

255 raise ValueError(

263 raise ValueError(f"invalid HTTP reason phrase: {d(raw_reason)}")

265

291 Parse a single line.
292

314def parse_headers(
315 read_line: Callable[[int], Generator[None, None, bytes]],
316) -> Generator[None, None, Headers]:
317 """
318 Parse HTTP headers.
319
320 Non-ASCII characters are represented with surrogate escapes.
321
322 Args:
323 read_line: Generator-based coroutine that reads a LF-terminated line
324 or raises an exception if there isn't enough data.
325
326 Raises:
327 EOFError: If the connection is closed without complete headers.
328 SecurityError: If the request exceeds a security limit.
329 ValueError: If the request isn't well formatted.
330
331 """
332 # https://datatracker.ietf.org/doc/html/rfc7230#section-3.2
333
334 # We don't attempt to support obsolete line folding.
335
336 headers = Headers()
337 for _ in range(MAX_NUM_HEADERS + 1):
338 try:
339 line = yield from parse_line(read_line)
340 except EOFError as exc:
341 raise EOFError("connection closed while reading HTTP headers") from exc
342 if line == b"":
343 break
344
345 try:
346 raw_name, raw_value = line.split(b":", 1)
347 except ValueError: # not enough values to unpack (expected 2, got 1)
348 raise ValueError(f"invalid HTTP header line: {d(line)}") from None
349 if not _token_re.fullmatch(raw_name):
350 raise ValueError(f"invalid HTTP header name: {d(raw_name)}")
351 raw_value = raw_value.strip(b" \t")
352 if not _value_re.fullmatch(raw_value):
353 raise ValueError(f"invalid HTTP header value: {d(raw_value)}")
354
355 name = raw_name.decode("ascii") # guaranteed to be ASCII at this point
356 value = raw_value.decode("ascii", "surrogateescape")
357 headers[name] = value
358
359 else:
360 raise SecurityError("too many HTTP headers")
361
362 return headers

375 # responses except 1xx, 204, and 304 include a message body.

377 return b""

380 # handle it when accessing headers. Instead we document that its parent
381 # class, LookupError, may be raised.
382 # Conversions from str to int are protected by sys.set_int_max_str_digits..
383
384 elif (coding := headers.get("Transfer-Encoding")) is not None:
385 if coding != "chunked":
386 raise NotImplementedError(f"transfer coding {coding} isn't supported")
387
388 body = b""
389 while True:
390 chunk_size_line = yield from parse_line(read_line)
391 raw_chunk_size = chunk_size_line.split(b";", 1)[0]
392 # Set a lower limit than default_max_str_digits; 1 EB is plenty.
393 if len(raw_chunk_size) > 15:

395 raise SecurityError(f"chunk too large: 0x{str_chunk_size} bytes")
396 chunk_size = int(raw_chunk_size, 16)
397 if chunk_size == 0:
398 break

400 raise SecurityError(
401 f"chunk too large: {chunk_size} bytes after {len(body)} bytes"
402 )
403 body += yield from read_exact(chunk_size)
404 if (yield from read_exact(2)) != b"\r\n":
405 raise ValueError("chunk without CRLF")
406 # Read the trailer.
407 yield from parse_headers(read_line)

411 # Set a lower limit than default_max_str_digits; 1 EiB is plenty.
412 if len(raw_content_length) > 18:
413 raise SecurityError(f"body too large: {raw_content_length} bytes")
414 content_length = int(raw_content_length)
415 if content_length > MAX_BODY_SIZE:
416 raise SecurityError(f"body too large: {content_length} bytes")
417 return (yield from read_exact(content_length))
418
