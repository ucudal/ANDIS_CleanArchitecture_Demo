213# LPCWSTR lpCmdLine,
214# int *pNumArgs

216def CommandLineToArgvW(lpCmdLine):
217 _CommandLineToArgvW = windll.shell32.CommandLineToArgvW
218 _CommandLineToArgvW.argtypes = [LPVOID, POINTER(ctypes.c_int)]
219 _CommandLineToArgvW.restype = LPVOID
220
221 if not lpCmdLine:
222 lpCmdLine = None
223 argc = ctypes.c_int(0)
224 vptr = ctypes.windll.shell32.CommandLineToArgvW(lpCmdLine, byref(argc))
225 if vptr == NULL:
226 raise ctypes.WinError()
227 argv = vptr
228 try:
229 argc = argc.value
230 if argc <= 0:
231 raise ctypes.WinError()
232 argv = ctypes.cast(argv, ctypes.POINTER(LPWSTR * argc))
233 argv = [argv.contents[i] for i in compat.xrange(0, argc)]
234 finally:
235 if vptr is not None:
236 LocalFree(vptr)
237 return argv
