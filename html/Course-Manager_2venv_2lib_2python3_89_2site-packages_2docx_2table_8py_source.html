161 @property
162 def _cells(self):
163 """
164 A sequence of |_Cell| objects, one for each cell of the layout grid.
165 If the table contains a span, one or more |_Cell| object references
166 are repeated.
167 """
168 col_count = self._column_count
169 cells = []
170 for tc in self._tbl.iter_tcs():
171 for grid_span_idx in range(tc.grid_span):
172 if tc.vMerge == ST_Merge.CONTINUE:
173 cells.append(cells[-col_count])
174 elif grid_span_idx > 0:
175 cells.append(cells[-1])
176 else:
177 cells.append(_Cell(tc, self))
178 return cells

199 def add_paragraph(self, text='', style=None):
200 """
201 Return a paragraph newly added to the end of the content in this
202 cell. If present, *text* is added to the paragraph in a single run.
203 If specified, the paragraph style *style* is applied. If *style* is
204 not specified or is |None|, the result is as though the 'Normal'
205 style was applied. Note that the formatting of text in a cell can be
206 influenced by the table style. *text* can contain tab (``\\t``)
207 characters, which are converted to the appropriate XML form for
208 a tab. *text* can also include newline (``\\n``) or carriage return
209 (``\\r``) characters, each of which is converted to a line break.
210 """
211 return super(_Cell, self).add_paragraph(text, style)
212
213 def add_table(self, rows, cols):
214 """
215 Return a table newly added to this cell after any existing cell
216 content, having *rows* rows and *cols* columns. An empty paragraph is
217 added after the table because Word requires a paragraph element as
218 the last element in every cell.
219 """
220 width = self.width if self.width is not None else Inches(1)
221 table = super(_Cell, self).add_table(rows, cols, width)
222 self.add_paragraph()
223 return table

225 def merge(self, other_cell):

231 tc, tc_2 = self._tc, other_cell._tc
