138 For Test Anything Protocol (TAP) output.

140 name = 'TAP'
141 url = 'https://testanything.org/'
142 aliases = ['tap']
143 filenames = ['*.tap']
144 version_added = '2.1'

146 tokens = {
147 'root': [
148 # A TAP version may be specified.
149 (r'^TAP version \d+\n', Name.Namespace),
150
151 # Specify a plan with a plan line.
152 (r'^1\.\.\d+', Keyword.Declaration, 'plan'),
153
154 # A test failure
155 (r'^(not ok)([^\S\n]*)(\d*)',
156 bygroups(Generic.Error, Text, Number.Integer), 'test'),
157
158 # A test success
159 (r'^(ok)([^\S\n]*)(\d*)',
160 bygroups(Keyword.Reserved, Text, Number.Integer), 'test'),
161
162 # Diagnostics start with a hash.
163 (r'^#.*\n', Comment),
164
165 # TAP's version of an abort statement.
166 (r'^Bail out!.*\n', Generic.Error),
167
168 # TAP ignores any unrecognized lines.
169 (r'^.*\n', Text),
170 ],
171 'plan': [
172 # Consume whitespace (but not newline).
173 (r'[^\S\n]+', Text),
174
175 # A plan may have a directive with it.
176 (r'#', Comment, 'directive'),
177
178 # Or it could just end.
179 (r'\n', Comment, '#pop'),
180
181 # Anything else is wrong.
182 (r'.*\n', Generic.Error, '#pop'),

185 # Consume whitespace (but not newline).
