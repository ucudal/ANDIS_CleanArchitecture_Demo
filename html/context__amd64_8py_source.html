138# //
139# // Context Frame
140# //
141# // This frame has a several purposes: 1) it is used as an argument to
142# // NtContinue, 2) is is used to constuct a call frame for APC delivery,
143# // and 3) it is used in the user level thread creation routines.
144# //
145# //
146# // The flags field within this record controls the contents of a CONTEXT
147# // record.
148# //
149# // If the context record is used as an input parameter, then for each
150# // portion of the context record controlled by a flag whose value is
151# // set, it is assumed that that portion of the context record contains
152# // valid context. If the context record is being used to modify a threads
153# // context, then only that portion of the threads context is modified.
154# //
155# // If the context record is used as an output parameter to capture the
156# // context of a thread, then only those portions of the thread's context
157# // corresponding to set flags will be returned.
158# //
159# // CONTEXT_CONTROL specifies SegSs, Rsp, SegCs, Rip, and EFlags.
160# //
161# // CONTEXT_INTEGER specifies Rax, Rcx, Rdx, Rbx, Rbp, Rsi, Rdi, and R8-R15.
162# //
163# // CONTEXT_SEGMENTS specifies SegDs, SegEs, SegFs, and SegGs.
164# //
165# // CONTEXT_DEBUG_REGISTERS specifies Dr0-Dr3 and Dr6-Dr7.
166# //
167# // CONTEXT_MMX_REGISTERS specifies the floating point and extended registers
168# // Mm0/St0-Mm7/St7 and Xmm0-Xmm15).
169# //
170#

444 @classmethod
445 def from_dict(cls, ctx):
446 "Instance a new structure from a Python native type."
447 ctx = Context(ctx)
448 s = cls()
449 ContextFlags = ctx["ContextFlags"]
450 s.ContextFlags = ContextFlags
451 for key in cls._others:
452 if key != "VectorRegister":
453 setattr(s, key, ctx[key])
454 else:
455 w = ctx[key]
456 v = (M128A * len(w))()
457 i = 0
458 for x in w:
459 y = M128A()
460 y.High = x >> 64
461 y.Low = x - (x >> 64)
462 v[i] = y
463 i += 1
464 setattr(s, key, v)
465 if (ContextFlags & CONTEXT_CONTROL) == CONTEXT_CONTROL:
466 for key in cls._control:
467 setattr(s, key, ctx[key])
468 if (ContextFlags & CONTEXT_INTEGER) == CONTEXT_INTEGER:
469 for key in cls._integer:
470 setattr(s, key, ctx[key])
471 if (ContextFlags & CONTEXT_SEGMENTS) == CONTEXT_SEGMENTS:
472 for key in cls._segments:
473 setattr(s, key, ctx[key])
474 if (ContextFlags & CONTEXT_DEBUG_REGISTERS) == CONTEXT_DEBUG_REGISTERS:
475 for key in cls._debug:
476 setattr(s, key, ctx[key])
477 if (ContextFlags & CONTEXT_MMX_REGISTERS) == CONTEXT_MMX_REGISTERS:
478 xmm = s.FltSave.xmm
479 for key in cls._mmx:
480 y = M128A()
481 y.High = x >> 64
482 y.Low = x - (x >> 64)
483 setattr(xmm, key, y)
484 return s
485
486 def to_dict(self):
487 "Convert a structure into a Python dictionary."
488 ctx = Context()
489 ContextFlags = self.ContextFlags
490 ctx["ContextFlags"] = ContextFlags
491 for key in self._others:
492 if key != "VectorRegister":
493 ctx[key] = getattr(self, key)
494 else:
495 ctx[key] = tuple([(x.Low + (x.High << 64)) for x in getattr(self, key)])
496 if (ContextFlags & CONTEXT_CONTROL) == CONTEXT_CONTROL:
497 for key in self._control:
498 ctx[key] = getattr(self, key)
499 if (ContextFlags & CONTEXT_INTEGER) == CONTEXT_INTEGER:
500 for key in self._integer:
501 ctx[key] = getattr(self, key)
502 if (ContextFlags & CONTEXT_SEGMENTS) == CONTEXT_SEGMENTS:
503 for key in self._segments:
504 ctx[key] = getattr(self, key)
505 if (ContextFlags & CONTEXT_DEBUG_REGISTERS) == CONTEXT_DEBUG_REGISTERS:
506 for key in self._debug:
507 ctx[key] = getattr(self, key)
508 if (ContextFlags & CONTEXT_MMX_REGISTERS) == CONTEXT_MMX_REGISTERS:
509 xmm = self.FltSave.xmm.to_dict()
510 for key in self._mmx:
511 ctx[key] = xmm.get(key)
512 return ctx
