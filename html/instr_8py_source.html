37def _pushes_back(opname):
38 if opname in ["CALL_FINALLY"]:
39 # CALL_FINALLY pushes the address of the "finally" block instead of a
40 # value, hence we don't treat it as pushing back op
41 return False
42 return (
43 opname.startswith("UNARY_")
44 or opname.startswith("GET_")
45 # BUILD_XXX_UNPACK have been removed in 3.9
46 or opname.startswith("BINARY_")
47 or opname.startswith("INPLACE_")
48 or opname.startswith("BUILD_")
49 or opname.startswith("CALL_")
50 ) or opname in (
51 "LIST_TO_TUPLE",

54 "DICT_UPDATE",

169 def _check_arg(self, name, opcode, arg):
170 if name == "EXTENDED_ARG":

172 "only concrete instruction can contain EXTENDED_ARG, " "highlevel instruction can represent arbitrary argument without it"
173 )
174
175 if opcode >= _opcode.HAVE_ARGUMENT:
176 if arg is UNSET:
177 raise ValueError("operation %s requires an argument" % name)
178 else:
179 if arg is not UNSET:
180 raise ValueError("operation %s has no argument" % name)
181
182 if self._has_jump(opcode):
183 if not isinstance(arg, (Label, _bytecode.BasicBlock)):
184 raise TypeError("operation %s argument type must be " "Label or BasicBlock, got %s" % (name, type(arg).__name__))

186 elif opcode in _opcode.hasfree:

191 if not isinstance(arg, str):

279 def stack_effect(self, jump=None):
280 if self._opcode < _opcode.HAVE_ARGUMENT:
281 arg = None
282 elif not isinstance(self._arg, int) or self._opcode in _opcode.hasconst:
283 # Argument is either a non-integer or an integer constant,
284 # not oparg.
285 arg = 0
286 else:
287 arg = self._arg
288
289 if sys.version_info < (3, 8):
290 effect = _stack_effects.get(self._opcode, None)
291 if effect is not None:
292 return max(effect) if jump is None else effect[jump]
293 return dis.stack_effect(self._opcode, arg)
294 else:
295 return dis.stack_effect(self._opcode, arg, jump=jump)

297 def pre_and_post_stack_effect(self, jump=None):
298 _effect = self.stack_effect(jump=jump)
299
300 # To compute pre size and post size to avoid segfault cause by not enough
301 # stack element
302 _opname = _opcode.opname[self._opcode]
303 if _opname.startswith("DUP_TOP"):
304 return _effect * -1, _effect * 2
305 if _pushes_back(_opname):
306 # if the op pushes value back to the stack, then the stack effect given
307 # by dis.stack_effect actually equals pre + post effect, therefore we need
308 # -1 from the stack effect as a pre condition
309 return _effect - 1, 1
310 if _opname.startswith("UNPACK_"):
311 # Instr(UNPACK_* , n) pops 1 and pushes n
312 # _effect = n - 1
313 # hence we return -1, _effect + 1
314 return -1, _effect + 1
315 if _opname == "FOR_ITER" and not jump:
316 # Since FOR_ITER needs TOS to be an iterator, which basically means
317 # a prerequisite of 1 on the stack
318 return -1, 2
319 if _opname == "ROT_N":
320 return (-self._arg, self._arg)
321 return {"ROT_TWO": (-2, 2), "ROT_THREE": (-3, 3), "ROT_FOUR": (-4, 4)}.get(_opname, (_effect, 0))
