1# -*- coding: utf-8 -*-
2"""
3h2/windows
4~~~~~~~~~~
5
6Defines tools for managing HTTP/2 flow control windows.
7
8The objects defined in this module are used to automatically manage HTTP/2
9flow control windows. Specifically, they keep track of what the size of the
10window is, how much data has been consumed from that window, and how much data
11the user has already used. It then implements a basic algorithm that attempts
12to manage the flow control window without user input, trying to ensure that it
13does not emit too many WINDOW_UPDATE frames.
14"""
15from __future__ import division
16
17from .exceptions import FlowControlError

20# The largest acceptable value for a HTTP/2 flow control window.
21LARGEST_FLOW_CONTROL_WINDOW = 2**31 - 1
22
23
24class WindowManager(object):
25 """
26 A basic HTTP/2 window manager.
27
28 :param max_window_size: The maximum size of the flow control window.
29 :type max_window_size: ``int``
30 """
31 def __init__(self, max_window_size):
32 assert max_window_size <= LARGEST_FLOW_CONTROL_WINDOW
33 self.max_window_size = max_window_size
34 self.current_window_size = max_window_size
35 self._bytes_processed = 0
36
37 def window_consumed(self, size):
38 """
39 We have received a certain number of bytes from the remote peer. This
40 necessarily shrinks the flow control window!
41
42 :param size: The number of flow controlled bytes we received from the
43 remote peer.
44 :type size: ``int``
45 :returns: Nothing.
46 :rtype: ``None``
47 """
48 self.current_window_size -= size
49 if self.current_window_size < 0:
50 raise FlowControlError("Flow control window shrunk below 0")
51
52 def window_opened(self, size):
53 """
54 The flow control window has been incremented, either because of manual
55 flow control management or because of the user changing the flow
56 control settings. This can have the effect of increasing what we
57 consider to be the "maximum" flow control window size.
58
59 This does not increase our view of how many bytes have been processed,
60 only of how much space is in the window.
61
62 :param size: The increment to the flow control window we received.
63 :type size: ``int``
64 :returns: Nothing
65 :rtype: ``None``
66 """
67 self.current_window_size += size
68
69 if self.current_window_size > LARGEST_FLOW_CONTROL_WINDOW:
70 raise FlowControlError(
71 "Flow control window mustn't exceed %d" %
72 LARGEST_FLOW_CONTROL_WINDOW
73 )
74
75 if self.current_window_size > self.max_window_size:
76 self.max_window_size = self.current_window_size
77
78 def process_bytes(self, size):
79 """
80 The application has informed us that it has processed a certain number
81 of bytes. This may cause us to want to emit a window update frame. If
82 we do want to emit a window update frame, this method will return the
83 number of bytes that we should increment the window by.
84
85 :param size: The number of flow controlled bytes that the application
86 has processed.
87 :type size: ``int``
88 :returns: The number of bytes to increment the flow control window by,
89 or ``None``.
90 :rtype: ``int`` or ``None``
91 """
92 self._bytes_processed += size
93 return self._maybe_update_window()
94
95 def _maybe_update_window(self):
96 """
97 Run the algorithm.
98
99 Our current algorithm can be described like this.
100
101 1. If no bytes have been processed, we immediately return 0. There is
102 no meaningful way for us to hand space in the window back to the
103 remote peer, so let's not even try.
104 2. If there is no space in the flow control window, and we have
105 processed at least 1024 bytes (or 1/4 of the window, if the window
106 is smaller), we will emit a window update frame. This is to avoid
107 the risk of blocking a stream altogether.
108 3. If there is space in the flow control window, and we have processed
109 at least 1/2 of the window worth of bytes, we will emit a window
110 update frame. This is to minimise the number of window update frames
111 we have to emit.
112
113 In a healthy system with large flow control windows, this will
114 irregularly emit WINDOW_UPDATE frames. This prevents us starving the
115 connection by emitting eleventy bajillion WINDOW_UPDATE frames,
116 especially in situations where the remote peer is sending a lot of very
117 small DATA frames.
118 """
119 # TODO: Can the window be smaller than 1024 bytes? If not, we can
120 # streamline this algorithm.
121 if not self._bytes_processed:
122 return None
123
124 max_increment = (self.max_window_size - self.current_window_size)
125 increment = 0
126
127 # Note that, even though we may increment less than _bytes_processed,
128 # we still want to set it to zero whenever we emit an increment. This
129 # is because we'll always increment up to the maximum we can.
130 if (self.current_window_size == 0) and (
131 self._bytes_processed > min(1024, self.max_window_size // 4)):
132 increment = min(self._bytes_processed, max_increment)
133 self._bytes_processed = 0
134 elif self._bytes_processed >= (self.max_window_size // 2):
135 increment = min(self._bytes_processed, max_increment)
136 self._bytes_processed = 0
137
138 self.current_window_size += increment
139 return increment
