28def _check_matches(patterns, paths):
29 if not patterns and not paths:
30 # Matched to the end.
31 return True
32
33 if (not patterns and paths) or (patterns and not paths):
34 return False
35
36 pattern = normcase(patterns[0])
37 path = normcase(paths[0])
38
39 if not glob.has_magic(pattern):
40 if pattern != path:
41 return False
42
43 elif pattern == "**":
44 if len(patterns) == 1:
45 return True # if ** is the last one it matches anything to the right.
46
47 for i in range(len(paths)):
48 # Recursively check the remaining patterns as the
49 # current pattern could match any number of paths.
50 if _check_matches(patterns[1:], paths[i:]):
51 return True
52
53 elif not fnmatch.fnmatch(path, pattern):
54 # Current part doesn't match.
55 return False
56
57 return _check_matches(patterns[1:], paths[1:])

60def glob_matches_path(path, pattern, sep=os.sep, altsep=os.altsep):
61 if altsep:
62 pattern = pattern.replace(altsep, sep)
63 path = path.replace(altsep, sep)
64
65 drive = ""
66 if len(path) > 1 and path[1] == ":":
67 drive, path = path[0], path[2:]
68
69 if drive and len(pattern) > 1:
70 if pattern[1] == ":":
71 if drive.lower() != pattern[0].lower():
72 return False
73 pattern = pattern[2:]
74
75 patterns = pattern.split(sep)
76 paths = path.split(sep)
77 if paths:
78 if paths[0] == "":
79 paths = paths[1:]
80 if patterns:
81 if patterns[0] == "":
82 patterns = patterns[1:]
83
84 return _check_matches(patterns, paths)

229 def in_project_roots(self, received_filename):
230 """
231 Note: don't call directly. Use PyDb.in_project_scope (there's no caching here and it doesn't
232 handle all possibilities for knowing whether a project is actually in the scope, it
233 just handles the heuristics based on the absolute_normalized_filename without the actual frame).
234 """

245 if DEBUG:
246 pydev_log.debug(

256 absolute_normalized_filename_as_dir = absolute_normalized_filename + ("\\" if IS_WINDOWS else "/")

258 found_in_project = []
259 for root in project_roots:
260 if root and (absolute_normalized_filename.startswith(root) or root == absolute_normalized_filename_as_dir):
261 if DEBUG:
262 pydev_log.debug("In project: %s (%s)", absolute_normalized_filename, root)
263 found_in_project.append(root)
264
265 found_in_library = []
266 library_roots = self._get_library_roots()
267 for root in library_roots:
268 if root and (absolute_normalized_filename.startswith(root) or root == absolute_normalized_filename_as_dir):
269 found_in_library.append(root)
270 if DEBUG:
271 pydev_log.debug("In library: %s (%s)", absolute_normalized_filename, root)
272 else:
273 if DEBUG:
274 pydev_log.debug("Not in library: %s (%s)", absolute_normalized_filename, root)
275
276 if not project_roots:
277 # If we have no project roots configured, consider it being in the project
278 # roots if it's not found in site-packages (because we have defaults for those
279 # and not the other way around).
280 in_project = not found_in_library

282 pydev_log.debug("Final in project (no project roots): %s (%s)", absolute_normalized_filename, in_project)

314 def exclude_by_filter(self, absolute_filename, module_name):
315 """
316 :return: True if it should be excluded, False if it should be included and None
317 if no rule matched the given file.
318 """
319 for exclude_filter in self._exclude_filters: # : :type exclude_filter: ExcludeFilter
320 if exclude_filter.is_path:
321 if glob_matches_path(absolute_filename, exclude_filter.name):
322 return exclude_filter.exclude
323 else:
324 # Module filter.
325 if exclude_filter.name == module_name or module_name.startswith(exclude_filter.name + "."):
326 return exclude_filter.exclude
327 return None
