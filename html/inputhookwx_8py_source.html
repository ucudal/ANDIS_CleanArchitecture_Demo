33def inputhook_wx1():
34 """Run the wx event loop by processing pending events only.
35
36 This approach seems to work, but its performance is not great as it
37 relies on having PyOS_InputHook called regularly.
38 """
39 try:
40 app = wx.GetApp() # @UndefinedVariable
41 if app is not None:
42 assert wx.Thread_IsMain() # @UndefinedVariable
43
44 # Make a temporary event loop and process system events until
45 # there are no more waiting, then allow idle events (which
46 # will also deal with pending or posted wx events.)
47 evtloop = wx.EventLoop() # @UndefinedVariable
48 ea = wx.EventLoopActivator(evtloop) # @UndefinedVariable
49 while evtloop.Pending():
50 evtloop.Dispatch()
51 app.ProcessIdle()
52 del ea
53 except KeyboardInterrupt:
54 pass
55 return 0

80def inputhook_wx2():
81 """Run the wx event loop, polling for stdin.
82
83 This version runs the wx eventloop for an undetermined amount of time,
84 during which it periodically checks to see if anything is ready on
85 stdin. If anything is ready on stdin, the event loop exits.
86
87 The argument to elr.Run controls how often the event loop looks at stdin.
88 This determines the responsiveness at the keyboard. A setting of 1000
89 enables a user to type at most 1 char per second. I have found that a
90 setting of 10 gives good keyboard response. We can shorten it further,
91 but eventually performance would suffer from calling select/kbhit too
92 often.
93 """
94 try:
95 app = wx.GetApp() # @UndefinedVariable
96 if app is not None:
97 assert wx.Thread_IsMain() # @UndefinedVariable
98 elr = EventLoopRunner()
99 # As this time is made shorter, keyboard response improves, but idle
100 # CPU load goes up. 10 ms seems like a good compromise.
101 elr.Run(time=10) # CHANGE time here to control polling interval
102 except KeyboardInterrupt:
103 pass
104 return 0

107def inputhook_wx3():
108 """Run the wx event loop by processing pending events only.
109
110 This is like inputhook_wx1, but it keeps processing pending events
111 until stdin is ready. After processing all pending events, a call to
112 time.sleep is inserted. This is needed, otherwise, CPU usage is at 100%.
113 This sleep time should be tuned though for best performance.
114 """
115 # We need to protect against a user pressing Control-C when IPython is
116 # idle and this is running. We trap KeyboardInterrupt and pass.
117 try:
118 app = wx.GetApp() # @UndefinedVariable
119 if app is not None:
120 if hasattr(wx, "IsMainThread"):
121 assert wx.IsMainThread() # @UndefinedVariable
122 else:
123 assert wx.Thread_IsMain() # @UndefinedVariable
124
125 # The import of wx on Linux sets the handler for signal.SIGINT
126 # to 0. This is a bug in wx or gtk. We fix by just setting it
127 # back to the Python default.
128 if not callable(signal.getsignal(signal.SIGINT)):
129 signal.signal(signal.SIGINT, signal.default_int_handler)
130
131 evtloop = wx.EventLoop() # @UndefinedVariable
132 ea = wx.EventLoopActivator(evtloop) # @UndefinedVariable
133 t = clock()
134 while not stdin_ready():
135 while evtloop.Pending():
136 t = clock()
137 evtloop.Dispatch()
138 app.ProcessIdle()
139 # We need to sleep at this point to keep the idle CPU load
140 # low. However, if sleep to long, GUI response is poor. As
141 # a compromise, we watch how often GUI events are being processed
142 # and switch between a short and long sleep time. Here are some
143 # stats useful in helping to tune this.
144 # time CPU load
145 # 0.001 13%
146 # 0.005 3%
147 # 0.01 1.5%
148 # 0.05 0.5%
149 used_time = clock() - t
150 if used_time > 10.0:
151 # print 'Sleep for 1 s' # dbg
152 time.sleep(1.0)
153 elif used_time > 0.1:
154 # Few GUI events coming in, so we can sleep longer
155 # print 'Sleep for 0.05 s' # dbg
156 time.sleep(0.05)
157 else:
158 # Many GUI events coming in, so sleep only very little
159 time.sleep(0.001)
160 del ea
161 except KeyboardInterrupt:
162 pass
163 return 0
