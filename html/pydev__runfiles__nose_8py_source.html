178def new_consolidate(self, result, batch_result):
179 """
180 Used so that it can work with the multiprocess plugin.
181 Monkeypatched because nose seems a bit unsupported at this time (ideally
182 the plugin would have this support by default).
183 """
184 ret = original(self, result, batch_result)
185
186 parent_frame = sys._getframe().f_back
187 # addr is something as D:\pytesting1\src\mod1\hello.py:TestCase.testMet4
188 # so, convert it to what report_cond expects
189 addr = parent_frame.f_locals["addr"]
190 i = addr.rindex(":")
191 addr = [addr[:i], addr[i + 1 :]]
192
193 output, testsRun, failures, errors, errorClasses = batch_result
194 if failures or errors:
195 for failure in failures:
196 PYDEV_NOSE_PLUGIN_SINGLETON.report_cond("fail", addr, output, failure)
197
198 for error in errors:
199 PYDEV_NOSE_PLUGIN_SINGLETON.report_cond("error", addr, output, error)
200 else:
201 PYDEV_NOSE_PLUGIN_SINGLETON.report_cond("ok", addr, output)
202
