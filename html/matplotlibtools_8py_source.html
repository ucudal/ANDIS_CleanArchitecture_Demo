153def flag_calls(func):
154 """Wrap a function to detect and flag when it gets called.
155
156 This is a decorator which takes a function and wraps it in a function with
157 a 'called' attribute. wrapper.called is initialized to False.
158
159 The wrapper.called attribute is set to False right before each call to the
160 wrapped function, so if the call fails it remains False. After the call
161 completes, wrapper.called is set to True and the output is returned.
162
163 Testing for truth in wrapper.called allows you to determine if a call to
164 func() was attempted and succeeded."""
165
166 # don't wrap twice
167 if hasattr(func, "called"):
168 return func
169
170 def wrapper(*args, **kw):
171 wrapper.called = False
172 out = func(*args, **kw)
173 wrapper.called = True
174 return out
175
176 wrapper.called = False
177 wrapper.__doc__ = func.__doc__
178 return wrapper
