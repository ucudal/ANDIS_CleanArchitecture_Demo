12def _get_dictionary(obj, replacements):
13 ret = dict()
14 cls = obj.__class__
15 for attr_name in dir(obj):
16 # This is interesting but it actually hides too much info from the dataframe.
17 # attr_type_in_cls = type(getattr(cls, attr_name, None))
18 # if attr_type_in_cls == property:
19 # ret[attr_name] = '<property (not computed)>'
20 # continue
21
22 timer = Timer()
23 try:
24 replacement = replacements.get(attr_name)
25 if replacement is not None:
26 ret[attr_name] = replacement
27 continue
28
29 attr_value = getattr(obj, attr_name, "<unable to get>")
30 if inspect.isroutine(attr_value) or isinstance(attr_value, MethodWrapperType):
31 continue
32 ret[attr_name] = attr_value
33 except Exception as e:
34 ret[attr_name] = "<error getting: %s>" % (e,)
35 finally:
36 timer.report_if_getting_attr_slow(cls, attr_name)
37
38 return ret

41@contextmanager
42def customize_pandas_options():
43 # The default repr depends on the settings of:
44 #
45 # pandas.set_option('display.max_columns', None)
46 # pandas.set_option('display.max_rows', None)
47 #
48 # which can make the repr **very** slow on some cases, so, we customize pandas to have
49 # smaller values if the current values are too big.
50 custom_options = []
51
52 from pandas import get_option
53
54 max_rows = get_option("display.max_rows")
55 max_cols = get_option("display.max_columns")
56 max_colwidth = get_option("display.max_colwidth")
57
58 if max_rows is None or max_rows > PANDAS_MAX_ROWS:
59 custom_options.append("display.max_rows")
60 custom_options.append(PANDAS_MAX_ROWS)
61
62 if max_cols is None or max_cols > PANDAS_MAX_COLS:
63 custom_options.append("display.max_columns")
64 custom_options.append(PANDAS_MAX_COLS)
65
66 if max_colwidth is None or max_colwidth > PANDAS_MAX_COLWIDTH:
67 custom_options.append("display.max_colwidth")
68 custom_options.append(PANDAS_MAX_COLWIDTH)
69
70 if custom_options:
71 from pandas import option_context
72
73 with option_context(*custom_options):
74 yield
75 else:
76 yield
