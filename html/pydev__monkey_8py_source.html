46def _get_setup_updated_with_protocol_and_ppid(setup, is_exec=False):
47 if setup is None:
48 setup = {}
49 setup = setup.copy()
50 # Discard anything related to the protocol (we'll set the the protocol based on the one
51 # currently set).
52 setup.pop(pydevd_constants.ARGUMENT_HTTP_JSON_PROTOCOL, None)
53 setup.pop(pydevd_constants.ARGUMENT_JSON_PROTOCOL, None)
54 setup.pop(pydevd_constants.ARGUMENT_QUOTED_LINE_PROTOCOL, None)
55
56 if not is_exec:
57 # i.e.: The ppid for the subprocess is the current pid.
58 # If it's an exec, keep it what it was.
59 setup[pydevd_constants.ARGUMENT_PPID] = os.getpid()
60
61 protocol = pydevd_constants.get_protocol()
62 if protocol == pydevd_constants.HTTP_JSON_PROTOCOL:
63 setup[pydevd_constants.ARGUMENT_HTTP_JSON_PROTOCOL] = True
64
65 elif protocol == pydevd_constants.JSON_PROTOCOL:
66 setup[pydevd_constants.ARGUMENT_JSON_PROTOCOL] = True
67
68 elif protocol == pydevd_constants.QUOTED_LINE_PROTOCOL:
69 setup[pydevd_constants.ARGUMENT_QUOTED_LINE_PROTOCOL] = True
70
71 elif protocol == pydevd_constants.HTTP_PROTOCOL:
72 setup[pydevd_constants.ARGUMENT_HTTP_PROTOCOL] = True
73
74 else:
75 pydev_log.debug("Unexpected protocol: %s", protocol)
76
77 mode = pydevd_defaults.PydevdCustomization.DEBUG_MODE
78 if mode:
79 setup["debug-mode"] = mode
80
81 preimport = pydevd_defaults.PydevdCustomization.PREIMPORT
82 if preimport:
83 setup["preimport"] = preimport
84
85 if DebugInfoHolder.PYDEVD_DEBUG_FILE:
86 setup["log-file"] = DebugInfoHolder.PYDEVD_DEBUG_FILE
87
88 if DebugInfoHolder.DEBUG_TRACE_LEVEL:
89 setup["log-level"] = DebugInfoHolder.DEBUG_TRACE_LEVEL
90
91 return setup

115def _separate_future_imports(code):
116 """
117 :param code:
118 The code from where we want to get the __future__ imports (note that it's possible that
119 there's no such entry).
120
121 :return tuple(str, str):
122 The return is a tuple(future_import, code).
123
124 If the future import is not available a return such as ('', code) is given, otherwise, the
125 future import will end with a ';' (so that it can be put right before the pydevd attach
126 code).
127 """
128 try:
129 node = ast.parse(code, "<string>", "exec")
130 visitor = _LastFutureImportFinder()
131 visitor.visit(node)
132
133 if visitor.last_future_import_found is None:
134 return "", code
135
136 node = visitor.last_future_import_found
137 offset = -1
138 if hasattr(node, "end_lineno") and hasattr(node, "end_col_offset"):
139 # Python 3.8 onwards has these (so, use when possible).
140 line, col = node.end_lineno, node.end_col_offset
141 offset = _get_offset_from_line_col(code, line - 1, col) # ast lines are 1-based, make it 0-based.
142
143 else:
144 # end line/col not available, let's just find the offset and then search
145 # for the alias from there.
146 line, col = node.lineno, node.col_offset
147 offset = _get_offset_from_line_col(code, line - 1, col) # ast lines are 1-based, make it 0-based.
148 if offset >= 0 and node.names:
149 from_future_import_name = node.names[-1].name
150 i = code.find(from_future_import_name, offset)
151 if i < 0:
152 offset = -1
153 else:
154 offset = i + len(from_future_import_name)
155
156 if offset >= 0:
157 for i in range(offset, len(code)):
158 if code[i] in (" ", "\t", ";", ")", "\n"):
159 offset += 1
160 else:
161 break
162
163 future_import = code[:offset]
164 code_remainder = code[offset:]
165
166 # Now, put '\n' lines back into the code remainder (we had to search for
167 # `\n)`, but in case we just got the `\n`, it should be at the remainder,
168 # not at the future import.
169 while future_import.endswith("\n"):
170 future_import = future_import[:-1]
171 code_remainder = "\n" + code_remainder
172
173 if not future_import.endswith(";"):
174 future_import += ";"
175 return future_import, code_remainder
176
177 # This shouldn't happen...
178 pydev_log.info("Unable to find line %s in code:\n%r", line, code)
179 return "", code
180
181 except:
182 pydev_log.exception("Error getting from __future__ imports from: %r", code)
183 return "", code

249def _get_str_type_compatible(s, args):
250 """
251 This method converts `args` to byte/unicode based on the `s' type.
252 """
253 if isinstance(args, (list, tuple)):
254 ret = []
255 for arg in args:
256 if type(s) == type(arg):
257 ret.append(arg)
258 else:
259 if isinstance(s, bytes):
260 ret.append(arg.encode("utf-8"))
261 else:
262 ret.append(arg.decode("utf-8"))
263 return ret
264 else:
265 if type(s) == type(args):
266 return args
267 else:
268 if isinstance(s, bytes):
269 return args.encode("utf-8")
270 else:
271 return args.decode("utf-8")

277def is_python(path):
278 single_quote, double_quote = _get_str_type_compatible(path, ["'", '"'])
279
280 if path.endswith(single_quote) or path.endswith(double_quote):
281 path = path[1 : len(path) - 1]
282 filename = os.path.basename(path).lower()
283 for name in _get_str_type_compatible(filename, ["python", "jython", "pypy"]):
284 if filename.find(name) != -1:
285 return True
286
287 return False

294def remove_quotes_from_args(args):
295 if sys.platform == "win32":
296 new_args = []
297
298 for x in args:
299 if isinstance(x, Path):
300 x = str(x)
301 else:
302 if not isinstance(x, (bytes, str)):
303 raise InvalidTypeInArgsException(str(type(x)))
304
305 double_quote, two_double_quotes = _get_str_type_compatible(x, ['"', '""'])
306
307 if x != two_double_quotes:
308 if len(x) > 1 and x.startswith(double_quote) and x.endswith(double_quote):
309 x = x[1:-1]
310
311 new_args.append(x)
312 return new_args
313 else:
314 new_args = []
315 for x in args:
316 if isinstance(x, Path):
317 x = x.as_posix()
318 else:
319 if not isinstance(x, (bytes, str)):
320 raise InvalidTypeInArgsException(str(type(x)))
321 new_args.append(x)
322
323 return new_args

326def quote_arg_win32(arg):
327 fix_type = lambda x: _get_str_type_compatible(arg, x)
328
329 # See if we need to quote at all - empty strings need quoting, as do strings
330 # with whitespace or quotes in them. Backslashes do not need quoting.
331 if arg and not set(arg).intersection(fix_type(' "\t\n\v')):
332 return arg
333
334 # Per https://docs.microsoft.com/en-us/windows/desktop/api/shellapi/nf-shellapi-commandlinetoargvw,
335 # the standard way to interpret arguments in double quotes is as follows:
336 #
337 # 2N backslashes followed by a quotation mark produce N backslashes followed by
338 # begin/end quote. This does not become part of the parsed argument, but toggles
339 # the "in quotes" mode.
340 #
341 # 2N+1 backslashes followed by a quotation mark again produce N backslashes followed
342 # by a quotation mark literal ("). This does not toggle the "in quotes" mode.
343 #
344 # N backslashes not followed by a quotation mark simply produce N backslashes.
345 #
346 # This code needs to do the reverse transformation, thus:
347 #
348 # N backslashes followed by " produce 2N+1 backslashes followed by "
349 #
350 # N backslashes at the end (i.e. where the closing " goes) produce 2N backslashes.
351 #
352 # N backslashes in any other position remain as is.
353
354 arg = re.sub(fix_type(r"(\\*)\""), fix_type(r'\1\1\\"'), arg)
355 arg = re.sub(fix_type(r"(\\*)$"), fix_type(r"\1\1"), arg)
356 return fix_type('"') + arg + fix_type('"')

366def patch_args(args, is_exec=False):
367 """
368 :param list args:
369 Arguments to patch.
370
371 :param bool is_exec:
372 If it's an exec, the current process will be replaced (this means we have
373 to keep the same ppid).
374 """
375 try:
376 pydev_log.debug("Patching args: %s", args)
377 original_args = args
378 try:
379 unquoted_args = remove_quotes_from_args(args)
380 except InvalidTypeInArgsException as e:
381 pydev_log.info("Unable to monkey-patch subprocess arguments because a type found in the args is invalid: %s", e)
382 return original_args
383
384 # Internally we should reference original_args (if we want to return them) or unquoted_args
385 # to add to the list which will be then quoted in the end.
386 del args
387
388 from pydevd import SetupHolder
389
390 if not unquoted_args:
391 return original_args
392
393 if not is_python(unquoted_args[0]):
394 pydev_log.debug("Process is not python, returning.")
395 return original_args
396
397 # Note: we create a copy as string to help with analyzing the arguments, but
398 # the final list should have items from the unquoted_args as they were initially.
399 args_as_str = _get_str_type_compatible("", unquoted_args)
400
401 params_with_value_in_separate_arg = (
402 "--check-hash-based-pycs",
403 "--jit", # pypy option
404 )
405
406 # All short switches may be combined together. The ones below require a value and the
407 # value itself may be embedded in the arg.
408 #
409 # i.e.: Python accepts things as:
410 #
411 # python -OQold -qmtest
412 #
413 # Which is the same as:
414 #
415 # python -O -Q old -q -m test
416 #
417 # or even:
418 #
419 # python -OQold "-vcimport sys;print(sys)"
420 #
421 # Which is the same as:
422 #
423 # python -O -Q old -v -c "import sys;print(sys)"
424
425 params_with_combinable_arg = set(("W", "X", "Q", "c", "m"))
426
427 module_name = None
428 before_module_flag = ""
429 module_name_i_start = -1
430 module_name_i_end = -1
431
432 code = None
433 code_i = -1
434 code_i_end = -1
435 code_flag = ""
436
437 filename = None
438 filename_i = -1
439
440 ignore_next = True # start ignoring the first (the first entry is the python executable)
441 for i, arg_as_str in enumerate(args_as_str):
442 if ignore_next:
443 ignore_next = False
444 continue
445
446 if arg_as_str.startswith("-"):
447 if arg_as_str == "-":
448 # Contents will be read from the stdin. This is not currently handled.
449 pydev_log.debug('Unable to fix arguments to attach debugger on subprocess when reading from stdin ("python ... -").')
450 return original_args
451
452 if arg_as_str.startswith(params_with_value_in_separate_arg):
453 if arg_as_str in params_with_value_in_separate_arg:
454 ignore_next = True
455 continue
456
457 break_out = False
458 for j, c in enumerate(arg_as_str):
459 # i.e.: Python supports -X faulthandler as well as -Xfaulthandler
460 # (in one case we have to ignore the next and in the other we don't
461 # have to ignore it).
462 if c in params_with_combinable_arg:
463 remainder = arg_as_str[j + 1 :]
464 if not remainder:
465 ignore_next = True
466
467 if c == "m":
468 # i.e.: Something as
469 # python -qm test
470 # python -m test
471 # python -qmtest
472 before_module_flag = arg_as_str[:j] # before_module_flag would then be "-q"
473 if before_module_flag == "-":
474 before_module_flag = ""
475 module_name_i_start = i
476 if not remainder:
477 module_name = unquoted_args[i + 1]
478 module_name_i_end = i + 1
479 else:
480 # i.e.: python -qmtest should provide 'test' as the module_name
481 module_name = unquoted_args[i][j + 1 :]
482 module_name_i_end = module_name_i_start
483 break_out = True
484 break
485
486 elif c == "c":
487 # i.e.: Something as
488 # python -qc "import sys"
489 # python -c "import sys"
490 # python "-qcimport sys"
491 code_flag = arg_as_str[: j + 1] # code_flag would then be "-qc"
492
493 if not remainder:
494 # arg_as_str is something as "-qc", "import sys"
495 code = unquoted_args[i + 1]
496 code_i_end = i + 2
497 else:
498 # if arg_as_str is something as "-qcimport sys"
499 code = remainder # code would be "import sys"
500 code_i_end = i + 1
501 code_i = i
502 break_out = True
503 break
504
505 else:
506 break
507
508 if break_out:
509 break
510
511 else:
512 # It doesn't start with '-' and we didn't ignore this entry:
513 # this means that this is the file to be executed.
514 filename = unquoted_args[i]
515
516 # Note that the filename is not validated here.
517 # There are cases where even a .exe is valid (xonsh.exe):
518 # https://github.com/microsoft/debugpy/issues/945
519 # So, we should support whatever runpy.run_path
520 # supports in this case.
521
522 filename_i = i
523
524 if _is_managed_arg(filename): # no need to add pydevd twice
525 pydev_log.debug("Skipped monkey-patching as pydevd.py is in args already.")
526 return original_args
527
528 break
529 else:
530 # We didn't find the filename (something is unexpected).
531 pydev_log.debug("Unable to fix arguments to attach debugger on subprocess (filename not found).")
532 return original_args
533
534 if code_i != -1:
535 host, port = _get_host_port()
536
537 if port is not None:
538 new_args = []
539 new_args.extend(unquoted_args[:code_i])
540 new_args.append(code_flag)
541 new_args.append(_get_python_c_args(host, port, code, unquoted_args, SetupHolder.setup))
542 new_args.extend(unquoted_args[code_i_end:])
543
544 return quote_args(new_args)
545
546 first_non_vm_index = max(filename_i, module_name_i_start)
547 if first_non_vm_index == -1:
548 pydev_log.debug("Unable to fix arguments to attach debugger on subprocess (could not resolve filename nor module name).")
549 return original_args
550
551 # Original args should be something as:
552 # ['X:\\pysrc\\pydevd.py', '--multiprocess', '--print-in-debugger-startup',
553 # '--vm_type', 'python', '--client', '127.0.0.1', '--port', '56352', '--file', 'x:\\snippet1.py']
554 from _pydevd_bundle.pydevd_command_line_handling import setup_to_argv
555
556 new_args = []
557 new_args.extend(unquoted_args[:first_non_vm_index])
558 if before_module_flag:
559 new_args.append(before_module_flag)
560
561 add_module_at = len(new_args) + 1
562
563 new_args.extend(
564 setup_to_argv(
565 _get_setup_updated_with_protocol_and_ppid(SetupHolder.setup, is_exec=is_exec), skip_names=set(("module", "cmd-line"))
566 )
567 )
568 new_args.append("--file")
569
570 if module_name is not None:
571 assert module_name_i_start != -1
572 assert module_name_i_end != -1
573 # Always after 'pydevd' (i.e.: pydevd "--module" --multiprocess ...)
574 new_args.insert(add_module_at, "--module")
575 new_args.append(module_name)
576 new_args.extend(unquoted_args[module_name_i_end + 1 :])
577
578 elif filename is not None:
579 assert filename_i != -1
580 new_args.append(filename)
581 new_args.extend(unquoted_args[filename_i + 1 :])
582
583 else:
584 raise AssertionError("Internal error (unexpected condition)")
585
586 return quote_args(new_args)
587 except:
588 pydev_log.exception("Error patching args (debugger not attached to subprocess).")
589 return original_args

592def str_to_args_windows(args):
593 # See https://docs.microsoft.com/en-us/cpp/c-language/parsing-c-command-line-arguments.
594 #
595 # Implemetation ported from DebugPlugin.parseArgumentsWindows:
596 # https://github.com/eclipse/eclipse.platform.debug/blob/master/org.eclipse.debug.core/core/org/eclipse/debug/core/DebugPlugin.java
597
598 result = []
599
600 DEFAULT = 0
601 ARG = 1
602 IN_DOUBLE_QUOTE = 2
603
604 state = DEFAULT
605 backslashes = 0
606 buf = ""
607
608 args_len = len(args)
609 for i in range(args_len):
610 ch = args[i]
611 if ch == "\\":
612 backslashes += 1
613 continue
614 elif backslashes != 0:
615 if ch == '"':
616 while backslashes >= 2:
617 backslashes -= 2
618 buf += "\\"
619 if backslashes == 1:
620 if state == DEFAULT:
621 state = ARG
622
623 buf += '"'
624 backslashes = 0
625 continue
626 # else fall through to switch
627 else:
628 # false alarm, treat passed backslashes literally...
629 if state == DEFAULT:
630 state = ARG
631
632 while backslashes > 0:
633 backslashes -= 1
634 buf += "\\"
635 # fall through to switch
636 if ch in (" ", "\t"):
637 if state == DEFAULT:
638 # skip
639 continue
640 elif state == ARG:
641 state = DEFAULT
642 result.append(buf)
643 buf = ""
644 continue
645
646 if state in (DEFAULT, ARG):
647 if ch == '"':
648 state = IN_DOUBLE_QUOTE
649 else:
650 state = ARG
651 buf += ch
652
653 elif state == IN_DOUBLE_QUOTE:
654 if ch == '"':
655 if i + 1 < args_len and args[i + 1] == '"':
656 # Undocumented feature in Windows:
657 # Two consecutive double quotes inside a double-quoted argument are interpreted as
658 # a single double quote.
659 buf += '"'
660 i += 1
661 else:
662 state = ARG
663 else:
664 buf += ch
665
666 else:
667 raise RuntimeError("Illegal condition")
668
669 if len(buf) > 0 or state != DEFAULT:
670 result.append(buf)
671
672 return result
