103 def put_string_line(self, txt):
104 font_engine = self.font_engine
105 shape_result = font_engine.shape(txt)
106 x_advance = shape_result.x_advance * self.style.font_size
107 y_advance = shape_result.y_advance * self.style.font_size
108 ops = shape_result.graphics_ops
109 if font_engine.uses_complex_positioning:
110 # Tm and Tlm will be at the same position, after the last glyph
111 newline_x = -x_advance
112 newline_y = -y_advance
113
114 # TODO deal with TTB scripts with LTR line order too, like Mongolian
115 vertical = self.style.vertical_text
116 leading = self.leading
117 if vertical:
118 newline_x -= leading
119 extent = abs(y_advance)
120 else:
121 newline_y -= leading
122 extent = abs(x_advance)
123
124 ops += b' %g %g Td' % (newline_x, newline_y)
125 else:
126 ops += b' T*'
127 extent = abs(x_advance)
128
129 return ops, extent

190 def render(self):
191
192 style = self.style
193
194 self.set_resource(
195 category=ResourceType.FONT, name=pdf_name('/' + self.font_name),
196 value=self.font_engine.as_resource()
197 )
198 leading = self.leading
199
200 nat_text_width = self._nat_text_width
201 nat_text_height = self._nat_text_height
202
203 vertical = self.style.vertical_text
204
205 box_layout = self.style.box_layout_rule
206 if box_layout is None:
207 margins = layout.Margins.uniform(DEFAULT_TEXT_BOX_MARGIN)
208 box_layout = DEFAULT_BOX_LAYOUT.substitute_margins(margins)
209
210 positioning = box_layout.fit(self.box, nat_text_width, nat_text_height)
211
212 command_stream = []
213
214 # draw border before scaling
215 if style.border_width:
216 command_stream.append(
217 b'q %g w 0 0 %g %g re S Q' % (
218 style.border_width, self.box.width, self.box.height
219 )
220 )
221
222 # reposition cursor
223 command_stream.append(positioning.as_cm())
224
225 command_stream += [
226 b'BT',
227 b'/%s %d Tf %d TL' % (
228 self.font_name.encode('latin1'),
229 style.font_size, leading
230 ),
231 ]
232
233 # start by moving the cursor to the starting position.
234 # In horizontal mode, that's the top left, accounting for leading.
235 # In vertical mode, we need the top right.
236 if vertical:
237 text_cursor_start = b'%g %g Td' % (
238 # V-mode leading/baselining is weird like that---the glyph
239 # origin is in the middle, so we chop off half of the leading
240 nat_text_width * positioning.x_scale - leading / 2,
241 nat_text_height * positioning.y_scale
242 )
243 else:
244 text_cursor_start = b'0 %g Td' % (
245 nat_text_height * positioning.y_scale - leading
246 )
247 command_stream.append(text_cursor_start)
248
249 command_stream.extend(self._wrapped_lines)
250 command_stream.append(b'ET')
251 return b' '.join(command_stream)
