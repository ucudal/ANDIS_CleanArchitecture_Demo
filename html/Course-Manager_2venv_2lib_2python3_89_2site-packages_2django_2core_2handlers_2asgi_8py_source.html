23class ASGIRequest(HttpRequest):
24 """
25 Custom request subclass that decodes from an ASGI-standard request dict
26 and wraps request body handling.
27 """
28 # Number of seconds until a Request gives up on trying to read a request

32 def __init__(self, scope, body_file):
33 self.scope = scope
34 self._post_parse_error = False
35 self._read_started = False
36 self.resolver_match = None
37 self.script_name = self.scope.get('root_path', '')
38 if self.script_name and scope['path'].startswith(self.script_name):
39 # TODO: Better is-prefix checking, slash handling?
40 self.path_info = scope['path'][len(self.script_name):]
41 else:
42 self.path_info = scope['path']
43 # The Django path is different from ASGI scope path args, it should
44 # combine with script name.
45 if self.script_name:
46 self.path = '%s/%s' % (
47 self.script_name.rstrip('/'),
48 self.path_info.replace('/', '', 1),
49 )
50 else:
51 self.path = scope['path']
52 # HTTP basics.
53 self.method = self.scope['method'].upper()
54 # Ensure query string is encoded correctly.
55 query_string = self.scope.get('query_string', '')
56 if isinstance(query_string, bytes):
57 query_string = query_string.decode()
58 self.META = {
59 'REQUEST_METHOD': self.method,
60 'QUERY_STRING': query_string,
61 'SCRIPT_NAME': self.script_name,
62 'PATH_INFO': self.path_info,
63 # WSGI-expecting code will need these for a while
64 'wsgi.multithread': True,
65 'wsgi.multiprocess': True,
66 }
67 if self.scope.get('client'):
68 self.META['REMOTE_ADDR'] = self.scope['client'][0]
69 self.META['REMOTE_HOST'] = self.META['REMOTE_ADDR']
70 self.META['REMOTE_PORT'] = self.scope['client'][1]
71 if self.scope.get('server'):
72 self.META['SERVER_NAME'] = self.scope['server'][0]
73 self.META['SERVER_PORT'] = str(self.scope['server'][1])
74 else:
75 self.META['SERVER_NAME'] = 'unknown'
76 self.META['SERVER_PORT'] = '0'
77 # Headers go into META.
78 for name, value in self.scope.get('headers', []):
79 name = name.decode('latin1')
80 if name == 'content-length':
81 corrected_name = 'CONTENT_LENGTH'
82 elif name == 'content-type':
83 corrected_name = 'CONTENT_TYPE'
84 else:
85 corrected_name = 'HTTP_%s' % name.upper().replace('-', '_')
86 # HTTP/2 say only ASCII chars are allowed in headers, but decode
87 # latin1 just in case.
88 value = value.decode('latin1')
89 if corrected_name in self.META:
90 value = self.META[corrected_name] + ',' + value
91 self.META[corrected_name] = value
92 # Pull out request encoding, if provided.
93 self._set_content_type_params(self.META)
94 # Directly assign the body file to be our stream.
95 self._stream = body_file
96 # Other bits.
97 self.resolver_match = None

128 """Handler for ASGI requests."""

137 async def __call__(self, scope, receive, send):
138 """
139 Async entrypoint - parses the request and hands off to get_response.
140 """
141 # Serve only HTTP connections.
142 # FIXME: Allow to override this.
143 if scope['type'] != 'http':
144 raise ValueError(
145 'Django can only handle ASGI/HTTP connections, not %s.'
146 % scope['type']
147 )
148 # Receive the HTTP request body as a stream object.
149 try:
150 body_file = await self.read_body(receive)
151 except RequestAborted:
152 return
153 # Request is complete and can be served.
154 set_script_prefix(self.get_script_prefix(scope))
155 await sync_to_async(signals.request_started.send)(sender=self.__class__, scope=scope)
156 # Get the request and check for basic issues.
157 request, error_response = self.create_request(scope, body_file)
158 if request is None:
159 await self.send_response(error_response, send)
160 return
161 # Get the response, using a threadpool via sync_to_async, if needed.
162 if asyncio.iscoroutinefunction(self.get_response):
163 response = await self.get_response(request)
164 else:
165 # If get_response is synchronous, run it non-blocking.
166 response = await sync_to_async(self.get_response)(request)
167 response._handler_class = self.__class__
168 # Increase chunk size on file responses (ASGI servers handles low-level
169 # chunking).
170 if isinstance(response, FileResponse):
171 response.block_size = self.chunk_size
172 # Send the response.
173 await self.send_response(response, send)

175 async def read_body(self, receive):
176 """Reads a HTTP body from an ASGI connection."""
177 # Use the tempfile that auto rolls-over to a disk file as it fills up.
178 body_file = tempfile.SpooledTemporaryFile(max_size=settings.FILE_UPLOAD_MAX_MEMORY_SIZE, mode='w+b')
179 while True:
180 message = await receive()
181 if message['type'] == 'http.disconnect':
182 # Early client disconnect.
183 raise RequestAborted()
184 # Add a body chunk from the message, if provided.
185 if 'body' in message:
186 body_file.write(message['body'])
187 # Quit out if that's the end.
188 if not message.get('more_body', False):
189 break
190 body_file.seek(0)
191 return body_file

222 async def send_response(self, response, send):
223 """Encode and send a response out over ASGI."""
224 # Collect cookies into headers. Have to preserve header case as there
225 # are some non-RFC compliant clients that require e.g. Content-Type.
226 response_headers = []
227 for header, value in response.items():
228 if isinstance(header, str):
229 header = header.encode('ascii')
230 if isinstance(value, str):
231 value = value.encode('latin1')
232 response_headers.append((bytes(header), bytes(value)))
233 for c in response.cookies.values():
234 response_headers.append(
235 (b'Set-Cookie', c.output(header='').encode('ascii').strip())
236 )
237 # Initial response message.
238 await send({
239 'type': 'http.response.start',
240 'status': response.status_code,
241 'headers': response_headers,
242 })
243 # Streaming responses need to be pinned to their iterator.
244 if response.streaming:
245 # Access `__iter__` and not `streaming_content` directly in case
246 # it has been overridden in a subclass.
247 for part in response:
248 for chunk, _ in self.chunk_bytes(part):
249 await send({
250 'type': 'http.response.body',
251 'body': chunk,
252 # Ignore "more" as there may be more parts; instead,
253 # use an empty final closing message with False.
254 'more_body': True,
255 })
256 # Final closing message.
257 await send({'type': 'http.response.body'})
258 # Other responses just need chunking.
259 else:
260 # Yield chunks of response.
261 for chunk, last in self.chunk_bytes(response.content):
262 await send({
263 'type': 'http.response.body',
264 'body': chunk,
265 'more_body': not last,
266 })
267 response.close()

269 @classmethod
270 def chunk_bytes(cls, data):
271 """
272 Chunks some data up so it can be sent in reasonable size messages.
273 Yields (chunk, last_chunk) tuples.
274 """
275 position = 0
276 if not data:
277 yield data, True
278 return
279 while position < len(data):
280 yield (
281 data[position:position + cls.chunk_size],
282 (position + cls.chunk_size) >= len(data),
283 )
284 position += cls.chunk_size
