19RunStatus = Literal["pending", "error", "success", "timeout", "interrupted"]
20"""
21Represents the status of a run:
22- "pending": The run is waiting to start.
23- "error": The run encountered an error and stopped.
24- "success": The run completed successfully.
25- "timeout": The run exceeded its time limit.
26- "interrupted": The run was manually stopped or interrupted.
27"""
28
29ThreadStatus = Literal["idle", "busy", "interrupted", "error"]
30"""
31Represents the status of a thread:
32- "idle": The thread is not currently processing any task.
33- "busy": The thread is actively processing a task.
34- "interrupted": The thread's execution was interrupted.
35- "error": An exception occurred during task processing.
36"""
37
38StreamMode = Literal[
39 "values", "messages", "updates", "events", "debug", "custom", "messages-tuple"
40]
41"""
42Defines the mode of streaming:
43- "values": Stream only the values.
44- "messages": Stream complete messages.
45- "updates": Stream updates to the state.
46- "events": Stream events occurring during execution.
47- "debug": Stream detailed debug information.
48- "custom": Stream custom events.
49"""
50
51DisconnectMode = Literal["cancel", "continue"]
52"""
53Specifies behavior on disconnection:
54- "cancel": Cancel the operation on disconnection.
55- "continue": Continue the operation even if disconnected.
56"""
57
58MultitaskStrategy = Literal["reject", "interrupt", "rollback", "enqueue"]
59"""
60Defines how to handle multiple tasks:
61- "reject": Reject new tasks when busy.
62- "interrupt": Interrupt current task for new ones.
63- "rollback": Roll back current task and start new one.
64- "enqueue": Queue new tasks for later execution.
65"""
66
67OnConflictBehavior = Literal["raise", "do_nothing"]
68"""
69Specifies behavior on conflict:
70- "raise": Raise an exception when a conflict occurs.
71- "do_nothing": Ignore conflicts and proceed.
72"""
73
74OnCompletionBehavior = Literal["delete", "keep"]
75"""
76Defines action after completion:
77- "delete": Delete resources after completion.
78- "keep": Retain resources after completion.
79"""
80
81All = Literal["*"]
82"""Represents a wildcard or 'all' selector."""
83
84IfNotExists = Literal["create", "reject"]
85"""
86Specifies behavior if the thread doesn't exist:
87- "create": Create a new thread if it doesn't exist.
88- "reject": Reject the operation if the thread doesn't exist.
89"""

91CancelAction = Literal["interrupt", "rollback"]
92"""
93Action to take when cancelling the run.
94- "interrupt": Simply cancel the run.
95- "rollback": Cancel the run. Then delete the run and associated checkpoints.
96"""
