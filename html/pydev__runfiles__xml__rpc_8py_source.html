95 # It is necessary to specify an encoding, that matches
96 # the encoding of all bytes-strings passed into an
97 # XMLRPC call: "All 8-bit strings in the data structure are assumed to use the
98 # packet encoding. Unicode strings are automatically converted,
99 # where necessary."
100 # Byte strings most likely come from file names.
101 encoding = file_system_encoding
102 if encoding == "mbcs":
103 # Windos symbolic name for the system encoding CP_ACP.
104 # We need to convert it into a encoding that is recognized by Java.
105 # Unfortunately this is not always possible. You could use
106 # GetCPInfoEx and get a name similar to "windows-1251". Then
107 # you need a table to translate on a best effort basis. Much to complicated.
108 # ISO-8859-1 is good enough.
109 encoding = "ISO-8859-1"
110
111 self.server = xmlrpclib.Server("http://%s:%s" % (pydev_localhost.get_localhost(), port), encoding=encoding)

113 def run(self):
114 while True:
115 kill_found = False
116 commands = []
117 command = self.notifications_queue.get(block=True)
118 if isinstance(command, KillServer):
119 kill_found = True
120 else:
121 assert isinstance(command, ParallelNotification)
122 commands.append(command.to_tuple())
123
124 try:
125 while True:
126 command = self.notifications_queue.get(block=False) # No block to create a batch.
127 if isinstance(command, KillServer):
128 kill_found = True
129 else:
130 assert isinstance(command, ParallelNotification)
131 commands.append(command.to_tuple())
132 except:
133 pass # That's OK, we're getting it until it becomes empty so that we notify multiple at once.
134
135 if commands:
136 try:
137 self.server.notifyCommands(commands)
138 except:
139 traceback.print_exc()
140
141 if kill_found:
142 self.finished = True
143 return

199def _encode_if_needed(obj):
200 # In the java side we expect strings to be ISO-8859-1 (org.python.pydev.debug.pyunit.PyUnitServer.initializeDispatches().new Dispatch() {...}.getAsStr(Object))
201 if isinstance(obj, str): # Unicode in py3
202 return xmlrpclib.Binary(obj.encode("ISO-8859-1", "xmlcharrefreplace"))
203
204 elif isinstance(obj, bytes):
205 try:
206 return xmlrpclib.Binary(obj.decode(sys.stdin.encoding, "replace").encode("ISO-8859-1", "xmlcharrefreplace"))
207 except:
208 return xmlrpclib.Binary(obj) # bytes already
209
210 return obj
