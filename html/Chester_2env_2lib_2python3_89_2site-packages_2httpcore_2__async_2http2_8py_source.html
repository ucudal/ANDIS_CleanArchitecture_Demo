123 async def send_connection_init(self, timeout: TimeoutDict) -> None:
124 """
125 The HTTP/2 connection requires some initial setup before we can start
126 using individual request/response streams on it.
127 """
128 # Need to set these manually here instead of manipulating via
129 # __setitem__() otherwise the H2Connection will emit SettingsUpdate
130 # frames in addition to sending the undesired defaults.
131 self.h2_state.local_settings = Settings(
132 client=True,
133 initial_values={
134 # Disable PUSH_PROMISE frames from the server since we don't do anything
135 # with them for now. Maybe when we support caching?
136 SettingCodes.ENABLE_PUSH: 0,
137 # These two are taken from h2 for safe defaults
138 SettingCodes.MAX_CONCURRENT_STREAMS: 100,
139 SettingCodes.MAX_HEADER_LIST_SIZE: 65536,
140 },
141 )
142
143 # Some websites (*cough* Yahoo *cough*) balk at this setting being
144 # present in the initial handshake since it's not defined in the original
145 # RFC despite the RFC mandating ignoring settings you don't know about.
146 del self.h2_state.local_settings[
147 h2.settings.SettingCodes.ENABLE_CONNECT_PROTOCOL
148 ]
149
150 logger.trace("initiate_connection=%r", self)
151 self.h2_state.initiate_connection()
152 self.h2_state.increment_flow_control_window(2 ** 24)
153 data_to_send = self.h2_state.data_to_send()
154 await self.socket.write(data_to_send, timeout)

170 async def wait_for_outgoing_flow(self, stream_id: int, timeout: TimeoutDict) -> int:
171 """
172 Returns the maximum allowable outgoing flow for a given stream.
173 If the allowable flow is zero, then waits on the network until
174 WindowUpdated frames have increased the flow rate.
175 https://tools.ietf.org/html/rfc7540#section-6.9
176 """
177 local_flow = self.h2_state.local_flow_control_window(stream_id)
178 connection_flow = self.h2_state.max_outbound_frame_size
179 flow = min(local_flow, connection_flow)
180 while flow == 0:
181 await self.receive_events(timeout)
182 local_flow = self.h2_state.local_flow_control_window(stream_id)
183 connection_flow = self.h2_state.max_outbound_frame_size
184 flow = min(local_flow, connection_flow)
185 return flow

200 async def receive_events(self, timeout: TimeoutDict) -> None:
201 """
202 Read some data from the network, and update the H2 state.
203 """
204 data = await self.socket.read(self.READ_NUM_BYTES, timeout)
205 events = self.h2_state.receive_data(data)
206 for event in events:

208 logger.trace("receive_event stream_id=%r event=%s", event_stream_id, event)

210 if hasattr(event, "error_code"):
211 raise ProtocolError(event)
212
213 if event_stream_id in self.events:
214 self.events[event_stream_id].append(event)
215
216 data_to_send = self.h2_state.data_to_send()
217 await self.socket.write(data_to_send, timeout)

337 async def receive_response(
338 self, timeout: TimeoutDict
339 ) -> Tuple[int, List[Tuple[bytes, bytes]]]:
340 """
341 Read the response status and headers from the network.
342 """
343 while True:
344 event = await self.connection.wait_for_event(self.stream_id, timeout)
345 if isinstance(event, h2.events.ResponseReceived):
346 break
347
348 status_code = 200
349 headers = []
350 for k, v in event.headers:
351 if k == b":status":
352 status_code = int(v.decode("ascii", errors="ignore"))
353 elif not k.startswith(b":"):
354 headers.append((k, v))
355
356 return (status_code, headers)
