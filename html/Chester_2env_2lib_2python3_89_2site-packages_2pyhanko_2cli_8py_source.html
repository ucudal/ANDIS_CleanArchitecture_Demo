64 log_config: LogConfig

383def _validate_detached(infile, sig_infile, validation_context,
384 key_usage_settings):
385 sig_bytes = sig_infile.read()
386 try:
387 if pem.detect(sig_bytes):
388 _, _, sig_bytes = pem.unarmor(sig_bytes)
389 content_info = cms.ContentInfo.load(sig_bytes)
390 if content_info['content_type'].native != 'signed_data':
391 raise click.ClickException("CMS content type is not signedData")
392 except ValueError as e:
393 raise click.ClickException("Could not parse CMS object") from e
394
395 validation_coro = validation.async_validate_detached_cms(
396 infile, signed_data=content_info['content'],
397 signer_validation_context=validation_context,
398 key_usage_settings=key_usage_settings
399 )
400 return asyncio.run(validation_coro)

403def _signature_status_str(status_callback, pretty_print, executive_summary):
404 try:
405 status = status_callback()
406 if executive_summary and not pretty_print:
407 return 'VALID' if status.bottom_line else 'INVALID', status.bottom_line
408 elif pretty_print:
409 return status.pretty_print_details(), status.bottom_line
410 else:
411 return status.summary(), status.bottom_line
412 except validation.ValidationInfoReadingError as e:
413 msg = (
414 'An error occurred while parsing the revocation information '
415 'for this signature: ' + str(e)
416 )
417 logger.error(msg)
418 if pretty_print:
419 return msg, False
420 else:
421 return 'REVINFO_FAILURE', False
422 except SignatureValidationError as e:
423 msg = 'An error occurred while validating this signature: ' + str(e)
424 logger.error(msg, exc_info=e)
425 if pretty_print:
426 return msg, False
427 else:
428 return 'INVALID', False

488
489 if pretty_print and executive_summary:
490 raise click.ClickException(
491 "--pretty-print is incompatible with --executive-summary."
492 )
493
494 if ltv_profile is not None:
495 ltv_profile = RevocationInfoValidationType(ltv_profile)
496
497 vc_kwargs = _build_vc_kwargs(
498 ctx, validation_context, trust, trust_replace, other_certs,
499 retroactive_revinfo,
500 allow_fetching=False if no_revocation_check else None
501 )

504 vc_kwargs = _prepare_vc(
505 vc_kwargs, soft_revocation_check=soft_revocation_check,
506 force_revinfo=force_revinfo
507 )
508 with pyhanko_exception_manager():
509 if detached is not None:
510 (status_str, signature_ok) = _signature_status_str(
511 status_callback=lambda: _validate_detached(
512 infile, detached, ValidationContext(**vc_kwargs),
513 key_usage_settings
514 ),
515 pretty_print=pretty_print, executive_summary=executive_summary
516 )
517 if signature_ok:
518 print(status_str)
519 else:
520 raise click.ClickException(status_str)
521 return
522
523 if no_strict_syntax:
524 logger.info(
525 "Strict PDF syntax is disabled; this could impact validation "
526 "results. Use caution."
527 )
528 r = PdfFileReader(infile, strict=False)
529 else:
530 r = PdfFileReader(infile)
531 sh = r.security_handler
532 if isinstance(sh, crypt.StandardSecurityHandler):
533 if password is None:
534 password = getpass.getpass(prompt='File password: ')
535 auth_result = r.decrypt(password)

541 "validating (for now)"
542 )

567
568 if not all_signatures_ok:
569 raise click.ClickException("Validation failed")

768def _open_for_signing(infile_path, lenient, signer_cert=None, signer_key=None):
769 infile = open(infile_path, 'rb')
770 writer = IncrementalPdfFileWriter(infile, strict=not lenient)

780 writer.encrypt(pdf_pass)
781 elif isinstance(sh, crypt.PubKeySecurityHandler) \
782 and signer_key is not None:

789 "signer's public key, but be aware that using the same key "
790 "for both signing and decryption is considered bad practice. "
791 "Never use the same RSA key that you use to decrypt messages to"
792 "sign hashes that you didn't compute yourself!" % infile_path

798 "credentials are not available."
