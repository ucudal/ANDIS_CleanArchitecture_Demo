48class Lexer(metaclass=LexerMeta):
49 """
50 Lexer for a specific language.
51
52 Basic options recognized:
53 ``stripnl``
54 Strip leading and trailing newlines from the input (default: True).
55 ``stripall``
56 Strip all leading and trailing whitespace from the input
57 (default: False).
58 ``ensurenl``
59 Make sure that the input ends with a newline (default: True). This
60 is required for some lexers that consume input linewise.
61
62 .. versionadded:: 1.3
63
64 ``tabsize``
65 If given and greater than 0, expand tabs in the input (default: 0).
66 ``encoding``
67 If given, must be an encoding name. This encoding will be used to
68 convert the input string to Unicode, if it is not already a Unicode
69 string (default: ``'guess'``, which uses a simple UTF-8 / Locale /
70 Latin1 detection. Can also be ``'chardet'`` to use the chardet
71 library, if it is installed.
72 ``inencoding``
73 Overrides the ``encoding`` if given.
74 """
75
76 #: Name of the lexer
77 name = None
78
79 #: URL of the language specification/definition
80 url = None
81
82 #: Shortcuts for the lexer
83 aliases = []
84
85 #: File name globs
86 filenames = []
87
88 #: Secondary file name globs
89 alias_filenames = []
90
91 #: MIME types
92 mimetypes = []
93
94 #: Priority, should multiple lexers match and no content is provided
95 priority = 0

97 def __init__(self, **options):
98 self.options = options
99 self.stripnl = get_bool_opt(options, 'stripnl', True)
100 self.stripall = get_bool_opt(options, 'stripall', False)
101 self.ensurenl = get_bool_opt(options, 'ensurenl', True)
102 self.tabsize = get_int_opt(options, 'tabsize', 0)
103 self.encoding = options.get('encoding', 'guess')
104 self.encoding = options.get('inencoding') or self.encoding
105 self.filters = []
106 for filter_ in get_list_opt(options, 'filters', ()):
107 self.add_filter(filter_)

124 def analyse_text(text):
125 """
126 Has to return a float between ``0`` and ``1`` that indicates
127 if a lexer wants to highlight this text. Used by ``guess_lexer``.
128 If this method returns ``0`` it won't highlight it in any case, if
129 it returns ``1`` highlighting with this lexer is guaranteed.
130
131 The `LexerMeta` metaclass automatically wraps this function so
132 that it works like a static method (no ``self`` or ``cls``
133 parameter) and the return value is automatically converted to
134 `float`. If the return value is an object that is boolean `False`
135 it's the same as if the return values was ``0.0``.
136 """
137
138 def get_tokens(self, text, unfiltered=False):
139 """
140 Return an iterable of (tokentype, value) pairs generated from
141 `text`. If `unfiltered` is set to `True`, the filtering mechanism
142 is bypassed even if filters are defined.
143
144 Also preprocess the text, i.e. expand tabs and strip it if
145 wanted and applies registered filters.
146 """
147 if not isinstance(text, str):
148 if self.encoding == 'guess':
149 text, _ = guess_decode(text)
150 elif self.encoding == 'chardet':
151 try:
152 from pip._vendor import chardet
153 except ImportError as e:
154 raise ImportError('To enable chardet encoding guessing, '
155 'please install the chardet library '
156 'from http://chardet.feedparser.org/') from e
157 # check for BOM first
158 decoded = None
159 for bom, encoding in _encoding_map:
160 if text.startswith(bom):
161 decoded = text[len(bom):].decode(encoding, 'replace')
162 break
163 # no BOM found, so use chardet
164 if decoded is None:
165 enc = chardet.detect(text[:1024]) # Guess using first 1KB
166 decoded = text.decode(enc.get('encoding') or 'utf-8',
167 'replace')
168 text = decoded
169 else:
170 text = text.decode(self.encoding)
171 if text.startswith('\ufeff'):
172 text = text[len('\ufeff'):]
173 else:
174 if text.startswith('\ufeff'):
175 text = text[len('\ufeff'):]
176
177 # text now *is* a unicode string
178 text = text.replace('\r\n', '\n')
179 text = text.replace('\r', '\n')
180 if self.stripall:
181 text = text.strip()
182 elif self.stripnl:
183 text = text.strip('\n')
184 if self.tabsize > 0:
185 text = text.expandtabs(self.tabsize)
186 if self.ensurenl and not text.endswith('\n'):
187 text += '\n'
188
189 def streamer():
190 for _, t, v in self.get_tokens_unprocessed(text):
191 yield t, v
192 stream = streamer()
193 if not unfiltered:
194 stream = apply_filters(stream, self.filters, self)
195 return stream

198 """
199 Return an iterable of (index, tokentype, value) pairs where "index"
200 is the starting position of the token within the input text.
201
202 In subclasses, implement this method as a generator to
203 maximize effectiveness.
204 """
205 raise NotImplementedError

304def bygroups(*args):
305 """
306 Callback that yields multiple actions for each group in the match.
307 """
308 def callback(lexer, match, ctx=None):
309 for i, action in enumerate(args):
310 if action is None:
311 continue
312 elif type(action) is _TokenType:
313 data = match.group(i + 1)
314 if data:
315 yield match.start(i + 1), action, data
316 else:
317 data = match.group(i + 1)
318 if data is not None:
319 if ctx:
320 ctx.pos = match.start(i + 1)
321 for item in action(lexer,
322 _PseudoMatch(match.start(i + 1), data), ctx):
323 if item:
324 yield item
325 if ctx:
326 ctx.pos = match.end()
327 return callback

339def using(_other, **kwargs):
340 """
341 Callback that processes the match with a different lexer.
342
343 The keyword arguments are forwarded to the lexer, except `state` which
344 is handled separately.
345
346 `state` specifies the state that the new lexer will start in, and can
347 be an enumerable such as ('root', 'inline', 'string') or a simple
348 string which is assumed to be on top of the root state.
349
350 Note: For that to work, `_other` must not be an `ExtendedRegexLexer`.
351 """
352 gt_kwargs = {}
353 if 'state' in kwargs:
354 s = kwargs.pop('state')
355 if isinstance(s, (list, tuple)):
356 gt_kwargs['stack'] = s
357 else:
358 gt_kwargs['stack'] = ('root', s)
359
360 if _other is this:
361 def callback(lexer, match, ctx=None):
362 # if keyword arguments are given the callback
363 # function has to create a new lexer instance
364 if kwargs:
365 # XXX: cache that somehow
366 kwargs.update(lexer.options)
367 lx = lexer.__class__(**kwargs)
368 else:
369 lx = lexer
370 s = match.start()
371 for i, t, v in lx.get_tokens_unprocessed(match.group(), **gt_kwargs):
372 yield i + s, t, v
373 if ctx:
374 ctx.pos = match.end()
375 else:
376 def callback(lexer, match, ctx=None):
377 # XXX: cache that somehow
378 kwargs.update(lexer.options)
379 lx = _other(**kwargs)
380
381 s = match.start()
382 for i, t, v in lx.get_tokens_unprocessed(match.group(), **gt_kwargs):
383 yield i + s, t, v
384 if ctx:
385 ctx.pos = match.end()
386 return callback

417class RegexLexerMeta(LexerMeta):
418 """
419 Metaclass for RegexLexer, creates the self._tokens attribute from
420 self.tokens on the first instantiation.
421 """
422
423 def _process_regex(cls, regex, rflags, state):
424 """Preprocess the regular expression component of a token definition."""
425 if isinstance(regex, Future):
426 regex = regex.get()
427 return re.compile(regex, rflags).match
428
429 def _process_token(cls, token):
430 """Preprocess the token component of a token definition."""
431 assert type(token) is _TokenType or callable(token), \
432 'token type must be simple type or callable, not %r' % (token,)
433 return token
434
435 def _process_new_state(cls, new_state, unprocessed, processed):
436 """Preprocess the state transition action of a token definition."""
437 if isinstance(new_state, str):
438 # an existing state
439 if new_state == '#pop':
440 return -1
441 elif new_state in unprocessed:
442 return (new_state,)
443 elif new_state == '#push':
444 return new_state
445 elif new_state[:5] == '#pop:':
446 return -int(new_state[5:])
447 else:
448 assert False, 'unknown new state %r' % new_state
449 elif isinstance(new_state, combined):
450 # combine a new state from existing ones
451 tmp_state = '_tmp_%d' % cls._tmpname
452 cls._tmpname += 1
453 itokens = []
454 for istate in new_state:
455 assert istate != new_state, 'circular state ref %r' % istate
456 itokens.extend(cls._process_state(unprocessed,
457 processed, istate))
458 processed[tmp_state] = itokens
459 return (tmp_state,)
460 elif isinstance(new_state, tuple):
461 # push more than one state
462 for istate in new_state:
463 assert (istate in unprocessed or
464 istate in ('#pop', '#push')), \
465 'unknown new state ' + istate
466 return new_state
467 else:
468 assert False, 'unknown new state def %r' % new_state
469
470 def _process_state(cls, unprocessed, processed, state):
471 """Preprocess a single state definition."""
472 assert type(state) is str, "wrong state name %r" % state
473 assert state[0] != '#', "invalid state name %r" % state
474 if state in processed:
475 return processed[state]
476 tokens = processed[state] = []
477 rflags = cls.flags
478 for tdef in unprocessed[state]:
479 if isinstance(tdef, include):
480 # it's a state reference
481 assert tdef != state, "circular state reference %r" % state
482 tokens.extend(cls._process_state(unprocessed, processed,
483 str(tdef)))
484 continue
485 if isinstance(tdef, _inherit):
486 # should be processed already, but may not in the case of:
487 # 1. the state has no counterpart in any parent
488 # 2. the state includes more than one 'inherit'
489 continue
490 if isinstance(tdef, default):
491 new_state = cls._process_new_state(tdef.state, unprocessed, processed)
492 tokens.append((re.compile('').match, None, new_state))
493 continue
494
495 assert type(tdef) is tuple, "wrong rule def %r" % tdef
496
497 try:
498 rex = cls._process_regex(tdef[0], rflags, state)
499 except Exception as err:
500 raise ValueError("uncompilable regex %r in state %r of %r: %s" %
501 (tdef[0], state, cls, err)) from err
502
503 token = cls._process_token(tdef[1])
504
505 if len(tdef) == 2:
506 new_state = None
507 else:
508 new_state = cls._process_new_state(tdef[2],
509 unprocessed, processed)
510
511 tokens.append((rex, token, new_state))
512 return tokens
513
514 def process_tokendef(cls, name, tokendefs=None):
515 """Preprocess a dictionary of token definitions."""
516 processed = cls._all_tokens[name] = {}
517 tokendefs = tokendefs or cls.tokens[name]
518 for state in list(tokendefs):
519 cls._process_state(tokendefs, processed, state)
520 return processed
521
522 def get_tokendefs(cls):
523 """
524 Merge tokens from superclasses in MRO order, returning a single tokendef
525 dictionary.
526
527 Any state that is not defined by a subclass will be inherited
528 automatically. States that *are* defined by subclasses will, by
529 default, override that state in the superclass. If a subclass wishes to
530 inherit definitions from a superclass, it can use the special value
531 "inherit", which will cause the superclass' state definition to be
532 included at that point in the state.
533 """
534 tokens = {}
535 inheritable = {}
536 for c in cls.__mro__:
537 toks = c.__dict__.get('tokens', {})
538
539 for state, items in toks.items():
540 curitems = tokens.get(state)
541 if curitems is None:
542 # N.b. because this is assigned by reference, sufficiently
543 # deep hierarchies are processed incrementally (e.g. for
544 # A(B), B(C), C(RegexLexer), B will be premodified so X(B)
545 # will not see any inherits in B).
546 tokens[state] = items
547 try:
548 inherit_ndx = items.index(inherit)
549 except ValueError:
550 continue
551 inheritable[state] = inherit_ndx
552 continue
553
554 inherit_ndx = inheritable.pop(state, None)
555 if inherit_ndx is None:
556 continue
557
558 # Replace the "inherit" value with the items
559 curitems[inherit_ndx:inherit_ndx+1] = items
560 try:
561 # N.b. this is the index in items (that is, the superclass
562 # copy), so offset required when storing below.
563 new_inh_ndx = items.index(inherit)
564 except ValueError:
565 pass
566 else:
567 inheritable[state] = inherit_ndx + new_inh_ndx
568
569 return tokens
570
571 def __call__(cls, *args, **kwds):
572 """Instantiate cls after preprocessing its token definitions."""
573 if '_tokens' not in cls.__dict__:
574 cls._all_tokens = {}
575 cls._tmpname = 0
576 if hasattr(cls, 'token_variants') and cls.token_variants:
577 # don't process yet
578 pass
579 else:
580 cls._tokens = cls.process_tokendef('', cls.get_tokendefs())
581
582 return type.__call__(cls, *args, **kwds)

585class RegexLexer(Lexer, metaclass=RegexLexerMeta):
586 """
587 Base for simple stateful regular expression-based lexers.
588 Simplifies the lexing process so that you need only
589 provide a list of states and regular expressions.
590 """
591
592 #: Flags for compiling the regular expressions.
593 #: Defaults to MULTILINE.
594 flags = re.MULTILINE
595
596 #: At all time there is a stack of states. Initially, the stack contains
597 #: a single state 'root'. The top of the stack is called "the current state".
598 #:
599 #: Dict of ``{'state': [(regex, tokentype, new_state), ...], ...}``
600 #:
601 #: ``new_state`` can be omitted to signify no state transition.
602 #: If ``new_state`` is a string, it is pushed on the stack. This ensure
603 #: the new current state is ``new_state``.
604 #: If ``new_state`` is a tuple of strings, all of those strings are pushed
605 #: on the stack and the current state will be the last element of the list.
606 #: ``new_state`` can also be ``combined('state1', 'state2', ...)``
607 #: to signify a new, anonymous state combined from the rules of two
608 #: or more existing ones.
609 #: Furthermore, it can be '#pop' to signify going back one step in
610 #: the state stack, or '#push' to push the current state on the stack
611 #: again. Note that if you push while in a combined state, the combined
612 #: state itself is pushed, and not only the state in which the rule is
613 #: defined.
614 #:
615 #: The tuple can also be replaced with ``include('state')``, in which
616 #: case the rules from the state named by the string are included in the
617 #: current one.
618 tokens = {}
619
620 def get_tokens_unprocessed(self, text, stack=('root',)):
621 """
622 Split ``text`` into (tokentype, text) pairs.
623
624 ``stack`` is the initial stack (default: ``['root']``)
625 """
626 pos = 0
627 tokendefs = self._tokens
628 statestack = list(stack)
629 statetokens = tokendefs[statestack[-1]]
630 while 1:
631 for rexmatch, action, new_state in statetokens:
632 m = rexmatch(text, pos)
633 if m:
634 if action is not None:
635 if type(action) is _TokenType:
636 yield pos, action, m.group()
637 else:
638 yield from action(self, m)
639 pos = m.end()
640 if new_state is not None:
641 # state transition
642 if isinstance(new_state, tuple):
643 for state in new_state:
644 if state == '#pop':
645 if len(statestack) > 1:
646 statestack.pop()
647 elif state == '#push':
648 statestack.append(statestack[-1])
649 else:
650 statestack.append(state)
651 elif isinstance(new_state, int):
652 # pop, but keep at least one state on the stack
653 # (random code leading to unexpected pops should
654 # not allow exceptions)
655 if abs(new_state) >= len(statestack):
656 del statestack[1:]
657 else:
658 del statestack[new_state:]
659 elif new_state == '#push':
660 statestack.append(statestack[-1])
661 else:
662 assert False, "wrong state def: %r" % new_state
663 statetokens = tokendefs[statestack[-1]]
664 break
665 else:
666 # We are here only if all state tokens have been considered
667 # and there was not a match on any of them.
668 try:
669 if text[pos] == '\n':
670 # at EOL, reset state to "root"
671 statestack = ['root']
672 statetokens = tokendefs['root']
673 yield pos, Text, '\n'
674 pos += 1
675 continue
676 yield pos, Error, text[pos]
677 pos += 1
678 except IndexError:
679 break

698class ExtendedRegexLexer(RegexLexer):
699 """
700 A RegexLexer that uses a context object to store its state.
701 """
702
703 def get_tokens_unprocessed(self, text=None, context=None):
704 """
705 Split ``text`` into (tokentype, text) pairs.
706 If ``context`` is given, use this lexer context instead.
707 """
708 tokendefs = self._tokens
709 if not context:
710 ctx = LexerContext(text, 0)
711 statetokens = tokendefs['root']
712 else:
713 ctx = context
714 statetokens = tokendefs[ctx.stack[-1]]
715 text = ctx.text
716 while 1:
717 for rexmatch, action, new_state in statetokens:
718 m = rexmatch(text, ctx.pos, ctx.end)
719 if m:
720 if action is not None:
721 if type(action) is _TokenType:
722 yield ctx.pos, action, m.group()
723 ctx.pos = m.end()
724 else:
725 yield from action(self, m, ctx)
726 if not new_state:
727 # altered the state stack?
728 statetokens = tokendefs[ctx.stack[-1]]
729 # CAUTION: callback must set ctx.pos!
730 if new_state is not None:
731 # state transition
732 if isinstance(new_state, tuple):
733 for state in new_state:
734 if state == '#pop':
735 if len(ctx.stack) > 1:
736 ctx.stack.pop()
737 elif state == '#push':
738 ctx.stack.append(ctx.stack[-1])
739 else:
740 ctx.stack.append(state)
741 elif isinstance(new_state, int):
742 # see RegexLexer for why this check is made
743 if abs(new_state) >= len(ctx.stack):
744 del ctx.stack[1:]
745 else:
746 del ctx.stack[new_state:]
747 elif new_state == '#push':
748 ctx.stack.append(ctx.stack[-1])
749 else:
750 assert False, "wrong state def: %r" % new_state
751 statetokens = tokendefs[ctx.stack[-1]]
752 break
753 else:
754 try:
755 if ctx.pos >= ctx.end:
756 break
757 if text[ctx.pos] == '\n':
758 # at EOL, reset state to "root"
759 ctx.stack = ['root']
760 statetokens = tokendefs['root']
761 yield ctx.pos, Text, '\n'
