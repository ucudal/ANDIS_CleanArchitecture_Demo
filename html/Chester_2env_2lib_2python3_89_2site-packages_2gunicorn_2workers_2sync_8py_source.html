59 def run_for_one(self, timeout):
60 listener = self.sockets[0]

62 self.notify()
63
64 # Accept a connection. If we get an error telling us
65 # that no connection is waiting we fall down to the
66 # select which is where we'll wait for a bit for new
67 # workers to come give us some love.

69 self.accept(listener)
70 # Keep processing clients until no one is waiting. This

112 def run(self):
113 # if no timeout is given the worker will never wait and will
114 # use the CPU for nothing. This minimal timeout prevent it.
115 timeout = self.timeout or 0.5
116
117 # self.socket appears to lose its blocking status after
118 # we fork in the arbiter. Reset it here.
119 for s in self.sockets:
120 s.setblocking(0)
121
122 if len(self.sockets) > 1:
123 self.run_for_multiple(timeout)
124 else:
