66 def track_module(self, filename_in_utf8, module_name, frame):
67 """
68 :return list(NetCommand):
69 Returns a list with the module events to be sent.
70 """
71 if filename_in_utf8 in self._modules:
72 return []
73
74 module_events = []
75 with self._lock:
76 # Must check again after getting the lock.
77 if filename_in_utf8 in self._modules:
78 return
79
80 try:
81 version = str(frame.f_globals.get("__version__", ""))
82 except:
83 version = "<unknown>"
84
85 try:
86 package_name = str(frame.f_globals.get("__package__", ""))
87 except:
88 package_name = "<unknown>"
89
90 module_id = self._next_id()
91
92 module = Module(module_id, module_name, filename_in_utf8)
93 if version:
94 module.version = version
95
96 if package_name:
97 # Note: package doesn't appear in the docs but seems to be expected?
98 module.kwargs["package"] = package_name
99
100 module_event = ModuleEvent(ModuleEventBody("new", module))
101
102 module_events.append(NetCommand(CMD_MODULE_EVENT, 0, module_event, is_json=True))
103
104 self._modules[filename_in_utf8] = module.to_dict()
105 return module_events

108 """
109 :return list(Module)

111 with self._lock:
