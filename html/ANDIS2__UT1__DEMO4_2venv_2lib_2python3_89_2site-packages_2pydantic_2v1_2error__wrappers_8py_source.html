149def _get_exc_type(cls: Type[Exception]) -> str:
150 if issubclass(cls, AssertionError):
151 return 'assertion_error'
152
153 base_name = 'type_error' if issubclass(cls, TypeError) else 'value_error'
154 if cls in (TypeError, ValueError):
155 # just TypeError or ValueError, no extra code
156 return base_name
157
158 # if it's not a TypeError or ValueError, we just take the lowercase of the exception name
159 # no chaining or snake case logic, use "code" for more complex error types.
160 code = getattr(cls, 'code', None) or cls.__name__.replace('Error', '').lower()
161 return base_name + '.' + code
