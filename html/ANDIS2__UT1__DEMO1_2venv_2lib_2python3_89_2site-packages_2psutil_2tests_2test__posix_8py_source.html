45def ps(fmt, pid=None):
46 """Wrapper for calling the ps command with a little bit of cross-platform
47 support for a narrow range of features.
48 """
49
50 cmd = ['ps']
51
52 if LINUX:
53 cmd.append('--no-headers')
54
55 if pid is not None:
56 cmd.extend(['-p', str(pid)])
57 elif SUNOS or AIX:
58 cmd.append('-A')
59 else:
60 cmd.append('ax')
61
62 if SUNOS:
63 fmt = fmt.replace("start", "stime")
64
65 cmd.extend(['-o', fmt])
66
67 output = sh(cmd)
68
69 output = output.splitlines() if LINUX else output.splitlines()[1:]
70
71 all_output = []
72 for line in output:
73 line = line.strip()
74
75 try:
76 line = int(line)
77 except ValueError:
78 pass
79
80 all_output.append(line)
81
82 if pid is None:
83 return all_output
84 else:
85 return all_output[0]
86

88# ps "-o" field names differ wildly between platforms.
89# "comm" means "only executable name" but is not available on BSD platforms.
90# "args" means "command with all its arguments", and is also not available
91# on BSD platforms.
92# "command" is like "args" on most platforms, but like "comm" on AIX,
93# and not available on SUNOS.
94# so for the executable name we can use "comm" on Solaris and split "command"
95# on other platforms.
96# to get the cmdline (with args) we have to use "args" on AIX and
97# Solaris, and can use "command" on all others.
98
99
100def ps_name(pid):
101 field = "command"
102 if SUNOS:
103 field = "comm"
104 command = ps(field, pid).split()
105 return command[0]

108def ps_args(pid):
109 field = "command"
110 if AIX or SUNOS:
111 field = "args"
112 out = ps(field, pid)
113 # observed on BSD + Github CI: '/usr/local/bin/python3 -E -O (python3.9)'
114 out = re.sub(r"\‍(python.*?\‍)$", "", out)
115 return out.strip()

132def df(device):
133 try:
134 out = sh(f"df -k {device}").strip()
135 except RuntimeError as err:
136 if "device busy" in str(err).lower():
137 raise pytest.skip("df returned EBUSY")
138 raise
139 line = out.split('\n')[1]
140 fields = line.split()
141 sys_total = int(fields[1]) * 1024
142 sys_used = int(fields[2]) * 1024
143 sys_free = int(fields[3]) * 1024
144 sys_percent = float(fields[4].replace('%', ''))
145 return (sys_total, sys_used, sys_free, sys_percent)
