132class RandomColor:
133 """Implement random color generation in a human-friendly way.
134
135 This helper class encapsulates the internal implementation and logic of the
136 :meth:`color() <faker.providers.color.Provider.color>` method.
137 """
138
139 def __init__(self, generator: Optional["Generator"] = None, seed: Optional[SeedType] = None) -> None:
140 self.colormap = COLOR_MAP
141
142 # Option to specify a seed was not removed so this class
143 # can still be tested independently w/o generators
144 if generator:
145 self.random = generator.random
146 else:
147 self.seed = seed if seed else random.randint(0, sys.maxsize)
148 self.random = random.Random(int(self.seed))
149
150 def generate(
151 self,
152 hue: Optional[HueType] = None,
153 luminosity: Optional[str] = None,
154 color_format: ColorFormat = "hex",
155 ) -> str:
156 """Generate and format a color.
157
158 Whenever :meth:`color() <faker.providers.color.Provider.color>` is
159 called, the arguments used are simply passed into this method, and this
160 method handles the rest.
161 """
162 # Generate HSV color tuple from picked hue and luminosity
163 hsv = self.generate_hsv(hue=hue, luminosity=luminosity)
164
165 # Return the HSB/V color in the desired string format
166 return self.set_format(hsv, color_format)
167
168 def generate_hsv(
169 self,
170 hue: Optional[HueType] = None,
171 luminosity: Optional[str] = None,
172 ) -> Tuple[int, int, int]:
173 """Generate a HSV color tuple."""
174 # First we pick a hue (H)
175 h = self.pick_hue(hue)
176
177 # Then use H to determine saturation (S)
178 s = self.pick_saturation(h, hue, luminosity)
179
180 # Then use S and H to determine brightness/value (B/V).
181 v = self.pick_brightness(h, s, luminosity)
182
183 return h, s, v
184
185 def generate_rgb(
186 self,
187 hue: Optional[HueType] = None,
188 luminosity: Optional[str] = None,
189 ) -> Tuple[int, int, int]:
190 """Generate a RGB color tuple of integers."""
191 return self.hsv_to_rgb(self.generate_hsv(hue=hue, luminosity=luminosity))
192
193 def generate_rgb_float(
194 self,
195 hue: Optional[HueType] = None,
196 luminosity: Optional[str] = None,
197 ) -> Tuple[float, float, float]:
198 """Generate a RGB color tuple of floats."""
199 return self.hsv_to_rgb_float(self.generate_hsv(hue=hue, luminosity=luminosity))
200
201 def generate_hsl(
202 self,
203 hue: Optional[HueType] = None,
204 luminosity: Optional[str] = None,
205 ) -> Tuple[int, int, int]:
206 """Generate a HSL color tuple."""
207 return self.hsv_to_hsl(self.generate_hsv(hue=hue, luminosity=luminosity))
208
209 def pick_hue(self, hue: Optional[HueType]) -> int:
210 """Return a numerical hue value."""
211 hue_ = self.random_within(self.get_hue_range(hue))
212
213 # Instead of storing red as two separate ranges,
214 # we group them, using negative numbers
215 if hue_ < 0:
216 hue_ += 360
217
218 return hue_
219
220 def pick_saturation(self, hue: int, hue_name: Optional[HueType], luminosity: Optional[str]) -> int:
221 """Return a numerical saturation value."""
222 if luminosity is None:
223 luminosity = ""
224 if luminosity == "random":
225 return self.random_within((0, 100))
226
227 if isinstance(hue_name, str) and hue_name == "monochrome":
228 return 0
229
230 s_min, s_max = self.get_saturation_range(hue)
231
232 if luminosity == "bright":
233 s_min = 55
234 elif luminosity == "dark":
235 s_min = s_max - 10
236 elif luminosity == "light":
237 s_max = 55
238
239 return self.random_within((s_min, s_max))
240
241 def pick_brightness(self, h: int, s: int, luminosity: Optional[str]) -> int:
242 """Return a numerical brightness value."""
243 if luminosity is None:
244 luminosity = ""
245
246 b_min = self.get_minimum_brightness(h, s)
247 b_max = 100
248
249 if luminosity == "dark":
250 b_max = b_min + 20
251 elif luminosity == "light":
252 b_min = (b_max + b_min) // 2
253 elif luminosity == "random":
254 b_min = 0
255 b_max = 100
256
257 return self.random_within((b_min, b_max))
258
259 def set_format(self, hsv: Tuple[int, int, int], color_format: ColorFormat) -> str:
260 """Handle conversion of HSV values into desired format."""
261 if color_format == "hsv":
262 color = f"hsv({hsv[0]}, {hsv[1]}, {hsv[2]})"
263
264 elif color_format == "hsl":
265 hsl = self.hsv_to_hsl(hsv)
266 color = f"hsl({hsl[0]}, {hsl[1]}, {hsl[2]})"
267
268 elif color_format == "rgb":
269 rgb = self.hsv_to_rgb(hsv)
270 color = f"rgb({rgb[0]}, {rgb[1]}, {rgb[2]})"
271
272 else:
273 rgb = self.hsv_to_rgb(hsv)
274 color = f"#{rgb[0]:02x}{rgb[1]:02x}{rgb[2]:02x}"
275
276 return color
277
278 def get_minimum_brightness(self, h: int, s: int) -> int:
279 """Return the minimum allowed brightness for ``h`` and ``s``."""
280 lower_bounds: Sequence[Tuple[int, int]] = self.get_color_info(h)["lower_bounds"]
281
282 for i in range(len(lower_bounds) - 1):
283 s1, v1 = lower_bounds[i]
284 s2, v2 = lower_bounds[i + 1]
285
286 if s1 <= s <= s2:
287 m: float = (v2 - v1) / (s2 - s1)
288 b: float = v1 - m * s1
289
290 return int(m * s + b)
291
292 return 0

294 def _validate_color_input(self, color_input: HueType) -> Tuple[int, int]:
295 if (
296 not isinstance(color_input, (list, tuple))
297 or len(color_input) != 2
298 or any(not isinstance(c, (float, int)) for c in color_input)
299 ):
300 raise TypeError("Hue must be a valid string, numeric type, or a tuple/list of 2 numeric types.")
301
302 return color_input[0], color_input[1]
303
304 def get_hue_range(self, color_input: Optional[HueType]) -> Tuple[int, int]:
305 """Return the hue range for a given ``color_input``."""
306 if color_input is None:
307 return 0, 360
308
309 if isinstance(color_input, (int, float)) and 0 <= color_input <= 360:
310 color_input = int(color_input)
311 return color_input, color_input
312
313 if isinstance(color_input, str) and color_input in self.colormap:
314 return self.colormap[color_input]["hue_range"][0]
315
316 color_input = self._validate_color_input(color_input)
317
318 v1 = int(color_input[0])
319 v2 = int(color_input[1])
320
321 if v2 < v1:
322 v1, v2 = v2, v1
323 v1 = max(v1, 0)
324 v2 = min(v2, 360)
325 return v1, v2
326
327 def get_saturation_range(self, hue: int) -> Tuple[int, int]:
328 """Return the saturation range for a given numerical ``hue`` value."""
329 saturation_bounds = [s for s, v in self.get_color_info(hue)["lower_bounds"]]
330 return min(saturation_bounds), max(saturation_bounds)
331
332 def get_color_info(self, hue: int) -> Dict[str, Sequence[Tuple[int, int]]]:
333 """Return the color info for a given numerical ``hue`` value."""
334 # Maps red colors to make picking hue easier
335 if 334 <= hue <= 360:
336 hue -= 360
337
338 for color_name, color in self.colormap.items():
339 hue_range: Tuple[int, int] = color["hue_range"][0]
340 if hue_range[0] <= hue <= hue_range[1]:
341 return self.colormap[color_name]
342 else:
343 raise ValueError("Value of hue `%s` is invalid." % hue)
344
345 def random_within(self, r: Sequence[int]) -> int:
346 """Return a random integer within the range ``r``."""
347 return self.random.randint(int(r[0]), int(r[1]))

350 def hsv_to_rgb_float(cls, hsv: Tuple[int, int, int]) -> Tuple[float, float, float]:
351 """Convert HSV to RGB.
352
353 This method expects ``hsv`` to be a 3-tuple of H, S, and V values, and
354 it will return a 3-tuple of the equivalent R, G, and B float values.
355 """
356 h, s, v = hsv
357 h = max(h, 1)
358 h = min(h, 359)
359
360 return colorsys.hsv_to_rgb(h / 360, s / 100, v / 100)

362 @classmethod
363 def hsv_to_rgb(cls, hsv: Tuple[int, int, int]) -> Tuple[int, int, int]:
364 """Convert HSV to RGB.
365
366 This method expects ``hsv`` to be a 3-tuple of H, S, and V values, and
367 it will return a 3-tuple of the equivalent R, G, and B integer values.
368 """
369 r, g, b = cls.hsv_to_rgb_float(hsv)
370 return int(r * 255), int(g * 255), int(b * 255)

372 @classmethod
373 def hsv_to_hsl(cls, hsv: Tuple[int, int, int]) -> Tuple[int, int, int]:
374 """Convert HSV to HSL.
375
376 This method expects ``hsv`` to be a 3-tuple of H, S, and V values, and
377 it will return a 3-tuple of the equivalent H, S, and L values.
378 """
379 h, s, v = hsv
380
381 s_: float = s / 100.0
382 v_: float = v / 100.0
383 l = 0.5 * v_ * (2 - s_) # noqa: E741
384
385 s_ = 0.0 if l in [0, 1] else v_ * s_ / (1 - math.fabs(2 * l - 1))
386 return int(h), int(s_ * 100), int(l * 100)
