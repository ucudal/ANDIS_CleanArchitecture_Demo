54class WebSocketCommonProtocol(asyncio.Protocol):
55 """
56 WebSocket connection.
57
58 :class:`WebSocketCommonProtocol` provides APIs shared between WebSocket
59 servers and clients. You shouldn't use it directly. Instead, use
60 :class:`~websockets.legacy.client.WebSocketClientProtocol` or
61 :class:`~websockets.legacy.server.WebSocketServerProtocol`.
62
63 This documentation focuses on low-level details that aren't covered in the
64 documentation of :class:`~websockets.legacy.client.WebSocketClientProtocol`
65 and :class:`~websockets.legacy.server.WebSocketServerProtocol` for the sake
66 of simplicity.
67
68 Once the connection is open, a Ping_ frame is sent every ``ping_interval``
69 seconds. This serves as a keepalive. It helps keeping the connection open,
70 especially in the presence of proxies with short timeouts on inactive
71 connections. Set ``ping_interval`` to :obj:`None` to disable this behavior.
72
73 .. _Ping: https://datatracker.ietf.org/doc/html/rfc6455#section-5.5.2
74
75 If the corresponding Pong_ frame isn't received within ``ping_timeout``
76 seconds, the connection is considered unusable and is closed with code 1011.
77 This ensures that the remote endpoint remains responsive. Set
78 ``ping_timeout`` to :obj:`None` to disable this behavior.
79
80 .. _Pong: https://datatracker.ietf.org/doc/html/rfc6455#section-5.5.3
81
82 See the discussion of :doc:`keepalive <../../topics/keepalive>` for details.
83
84 The ``close_timeout`` parameter defines a maximum wait time for completing
85 the closing handshake and terminating the TCP connection. For legacy
86 reasons, :meth:`close` completes in at most ``5 * close_timeout`` seconds
87 for clients and ``4 * close_timeout`` for servers.
88
89 ``close_timeout`` is a parameter of the protocol because websockets usually
90 calls :meth:`close` implicitly upon exit:
91
92 * on the client side, when using :func:`~websockets.legacy.client.connect`
93 as a context manager;
94 * on the server side, when the connection handler terminates.
95
96 To apply a timeout to any other API, wrap it in :func:`~asyncio.timeout` or
97 :func:`~asyncio.wait_for`.
98
99 The ``max_size`` parameter enforces the maximum size for incoming messages
100 in bytes. The default value is 1 MiB. If a larger message is received,
101 :meth:`recv` will raise :exc:`~websockets.exceptions.ConnectionClosedError`
102 and the connection will be closed with code 1009.
103
104 The ``max_queue`` parameter sets the maximum length of the queue that
105 holds incoming messages. The default value is ``32``. Messages are added
106 to an in-memory queue when they're received; then :meth:`recv` pops from
107 that queue. In order to prevent excessive memory consumption when
108 messages are received faster than they can be processed, the queue must
109 be bounded. If the queue fills up, the protocol stops processing incoming
110 data until :meth:`recv` is called. In this situation, various receive
111 buffers (at least in :mod:`asyncio` and in the OS) will fill up, then the
112 TCP receive window will shrink, slowing down transmission to avoid packet
113 loss.
114
115 Since Python can use up to 4 bytes of memory to represent a single
116 character, each connection may use up to ``4 * max_size * max_queue``
117 bytes of memory to store incoming messages. By default, this is 128 MiB.
118 You may want to lower the limits, depending on your application's
119 requirements.
120
121 The ``read_limit`` argument sets the high-water limit of the buffer for
122 incoming bytes. The low-water limit is half the high-water limit. The
123 default value is 64 KiB, half of asyncio's default (based on the current
124 implementation of :class:`~asyncio.StreamReader`).
125
126 The ``write_limit`` argument sets the high-water limit of the buffer for
127 outgoing bytes. The low-water limit is a quarter of the high-water limit.
128 The default value is 64 KiB, equal to asyncio's default (based on the
129 current implementation of ``FlowControlMixin``).
130
131 See the discussion of :doc:`memory usage <../../topics/memory>` for details.
132
133 Args:
134 logger: Logger for this server.
135 It defaults to ``logging.getLogger("websockets.protocol")``.
136 See the :doc:`logging guide <../../topics/logging>` for details.
137 ping_interval: Interval between keepalive pings in seconds.
138 :obj:`None` disables keepalive.
139 ping_timeout: Timeout for keepalive pings in seconds.
140 :obj:`None` disables timeouts.
141 close_timeout: Timeout for closing the connection in seconds.
142 For legacy reasons, the actual timeout is 4 or 5 times larger.
143 max_size: Maximum size of incoming messages in bytes.
144 :obj:`None` disables the limit.
145 max_queue: Maximum number of incoming messages in receive buffer.
146 :obj:`None` disables the limit.
147 read_limit: High-water mark of read buffer in bytes.
148 write_limit: High-water mark of write buffer in bytes.
149
150 """

488 async def recv(self) -> Data:
489 """
490 Receive the next message.
491
492 When the connection is closed, :meth:`recv` raises
493 :exc:`~websockets.exceptions.ConnectionClosed`. Specifically, it raises
494 :exc:`~websockets.exceptions.ConnectionClosedOK` after a normal
495 connection closure and
496 :exc:`~websockets.exceptions.ConnectionClosedError` after a protocol
497 error or a network failure. This is how you detect the end of the
498 message stream.
499
500 Canceling :meth:`recv` is safe. There's no risk of losing the next
501 message. The next invocation of :meth:`recv` will return it.
502
503 This makes it possible to enforce a timeout by wrapping :meth:`recv` in
504 :func:`~asyncio.timeout` or :func:`~asyncio.wait_for`.
505
506 Returns:
507 A string (:class:`str`) for a Text_ frame. A bytestring
508 (:class:`bytes`) for a Binary_ frame.
509
510 .. _Text: https://datatracker.ietf.org/doc/html/rfc6455#section-5.6
511 .. _Binary: https://datatracker.ietf.org/doc/html/rfc6455#section-5.6
512
513 Raises:
514 ConnectionClosed: When the connection is closed.
515 RuntimeError: If two coroutines call :meth:`recv` concurrently.
516
517 """
518 if self._pop_message_waiter is not None:
519 raise RuntimeError(
520 "cannot call recv while another coroutine "
521 "is already waiting for the next message"
522 )
523
524 # Don't await self.ensure_open() here:
525 # - messages could be available in the queue even if the connection
526 # is closed;
527 # - messages could be received before the closing frame even if the
528 # connection is closing.
529
530 # Wait until there's a message in the queue (if necessary) or the
531 # connection is closed.
532 while len(self.messages) <= 0:
533 pop_message_waiter: asyncio.Future[None] = self.loop.create_future()
534 self._pop_message_waiter = pop_message_waiter
535 try:
536 # If asyncio.wait() is canceled, it doesn't cancel
537 # pop_message_waiter and self.transfer_data_task.
538 await asyncio.wait(
539 [pop_message_waiter, self.transfer_data_task],
540 return_when=asyncio.FIRST_COMPLETED,
541 )
542 finally:
543 self._pop_message_waiter = None
544
545 # If asyncio.wait(...) exited because self.transfer_data_task
546 # completed before receiving a new message, raise a suitable
547 # exception (or return None if legacy_recv is enabled).
548 if not pop_message_waiter.done():
549 if self.legacy_recv:
550 return None # type: ignore
551 else:
552 # Wait until the connection is closed to raise
553 # ConnectionClosed with the correct code and reason.
554 await self.ensure_open()
555
556 # Pop a message from the queue.
557 message = self.messages.popleft()
558
559 # Notify transfer_data().
560 if self._put_message_waiter is not None:
561 self._put_message_waiter.set_result(None)
562 self._put_message_waiter = None
563
564 return message

566 async def send(
567 self,
568 message: Data | Iterable[Data] | AsyncIterable[Data],
569 ) -> None:
570 """
571 Send a message.
572
573 A string (:class:`str`) is sent as a Text_ frame. A bytestring or
574 bytes-like object (:class:`bytes`, :class:`bytearray`, or
575 :class:`memoryview`) is sent as a Binary_ frame.
576
577 .. _Text: https://datatracker.ietf.org/doc/html/rfc6455#section-5.6
578 .. _Binary: https://datatracker.ietf.org/doc/html/rfc6455#section-5.6
579
580 :meth:`send` also accepts an iterable or an asynchronous iterable of
581 strings, bytestrings, or bytes-like objects to enable fragmentation_.
582 Each item is treated as a message fragment and sent in its own frame.
583 All items must be of the same type, or else :meth:`send` will raise a
584 :exc:`TypeError` and the connection will be closed.
585
586 .. _fragmentation: https://datatracker.ietf.org/doc/html/rfc6455#section-5.4
587
588 :meth:`send` rejects dict-like objects because this is often an error.
589 (If you want to send the keys of a dict-like object as fragments, call
590 its :meth:`~dict.keys` method and pass the result to :meth:`send`.)
591
592 Canceling :meth:`send` is discouraged. Instead, you should close the
593 connection with :meth:`close`. Indeed, there are only two situations
594 where :meth:`send` may yield control to the event loop and then get
595 canceled; in both cases, :meth:`close` has the same effect and is
596 more clear:
597
598 1. The write buffer is full. If you don't want to wait until enough
599 data is sent, your only alternative is to close the connection.
600 :meth:`close` will likely time out then abort the TCP connection.
601 2. ``message`` is an asynchronous iterator that yields control.
602 Stopping in the middle of a fragmented message will cause a
603 protocol error and the connection will be closed.
604
605 When the connection is closed, :meth:`send` raises
606 :exc:`~websockets.exceptions.ConnectionClosed`. Specifically, it
607 raises :exc:`~websockets.exceptions.ConnectionClosedOK` after a normal
608 connection closure and
609 :exc:`~websockets.exceptions.ConnectionClosedError` after a protocol
610 error or a network failure.
611
612 Args:
613 message: Message to send.
614
615 Raises:
616 ConnectionClosed: When the connection is closed.
617 TypeError: If ``message`` doesn't have a supported type.
618
619 """
620 await self.ensure_open()
621
622 # While sending a fragmented message, prevent sending other messages
623 # until all fragments are sent.
624 while self._fragmented_message_waiter is not None:
625 await asyncio.shield(self._fragmented_message_waiter)
626
627 # Unfragmented message -- this case must be handled first because
628 # strings and bytes-like objects are iterable.
629
630 if isinstance(message, (str, bytes, bytearray, memoryview)):
631 opcode, data = prepare_data(message)
632 await self.write_frame(True, opcode, data)
633
634 # Catch a common mistake -- passing a dict to send().
635
636 elif isinstance(message, Mapping):
637 raise TypeError("data is a dict-like object")
638
639 # Fragmented message -- regular iterator.
640
641 elif isinstance(message, Iterable):
642 # Work around https://github.com/python/mypy/issues/6227
643 message = cast(Iterable[Data], message)
644
645 iter_message = iter(message)
646 try:
647 fragment = next(iter_message)
648 except StopIteration:
649 return
650 opcode, data = prepare_data(fragment)
651
652 self._fragmented_message_waiter = self.loop.create_future()
653 try:
654 # First fragment.
655 await self.write_frame(False, opcode, data)
656
657 # Other fragments.
658 for fragment in iter_message:
659 confirm_opcode, data = prepare_data(fragment)
660 if confirm_opcode != opcode:
661 raise TypeError("data contains inconsistent types")
662 await self.write_frame(False, OP_CONT, data)
663
664 # Final fragment.
665 await self.write_frame(True, OP_CONT, b"")
666
667 except (Exception, asyncio.CancelledError):
668 # We're half-way through a fragmented message and we can't
669 # complete it. This makes the connection unusable.
670 self.fail_connection(CloseCode.INTERNAL_ERROR)
671 raise
672
673 finally:
674 self._fragmented_message_waiter.set_result(None)
675 self._fragmented_message_waiter = None
676
677 # Fragmented message -- asynchronous iterator
678
679 elif isinstance(message, AsyncIterable):
680 # Implement aiter_message = aiter(message) without aiter
681 # Work around https://github.com/python/mypy/issues/5738
682 aiter_message = cast(

684 type(message).__aiter__,
685 )(message)

687 # Implement fragment = anext(aiter_message) without anext
688 # Work around https://github.com/python/mypy/issues/5738
689 fragment = await cast(
690 Callable[[AsyncIterator[Data]], Awaitable[Data]],
691 type(aiter_message).__anext__,
692 )(aiter_message)
693 except StopAsyncIteration:
694 return

696
697 self._fragmented_message_waiter = self.loop.create_future()
698 try:
699 # First fragment.
700 await self.write_frame(False, opcode, data)
701
702 # Other fragments.
703 async for fragment in aiter_message:
704 confirm_opcode, data = prepare_data(fragment)
705 if confirm_opcode != opcode:
706 raise TypeError("data contains inconsistent types")
707 await self.write_frame(False, OP_CONT, data)
708
709 # Final fragment.
710 await self.write_frame(True, OP_CONT, b"")
711
712 except (Exception, asyncio.CancelledError):
713 # We're half-way through a fragmented message and we can't
714 # complete it. This makes the connection unusable.
715 self.fail_connection(CloseCode.INTERNAL_ERROR)
716 raise
717
718 finally:
719 self._fragmented_message_waiter.set_result(None)
720 self._fragmented_message_waiter = None
721
722 else:
723 raise TypeError("data must be str, bytes-like, or iterable")

725 async def close(
726 self,
727 code: int = CloseCode.NORMAL_CLOSURE,
728 reason: str = "",
729 ) -> None:
730 """
731 Perform the closing handshake.
732
733 :meth:`close` waits for the other end to complete the handshake and
734 for the TCP connection to terminate. As a consequence, there's no need
735 to await :meth:`wait_closed` after :meth:`close`.
736
737 :meth:`close` is idempotent: it doesn't do anything once the
738 connection is closed.
739
740 Wrapping :func:`close` in :func:`~asyncio.create_task` is safe, given
741 that errors during connection termination aren't particularly useful.
742
743 Canceling :meth:`close` is discouraged. If it takes too long, you can
744 set a shorter ``close_timeout``. If you don't want to wait, let the
745 Python process exit, then the OS will take care of closing the TCP
746 connection.
747
748 Args:
749 code: WebSocket close code.
750 reason: WebSocket close reason.
751
752 """
753 try:
754 async with asyncio_timeout(self.close_timeout):
755 await self.write_close_frame(Close(code, reason))
756 except asyncio.TimeoutError:
757 # If the close frame cannot be sent because the send buffers
758 # are full, the closing handshake won't complete anyway.
759 # Fail the connection to shut down faster.
760 self.fail_connection()
761
762 # If no close frame is received within the timeout, asyncio_timeout()
763 # cancels the data transfer task and raises TimeoutError.
764
765 # If close() is called multiple times concurrently and one of these
766 # calls hits the timeout, the data transfer task will be canceled.
767 # Other calls will receive a CancelledError here.
768
769 try:
770 # If close() is canceled during the wait, self.transfer_data_task
771 # is canceled before the timeout elapses.
772 async with asyncio_timeout(self.close_timeout):
773 await self.transfer_data_task
774 except (asyncio.TimeoutError, asyncio.CancelledError):
775 pass
776
777 # Wait for the close connection task to close the TCP connection.
778 await asyncio.shield(self.close_connection_task)

791 await asyncio.shield(self.connection_lost_waiter)

793 async def ping(self, data: Data | None = None) -> Awaitable[float]:
794 """
795 Send a Ping_.
796
797 .. _Ping: https://datatracker.ietf.org/doc/html/rfc6455#section-5.5.2
798
799 A ping may serve as a keepalive, as a check that the remote endpoint
800 received all messages up to this point, or to measure :attr:`latency`.
801
802 Canceling :meth:`ping` is discouraged. If :meth:`ping` doesn't return
803 immediately, it means the write buffer is full. If you don't want to
804 wait, you should close the connection.
805
806 Canceling the :class:`~asyncio.Future` returned by :meth:`ping` has no
807 effect.
808
809 Args:
810 data: Payload of the ping. A string will be encoded to UTF-8.
811 If ``data`` is :obj:`None`, the payload is four random bytes.
812
813 Returns:
814 A future that will be completed when the corresponding pong is
815 received. You can ignore it if you don't intend to wait. The result
816 of the future is the latency of the connection in seconds.
817
818 ::
819
820 pong_waiter = await ws.ping()
821 # only if you want to wait for the corresponding pong
822 latency = await pong_waiter
823
824 Raises:
825 ConnectionClosed: When the connection is closed.
826 RuntimeError: If another ping was sent with the same data and
827 the corresponding pong wasn't received yet.
828
829 """
830 await self.ensure_open()
831
832 if data is not None:
833 data = prepare_ctrl(data)
834
835 # Protect against duplicates if a payload is explicitly set.
836 if data in self.pings:
837 raise RuntimeError("already waiting for a pong with the same data")
838
839 # Generate a unique random payload otherwise.
840 while data is None or data in self.pings:
841 data = struct.pack("!I", random.getrandbits(32))
842
843 pong_waiter = self.loop.create_future()
844 # Resolution of time.monotonic() may be too low on Windows.
845 ping_timestamp = time.perf_counter()
846 self.pings[data] = (pong_waiter, ping_timestamp)
847
848 await self.write_frame(True, OP_PING, data)
849
850 return asyncio.shield(pong_waiter)

858 An unsolicited pong may serve as a unidirectional heartbeat.
859
860 Canceling :meth:`pong` is discouraged. If :meth:`pong` doesn't return

902 async def ensure_open(self) -> None:
903 """
904 Check that the WebSocket connection is open.
905
906 Raise :exc:`~websockets.exceptions.ConnectionClosed` if it isn't.
907
908 """
909 # Handle cases from most common to least common for performance.
910 if self.state is State.OPEN:
911 # If self.transfer_data_task exited without a closing handshake,
912 # self.close_connection_task may be closing the connection, going
913 # straight from OPEN to CLOSED.
914 if self.transfer_data_task.done():
915 await asyncio.shield(self.close_connection_task)
916 raise self.connection_closed_exc()
917 else:
918 return
919
920 if self.state is State.CLOSED:
921 raise self.connection_closed_exc()
922
923 if self.state is State.CLOSING:
924 # If we started the closing handshake, wait for its completion to
925 # get the proper close code and reason. self.close_connection_task
926 # will complete within 4 or 5 * close_timeout after close(). The
927 # CLOSING state also occurs when failing the connection. In that
928 # case self.close_connection_task will complete even faster.
929 await asyncio.shield(self.close_connection_task)
930 raise self.connection_closed_exc()
931
932 # Control may only reach this point in buggy third-party subclasses.
933 assert self.state is State.CONNECTING
934 raise InvalidState("WebSocket connection isn't established yet")

936 async def transfer_data(self) -> None:
937 """
938 Read incoming messages and put them in a queue.
939
940 This coroutine runs in a task until the closing handshake is started.
941
942 """
943 try:
944 while True:
945 message = await self.read_message()
946
947 # Exit the loop when receiving a close frame.
948 if message is None:
949 break
950
951 # Wait until there's room in the queue (if necessary).
952 if self.max_queue is not None:
953 while len(self.messages) >= self.max_queue:
954 self._put_message_waiter = self.loop.create_future()
955 try:
956 await asyncio.shield(self._put_message_waiter)
957 finally:
958 self._put_message_waiter = None
959
960 # Put the message in the queue.
961 self.messages.append(message)
962
963 # Notify recv().
964 if self._pop_message_waiter is not None:
965 self._pop_message_waiter.set_result(None)
966 self._pop_message_waiter = None
967
968 except asyncio.CancelledError as exc:
969 self.transfer_data_exc = exc
970 # If fail_connection() cancels this task, avoid logging the error
971 # twice and failing the connection again.
972 raise
973
974 except ProtocolError as exc:
975 self.transfer_data_exc = exc
976 self.fail_connection(CloseCode.PROTOCOL_ERROR)
977
978 except (ConnectionError, TimeoutError, EOFError, ssl.SSLError) as exc:
979 # Reading data with self.reader.readexactly may raise:
980 # - most subclasses of ConnectionError if the TCP connection
981 # breaks, is reset, or is aborted;
982 # - TimeoutError if the TCP connection times out;
983 # - IncompleteReadError, a subclass of EOFError, if fewer
984 # bytes are available than requested;
985 # - ssl.SSLError if the other side infringes the TLS protocol.
986 self.transfer_data_exc = exc

989 except UnicodeDecodeError as exc:
990 self.transfer_data_exc = exc
991 self.fail_connection(CloseCode.INVALID_DATA)
992
993 except PayloadTooBig as exc:

995 self.fail_connection(CloseCode.MESSAGE_TOO_BIG)
996
997 except Exception as exc:

999 # regular circumstances are handled above. If it does, consider
1000 # catching and handling more exceptions.

1003 self.transfer_data_exc = exc
1004 self.fail_connection(CloseCode.INTERNAL_ERROR)

1006 async def read_message(self) -> Data | None:
1007 """
1008 Read a single message from the connection.
1009
1010 Re-assemble data frames if the message is fragmented.
1011
1012 Return :obj:`None` when the closing handshake is started.
1013
1014 """
1015 frame = await self.read_data_frame(max_size=self.max_size)
1016
1017 # A close frame was received.
1018 if frame is None:
1019 return None
1020
1021 if frame.opcode == OP_TEXT:
1022 text = True
1023 elif frame.opcode == OP_BINARY:
1024 text = False
1025 else: # frame.opcode == OP_CONT
1026 raise ProtocolError("unexpected opcode")
1027
1028 # Shortcut for the common case - no fragmentation
1029 if frame.fin:
1030 return frame.data.decode() if text else frame.data
1031
1032 # 5.4. Fragmentation
1033 fragments: list[Data] = []
1034 max_size = self.max_size
1035 if text:
1036 decoder_factory = codecs.getincrementaldecoder("utf-8")
1037 decoder = decoder_factory(errors="strict")
1038 if max_size is None:
1039

1041 nonlocal fragments
1042 fragments.append(decoder.decode(frame.data, frame.fin))
1043
1044 else:
1045

1047 nonlocal fragments, max_size
1048 fragments.append(decoder.decode(frame.data, frame.fin))
1049 assert isinstance(max_size, int)
1050 max_size -= len(frame.data)
1051
1052 else:
1053 if max_size is None:
1054
1055 def append(frame: Frame) -> None:
1056 nonlocal fragments
1057 fragments.append(frame.data)
1058
1059 else:
1060

1062 nonlocal fragments, max_size
1063 fragments.append(frame.data)
1064 assert isinstance(max_size, int)
1065 max_size -= len(frame.data)

1072 raise ProtocolError("incomplete fragmented message")

1074 raise ProtocolError("unexpected opcode")

1079 async def read_data_frame(self, max_size: int | None) -> Frame | None:
1080 """
1081 Read a single data frame from the connection.
1082
1083 Process control frames received before the next data frame.
1084
1085 Return :obj:`None` if a close frame is encountered before any data frame.
1086
1087 """
1088 # 6.2. Receiving Data
1089 while True:
1090 frame = await self.read_frame(max_size)
1091
1092 # 5.5. Control Frames
1093 if frame.opcode == OP_CLOSE:
1094 # 7.1.5. The WebSocket Connection Close Code
1095 # 7.1.6. The WebSocket Connection Close Reason
1096 self.close_rcvd = Close.parse(frame.data)
1097 if self.close_sent is not None:
1098 self.close_rcvd_then_sent = False
1099 try:
1100 # Echo the original data instead of re-serializing it with
1101 # Close.serialize() because that fails when the close frame
1102 # is empty and Close.parse() synthesizes a 1005 close code.
1103 await self.write_close_frame(self.close_rcvd, frame.data)
1104 except ConnectionClosed:
1105 # Connection closed before we could echo the close frame.
1106 pass
1107 return None
1108
1109 elif frame.opcode == OP_PING:
1110 # Answer pings, unless connection is CLOSING.
1111 if self.state is State.OPEN:
1112 try:
1113 await self.pong(frame.data)
1114 except ConnectionClosed:
1115 # Connection closed while draining write buffer.
1116 pass
1117
1118 elif frame.opcode == OP_PONG:
1119 if frame.data in self.pings:
1120 pong_timestamp = time.perf_counter()
1121 # Sending a pong for only the most recent ping is legal.
1122 # Acknowledge all previous pings too in that case.
1123 ping_id = None
1124 ping_ids = []
1125 for ping_id, (pong_waiter, ping_timestamp) in self.pings.items():
1126 ping_ids.append(ping_id)
1127 if not pong_waiter.done():
1128 pong_waiter.set_result(pong_timestamp - ping_timestamp)
1129 if ping_id == frame.data:
1130 self.latency = pong_timestamp - ping_timestamp
1131 break
1132 else:
1133 raise AssertionError("solicited pong not found in pings")
1134 # Remove acknowledged pings from self.pings.
1135 for ping_id in ping_ids:
1136 del self.pings[ping_id]
1137
1138 # 5.6. Data Frames
1139 else:
1140 return frame

1263 async def close_connection(self) -> None:
1264 """
1265 7.1.1. Close the WebSocket Connection
1266
1267 When the opening handshake succeeds, :meth:`connection_open` starts
1268 this coroutine in a task. It waits for the data transfer phase to
1269 complete then it closes the TCP connection cleanly.
1270
1271 When the opening handshake fails, :meth:`fail_connection` does the
1272 same. There's no data transfer phase in that case.
1273
1274 """
1275 try:
1276 # Wait for the data transfer phase to complete.
1277 if hasattr(self, "transfer_data_task"):
1278 try:
1279 await self.transfer_data_task
1280 except asyncio.CancelledError:
1281 pass
1282
1283 # Cancel the keepalive ping task.
1284 if hasattr(self, "keepalive_ping_task"):
1285 self.keepalive_ping_task.cancel()
1286
1287 # A client should wait for a TCP close from the server.
1288 if self.is_client and hasattr(self, "transfer_data_task"):
1289 if await self.wait_for_connection_lost():
1290 return
1291 if self.debug:
1292 self.logger.debug("- timed out waiting for TCP close")
1293
1294 # Half-close the TCP connection if possible (when there's no TLS).
1295 if self.transport.can_write_eof():
1296 if self.debug:
1297 self.logger.debug("x half-closing TCP connection")
1298 # write_eof() doesn't document which exceptions it raises.
1299 # "[Errno 107] Transport endpoint is not connected" happens
1300 # but it isn't completely clear under which circumstances.
1301 # uvloop can raise RuntimeError here.
1302 try:
1303 self.transport.write_eof()
1304 except (OSError, RuntimeError): # pragma: no cover
1305 pass
1306
1307 if await self.wait_for_connection_lost():
1308 return
1309 if self.debug:
1310 self.logger.debug("- timed out waiting for TCP close")
1311
1312 finally:
1313 # The try/finally ensures that the transport never remains open,
1314 # even if this coroutine is canceled (for example).
1315 await self.close_transport()

1317 async def close_transport(self) -> None:
1318 """
1319 Close the TCP connection.
1320
1321 """
1322 # If connection_lost() was called, the TCP connection is closed.
1323 # However, if TLS is enabled, the transport still needs closing.
1324 # Else asyncio complains: ResourceWarning: unclosed transport.
1325 if self.connection_lost_waiter.done() and self.transport.is_closing():
1326 return
1327
1328 # Close the TCP connection. Buffers are flushed asynchronously.
1329 if self.debug:
1330 self.logger.debug("x closing TCP connection")
1331 self.transport.close()
1332
1333 if await self.wait_for_connection_lost():
1334 return
1335 if self.debug:
1336 self.logger.debug("- timed out waiting for TCP close")
1337
1338 # Abort the TCP connection. Buffers are discarded.
1339 if self.debug:
1340 self.logger.debug("x aborting TCP connection")
1341 self.transport.abort()
1342
1343 # connection_lost() is called quickly after aborting.
1344 await self.wait_for_connection_lost()

1346 async def wait_for_connection_lost(self) -> bool:
1347 """
1348 Wait until the TCP connection is closed or ``self.close_timeout`` elapses.
1349
1350 Return :obj:`True` if the connection is closed and :obj:`False`
1351 otherwise.
1352
1353 """
1354 if not self.connection_lost_waiter.done():
1355 try:

1357 await asyncio.shield(self.connection_lost_waiter)

1365 def fail_connection(
1366 self,
1367 code: int = CloseCode.ABNORMAL_CLOSURE,
1368 reason: str = "",
1369 ) -> None:
1370 """
1371 7.1.7. Fail the WebSocket Connection
1372
1373 This requires:
1374
1375 1. Stopping all processing of incoming data, which means cancelling
1376 :attr:`transfer_data_task`. The close code will be 1006 unless a
1377 close frame was received earlier.
1378
1379 2. Sending a close frame with an appropriate code if the opening
1380 handshake succeeded and the other side is likely to process it.
1381
1382 3. Closing the connection. :meth:`close_connection` takes care of
1383 this once :attr:`transfer_data_task` exits after being canceled.
1384
1385 (The specification describes these steps in the opposite order.)
1386
1387 """
1388 if self.debug:
1389 self.logger.debug("! failing connection with code %d", code)
1390
1391 # Cancel transfer_data_task if the opening handshake succeeded.
1392 # cancel() is idempotent and ignored if the task is done already.
1393 if hasattr(self, "transfer_data_task"):
1394 self.transfer_data_task.cancel()
1395
1396 # Send a close frame when the state is OPEN (a close frame was already
1397 # sent if it's CLOSING), except when failing the connection because of
1398 # an error reading from or writing to the network.
1399 # Don't send a close frame if the connection is broken.
1400 if code != CloseCode.ABNORMAL_CLOSURE and self.state is State.OPEN:
1401 close = Close(code, reason)
1402
1403 # Write the close frame without draining the write buffer.
1404
1405 # Keeping fail_connection() synchronous guarantees it can't
1406 # get stuck and simplifies the implementation of the callers.
1407 # Not drainig the write buffer is acceptable in this context.
1408
1409 # This duplicates a few lines of code from write_close_frame().
1410
1411 self.state = State.CLOSING
1412 if self.debug:
1413 self.logger.debug("= connection is CLOSING")
1414
1415 # If self.close_rcvd was set, the connection state would be
1416 # CLOSING. Therefore self.close_rcvd isn't set and we don't
1417 # have to set self.close_rcvd_then_sent.
1418 assert self.close_rcvd is None
