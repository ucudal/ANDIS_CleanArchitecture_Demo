61class URL:
62 def __init__(
63 self,
64 url: URLTypes,
65 allow_relative: bool = False,
66 params: QueryParamTypes = None,
67 ) -> None:
68 if isinstance(url, str):
69 self._uri_reference = rfc3986.api.iri_reference(url).encode()
70 else:
71 self._uri_reference = url._uri_reference
72
73 # Normalize scheme and domain name.
74 if self.is_absolute_url:
75 self._uri_reference = self._uri_reference.normalize()
76
77 # Add any query parameters, merging with any in the URL if needed.
78 if params:
79 if self._uri_reference.query:
80 url_params = QueryParams(self._uri_reference.query)
81 url_params.update(params)
82 query_string = str(url_params)
83 else:
84 query_string = str(QueryParams(params))
85 self._uri_reference = self._uri_reference.copy_with(query=query_string)
86
87 # Enforce absolute URLs by default.
88 if not allow_relative:
89 if not self.scheme:
90 raise InvalidURL("No scheme included in URL.")
91 if not self.host:
92 raise InvalidURL("No host included in URL.")
93
94 # Allow setting full_path to custom attributes requests
95 # like OPTIONS, CONNECT, and forwarding proxy requests.
96 self._full_path: typing.Optional[str] = None

98 @property
99 def scheme(self) -> str:
100 return self._uri_reference.scheme or ""
101
102 @property
103 def authority(self) -> str:
104 port_str = self._uri_reference.port
105 default_port_str = {"https": "443", "http": "80"}.get(self.scheme, "")
106 if port_str is None or port_str == default_port_str:
107 return self._uri_reference.host or ""
108 return self._uri_reference.authority or ""

169 @property

173 @property
174 def is_absolute_url(self) -> bool:
175 """
176 Return `True` for absolute URLs such as 'http://example.com/path',
177 and `False` for relative URLs such as '/path'.
178 """
179 # We don't use `.is_absolute` from `rfc3986` because it treats
180 # URLs with a fragment portion as not absolute.
181 # What we actually care about is if the URL provides
182 # a scheme and hostname to which connections should be made.
183 return bool(self.scheme and self.host)

185 @property

187 return not self.is_absolute_url

189 def copy_with(self, **kwargs: typing.Any) -> "URL":
190 if (
191 "username" in kwargs
192 or "password" in kwargs
193 or "host" in kwargs
194 or "port" in kwargs
195 ):
196 host = kwargs.pop("host", self.host)
197 port = kwargs.pop("port", None if self.is_relative_url else self.port)
198 username = kwargs.pop("username", self.username)
199 password = kwargs.pop("password", self.password)
200
201 authority = host
202 if port is not None:
203 authority += f":{port}"
204 if username:
205 userpass = username
206 if password:
207 userpass += f":{password}"
208 authority = f"{userpass}@{authority}"
209
210 kwargs["authority"] = authority
211
212 return URL(
213 self._uri_reference.copy_with(**kwargs).unsplit(),
214 allow_relative=self.is_relative_url,
215 )
216
217 def join(self, relative_url: URLTypes) -> "URL":
218 """
219 Return an absolute URL, using given this URL as the base.
220 """
221 if self.is_relative_url:
222 return URL(relative_url)
223
224 # We drop any fragment portion, because RFC 3986 strictly
225 # treats URLs with a fragment portion as not being absolute URLs.
226 base_uri = self._uri_reference.copy_with(fragment=None)
227 relative_url = URL(relative_url, allow_relative=True)
228 return URL(relative_url._uri_reference.resolve_with(base_uri).unsplit())

234 return isinstance(other, (URL, str)) and str(self) == str(other)

236 def __str__(self) -> str:
237 return self._uri_reference.unsplit()
238

240 class_name = self.__class__.__name__
241 url_str = str(self)
242 if self._uri_reference.userinfo:
243 url_str = (
244 rfc3986.urlparse(url_str)
245 .copy_with(userinfo=f"{self.username}:[secure]")
246 .unsplit()
247 )
248 return f"{class_name}({url_str!r})"

382class Headers(typing.MutableMapping[str, str]):
383 """
384 HTTP headers, as a case-insensitive multi-dict.
385 """
386
387 def __init__(self, headers: HeaderTypes = None, encoding: str = None) -> None:
388 if headers is None:
389 self._list = [] # type: typing.List[typing.Tuple[bytes, bytes]]
390 elif isinstance(headers, Headers):
391 self._list = list(headers.raw)
392 elif isinstance(headers, dict):
393 self._list = [
394 (normalize_header_key(k, encoding), normalize_header_value(v, encoding))
395 for k, v in headers.items()
396 ]
397 else:
398 self._list = [
399 (normalize_header_key(k, encoding), normalize_header_value(v, encoding))
400 for k, v in headers
401 ]
402 self._encoding = encoding
403
404 @property
405 def encoding(self) -> str:
406 """
407 Header encoding is mandated as ascii, but we allow fallbacks to utf-8
408 or iso-8859-1.
409 """
410 if self._encoding is None:
411 for encoding in ["ascii", "utf-8"]:
412 for key, value in self.raw:
413 try:
414 key.decode(encoding)
415 value.decode(encoding)
416 except UnicodeDecodeError:
417 break
418 else:
419 # The else block runs if 'break' did not occur, meaning
420 # all values fitted the encoding.
421 self._encoding = encoding
422 break
423 else:
424 # The ISO-8859-1 encoding covers all 256 code points in a byte,
425 # so will never raise decode errors.
426 self._encoding = "iso-8859-1"
427 return self._encoding
428
429 @encoding.setter
430 def encoding(self, value: str) -> None:
431 self._encoding = value

433 @property
434 def raw(self) -> typing.List[typing.Tuple[bytes, bytes]]:
435 """
436 Returns a list of the raw header items, as byte pairs.
437 May be mutated in-place.

439 return self._list

459 def getlist(self, key: str, split_commas: bool = False) -> typing.List[str]:
460 """
461 Return multiple header values.
462 """
463 get_header_key = key.lower().encode(self.encoding)
464
465 values = [
466 item_value.decode(self.encoding)
467 for item_key, item_value in self._list
468 if item_key == get_header_key
469 ]
470
471 if not split_commas:
472 return values
473
474 split_values = []
475 for value in values:
476 split_values.extend([item.strip() for item in value.split(",")])
477 return split_values

479 def update(self, headers: HeaderTypes = None) -> None: # type: ignore
480 headers = Headers(headers)
481 for header in headers:
482 self[header] = headers[header]

484 def copy(self) -> "Headers":
485 return Headers(self.items(), encoding=self.encoding)
486
487 def __getitem__(self, key: str) -> str:
488 """
489 Return a single header value.
490
491 If there are multiple headers with the same key, then we concatenate
492 them with commas. See: https://tools.ietf.org/html/rfc7230#section-3.2.2
493 """
494 normalized_key = key.lower().encode(self.encoding)
495
496 items = []
497 for header_key, header_value in self._list:
498 if header_key == normalized_key:
499 items.append(header_value.decode(self.encoding))
500
501 if items:
502 return ", ".join(items)
503
504 raise KeyError(key)
505
506 def __setitem__(self, key: str, value: str) -> None:
507 """
508 Set the header `key` to `value`, removing any duplicate entries.
509 Retains insertion order.
510 """
511 set_key = key.lower().encode(self._encoding or "utf-8")
512 set_value = value.encode(self._encoding or "utf-8")
513
514 found_indexes = []
515 for idx, (item_key, _) in enumerate(self._list):
516 if item_key == set_key:
517 found_indexes.append(idx)
518
519 for idx in reversed(found_indexes[1:]):
520 del self._list[idx]
521
522 if found_indexes:
523 idx = found_indexes[0]
524 self._list[idx] = (set_key, set_value)
525 else:
526 self._list.append((set_key, set_value))
527
528 def __delitem__(self, key: str) -> None:
529 """
530 Remove the header `key`.
531 """
532 del_key = key.lower().encode(self.encoding)
533
534 pop_indexes = []
535 for idx, (item_key, _) in enumerate(self._list):
536 if item_key == del_key:
537 pop_indexes.append(idx)
538 if not pop_indexes:
539 raise KeyError(key)
540
541 for idx in reversed(pop_indexes):
542 del self._list[idx]
543
544 def __contains__(self, key: typing.Any) -> bool:
545 get_header_key = key.lower().encode(self.encoding)
546 for header_key, _ in self._list:
547 if header_key == get_header_key:
548 return True
549 return False
550
551 def __iter__(self) -> typing.Iterator[typing.Any]:
552 return iter(self.keys())
553
554 def __len__(self) -> int:
555 return len(self._list)
556
557 def __eq__(self, other: typing.Any) -> bool:
558 if not isinstance(other, Headers):
559 return False
560 return sorted(self._list) == sorted(other._list)
561
562 def __repr__(self) -> str:
563 class_name = self.__class__.__name__
564
565 encoding_str = ""
566 if self.encoding != "ascii":
567 encoding_str = f", encoding={self.encoding!r}"
568
569 as_list = list(obfuscate_sensitive_headers(self.items()))
570 as_dict = dict(as_list)
571
572 no_duplicate_keys = len(as_dict) == len(as_list)
573 if no_duplicate_keys:
574 return f"{class_name}({as_dict!r}{encoding_str})"
575 return f"{class_name}({as_list!r}{encoding_str})"
