116int AttachCallback(void *voidInitializeThreadingInfo) {
117 // initialize us for threading, this will acquire the GIL if not already created, and is a nop if the GIL is created.
118 // This leaves us in the proper state when we return back to the runtime whether the GIL was created or not before
119 // we were called.
120 InitializeThreadingInfo* initializeThreadingInfo = reinterpret_cast<InitializeThreadingInfo*>(voidInitializeThreadingInfo);
121 initializeThreadingInfo->initThreads(); // Note: calling multiple times is ok.
122 initializeThreadingInfo->pyImportMod("threading");
123
124 EnterCriticalSection(&initializeThreadingInfo->cs);
125 initializeThreadingInfo->completed = true;
126 if(initializeThreadingInfo->initedEvent != nullptr) {
127 SetEvent(initializeThreadingInfo->initedEvent);
128 }
129 LeaveCriticalSection(&initializeThreadingInfo->cs);
130 return 0;
131}

134// create a custom heap for our unordered map. This is necessary because if we suspend a thread while in a heap function
135// then we could deadlock here. We need to be VERY careful about what we do while the threads are suspended.
136static HANDLE g_heap = 0;

138template<typename T>
139class PrivateHeapAllocator {
140public:
141 typedef size_t size_type;
142 typedef ptrdiff_t difference_type;
143 typedef T* pointer;
144 typedef const T* const_pointer;
145 typedef T& reference;
146 typedef const T& const_reference;
147 typedef T value_type;
148
149 template<class U>
150 struct rebind {
151 typedef PrivateHeapAllocator<U> other;
152 };
153
154 explicit PrivateHeapAllocator() {}
155
156 PrivateHeapAllocator(PrivateHeapAllocator const&) {}
157
158 ~PrivateHeapAllocator() {}
159
160 template<typename U>
161 PrivateHeapAllocator(PrivateHeapAllocator<U> const&) {}
162
163 pointer allocate(size_type size, std::allocator<void>::const_pointer hint = 0) {
164 UNREFERENCED_PARAMETER(hint);
165
166 if (g_heap == nullptr) {
167 g_heap = HeapCreate(0, 0, 0);
168 }
169 auto mem = HeapAlloc(g_heap, 0, size * sizeof(T));
170 return static_cast<pointer>(mem);
171 }
172
173 void deallocate(pointer p, size_type n) {
174 UNREFERENCED_PARAMETER(n);
175
176 HeapFree(g_heap, 0, p);
177 }
178
179 size_type max_size() const {
180 return (std::numeric_limits<size_type>::max)() / sizeof(T);
181 }
182
183 void construct(pointer p, const T& t) {
184 new(p) T(t);
185 }
186
187 void destroy(pointer p) {
188 p->~T();
189 }
190};

192typedef std::unordered_map<DWORD, HANDLE, std::hash<DWORD>, std::equal_to<DWORD>, PrivateHeapAllocator<std::pair<DWORD, HANDLE>>> ThreadMap;

194void ResumeThreads(ThreadMap &suspendedThreads) {
195 for (auto start = suspendedThreads.begin(); start != suspendedThreads.end(); start++) {
196 ResumeThread((*start).second);
197 CloseHandle((*start).second);
198 }
199 suspendedThreads.clear();
200}

202// Suspends all threads ensuring that they are not currently in a call to Py_AddPendingCall.
203void SuspendThreads(ThreadMap &suspendedThreads, Py_AddPendingCall* addPendingCall, PyEval_ThreadsInitialized* threadsInited) {
204 DWORD curThreadId = GetCurrentThreadId();
205 DWORD curProcess = GetCurrentProcessId();
206 // suspend all the threads in the process so we can do things safely...
207 bool suspended;
208
209 do {
210 suspended = false;
211 HANDLE h = CreateToolhelp32Snapshot(TH32CS_SNAPTHREAD, 0);
212 if (h != INVALID_HANDLE_VALUE) {
213
214 THREADENTRY32 te;
215 te.dwSize = sizeof(te);
216 if (Thread32First(h, &te)) {
217 do {
218 if (te.dwSize >= FIELD_OFFSET(THREADENTRY32, th32OwnerProcessID) + sizeof(te.th32OwnerProcessID) && te.th32OwnerProcessID == curProcess) {
219
220
221 if (te.th32ThreadID != curThreadId && suspendedThreads.find(te.th32ThreadID) == suspendedThreads.end()) {
222 auto hThread = OpenThread(THREAD_ALL_ACCESS, FALSE, te.th32ThreadID);
223 if (hThread != nullptr) {
224 SuspendThread(hThread);
225
226 bool addingPendingCall = false;
227
228 CONTEXT context;
229 memset(&context, 0x00, sizeof(CONTEXT));
230 context.ContextFlags = CONTEXT_ALL;
231 GetThreadContext(hThread, &context);
232
233#if defined(_X86_)
234 if (context.Eip >= *(reinterpret_cast<DWORD*>(addPendingCall)) && context.Eip <= (*(reinterpret_cast<DWORD*>(addPendingCall))) + 0x100) {
235 addingPendingCall = true;
236 }
237#elif defined(_AMD64_)
238 if (context.Rip >= *(reinterpret_cast<DWORD64*>(addPendingCall)) && context.Rip <= *(reinterpret_cast<DWORD64*>(addPendingCall) + 0x100)) {
239 addingPendingCall = true;
240 }
241#endif
242
243 if (addingPendingCall) {
244 // we appear to be adding a pending call via this thread - wait for this to finish so we can add our own pending call...
245 ResumeThread(hThread);
246 SwitchToThread(); // yield to the resumed thread if it's on our CPU...
247 CloseHandle(hThread);
248 } else {
249 suspendedThreads[te.th32ThreadID] = hThread;
250 }
251 suspended = true;
252 }
253 }
254 }
255
256 te.dwSize = sizeof(te);
257 } while (Thread32Next(h, &te) && !threadsInited());
258 }
259 CloseHandle(h);
260 }
261 } while (suspended && !threadsInited());
262}

281 int DoAttach(HMODULE module, bool isDebug, const char *command, bool showDebugInfo )
282 {
283 auto isInit = reinterpret_cast<Py_IsInitialized*>(GetProcAddress(module, "Py_IsInitialized"));
284
285 if (isInit == nullptr) {
286 std::cerr << "Py_IsInitialized not found. " << std::endl << std::flush;
287 return 1;
288 }
289 if (!isInit()) {
290 std::cerr << "Py_IsInitialized returned false. " << std::endl << std::flush;
291 return 2;
292 }
293
294 auto version = GetPythonVersion(module);
295
296 // found initialized Python runtime, gather and check the APIs we need for a successful attach...
297 DEFINE_PROC(addPendingCall, Py_AddPendingCall*, "Py_AddPendingCall", -100);
298 DEFINE_PROC(interpHead, PyInterpreterState_Head*, "PyInterpreterState_Head", -110);
299 DEFINE_PROC(gilEnsure, PyGILState_Ensure*, "PyGILState_Ensure", -120);
300 DEFINE_PROC(gilRelease, PyGILState_Release*, "PyGILState_Release", -130);
301 DEFINE_PROC(threadHead, PyInterpreterState_ThreadHead*, "PyInterpreterState_ThreadHead", -140);
302 DEFINE_PROC(initThreads, PyEval_Lock*, "PyEval_InitThreads", -150);
303 DEFINE_PROC(releaseLock, PyEval_Lock*, "PyEval_ReleaseLock", -160);
304 DEFINE_PROC(threadsInited, PyEval_ThreadsInitialized*, "PyEval_ThreadsInitialized", -170);
305 DEFINE_PROC(threadNext, PyThreadState_Next*, "PyThreadState_Next", -180);
306 DEFINE_PROC(pyImportMod, PyImport_ImportModule*, "PyImport_ImportModule", -190);
307 DEFINE_PROC(pyNone, PyObject*, "_Py_NoneStruct", -2000);
308 DEFINE_PROC(pyRun_SimpleString, PyRun_SimpleString*, "PyRun_SimpleString", -210);
309
310 // Either _PyThreadState_Current or _PyThreadState_UncheckedGet are required
311 DEFINE_PROC_NO_CHECK(curPythonThread, PyThreadState**, "_PyThreadState_Current", -220); // optional
312 DEFINE_PROC_NO_CHECK(getPythonThread, _PyThreadState_UncheckedGet*, "_PyThreadState_UncheckedGet", -230); // optional
313 DEFINE_PROC_NO_CHECK(getPythonThread13, _PyThreadState_GetCurrent*, "_PyThreadState_GetCurrent", -231); // optional
314 if (getPythonThread == nullptr && getPythonThread13 != nullptr) {
315 std::cout << "Using Python 3.13 or later, using _PyThreadState_GetCurrent" << std::endl << std::flush;
316 getPythonThread = getPythonThread13;
317 }
318
319 if (curPythonThread == nullptr && getPythonThread == nullptr) {
320 // we're missing some APIs, we cannot attach.
321 std::cerr << "Error, missing Python threading API!!" << std::endl << std::flush;
322 return -240;
323 }
324
325 // Either _Py_CheckInterval or _PyEval_[GS]etSwitchInterval are useful, but not required
326 DEFINE_PROC_NO_CHECK(intervalCheck, int*, "_Py_CheckInterval", -250); // optional
327 DEFINE_PROC_NO_CHECK(getSwitchInterval, _PyEval_GetSwitchInterval*, "_PyEval_GetSwitchInterval", -260); // optional
328 DEFINE_PROC_NO_CHECK(setSwitchInterval, _PyEval_SetSwitchInterval*, "_PyEval_SetSwitchInterval", -270); // optional
329
330 auto head = interpHead();
331 if (head == nullptr) {
332 // this interpreter is loaded but not initialized.
333 std::cerr << "Interpreter not initialized! " << std::endl << std::flush;
334 return 4;
335 }
336
337 // check that we're a supported version
338 if (version == PythonVersion_Unknown) {
339 std::cerr << "Python version unknown! " << std::endl << std::flush;
340 return 5;
341 } else if (version == PythonVersion_25 || version == PythonVersion_26 ||
342 version == PythonVersion_30 || version == PythonVersion_31 || version == PythonVersion_32) {
343 std::cerr << "Python version unsupported! " << std::endl << std::flush;
344 return 5;
345 }
346
347
348 // We always try to initialize threading and import the threading module in the main thread in the code
349 // below...
350 //
351 // We need to initialize multiple threading support but we need to do so safely, so we call
352 // Py_AddPendingCall and have our callback then initialize multi threading. This is completely safe on 2.7
353 // and up. Unfortunately that doesn't work if we're not actively running code on the main thread (blocked on a lock
354 // or reading input).
355 //
356 // Another option is to make sure no code is running - if there is no active thread then we can safely call
357 // PyEval_InitThreads and we're in business. But to know this is safe we need to first suspend all the other
358 // threads in the process and then inspect if any code is running (note that this is still not ideal because
359 // this thread will be the thread head for Python, but still better than not attach at all).
360 //
361 // Finally if code is running after we've suspended the threads then we can go ahead and do Py_AddPendingCall
362 // on down-level interpreters as long as we're sure no one else is making a call to Py_AddPendingCall at the same
363 // time.
364 //
365 // Therefore our strategy becomes: Make the Py_AddPendingCall on interpreters and wait for it. If it doesn't
366 // call after a timeout, suspend all threads - if a threads is in Py_AddPendingCall resume and try again. Once we've got all of the threads
367 // stopped and not in Py_AddPendingCall (which calls no functions its self, you can see this and it's size in the
368 // debugger) then see if we have a current thread. If not go ahead and initialize multiple threading (it's now safe,
369 // no Python code is running).
370
371 InitializeThreadingInfo *initializeThreadingInfo = new InitializeThreadingInfo();
372 initializeThreadingInfo->pyImportMod = pyImportMod;
373 initializeThreadingInfo->initThreads = initThreads;
374 initializeThreadingInfo->initedEvent = CreateEvent(nullptr, TRUE, FALSE, nullptr);
375 InitializeCriticalSection(&initializeThreadingInfo->cs);
376
377 // Add the call to initialize threading.
378 addPendingCall(&AttachCallback, initializeThreadingInfo);
379
380 ::WaitForSingleObject(initializeThreadingInfo->initedEvent, 5000);
381
382 // Whether this completed or not, release the event handle as we won't use it anymore.
383 EnterCriticalSection(&initializeThreadingInfo->cs);
384 CloseHandle(initializeThreadingInfo->initedEvent);
385 bool completed = initializeThreadingInfo->completed;
386 initializeThreadingInfo->initedEvent = nullptr;
387 LeaveCriticalSection(&initializeThreadingInfo->cs);
388
389 if(completed) {
390 // Note that this structure will leak if addPendingCall did not complete in the timeout
391 // (we can't release now because it's possible that it'll still be called).
392 DeleteCriticalSection(&initializeThreadingInfo->cs);
393 delete initializeThreadingInfo;
394 if (showDebugInfo) {
395 std::cout << "addPendingCall to initialize threads/import threading completed. " << std::endl << std::flush;
396 }
397 } else {
398 if (showDebugInfo) {
399 std::cout << "addPendingCall to initialize threads/import threading did NOT complete. " << std::endl << std::flush;
400 }
401 }
402
403 if (threadsInited()) {
404 // Note that since Python 3.7, threads are *always* initialized!
405 if (showDebugInfo) {
406 std::cout << "Threads initialized! " << std::endl << std::flush;
407 }
408
409 } else {
410 int saveIntervalCheck;
411 unsigned long saveLongIntervalCheck;
412 if (intervalCheck != nullptr) {
413 // not available on 3.2
414 saveIntervalCheck = *intervalCheck;
415 *intervalCheck = -1; // lower the interval check so pending calls are processed faster
416 saveLongIntervalCheck = 0; // prevent compiler warning
417 } else if (getSwitchInterval != nullptr && setSwitchInterval != nullptr) {
418 saveLongIntervalCheck = getSwitchInterval();
419 setSwitchInterval(0);
420 saveIntervalCheck = 0; // prevent compiler warning
421 }
422 else {
423 saveIntervalCheck = 0; // prevent compiler warning
424 saveLongIntervalCheck = 0; // prevent compiler warning
425 }
426
427 // If threads weren't initialized in our pending call, instead of giving a timeout, try
428 // to initialize it in this thread.
429 for(int attempts = 0; !threadsInited() && attempts < 20; attempts++) {
430 if(attempts > 0){
431 // If we haven't been able to do it in the first time, wait a bit before retrying.
432 Sleep(10);
433 }
434
435 ThreadMap suspendedThreads;
436 if (showDebugInfo) {
437 std::cout << "SuspendThreads(suspendedThreads, addPendingCall, threadsInited);" << std::endl << std::flush;
438 }
439 SuspendThreads(suspendedThreads, addPendingCall, threadsInited);
440
441 if(!threadsInited()){ // Check again with threads suspended.
442 if (showDebugInfo) {
443 std::cout << "ENTERED if (!threadsInited()) {" << std::endl << std::flush;
444 }
445 auto curPyThread = getPythonThread ? getPythonThread() : *curPythonThread;
446
447 if (curPyThread == nullptr) {
448 if (showDebugInfo) {
449 std::cout << "ENTERED if (curPyThread == nullptr) {" << std::endl << std::flush;
450 }
451 // no threads are currently running, it is safe to initialize multi threading.
452 PyGILState_STATE gilState;
453 if (version >= PythonVersion_34) {
454 // in 3.4 due to http://bugs.python.org/issue20891,
455 // we need to create our thread state manually
456 // before we can call PyGILState_Ensure() before we
457 // can call PyEval_InitThreads().
458
459 // Don't require this function unless we need it.
460 auto threadNew = (PyThreadState_NewFunc*)GetProcAddress(module, "PyThreadState_New");
461 if (threadNew != nullptr) {
462 threadNew(head);
463 }
464 }
465
466 if (version >= PythonVersion_32) {
467 // in 3.2 due to the new GIL and later we can't call Py_InitThreads
468 // without a thread being initialized.
469 // So we use PyGilState_Ensure here to first
470 // initialize the current thread, and then we use
471 // Py_InitThreads to bring up multi-threading.
472 // Some context here: http://bugs.python.org/issue11329
473 // http://pytools.codeplex.com/workitem/834
474 gilState = gilEnsure();
475 }
476 else {
477 gilState = PyGILState_LOCKED; // prevent compiler warning
478 }
479
480 if (showDebugInfo) {
481 std::cout << "Called initThreads()" << std::endl << std::flush;
482 }
483 // Initialize threads in our secondary thread (this is NOT ideal because
484 // this thread will be the thread head), but is still better than not being
485 // able to attach if the main thread is not actually running any code.
486 initThreads();
487
488 if (version >= PythonVersion_32) {
489 // we will release the GIL here
490 gilRelease(gilState);
491 } else {
492 releaseLock();
493 }
494 }
495 }
496 ResumeThreads(suspendedThreads);
497 }
498
499
500 if (intervalCheck != nullptr) {
501 *intervalCheck = saveIntervalCheck;
502 } else if (setSwitchInterval != nullptr) {
503 setSwitchInterval(saveLongIntervalCheck);
504 }
505
506 }
507
508 if (g_heap != nullptr) {
509 HeapDestroy(g_heap);
510 g_heap = nullptr;
511 }
512
513 if (!threadsInited()) {
514 std::cerr << "Unable to initialize threads in the given timeout! " << std::endl << std::flush;
515 return 8;
516 }
517
518 GilHolder gilLock(gilEnsure, gilRelease); // acquire and hold the GIL until done...
519
520 pyRun_SimpleString(command);
521 return 0;
522
523 }

563 DECLDIR int PrintDebugInfo() {
564 PRINT("Getting debug info...");
565 ModuleInfo moduleInfo = GetPythonModule();
566 if (moduleInfo.errorGettingModule != 0) {
567 PRINT("Error getting python module");
568 return 0;
569 }
570 HMODULE module = moduleInfo.module;
571
572 DEFINE_PROC(interpHead, PyInterpreterState_Head*, "PyInterpreterState_Head", 0);
573 DEFINE_PROC(threadHead, PyInterpreterState_ThreadHead*, "PyInterpreterState_ThreadHead", 0);
574 DEFINE_PROC(threadNext, PyThreadState_Next*, "PyThreadState_Next", 160);
575 DEFINE_PROC(gilEnsure, PyGILState_Ensure*, "PyGILState_Ensure", 0);
576 DEFINE_PROC(gilRelease, PyGILState_Release*, "PyGILState_Release", 0);
577
578 auto head = interpHead();
579 if (head == nullptr) {
580 // this interpreter is loaded but not initialized.
581 PRINT("Interpreter not initialized!");
582 return 0;
583 }
584
585 auto version = GetPythonVersion(module);
586 printf("Python version: %d\n", version);
587
588 GilHolder gilLock(gilEnsure, gilRelease); // acquire and hold the GIL until done...
589 auto curThread = threadHead(head);
590 if (curThread == nullptr) {
591 PRINT("Thread head is NULL.")
592 return 0;
593 }
594
595 for (auto curThread = threadHead(head); curThread != nullptr; curThread = threadNext(curThread)) {
596 printf("Found thread id: %d\n", GetPythonThreadId(version, curThread));
597 }
598
599 PRINT("Finished getting debug info.")
600 return 0;
601 }

607 DECLDIR int AttachDebuggerTracing(bool showDebugInfo, void* pSetTraceFunc, void* pTraceFunc, unsigned int threadId, void* pPyNone)
608 {
609 ModuleInfo moduleInfo = GetPythonModule();
610 if (moduleInfo.errorGettingModule != 0) {
611 return moduleInfo.errorGettingModule;
612 }
613 HMODULE module = moduleInfo.module;
614 if (showDebugInfo) {
615 std::cout << "Setting sys trace for existing threads." << std::endl << std::flush;
616 }
617 int attached = 0;
618 PyObjectHolder traceFunc(moduleInfo.isDebug, reinterpret_cast<PyObject*>(pTraceFunc), true);
619 PyObjectHolder setTraceFunc(moduleInfo.isDebug, reinterpret_cast<PyObject*>(pSetTraceFunc), true);
620 PyObjectHolder pyNone(moduleInfo.isDebug, reinterpret_cast<PyObject*>(pPyNone), true);
621
622 int temp = InternalSetSysTraceFunc(module, moduleInfo.isDebug, showDebugInfo, &traceFunc, &setTraceFunc, threadId, &pyNone);
623 if (temp == 0) {
624 // we've successfully attached the debugger
625 return 0;
626 } else {
627 if (temp > attached) {
628 //I.e.: the higher the value the more significant it is.
629 attached = temp;
630 }
631 }
632
633 if (showDebugInfo) {
634 std::cout << "Setting sys trace for existing threads failed with code: " << attached << "." << std::endl << std::flush;
635 }
636 return attached;
637 }
