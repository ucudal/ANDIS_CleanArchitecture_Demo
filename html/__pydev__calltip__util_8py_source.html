138def create_function_stub(fn_name, fn_argspec, fn_docstring, indent=0):
139 def shift_right(string, prefix):
140 return "".join(prefix + line for line in string.splitlines(True))
141
142 fn_docstring = shift_right(inspect.cleandoc(fn_docstring), " " * (indent + 1))
143 ret = '''
144def %s%s:
145 """%s"""
146 pass
147''' % (fn_name, fn_argspec, fn_docstring)
148 ret = ret[1:] # remove first /n
149 ret = ret.replace("\t", " ")
150 if indent:
151 prefix = " " * indent
