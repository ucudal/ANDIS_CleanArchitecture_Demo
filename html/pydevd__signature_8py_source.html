44def get_type_of_value(value, ignore_module_name=("__main__", "__builtin__", "builtins"), recursive=False):
45 tp = type(value)
46 class_name = tp.__name__
47 if class_name == "instance": # old-style classes
48 tp = value.__class__
49 class_name = tp.__name__
50
51 if hasattr(tp, "__module__") and tp.__module__ and tp.__module__ not in ignore_module_name:
52 class_name = "%s.%s" % (tp.__module__, class_name)
53
54 if class_name == "list":
55 class_name = "List"
56 if len(value) > 0 and recursive:
57 class_name += "[%s]" % get_type_of_value(value[0], recursive=recursive)
58 return class_name
59
60 if class_name == "dict":
61 class_name = "Dict"
62 if len(value) > 0 and recursive:
63 for k, v in value.items():
64 class_name += "[%s, %s]" % (get_type_of_value(k, recursive=recursive), get_type_of_value(v, recursive=recursive))
65 break
66 return class_name
67
68 if class_name == "tuple":
69 class_name = "Tuple"
70 if len(value) > 0 and recursive:
71 class_name += "["
72 class_name += ", ".join(get_type_of_value(v, recursive=recursive) for v in value)
73 class_name += "]"
74
75 return class_name
