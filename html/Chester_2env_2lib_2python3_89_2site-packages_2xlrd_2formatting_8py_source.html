129def nearest_colour_index(colour_map, rgb, debug=0):
130 """
131 General purpose function. Uses Euclidean distance.
132 So far used only for pre-BIFF8 ``WINDOW2`` record.
133 Doesn't have to be fast.
134 Doesn't have to be fancy.
135 """
136 best_metric = 3 * 256 * 256
137 best_colourx = 0
138 for colourx, cand_rgb in colour_map.items():
139 if cand_rgb is None:
140 continue
141 metric = 0
142 for v1, v2 in zip(rgb, cand_rgb):
143 metric += (v1 - v2) * (v1 - v2)
144 if metric < best_metric:
145 best_metric = metric
146 best_colourx = colourx
147 if metric == 0:
148 break
149 if 0 and debug:
150 print("nearest_colour_index for %r is %r -> %r; best_metric is %d"
151 % (rgb, best_colourx, colour_map[best_colourx], best_metric))
152 return best_colourx

459def is_date_format_string(book, fmt):
460 # Heuristics:
461 # Ignore "text" and [stuff in square brackets (aarrgghh -- see below)].
462 # Handle backslashed-escaped chars properly.
463 # E.g. hh\hmm\mss\s should produce a display like 23h59m59s
464 # Date formats have one or more of ymdhs (caseless) in them.
465 # Numeric formats have # and 0.
466 # N.B. 'General"."' hence get rid of "text" first.
467 # TODO: Find where formats are interpreted in Gnumeric
468 # TODO: '[h]\\ \\h\\o\\u\\r\\s' ([h] means don't care about hours > 23)
469 state = 0
470 s = ''
471
472 for c in fmt:
473 if state == 0:
474 if c == UNICODE_LITERAL('"'):
475 state = 1
476 elif c in UNICODE_LITERAL(r"\_*"):
477 state = 2
478 elif c in skip_char_dict:
479 pass
480 else:
481 s += c
482 elif state == 1:
483 if c == UNICODE_LITERAL('"'):
484 state = 0
485 elif state == 2:
486 # Ignore char after backslash, underscore or asterisk
487 state = 0
488 assert 0 <= state <= 2
489 if book.verbosity >= 4:
490 print("is_date_format_string: reduced format is %s" % REPR(s), file=book.logfile)
491 s = fmt_bracketed_sub('', s)
492 if s in non_date_formats:
493 return False
494 state = 0
495 separator = ";"
496 got_sep = 0
497 date_count = num_count = 0
498 for c in s:
499 if c in date_char_dict:
500 date_count += date_char_dict[c]
501 elif c in num_char_dict:
502 num_count += num_char_dict[c]
503 elif c == separator:
504 got_sep = 1
505 # print num_count, date_count, repr(fmt)
506 if date_count and not num_count:
507 return True
508 if num_count and not date_count:
509 return False
510 if date_count:
511 if book.verbosity:
512 fprintf(book.logfile,
513 'WARNING *** is_date_format: ambiguous d=%d n=%d fmt=%r\n',
514 date_count, num_count, fmt)
515 elif not got_sep:
516 if book.verbosity:
517 fprintf(book.logfile,
518 "WARNING *** format %r produces constant result\n",
519 fmt)
520 return date_count > num_count

574def handle_palette(book, data):
575 if not book.formatting_info:
576 return
577 blah = DEBUG or book.verbosity >= 2
578 n_colours, = unpack('<H', data[:2])
579 expected_n_colours = (16, 56)[book.biff_version >= 50]
580 if (DEBUG or book.verbosity >= 1) and n_colours != expected_n_colours:
581 fprintf(book.logfile,
582 "NOTE *** Expected %d colours in PALETTE record, found %d\n",
583 expected_n_colours, n_colours)
584 elif blah:
585 fprintf(book.logfile,
586 "PALETTE record with %d colours\n", n_colours)
587 fmt = '<xx%di' % n_colours # use i to avoid long integers
588 expected_size = 4 * n_colours + 2
589 actual_size = len(data)
590 tolerance = 4
591 if not expected_size <= actual_size <= expected_size + tolerance:
592 raise XLRDError('PALETTE record: expected size %d, actual size %d' % (expected_size, actual_size))
593 colours = unpack(fmt, data[:expected_size])
594 assert book.palette_record == [] # There should be only 1 PALETTE record
595 # a colour will be 0xbbggrr
596 # IOW, red is at the little end
597 for i in xrange(n_colours):
598 c = colours[i]
599 red = c & 0xff
600 green = (c >> 8) & 0xff
601 blue = (c >> 16) & 0xff
602 old_rgb = book.colour_map[8+i]
603 new_rgb = (red, green, blue)
604 book.palette_record.append(new_rgb)
605 book.colour_map[8+i] = new_rgb
606 if blah:
607 if new_rgb != old_rgb:
608 print("%2d: %r -> %r" % (i, old_rgb, new_rgb), file=book.logfile)

630def handle_style(book, data):
631 if not book.formatting_info:
632 return
633 blah = DEBUG or book.verbosity >= 2
634 bv = book.biff_version
635 flag_and_xfx, built_in_id, level = unpack('<HBB', data[:4])
636 xf_index = flag_and_xfx & 0x0fff
637 if data == b"\0\0\0\0" and "Normal" not in book.style_name_map:
638 # Erroneous record (doesn't have built-in bit set).
639 # Example file supplied by Jeff Bell.
640 built_in = 1
641 built_in_id = 0
642 xf_index = 0
643 name = "Normal"
644 level = 255
645 elif flag_and_xfx & 0x8000:
646 # built-in style
647 built_in = 1
648 name = built_in_style_names[built_in_id]
649 if 1 <= built_in_id <= 2:
650 name += str(level + 1)
651 else:
652 # user-defined style
653 built_in = 0
654 built_in_id = 0
655 level = 0
656 if bv >= 80:
657 try:
658 name = unpack_unicode(data, 2, lenlen=2)
659 except UnicodeDecodeError:
660 print("STYLE: built_in=%d xf_index=%d built_in_id=%d level=%d"
661 % (built_in, xf_index, built_in_id, level), file=book.logfile)
662 print("raw bytes:", repr(data[2:]), file=book.logfile)
663 raise
664 else:
665 name = unpack_string(data, 2, book.encoding, lenlen=1)
666 if blah and not name:
667 print("WARNING *** A user-defined style has a zero-length name", file=book.logfile)
668 book.style_name_map[name] = (built_in, xf_index)
669 if blah:
670 fprintf(book.logfile, "STYLE: built_in=%d xf_index=%d built_in_id=%d level=%d name=%r\n",
671 built_in, xf_index, built_in_id, level, name)
