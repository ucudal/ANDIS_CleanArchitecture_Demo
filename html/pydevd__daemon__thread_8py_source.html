173def mark_as_pydevd_daemon_thread(thread):
174 if not IS_JYTHON and not IS_IRONPYTHON and PYDEVD_APPLY_PATCHING_TO_HIDE_PYDEVD_THREADS:

178 # functions to hide pydevd threads.

180 # (i.e.: just using the `thread.start_new_thread` instead of `threading.Thread`)
181 # because there's 1 thread (the `CheckAliveThread`) which is a pydevd thread but

183 # in which case it needs to be in `threading` and the patching would be needed anyways).
