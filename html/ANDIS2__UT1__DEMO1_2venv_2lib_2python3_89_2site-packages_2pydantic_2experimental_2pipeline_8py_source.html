134 def transform(
135 self,
136 func: Callable[[_OutT], _NewOutT],
137 ) -> _Pipeline[_InT, _NewOutT]:

139
140 If used as the first step in a pipeline, the type of the field is used.

224 """Constrain a value to meet a certain condition.
225
226 We support most conditions from `annotated_types`, as well as regular expressions.
227

229 so you don't need to call this directly.
230 """
231 return _Pipeline[_InT, _OutT](self._steps + (_Constraint(constraint),))

338 def __get_pydantic_core_schema__(self, source_type: Any, handler: GetCoreSchemaHandler) -> cs.CoreSchema:
339 from pydantic_core import core_schema as cs
340
341 queue = deque(self._steps)
342
343 s = None
344
345 while queue:
346 step = queue.popleft()
347 s = _apply_step(step, s, handler, source_type)
348
349 s = s or cs.any_schema()
350 return s

361def _check_func(
362 func: Callable[[Any], bool], predicate_err: str | Callable[[], str], s: cs.CoreSchema | None
363) -> cs.CoreSchema:
364 from pydantic_core import core_schema as cs
365
366 def handler(v: Any) -> Any:
367 if func(v):
368 return v
369 raise ValueError(f'Expected {predicate_err if isinstance(predicate_err, str) else predicate_err()}')
370
371 if s is None:
372 return cs.no_info_plain_validator_function(handler)
373 else:
374 return cs.no_info_after_validator_function(handler, s)

377def _apply_step(step: _Step, s: cs.CoreSchema | None, handler: GetCoreSchemaHandler, source_type: Any) -> cs.CoreSchema:
378 from pydantic_core import core_schema as cs
379
380 if isinstance(step, _ValidateAs):
381 s = _apply_parse(s, step.tp, step.strict, handler, source_type)
382 elif isinstance(step, _ValidateAsDefer):
383 s = _apply_parse(s, step.tp, False, handler, source_type)
384 elif isinstance(step, _Transform):
385 s = _apply_transform(s, step.func, handler)
386 elif isinstance(step, _Constraint):
387 s = _apply_constraint(s, step.constraint)
388 elif isinstance(step, _PipelineOr):
389 s = cs.union_schema([handler(step.left), handler(step.right)])
390 else:
391 assert isinstance(step, _PipelineAnd)
392 s = cs.chain_schema([handler(step.left), handler(step.right)])
393 return s

396def _apply_parse(
397 s: cs.CoreSchema | None,
398 tp: type[Any],
399 strict: bool,
400 handler: GetCoreSchemaHandler,
401 source_type: Any,
402) -> cs.CoreSchema:
403 from pydantic_core import core_schema as cs
404
405 from pydantic import Strict
406
407 if tp is _FieldTypeMarker:
408 return cs.chain_schema([s, handler(source_type)]) if s else handler(source_type)
409
410 if strict:
411 tp = Annotated[tp, Strict()] # type: ignore
412
413 if s and s['type'] == 'any':
414 return handler(tp)
415 else:
416 return cs.chain_schema([s, handler(tp)]) if s else handler(tp)

419def _apply_transform(
420 s: cs.CoreSchema | None, func: Callable[[Any], Any], handler: GetCoreSchemaHandler
421) -> cs.CoreSchema:
422 from pydantic_core import core_schema as cs
423
424 if s is None:
425 return cs.no_info_plain_validator_function(func)
426
427 if s['type'] == 'str':
428 if func is str.strip:
429 s = s.copy()
430 s['strip_whitespace'] = True
431 return s
432 elif func is str.lower:
433 s = s.copy()
434 s['to_lower'] = True
435 return s
436 elif func is str.upper:
437 s = s.copy()
438 s['to_upper'] = True
439 return s
440
441 return cs.no_info_after_validator_function(func, s)

444def _apply_constraint( # noqa: C901
445 s: cs.CoreSchema | None, constraint: _ConstraintAnnotation
446) -> cs.CoreSchema:
447 """Apply a single constraint to a schema."""
448 if isinstance(constraint, annotated_types.Gt):
449 gt = constraint.gt
450 if s and s['type'] in {'int', 'float', 'decimal'}:
451 s = s.copy()
452 if s['type'] == 'int' and isinstance(gt, int):
453 s['gt'] = gt
454 elif s['type'] == 'float' and isinstance(gt, float):
455 s['gt'] = gt
456 elif s['type'] == 'decimal' and isinstance(gt, Decimal):
457 s['gt'] = gt
458 else:
459
460 def check_gt(v: Any) -> bool:
461 return v > gt
462
463 s = _check_func(check_gt, f'> {gt}', s)
464 elif isinstance(constraint, annotated_types.Ge):
465 ge = constraint.ge
466 if s and s['type'] in {'int', 'float', 'decimal'}:
467 s = s.copy()
468 if s['type'] == 'int' and isinstance(ge, int):
469 s['ge'] = ge
470 elif s['type'] == 'float' and isinstance(ge, float):
471 s['ge'] = ge
472 elif s['type'] == 'decimal' and isinstance(ge, Decimal):
473 s['ge'] = ge
474
475 def check_ge(v: Any) -> bool:
476 return v >= ge
477
478 s = _check_func(check_ge, f'>= {ge}', s)
479 elif isinstance(constraint, annotated_types.Lt):
480 lt = constraint.lt
481 if s and s['type'] in {'int', 'float', 'decimal'}:
482 s = s.copy()
483 if s['type'] == 'int' and isinstance(lt, int):
484 s['lt'] = lt
485 elif s['type'] == 'float' and isinstance(lt, float):
486 s['lt'] = lt
487 elif s['type'] == 'decimal' and isinstance(lt, Decimal):
488 s['lt'] = lt
489
490 def check_lt(v: Any) -> bool:
491 return v < lt
492
493 s = _check_func(check_lt, f'< {lt}', s)
494 elif isinstance(constraint, annotated_types.Le):
495 le = constraint.le
496 if s and s['type'] in {'int', 'float', 'decimal'}:
497 s = s.copy()
498 if s['type'] == 'int' and isinstance(le, int):
499 s['le'] = le
500 elif s['type'] == 'float' and isinstance(le, float):
501 s['le'] = le
502 elif s['type'] == 'decimal' and isinstance(le, Decimal):
503 s['le'] = le
504
505 def check_le(v: Any) -> bool:
506 return v <= le
507
508 s = _check_func(check_le, f'<= {le}', s)
509 elif isinstance(constraint, annotated_types.Len):
510 min_len = constraint.min_length
511 max_len = constraint.max_length
512
513 if s and s['type'] in {'str', 'list', 'tuple', 'set', 'frozenset', 'dict'}:
514 assert (
515 s['type'] == 'str'
516 or s['type'] == 'list'
517 or s['type'] == 'tuple'
518 or s['type'] == 'set'
519 or s['type'] == 'dict'
520 or s['type'] == 'frozenset'
521 )
522 s = s.copy()
523 if min_len != 0:
524 s['min_length'] = min_len
525 if max_len is not None:
526 s['max_length'] = max_len
527
528 def check_len(v: Any) -> bool:
529 if max_len is not None:
530 return (min_len <= len(v)) and (len(v) <= max_len)
531 return min_len <= len(v)
532
533 s = _check_func(check_len, f'length >= {min_len} and length <= {max_len}', s)
534 elif isinstance(constraint, annotated_types.MultipleOf):
535 multiple_of = constraint.multiple_of
536 if s and s['type'] in {'int', 'float', 'decimal'}:
537 s = s.copy()
538 if s['type'] == 'int' and isinstance(multiple_of, int):
539 s['multiple_of'] = multiple_of
540 elif s['type'] == 'float' and isinstance(multiple_of, float):
541 s['multiple_of'] = multiple_of
542 elif s['type'] == 'decimal' and isinstance(multiple_of, Decimal):
543 s['multiple_of'] = multiple_of
544
545 def check_multiple_of(v: Any) -> bool:
546 return v % multiple_of == 0
547
548 s = _check_func(check_multiple_of, f'% {multiple_of} == 0', s)
549 elif isinstance(constraint, annotated_types.Timezone):
550 tz = constraint.tz
551
552 if tz is ...:
553 if s and s['type'] == 'datetime':
554 s = s.copy()
555 s['tz_constraint'] = 'aware'
556 else:
557
558 def check_tz_aware(v: object) -> bool:
559 assert isinstance(v, datetime.datetime)
560 return v.tzinfo is not None
561
562 s = _check_func(check_tz_aware, 'timezone aware', s)
563 elif tz is None:
564 if s and s['type'] == 'datetime':
565 s = s.copy()
566 s['tz_constraint'] = 'naive'
567 else:
568
569 def check_tz_naive(v: object) -> bool:
570 assert isinstance(v, datetime.datetime)
571 return v.tzinfo is None
572
573 s = _check_func(check_tz_naive, 'timezone naive', s)
574 else:
575 raise NotImplementedError('Constraining to a specific timezone is not yet supported')
576 elif isinstance(constraint, annotated_types.Interval):
577 if constraint.ge:
578 s = _apply_constraint(s, annotated_types.Ge(constraint.ge))
579 if constraint.gt:
580 s = _apply_constraint(s, annotated_types.Gt(constraint.gt))
581 if constraint.le:
582 s = _apply_constraint(s, annotated_types.Le(constraint.le))
583 if constraint.lt:
584 s = _apply_constraint(s, annotated_types.Lt(constraint.lt))
585 assert s is not None
586 elif isinstance(constraint, annotated_types.Predicate):
587 func = constraint.func
588
589 if func.__name__ == '<lambda>':
590 # attempt to extract the source code for a lambda function
591 # to use as the function name in error messages
592 # TODO: is there a better way? should we just not do this?
593 import inspect
594
595 try:
596 source = inspect.getsource(func).strip()
597 source = source.removesuffix(')')
598 lambda_source_code = '`' + ''.join(''.join(source.split('lambda ')[1:]).split(':')[1:]).strip() + '`'
599 except OSError:
600 # stringified annotations
601 lambda_source_code = 'lambda'
602
603 s = _check_func(func, lambda_source_code, s)
604 else:
605 s = _check_func(func, func.__name__, s)
606 elif isinstance(constraint, _NotEq):
607 value = constraint.value
608
609 def check_not_eq(v: Any) -> bool:
610 return operator.__ne__(v, value)
611
612 s = _check_func(check_not_eq, f'!= {value}', s)
613 elif isinstance(constraint, _Eq):
614 value = constraint.value
615
616 def check_eq(v: Any) -> bool:
617 return operator.__eq__(v, value)
618
619 s = _check_func(check_eq, f'== {value}', s)
620 elif isinstance(constraint, _In):
621 values = constraint.values
622
623 def check_in(v: Any) -> bool:
624 return operator.__contains__(values, v)
625
626 s = _check_func(check_in, f'in {values}', s)
627 elif isinstance(constraint, _NotIn):
628 values = constraint.values
629
630 def check_not_in(v: Any) -> bool:
631 return operator.__not__(operator.__contains__(values, v))
632
633 s = _check_func(check_not_in, f'not in {values}', s)
634 else:
635 assert isinstance(constraint, Pattern)
636 if s and s['type'] == 'str':
637 s = s.copy()
638 s['pattern'] = constraint.pattern
639 else:
640
641 def check_pattern(v: object) -> bool:
642 assert isinstance(v, str)
643 return constraint.match(v) is not None
644
645 s = _check_func(check_pattern, f'~ {constraint.pattern}', s)
646 return s
