31def read_python_source(filename: str) -> bytes:
32 """Read the Python source text from `filename`.
33
34 Returns bytes.
35
36 """
37 with open(filename, "rb") as f:
38 source = f.read()
39
40 return source.replace(b"\r\n", b"\n").replace(b"\r", b"\n")

43def get_python_source(filename: str) -> str:
44 """Return the source code, as unicode."""
45 base, ext = os.path.splitext(filename)
46 if ext == ".py" and env.WINDOWS:
47 exts = [".py", ".pyw"]
48 else:
49 exts = [ext]
50
51 source_bytes: bytes | None
52 for ext in exts:
53 try_filename = base + ext
54 if os.path.exists(try_filename):
55 # A regular text file: open it.
56 source_bytes = read_python_source(try_filename)
57 break
58
59 # Maybe it's in a zip file?
60 source_bytes = get_zip_bytes(try_filename)
61 if source_bytes is not None:
62 break
63 else:
64 # Couldn't find source.
65 raise NoSource(f"No source for code: '{filename}'.")
66
67 # Replace \f because of http://bugs.python.org/issue19035
68 source_bytes = source_bytes.replace(b"\f", b" ")
69 source = source_bytes.decode(source_encoding(source_bytes), "replace")
70
71 # Python code should always end with a line with a newline.
72 if source and source[-1] != "\n":
73 source += "\n"
74
75 return source

78def get_zip_bytes(filename: str) -> bytes | None:
79 """Get data from `filename` if it is a zip file path.
80
81 Returns the bytestring data read from the zip file, or None if no zip file
82 could be found or `filename` isn't in it. The data returned will be
83 an empty string if the file is empty.
84
85 """
86 zipfile_inner = zip_location(filename)
87 if zipfile_inner is not None:
88 zipfile, inner = zipfile_inner
89 try:
90 zi = zipimport.zipimporter(zipfile)
91 except zipimport.ZipImportError:
92 return None
93 try:
94 data = zi.get_data(inner)
95 except OSError:
96 return None
97 return data
98 return None

101def source_for_file(filename: str) -> str:
102 """Return the source filename for `filename`.
103
104 Given a file name being traced, return the best guess as to the source
105 file to attribute it to.
106
107 """
108 if filename.endswith(".py"):
109 # .py files are themselves source files.
110 return filename
111
112 elif filename.endswith((".pyc", ".pyo")):
113 # Bytecode files probably have source files near them.
114 py_filename = filename[:-1]
115 if os.path.exists(py_filename):
116 # Found a .py file, use that.
117 return py_filename
118 if env.WINDOWS:
119 # On Windows, it could be a .pyw file.
120 pyw_filename = py_filename + "w"
121 if os.path.exists(pyw_filename):
122 return pyw_filename
123 # Didn't find source, but it's probably the .py file we want.
124 return py_filename
125
126 # No idea, just use the file name as-is.
127 return filename

130def source_for_morf(morf: TMorf) -> str:
131 """Get the source filename for the module-or-file `morf`."""
132 if hasattr(morf, "__file__") and morf.__file__:
133 filename = morf.__file__
134 elif isinstance(morf, types.ModuleType):
135 # A module should have had .__file__, otherwise we can't use it.
136 # This could be a PEP-420 namespace package.
137 raise CoverageException(f"Module {morf} has no file")
138 else:
139 filename = morf
140
141 filename = source_for_file(filename)
142 return filename

243 def should_be_python(self) -> bool:
244 """Does it seem like this file should contain Python?
245
246 This is used to decide if a file reported as part of the execution of
247 a program was really likely to have contained Python in the first
248 place.
249
250 """
251 # Get the file extension.
252 _, ext = os.path.splitext(self.filename)
253
254 # Anything named *.py* should be Python.
255 if ext.startswith(".py"):
256 return True
257 # A file with no extension should be Python.
258 if not ext:
259 return True
260 # Everything else is probably not Python.
261 return False
