266 children_variables = []
267 if scope is not None:
268 assert isinstance(scope, ScopeRequest)
269 scope = scope.scope
270
271 if scope in ("locals", None):
272 dct = self.frame.f_locals
273 elif scope == "globals":
274 dct = self.frame.f_globals

276 raise AssertionError("Unexpected scope: %s" % (scope,))

278 lst, group_entries = self._group_entries(

280 )
281 group_variables = []
282
283 for key, val, _ in group_entries:
284 # Make sure that the contents in the group are also sorted.
285 val.contents_debug_adapter_protocol.sort(key=lambda v: sorted_attributes_key(v[0]))
286 variable = _ObjectVariable(self.py_db, key, val, self._register_variable, False, key, frame=self.frame)
287 group_variables.append(variable)
288
289 for key, val, _ in lst:
290 is_return_value = key == RETURN_VALUES_DICT
291 if is_return_value:
292 for return_key, return_value in val.items():
293 variable = _ObjectVariable(
294 self.py_db,
295 return_key,
296 return_value,
297 self._register_variable,
298 is_return_value,
299 "%s[%r]" % (key, return_key),
300 frame=self.frame,
301 )
302 children_variables.append(variable)
303 else:
304 variable = _ObjectVariable(self.py_db, key, val, self._register_variable, is_return_value, key, frame=self.frame)
305 children_variables.append(variable)
306
307 # Frame variables always sorted.
308 children_variables.sort(key=sorted_variables_key)
309 if group_variables:
310 # Groups have priority over other variables.
311 children_variables = group_variables + children_variables

374 def track(self, thread_id, frames_list, frame_custom_thread_id=None):
375 """
376 :param thread_id:
377 The thread id to be used for this frame.
378
379 :param FramesList frames_list:
380 A list of frames to be tracked (the first is the topmost frame which is suspended at the given thread).

387 coroutine_or_main_thread_id = frame_custom_thread_id or thread_id
388
389 if coroutine_or_main_thread_id in self._suspended_frames_manager._thread_id_to_tracker:
390 sys.stderr.write("pydevd: Something is wrong. Tracker being added twice to the same thread id.\n")
391

393 self._main_thread_id = thread_id
394
395 frame_ids_from_thread = self._thread_id_to_frame_ids.setdefault(coroutine_or_main_thread_id, [])

397 self._thread_id_to_frames_list[coroutine_or_main_thread_id] = frames_list
398 for frame in frames_list:
399 frame_id = id(frame)
400 self._frame_id_to_frame[frame_id] = frame
401 _FrameVariable(self.py_db, frame, self._register_variable) # Instancing is enough to register.
402 self._suspended_frames_manager._variable_reference_to_frames_tracker[frame_id] = self
403 frame_ids_from_thread.append(frame_id)
404
405 self._frame_id_to_main_thread_id[frame_id] = thread_id

407 frame = None
