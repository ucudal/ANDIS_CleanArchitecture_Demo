50 def assert_stdout(exe, *args, **kwargs):
51 kwargs.setdefault("env", PYTHON_EXE_ENV)
52 exe = os.path.join(SCRIPTS_DIR, exe)
53 cmd = [PYTHON_EXE, exe]
54 for arg in args:
55 cmd.append(arg)
56 try:
57 out = sh(cmd, **kwargs).strip()
58 except RuntimeError as err:
59 if 'AccessDenied' in str(err):
60 return str(err)
61 else:
62 raise
63 assert out, out
64 return out
