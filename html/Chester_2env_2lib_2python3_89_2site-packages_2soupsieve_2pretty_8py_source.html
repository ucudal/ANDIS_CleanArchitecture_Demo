71RE_CLASS = re.compile(r'(?i)[a-z_][_a-z\d\.]+\‍(')
72RE_PARAM = re.compile(r'(?i)[_a-z][_a-z\d]+=')
73RE_EMPTY = re.compile(r'\‍(\‍)|\‍[\‍]|\{\}')
74RE_LSTRT = re.compile(r'\‍[')
75RE_DSTRT = re.compile(r'\{')
76RE_TSTRT = re.compile(r'\‍(')
77RE_LEND = re.compile(r'\‍]')
78RE_DEND = re.compile(r'\}')
79RE_TEND = re.compile(r'\‍)')
80RE_INT = re.compile(r'\d+')
81RE_KWORD = re.compile(r'(?i)[_a-z][_a-z\d]+')
82RE_DQSTR = re.compile(r'"(?:\\.|[^"\\])*"')
83RE_SQSTR = re.compile(r"'(?:\\.|[^'\\])*'")
84RE_SEP = re.compile(r'\s*(,)\s*')
85RE_DSEP = re.compile(r'\s*(:)\s*')

106def pretty(obj: Any) -> str: # pragma: no cover
107 """Make the object output string pretty."""
108
109 sel = str(obj)
110 index = 0
111 end = len(sel) - 1
112 indent = 0
113 output = []
114
115 while index <= end:
116 m = None
117 for k, v in TOKENS.items():
118 m = v.match(sel, index)
119
120 if m:
121 name = k
122 index = m.end(0)
123 if name in ('class', 'lstrt', 'dstrt', 'tstrt'):
124 indent += 4
125 output.append('{}\n{}'.format(m.group(0), " " * indent))
126 elif name in ('param', 'int', 'kword', 'sqstr', 'dqstr', 'empty'):
127 output.append(m.group(0))
128 elif name in ('lend', 'dend', 'tend'):
129 indent -= 4
130 output.append(m.group(0))
131 elif name in ('sep',):
132 output.append('{}\n{}'.format(m.group(1), " " * indent))
133 elif name in ('dsep',):
134 output.append('{} '.format(m.group(1)))
135 break
136
137 return ''.join(output)
