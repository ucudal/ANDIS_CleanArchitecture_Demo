34 def _print_bar(self) -> None:
35 """Print the progress bar to the console."""
36 progress = self.counter / self.total
37 arrow = "-" * int(round(progress * self.ncols) - 1) + ">"
38 spaces = " " * (self.ncols - len(arrow))
39 print(f"\r[{arrow + spaces}] {self.counter}/{self.total}", end="") # noqa: T201
