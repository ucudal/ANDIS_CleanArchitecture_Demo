298 # Token is a thread ID or name.
299 def input_thread(self, token):
300 targets = self.input_thread_list([token])
301 if len(targets) == 0:
302 raise CmdError("missing thread name or ID")
303 if len(targets) > 1:
304 msg = "more than one thread with that name:\n"
305 for tid in targets:
306 msg += "\t%d\n" % tid
307 msg = msg[: -len("\n")]
308 raise CmdError(msg)

311 # Token list is a list of thread IDs or names.
312 def input_thread_list(self, token_list):
313 targets = set()
314 system = self.debug.system
315 for token in token_list:
316 try:
317 tid = self.input_integer(token)
318 if not system.has_thread(tid):
319 raise CmdError("thread not found (%d)" % tid)
320 targets.add(tid)
321 except ValueError:
322 found = set()
323 for process in system.iter_processes():
324 found.update(system.find_threads_by_name(token))
325 if not found:
326 raise CmdError("thread not found (%s)" % token)
327 for thread in found:
328 targets.add(thread.get_tid())
329 targets = list(targets)
330 targets.sort()
331 return targets

393 # Token is an address.
394 # The address can be a integer, a label or a register.
395 def input_address(self, token, pid=None, tid=None):
396 address = None
397 if self.is_register(token):
398 if tid is None:
399 if self.lastEvent is None or pid != self.lastEvent.get_pid():
400 msg = "can't resolve register (%s) for unknown thread"
401 raise CmdError(msg % token)
402 tid = self.lastEvent.get_tid()
403 address = self.input_register(token, tid)
404 if address is None:
405 try:
406 address = self.input_hexadecimal_integer(token)
407 except ValueError:
408 if pid is None:
409 if self.lastEvent is None:
410 raise CmdError("no current process set")
411 process = self.lastEvent.get_process()
412 elif self.lastEvent is not None and pid == self.lastEvent.get_pid():
413 process = self.lastEvent.get_process()
414 else:
415 try:
416 process = self.debug.system.get_process(pid)
417 except KeyError:
418 raise CmdError("process not found (%d)" % pid)
419 try:
420 address = process.resolve_label(token)
421 except Exception:
422 raise CmdError("unknown address (%s)" % token)
423 return address

425 # Token is an address range, or a single address.
426 # The addresses can be integers, labels or registers.
427 def input_address_range(self, token_list, pid=None, tid=None):
428 if len(token_list) == 2:
429 token_1, token_2 = token_list
430 address = self.input_address(token_1, pid, tid)
431 try:
432 size = self.input_integer(token_2)
433 except ValueError:
434 raise CmdError("bad address range: %s %s" % (token_1, token_2))
435 elif len(token_list) == 1:
436 token = token_list[0]
437 if "-" in token:
438 try:
439 token_1, token_2 = token.split("-")
440 except Exception:
441 raise CmdError("bad address range: %s" % token)
442 address = self.input_address(token_1, pid, tid)
443 size = self.input_address(token_2, pid, tid) - address
444 else:
445 address = self.input_address(token, pid, tid)
446 size = None
447 return address, size

451 def is_register(self, token):
452 if win32.arch == "i386":

461 return True

464 if name == token:

468 # The token is a register name.
469 # Returns None if no register name is matched.
470 def input_register(self, token, tid=None):
471 if tid is None:
472 if self.lastEvent is None:
473 raise CmdError("no current process set")
474 thread = self.lastEvent.get_thread()
475 else:
476 thread = self.debug.system.get_thread(tid)
477 ctx = thread.get_context()
478
479 token = token.lower()
480 title = token.title()
481
482 if title in ctx:
483 return ctx.get(title) # eax -> Eax
484
485 if ctx.arch == "i386":
486 if token in self.segment_names:
487 return ctx.get("Seg%s" % title) # cs -> SegCs
488
489 if token in self.register_alias_32_to_16:
490 return ctx.get(self.register_alias_32_to_16[token]) & 0xFFFF
491
492 if token in self.register_alias_32_to_8_low:
493 return ctx.get(self.register_alias_32_to_8_low[token]) & 0xFF
494
495 if token in self.register_alias_32_to_8_high:
496 return (ctx.get(self.register_alias_32_to_8_high[token]) & 0xFF00) >> 8
497
498 elif ctx.arch == "amd64":
499 if token in self.segment_names:
500 return ctx.get("Seg%s" % title) # cs -> SegCs
501
502 if token in self.register_alias_64_to_32:
503 return ctx.get(self.register_alias_64_to_32[token]) & 0xFFFFFFFF
504
505 if token in self.register_alias_64_to_16:
506 return ctx.get(self.register_alias_64_to_16[token]) & 0xFFFF
507
508 if token in self.register_alias_64_to_8_low:
509 return ctx.get(self.register_alias_64_to_8_low[token]) & 0xFF
510
511 if token in self.register_alias_64_to_8_high:
512 return (ctx.get(self.register_alias_64_to_8_high[token]) & 0xFF00) >> 8
513
514 return None

819 # Change a register value.
820 def change_register(self, register, value, tid=None):
821 # Get the thread.
822 if tid is None:
823 if self.lastEvent is None:
824 raise CmdError("no current process set")
825 thread = self.lastEvent.get_thread()
826 else:
827 try:
828 thread = self.debug.system.get_thread(tid)
829 except KeyError:
830 raise CmdError("thread not found (%d)" % tid)

832 # Convert the value to integer type.
833 try:
834 value = self.input_integer(value)
835 except ValueError:
836 pid = thread.get_pid()
837 value = self.input_address(value, pid, tid)
838
839 # Suspend the thread.
840 # The finally clause ensures the thread is resumed before returning.
841 thread.suspend()
842 try:
843 # Get the current context.
844 ctx = thread.get_context()
845
846 # Register name matching is case insensitive.
847 register = register.lower()
848
849 # Integer 32 bits registers.
850 if register in self.register_names:
851 register = register.title() # eax -> Eax
852
853 # Segment (16 bit) registers.
854 if register in self.segment_names:
855 register = "Seg%s" % register.title() # cs -> SegCs
856 value = value & 0x0000FFFF
857
858 # Integer 16 bits registers.
859 if register in self.register_alias_16:
860 register = self.register_alias_16[register]
861 previous = ctx.get(register) & 0xFFFF0000
862 value = (value & 0x0000FFFF) | previous
863
864 # Integer 8 bits registers (low part).
865 if register in self.register_alias_8_low:
866 register = self.register_alias_8_low[register]
867 previous = ctx.get(register) % 0xFFFFFF00
868 value = (value & 0x000000FF) | previous
869
870 # Integer 8 bits registers (high part).
871 if register in self.register_alias_8_high:
872 register = self.register_alias_8_high[register]
873 previous = ctx.get(register) % 0xFFFF00FF
874 value = ((value & 0x000000FF) << 8) | previous
875
876 # Set the new context.
877 ctx.__setitem__(register, value)
878 thread.set_context(ctx)
879
880 # Resume the thread.
881 finally:
882 thread.resume()
