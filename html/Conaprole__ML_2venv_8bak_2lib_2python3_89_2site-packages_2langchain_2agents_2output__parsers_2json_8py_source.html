15class JSONAgentOutputParser(AgentOutputParser):
16 """Parses tool invocations and final answers in JSON format.
17
18 Expects output to be in one of two formats.
19
20 If the output signals that an action should be taken,
21 should be in the below format. This will result in an AgentAction
22 being returned.
23
24 ```
25 {
26 "action": "search",
27 "action_input": "2+2"
28 }
29 ```
30
31 If the output signals that a final answer should be given,
32 should be in the below format. This will result in an AgentFinish
33 being returned.
34
35 ```
36 {
37 "action": "Final Answer",
38 "action_input": "4"
39 }
40 ```
41 """
42
43 def parse(self, text: str) -> Union[AgentAction, AgentFinish]:
44 try:
45 response = parse_json_markdown(text)
46 if isinstance(response, list):
47 # gpt turbo frequently ignores the directive to emit a single action
48 logger.warning("Got multiple action responses: %s", response)
49 response = response[0]
50 if response["action"] == "Final Answer":
51 return AgentFinish({"output": response["action_input"]}, text)
52 else:
53 return AgentAction(
54 response["action"], response.get("action_input", {}), text
55 )
56 except Exception as e:
57 raise OutputParserException(f"Could not parse LLM output: {text}") from e
