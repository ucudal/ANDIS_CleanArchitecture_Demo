88 @staticmethod
89 def _accumulate_metrics(metrics, accumulate):
90 for metric in metrics.values():
91 samples = defaultdict(float)
92 sample_timestamps = defaultdict(float)
93 buckets = defaultdict(lambda: defaultdict(float))
94 samples_setdefault = samples.setdefault
95 for s in metric.samples:
96 name, labels, value, timestamp, exemplar, native_histogram_value = s
97 if metric.type == 'gauge':
98 without_pid_key = (name, tuple(l for l in labels if l[0] != 'pid'))
99 if metric._multiprocess_mode in ('min', 'livemin'):
100 current = samples_setdefault(without_pid_key, value)
101 if value < current:
102 samples[without_pid_key] = value
103 elif metric._multiprocess_mode in ('max', 'livemax'):
104 current = samples_setdefault(without_pid_key, value)
105 if value > current:
106 samples[without_pid_key] = value
107 elif metric._multiprocess_mode in ('sum', 'livesum'):
108 samples[without_pid_key] += value
109 elif metric._multiprocess_mode in ('mostrecent', 'livemostrecent'):
110 current_timestamp = sample_timestamps[without_pid_key]
111 timestamp = float(timestamp or 0)
112 if current_timestamp < timestamp:
113 samples[without_pid_key] = value
114 sample_timestamps[without_pid_key] = timestamp
115 else: # all/liveall
116 samples[(name, labels)] = value
117
118 elif metric.type == 'histogram':
119 # A for loop with early exit is faster than a genexpr
120 # or a listcomp that ends up building unnecessary things
121 for l in labels:
122 if l[0] == 'le':
123 bucket_value = float(l[1])
124 # _bucket
125 without_le = tuple(l for l in labels if l[0] != 'le')
126 buckets[without_le][bucket_value] += value
127 break
128 else: # did not find the `le` key
129 # _sum/_count
130 samples[(name, labels)] += value
131 else:
132 # Counter and Summary.
133 samples[(name, labels)] += value
134
135 # Accumulate bucket values.
136 if metric.type == 'histogram':
137 for labels, values in buckets.items():
138 acc = 0.0
139 for bucket, value in sorted(values.items()):
140 sample_key = (
141 metric.name + '_bucket',
142 labels + (('le', floatToGoString(bucket)),),
143 )
144 if accumulate:
145 acc += value
146 samples[sample_key] = acc
147 else:
148 samples[sample_key] = value
149 if accumulate:
150 samples[(metric.name + '_count', labels)] = acc
151
152 # Convert to correct sample format.
153 metric.samples = [Sample(name_, dict(labels), value) for (name_, labels), value in samples.items()]
154 return metrics.values()
