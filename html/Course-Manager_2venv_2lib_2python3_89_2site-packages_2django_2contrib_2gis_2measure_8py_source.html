172 def default_units(self, kwargs):
173 """
174 Return the unit value and the default units specified
175 from the given keyword arguments dictionary.
176 """
177 val = 0.0
178 default_unit = self.STANDARD_UNIT
179 for unit, value in kwargs.items():
180 if not isinstance(value, float):
181 value = float(value)
182 if unit in self.UNITS:
183 val += self.UNITS[unit] * value
184 default_unit = unit
185 elif unit in self.ALIAS:
186 u = self.ALIAS[unit]
187 val += self.UNITS[u] * value
188 default_unit = u
189 else:
190 lower = unit.lower()
191 if lower in self.UNITS:
192 val += self.UNITS[lower] * value
193 default_unit = lower
194 elif lower in self.LALIAS:
195 u = self.LALIAS[lower]
196 val += self.UNITS[u] * value
197 default_unit = u
198 else:
199 raise AttributeError('Unknown unit type: %s' % unit)
200 return val, default_unit

202 @classmethod
203 def unit_attname(cls, unit_str):
204 """
205 Retrieve the unit attribute name for the given unit string.
206 For example, if the given unit string is 'metre', return 'm'.
207 Raise an exception if an attribute cannot be found.
208 """
209 lower = unit_str.lower()
210 if unit_str in cls.UNITS:
211 return unit_str
212 elif lower in cls.UNITS:
213 return lower
214 elif lower in cls.LALIAS:
215 return cls.LALIAS[lower]
216 else:
217 raise Exception('Could not find a unit keyword associated with "%s"' % unit_str)
