12class SafeRepr(object):
13 # Can be used to override the encoding from locale.getpreferredencoding()
14 locale_preferred_encoding = None

19 # String types are truncated to maxstring_outer when at the outer-
20 # most level, and truncated to maxstring_inner characters inside
21 # collections.
22 maxstring_outer = 2**16
23 maxstring_inner = 128
24 string_types = (str, bytes)
25 bytes = bytes
26 set_info = (set, "{", "}", False)
27 frozenset_info = (frozenset, "frozenset({", "})", False)
28 int_types = (int,)
29 long_iter_types = (list, tuple, bytearray, range, dict, set, frozenset)
30
31 # Collection types are recursively iterated for each limit in
32 # maxcollection.
33 maxcollection = (60, 20)
34
35 # Specifies type, prefix string, suffix string, and whether to include a
36 # comma if there is only one element. (Using a sequence rather than a
37 # mapping because we use isinstance() to determine the matching type.)
38 collection_types = [
39 (tuple, "(", ")", True),
40 (list, "[", "]", False),
41 frozenset_info,
42 set_info,
43 ]
44 try:
45 from collections import deque
46
47 collection_types.append((deque, "deque([", "])", False))
48 except Exception:
49 pass
50
51 # type, prefix string, suffix string, item prefix string,
52 # item key/value separator, item suffix string
53 dict_types = [(dict, "{", "}", "", ": ", "")]
54 try:
55 from collections import OrderedDict
56
57 dict_types.append((OrderedDict, "OrderedDict([", "])", "(", ", ", ")"))
58 except Exception:
59 pass
60
61 # All other types are treated identically to strings, but using
62 # different limits.
63 maxother_outer = 2**16
64 maxother_inner = 128

66 convert_to_hex = False
67 raw_value = False

69 def __call__(self, obj):
70 """
71 :param object obj:
72 The object for which we want a representation.
73
74 :return str:
75 Returns bytes encoded as utf-8 on py2 and str on py3.
76 """
77 try:
78 return "".join(self._repr(obj, 0))
79 except Exception:
80 try:
81 return "An exception was raised: %r" % sys.exc_info()[1]

85 def _repr(self, obj, level):
86 """Returns an iterable of the parts in the final repr string."""
87
88 try:
89 obj_repr = type(obj).__repr__
90 except Exception:
91 obj_repr = None
92
93 def has_obj_repr(t):
94 r = t.__repr__
95 try:
96 return obj_repr == r
97 except Exception:
98 return obj_repr is r
99
100 for t, prefix, suffix, comma in self.collection_types:
101 if isinstance(obj, t) and has_obj_repr(t):
102 return self._repr_iter(obj, level, prefix, suffix, comma)
103
104 for t, prefix, suffix, item_prefix, item_sep, item_suffix in self.dict_types: # noqa
105 if isinstance(obj, t) and has_obj_repr(t):
106 return self._repr_dict(obj, level, prefix, suffix, item_prefix, item_sep, item_suffix)
107
108 for t in self.string_types:
109 if isinstance(obj, t) and has_obj_repr(t):
110 return self._repr_str(obj, level)
111
112 if self._is_long_iter(obj):
113 return self._repr_long_iter(obj)
114
115 return self._repr_other(obj, level)

117 # Determines whether an iterable exceeds the limits set in
118 # maxlimits, and is therefore unsafe to repr().
119 def _is_long_iter(self, obj, level=0):
120 try:
121 # Strings have their own limits (and do not nest). Because
122 # they don't have __iter__ in 2.x, this check goes before
123 # the next one.
124 if isinstance(obj, self.string_types):
125 return len(obj) > self.maxstring_inner
126
127 # If it's not an iterable (and not a string), it's fine.
128 if not hasattr(obj, "__iter__"):
129 return False
130
131 # If it's not an instance of these collection types then it
132 # is fine. Note: this is a fix for
133 # https://github.com/Microsoft/ptvsd/issues/406
134 if not isinstance(obj, self.long_iter_types):
135 return False
136
137 # Iterable is its own iterator - this is a one-off iterable
138 # like generator or enumerate(). We can't really count that,
139 # but repr() for these should not include any elements anyway,
140 # so we can treat it the same as non-iterables.
141 if obj is iter(obj):
142 return False
143
144 # range reprs fine regardless of length.
145 if isinstance(obj, range):
146 return False
147
148 # numpy and scipy collections (ndarray etc) have
149 # self-truncating repr, so they're always safe.
150 try:
151 module = type(obj).__module__.partition(".")[0]
152 if module in ("numpy", "scipy"):
153 return False
154 except Exception:
155 pass
156
157 # Iterables that nest too deep are considered long.
158 if level >= len(self.maxcollection):
159 return True
160
161 # It is too long if the length exceeds the limit, or any
162 # of its elements are long iterables.
163 if hasattr(obj, "__len__"):
164 try:
165 size = len(obj)
166 except Exception:
167 size = None
168 if size is not None and size > self.maxcollection[level]:
169 return True
170 return any((self._is_long_iter(item, level + 1) for item in obj)) # noqa
171 return any(i > self.maxcollection[level] or self._is_long_iter(item, level + 1) for i, item in enumerate(obj)) # noqa
172
173 except Exception:
174 # If anything breaks, assume the worst case.
175 return True

177 def _repr_iter(self, obj, level, prefix, suffix, comma_after_single_element=False):
178 yield prefix
179
180 if level >= len(self.maxcollection):
181 yield "..."
182 else:
183 count = self.maxcollection[level]
184 yield_comma = False
185 for item in obj:
186 if yield_comma:
187 yield ", "
188 yield_comma = True
189
190 count -= 1
191 if count <= 0:
192 yield "..."
193 break
194
195 for p in self._repr(item, 100 if item is obj else level + 1):
196 yield p
197 else:
198 if comma_after_single_element:
199 if count == self.maxcollection[level] - 1:
200 yield ","
201 yield suffix

203 def _repr_long_iter(self, obj):
204 try:
205 length = hex(len(obj)) if self.convert_to_hex else len(obj)
206 obj_repr = "<%s, len() = %s>" % (type(obj).__name__, length)
207 except Exception:
208 try:
209 obj_repr = "<" + type(obj).__name__ + ">"
210 except Exception:
211 obj_repr = "<no repr available for object>"
212 yield obj_repr

214 def _repr_dict(self, obj, level, prefix, suffix, item_prefix, item_sep, item_suffix):
215 if not obj:
216 yield prefix + suffix
217 return
218 if level >= len(self.maxcollection):
219 yield prefix + "..." + suffix
220 return
221
222 yield prefix
223
224 count = self.maxcollection[level]
225 yield_comma = False
226
227 if IS_PY36_OR_GREATER:
228 # On Python 3.6 (onwards) dictionaries now keep
229 # insertion order.
230 sorted_keys = list(obj)
231 else:
232 try:
233 sorted_keys = sorted(obj)
234 except Exception:
235 sorted_keys = list(obj)
236
237 for key in sorted_keys:
238 if yield_comma:
239 yield ", "
240 yield_comma = True
241
242 count -= 1
243 if count <= 0:
244 yield "..."
245 break
246
247 yield item_prefix
248 for p in self._repr(key, level + 1):
249 yield p
250
251 yield item_sep
252
253 try:
254 item = obj[key]
255 except Exception:
256 yield "<?>"
257 else:
258 for p in self._repr(item, 100 if item is obj else level + 1):
259 yield p
260 yield item_suffix
261
262 yield suffix

264 def _repr_str(self, obj, level):
265 try:
266 if self.raw_value:
267 # For raw value retrieval, ignore all limits.
268 if isinstance(obj, bytes):
269 yield obj.decode("latin-1")
270 else:
271 yield obj
272 return
273
274 limit_inner = self.maxother_inner
275 limit_outer = self.maxother_outer
276 limit = limit_inner if level > 0 else limit_outer
277 if len(obj) <= limit:
278 # Note that we check the limit before doing the repr (so, the final string
279 # may actually be considerably bigger on some cases, as besides
280 # the additional u, b, ' chars, some chars may be escaped in repr, so
281 # even a single char such as \U0010ffff may end up adding more
282 # chars than expected).
283 yield self._convert_to_unicode_or_bytes_repr(repr(obj))
284 return
285
286 # Slightly imprecise calculations - we may end up with a string that is
287 # up to 6 characters longer than limit. If you need precise formatting,
288 # you are using the wrong class.
289 left_count, right_count = max(1, int(2 * limit / 3)), max(1, int(limit / 3)) # noqa
290
291 # Important: only do repr after slicing to avoid duplicating a byte array that could be
292 # huge.
293
294 # Note: we don't deal with high surrogates here because we're not dealing with the
295 # repr() of a random object.
296 # i.e.: A high surrogate unicode char may be splitted on Py2, but as we do a `repr`
297 # afterwards, that's ok.
298
299 # Also, we just show the unicode/string/bytes repr() directly to make clear what the
300 # input type was (so, on py2 a unicode would start with u' and on py3 a bytes would
301 # start with b').
302
303 part1 = obj[:left_count]
304 part1 = repr(part1)
305 part1 = part1[: part1.rindex("'")] # Remove the last '
306
307 part2 = obj[-right_count:]
308 part2 = repr(part2)
309 part2 = part2[part2.index("'") + 1 :] # Remove the first ' (and possibly u or b).
310
311 yield part1
312 yield "..."
313 yield part2
314 except:
315 # This shouldn't really happen, but let's play it safe.
316 pydev_log.exception("Error getting string representation to show.")
317 for part in self._repr_obj(obj, level, self.maxother_inner, self.maxother_outer):
318 yield part

323 def _repr_obj(self, obj, level, limit_inner, limit_outer):
324 try:
325 if self.raw_value:
326 # For raw value retrieval, ignore all limits.
327 if isinstance(obj, bytes):
328 yield obj.decode("latin-1")
329 return
330
331 try:
332 mv = memoryview(obj)
333 except Exception:
334 yield self._convert_to_unicode_or_bytes_repr(repr(obj))
335 return
336 else:
337 # Map bytes to Unicode codepoints with same values.
338 yield mv.tobytes().decode("latin-1")
339 return
340 elif self.convert_to_hex and isinstance(obj, self.int_types):
341 obj_repr = hex(obj)
342 else:
343 obj_repr = repr(obj)
344 except Exception:
345 try:
346 obj_repr = object.__repr__(obj)
347 except Exception:
348 try:
349 obj_repr = "<no repr available for " + type(obj).__name__ + ">" # noqa
350 except Exception:
351 obj_repr = "<no repr available for object>"
352
353 limit = limit_inner if level > 0 else limit_outer
354
355 if limit >= len(obj_repr):
356 yield self._convert_to_unicode_or_bytes_repr(obj_repr)
357 return
358
359 # Slightly imprecise calculations - we may end up with a string that is
360 # up to 3 characters longer than limit. If you need precise formatting,
361 # you are using the wrong class.
362 left_count, right_count = max(1, int(2 * limit / 3)), max(1, int(limit / 3)) # noqa
363
364 yield obj_repr[:left_count]
365 yield "..."
366 yield obj_repr[-right_count:]

371 def _bytes_as_unicode_if_possible(self, obj_repr):
372 # We try to decode with 3 possible encoding (sys.stdout.encoding,
373 # locale.getpreferredencoding() and 'utf-8). If no encoding can decode
374 # the input, we return the original bytes.
375 try_encodings = []
376 encoding = self.sys_stdout_encoding or getattr(sys.stdout, "encoding", "")
377 if encoding:
378 try_encodings.append(encoding.lower())
379
380 preferred_encoding = self.locale_preferred_encoding or locale.getpreferredencoding()
381 if preferred_encoding:
382 preferred_encoding = preferred_encoding.lower()
383 if preferred_encoding not in try_encodings:
384 try_encodings.append(preferred_encoding)
385
386 if "utf-8" not in try_encodings:
387 try_encodings.append("utf-8")
388
389 for encoding in try_encodings:
390 try:
391 return obj_repr.decode(encoding)
392 except UnicodeDecodeError:
393 pass
394
395 return obj_repr # Return the original version (in bytes)
