22class GenerateJsonSchemaHandler(GetJsonSchemaHandler):

40 def resolve_ref_schema(self, maybe_ref_json_schema: JsonSchemaValue) -> JsonSchemaValue:
41 """Resolves `$ref` in the json schema.
42
43 This returns the input json schema if there is no `$ref` in json schema.
44
45 Args:
46 maybe_ref_json_schema: The input json schema that may contains `$ref`.
47
48 Returns:
49 Resolved json schema.
50
51 Raises:
52 LookupError: If it can't find the definition for `$ref`.
53 """
54 if '$ref' not in maybe_ref_json_schema:
55 return maybe_ref_json_schema
56 ref = maybe_ref_json_schema['$ref']
57 json_schema = self.generate_json_schema.get_schema_from_definitions(ref)
58 if json_schema is None:
59 raise LookupError(
60 f'Could not find a ref for {ref}.'
61 ' Maybe you tried to call resolve_ref_schema from within a recursive model?'
62 )
63 return json_schema
