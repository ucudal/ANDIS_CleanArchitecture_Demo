74 class WinDllHook(object):
75 def __getattr__(self, name):
76 if name.startswith("_"):
77 return object.__getattr__(self, name)
78 return WinFuncHook(name)

80 class WinFuncHook(object):
81 def __init__(self, name):
82 self.__name = name
83
84 def __getattr__(self, name):
85 if name.startswith("_"):
86 return object.__getattr__(self, name)
87 return WinCallHook(self.__name, name)

89 class WinCallHook(object):
90 def __init__(self, dllname, funcname):
91 self.__dllname = dllname
92 self.__funcname = funcname
93 self.__func = getattr(getattr(ctypes.windll, dllname), funcname)
94
95 def __copy_attribute(self, attribute):
96 try:
97 value = getattr(self, attribute)
98 setattr(self.__func, attribute, value)
99 except AttributeError:
100 try:
101 delattr(self.__func, attribute)
102 except AttributeError:
103 pass
104
105 def __call__(self, *argv):
106 self.__copy_attribute("argtypes")
107 self.__copy_attribute("restype")
108 self.__copy_attribute("errcheck")
109 print("-" * 10)
110 print("%s ! %s %r" % (self.__dllname, self.__funcname, argv))
111 retval = self.__func(*argv)
112 print("== %r" % (retval,))
113 return retval

163class GuessStringType(object):
164 """
165 Decorator that guesses the correct version (A or W) to call
166 based on the types of the strings passed as parameters.
167
168 Calls the B{ANSI} version if the only string types are ANSI.
169
170 Calls the B{Unicode} version if Unicode or mixed string types are passed.
171
172 The default if no string arguments are passed depends on the value of the
173 L{t_default} class variable.
174
175 @type fn_ansi: function
176 @ivar fn_ansi: ANSI version of the API function to call.
177 @type fn_unicode: function
178 @ivar fn_unicode: Unicode (wide) version of the API function to call.
179
180 @type t_default: type
181 @cvar t_default: Default string type to use.
182 Possible values are:
183 - type('') for ANSI
184 - type(u'') for Unicode
185 """
186
187 # ANSI and Unicode types
188 t_ansi = type("")
189 t_unicode = type("")
190
191 # Default is ANSI for Python 2.x
192 t_default = t_ansi
193
194 def __init__(self, fn_ansi, fn_unicode):
195 """
196 @type fn_ansi: function
197 @param fn_ansi: ANSI version of the API function to call.
198 @type fn_unicode: function
199 @param fn_unicode: Unicode (wide) version of the API function to call.
200 """
201 self.fn_ansi = fn_ansi
202 self.fn_unicode = fn_unicode
203
204 # Copy the wrapped function attributes.
205 try:
206 self.__name__ = self.fn_ansi.__name__[:-1] # remove the A or W
207 except AttributeError:
208 pass
209 try:
210 self.__module__ = self.fn_ansi.__module__
211 except AttributeError:
212 pass
213 try:
214 self.__doc__ = self.fn_ansi.__doc__
215 except AttributeError:
216 pass
217
218 def __call__(self, *argv, **argd):
219 # Shortcut to self.t_ansi
220 t_ansi = self.t_ansi
221
222 # Get the types of all arguments for the function
223 v_types = [type(item) for item in argv]
224 v_types.extend([type(value) for (key, value) in compat.iteritems(argd)])
225
226 # Get the appropriate function for the default type
227 if self.t_default == t_ansi:
228 fn = self.fn_ansi
229 else:
230 fn = self.fn_unicode
231
232 # If at least one argument is a Unicode string...
233 if self.t_unicode in v_types:
234 # If al least one argument is an ANSI string,
235 # convert all ANSI strings to Unicode
236 if t_ansi in v_types:
237 argv = list(argv)
238 for index in compat.xrange(len(argv)):
239 if v_types[index] == t_ansi:
240 argv[index] = compat.unicode(argv[index])
241 for key, value in argd.items():
242 if type(value) == t_ansi:
243 argd[key] = compat.unicode(value)
244
245 # Use the W version
246 fn = self.fn_unicode
247
248 # If at least one argument is an ANSI string,
249 # but there are no Unicode strings...
250 elif t_ansi in v_types:
251 # Use the A version
252 fn = self.fn_ansi
253
254 # Call the function and return the result
255 return fn(*argv, **argd)

306def MakeANSIVersion(fn):
307 """
308 Decorator that generates an ANSI version of a Unicode (wide) only API call.
309
310 @type fn: callable
311 @param fn: Unicode (wide) version of the API function to call.
312 """
313
314 @functools.wraps(fn)
315 def wrapper(*argv, **argd):
316 t_ansi = GuessStringType.t_ansi
317 t_unicode = GuessStringType.t_unicode
318 v_types = [type(item) for item in argv]
319 v_types.extend([type(value) for (key, value) in compat.iteritems(argd)])
320 if t_ansi in v_types:
321 argv = list(argv)
322 for index in compat.xrange(len(argv)):
323 if v_types[index] == t_ansi:
324 argv[index] = t_unicode(argv[index])
325 for key, value in argd.items():
326 if type(value) == t_ansi:
327 argd[key] = t_unicode(value)
328 return fn(*argv, **argd)
329
330 return wrapper
