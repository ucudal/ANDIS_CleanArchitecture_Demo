616 def get_raw_exception_record_list(self):
617 """
618 Traverses the exception record linked list and builds a Python list.
619
620 Nested exception records are received for nested exceptions. This
621 happens when an exception is raised in the debugee while trying to
622 handle a previous exception.
623
624 @rtype: list( L{win32.EXCEPTION_RECORD} )
625 @return:
626 List of raw exception record structures as used by the Win32 API.
627
628 There is always at least one exception record, so the list is
629 never empty. All other methods of this class read from the first
630 exception record only, that is, the most recent exception.
631 """
632 # The first EXCEPTION_RECORD is contained in EXCEPTION_DEBUG_INFO.
633 # The remaining EXCEPTION_RECORD structures are linked by pointers.
634 nested = list()
635 record = self.raw.u.Exception
636 while True:
637 record = record.ExceptionRecord
638 if not record:
639 break
640 nested.append(record)
641 return nested

643 def get_nested_exceptions(self):
644 """
645 Traverses the exception record linked list and builds a Python list.
646
647 Nested exception records are received for nested exceptions. This
648 happens when an exception is raised in the debugee while trying to
649 handle a previous exception.
650
651 @rtype: list( L{ExceptionEvent} )
652 @return:
653 List of ExceptionEvent objects representing each exception record
654 found in this event.
655
656 There is always at least one exception record, so the list is
657 never empty. All other methods of this class read from the first
658 exception record only, that is, the most recent exception.
659 """
660 # The list always begins with ourselves.
661 # Just put a reference to "self" as the first element,
662 # and start looping from the second exception record.
663 nested = [self]
664 raw = self.raw
665 dwDebugEventCode = raw.dwDebugEventCode
666 dwProcessId = raw.dwProcessId
667 dwThreadId = raw.dwThreadId
668 dwFirstChance = raw.u.Exception.dwFirstChance
669 record = raw.u.Exception.ExceptionRecord
670 while True:
671 record = record.ExceptionRecord
672 if not record:
673 break
674 raw = win32.DEBUG_EVENT()
675 raw.dwDebugEventCode = dwDebugEventCode
676 raw.dwProcessId = dwProcessId
677 raw.dwThreadId = dwThreadId
678 raw.u.Exception.ExceptionRecord = record
679 raw.u.Exception.dwFirstChance = dwFirstChance
680 event = EventFactory.get(self.debug, raw)
681 nested.append(event)
682 return nested

1197class EventHandler(object):
1198 """
1199 Base class for debug event handlers.
1200
1201 Your program should subclass it to implement it's own event handling.
1202
1203 The constructor can be overriden as long as you call the superclass
1204 constructor. The special method L{__call__} B{MUST NOT} be overriden.
1205
1206 The signature for event handlers is the following::
1207
1208 def event_handler(self, event):
1209
1210 Where B{event} is an L{Event} object.
1211
1212 Each event handler is named after the event they handle.
1213 This is the list of all valid event handler names:
1214
1215 - I{event}
1216
1217 Receives an L{Event} object or an object of any of it's subclasses,
1218 and handles any event for which no handler was defined.
1219
1220 - I{unknown_event}
1221
1222 Receives an L{Event} object or an object of any of it's subclasses,
1223 and handles any event unknown to the debugging engine. (This is not
1224 likely to happen unless the Win32 debugging API is changed in future
1225 versions of Windows).
1226
1227 - I{exception}
1228
1229 Receives an L{ExceptionEvent} object and handles any exception for
1230 which no handler was defined. See above for exception handlers.
1231
1232 - I{unknown_exception}
1233
1234 Receives an L{ExceptionEvent} object and handles any exception unknown
1235 to the debugging engine. This usually happens for C++ exceptions, which
1236 are not standardized and may change from one compiler to the next.
1237
1238 Currently we have partial support for C++ exceptions thrown by Microsoft
1239 compilers.
1240
1241 Also see: U{RaiseException()
1242 <http://msdn.microsoft.com/en-us/library/ms680552(VS.85).aspx>}
1243
1244 - I{create_thread}
1245
1246 Receives a L{CreateThreadEvent} object.
1247
1248 - I{create_process}
1249
1250 Receives a L{CreateProcessEvent} object.
1251
1252 - I{exit_thread}
1253
1254 Receives a L{ExitThreadEvent} object.
1255
1256 - I{exit_process}
1257
1258 Receives a L{ExitProcessEvent} object.
1259
1260 - I{load_dll}
1261
1262 Receives a L{LoadDLLEvent} object.
1263
1264 - I{unload_dll}
1265
1266 Receives an L{UnloadDLLEvent} object.
1267
1268 - I{output_string}
1269
1270 Receives an L{OutputDebugStringEvent} object.
1271
1272 - I{rip}
1273
1274 Receives a L{RIPEvent} object.
1275
1276 This is the list of all valid exception handler names
1277 (they all receive an L{ExceptionEvent} object):
1278
1279 - I{access_violation}
1280 - I{array_bounds_exceeded}
1281 - I{breakpoint}
1282 - I{control_c_exit}
1283 - I{datatype_misalignment}
1284 - I{debug_control_c}
1285 - I{float_denormal_operand}
1286 - I{float_divide_by_zero}
1287 - I{float_inexact_result}
1288 - I{float_invalid_operation}
1289 - I{float_overflow}
1290 - I{float_stack_check}
1291 - I{float_underflow}
1292 - I{guard_page}
1293 - I{illegal_instruction}
1294 - I{in_page_error}
1295 - I{integer_divide_by_zero}
1296 - I{integer_overflow}
1297 - I{invalid_disposition}
1298 - I{invalid_handle}
1299 - I{ms_vc_exception}
1300 - I{noncontinuable_exception}
1301 - I{possible_deadlock}
1302 - I{privileged_instruction}
1303 - I{single_step}
1304 - I{stack_overflow}
1305 - I{wow64_breakpoint}
1306
1307
1308
1309 @type apiHooks: dict( str S{->} list( tuple( str, int ) ) )
1310 @cvar apiHooks:
1311 Dictionary that maps module names to lists of
1312 tuples of ( procedure name, parameter count ).
1313
1314 All procedures listed here will be hooked for calls from the debugee.
1315 When this happens, the corresponding event handler can be notified both
1316 when the procedure is entered and when it's left by the debugee.
1317
1318 For example, let's hook the LoadLibraryEx() API call.
1319 This would be the declaration of apiHooks::
1320
1321 from winappdbg import EventHandler
1322 from winappdbg.win32 import *
1323
1324 # (...)
1325
1326 class MyEventHandler (EventHandler):
1327
1328 apiHook = {
1329
1330 "kernel32.dll" : (
1331
1332 # Procedure name Signature
1333 ( "LoadLibraryEx", (PVOID, HANDLE, DWORD) ),
1334
1335 # (more procedures can go here...)
1336 ),
1337
1338 # (more libraries can go here...)
1339 }
1340
1341 # (your method definitions go here...)
1342
1343 Note that all pointer types are treated like void pointers, so your
1344 callback won't get the string or structure pointed to by it, but the
1345 remote memory address instead. This is so to prevent the ctypes library
1346 from being "too helpful" and trying to dereference the pointer. To get
1347 the actual data being pointed to, use one of the L{Process.read}
1348 methods.
1349
1350 Now, to intercept calls to LoadLibraryEx define a method like this in
1351 your event handler class::
1352
1353 def pre_LoadLibraryEx(self, event, ra, lpFilename, hFile, dwFlags):
1354 szFilename = event.get_process().peek_string(lpFilename)
1355
1356 # (...)
1357
1358 Note that the first parameter is always the L{Event} object, and the
1359 second parameter is the return address. The third parameter and above
1360 are the values passed to the hooked function.
1361
1362 Finally, to intercept returns from calls to LoadLibraryEx define a
1363 method like this::
1364
1365 def post_LoadLibraryEx(self, event, retval):
1366 # (...)
1367
1368 The first parameter is the L{Event} object and the second is the
1369 return value from the hooked function.
1370 """
1371

1374 # Default (empty) API hooks dictionary.
1375 apiHooks = {}

1377 def __init__(self):
1378 """
1379 Class constructor. Don't forget to call it when subclassing!
1380
1381 Forgetting to call the superclass constructor is a common mistake when
1382 you're new to Python. :)
1383
1384 Example::
1385 class MyEventHandler (EventHandler):
1386
1387 # Override the constructor to use an extra argument.
1388 def __init__(self, myArgument):
1389
1390 # Do something with the argument, like keeping it
1391 # as an instance variable.
1392 self.myVariable = myArgument
1393
1394 # Call the superclass constructor.
1395 super(MyEventHandler, self).__init__()
1396
1397 # The rest of your code below...
1398 """
1399
1400 # TODO
1401 # All this does is set up the hooks.
1402 # This code should be moved to the EventDispatcher class.
1403 # Then the hooks can be set up at set_event_handler() instead, making
1404 # this class even simpler. The downside here is deciding where to store
1405 # the ApiHook objects.
1406
1407 # Convert the tuples into instances of the ApiHook class.
1408 # A new dictionary must be instanced, otherwise we could also be
1409 # affecting all other instances of the EventHandler.
1410 apiHooks = dict()
1411 for lib, hooks in compat.iteritems(self.apiHooks):
1412 hook_objs = []
1413 for proc, args in hooks:
1414 if type(args) in (int, long):
1415 h = ApiHook(self, lib, proc, paramCount=args)
1416 else:
1417 h = ApiHook(self, lib, proc, signature=args)
1418 hook_objs.append(h)
1419 apiHooks[lib] = hook_objs
1420 self.__apiHooks = apiHooks

1422 def __get_hooks_for_dll(self, event):
1423 """
1424 Get the requested API hooks for the current DLL.
1425
1426 Used by L{__hook_dll} and L{__unhook_dll}.
1427 """
1428 result = []
1429 if self.__apiHooks:
1430 path = event.get_module().get_filename()
1431 if path:
1432 lib_name = PathOperations.pathname_to_filename(path).lower()
1433 for hook_lib, hook_api_list in compat.iteritems(self.__apiHooks):
1434 if hook_lib == lib_name:
1435 result.extend(hook_api_list)
1436 return result

1438 def __hook_dll(self, event):
1439 """
1440 Hook the requested API calls (in self.apiHooks).
1441
1442 This method is called automatically whenever a DLL is loaded.
1443 """
1444 debug = event.debug
1445 pid = event.get_pid()
1446 for hook_api_stub in self.__get_hooks_for_dll(event):
1447 hook_api_stub.hook(debug, pid)

1449 def __unhook_dll(self, event):
1450 """
1451 Unhook the requested API calls (in self.apiHooks).
1452
1453 This method is called automatically whenever a DLL is unloaded.
1454 """
1455 debug = event.debug
1456 pid = event.get_pid()
1457 for hook_api_stub in self.__get_hooks_for_dll(event):
1458 hook_api_stub.unhook(debug, pid)

1460 def __call__(self, event):
1461 """
1462 Dispatch debug events.
1463
1464 @warn: B{Don't override this method!}
1465
1466 @type event: L{Event}
1467 @param event: Event object.
1468 """
1469 try:
1470 code = event.get_event_code()
1471 if code == win32.LOAD_DLL_DEBUG_EVENT:
1472 self.__hook_dll(event)
1473 elif code == win32.UNLOAD_DLL_DEBUG_EVENT:
1474 self.__unhook_dll(event)
1475 finally:
1476 method = EventDispatcher.get_handler_method(self, event)
1477 if method is not None:
1478 return method(event)
