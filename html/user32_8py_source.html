509class Rect(object):
510 """
511 Python wrapper over the L{RECT} class.
512
513 @type left: int
514 @ivar left: Horizontal coordinate for the top left corner.
515 @type top: int
516 @ivar top: Vertical coordinate for the top left corner.
517 @type right: int
518 @ivar right: Horizontal coordinate for the bottom right corner.
519 @type bottom: int
520 @ivar bottom: Vertical coordinate for the bottom right corner.
521
522 @type width: int
523 @ivar width: Width in pixels. Same as C{right - left}.
524 @type height: int
525 @ivar height: Height in pixels. Same as C{bottom - top}.
526 """
527
528 def __init__(self, left=0, top=0, right=0, bottom=0):
529 """
530 @see: L{RECT}
531 @type left: int
532 @param left: Horizontal coordinate for the top left corner.
533 @type top: int
534 @param top: Vertical coordinate for the top left corner.
535 @type right: int
536 @param right: Horizontal coordinate for the bottom right corner.
537 @type bottom: int
538 @param bottom: Vertical coordinate for the bottom right corner.
539 """
540 self.left = left
541 self.top = top
542 self.right = right
543 self.bottom = bottom
544
545 def __iter__(self):
546 return (self.left, self.top, self.right, self.bottom).__iter__()
547
548 def __len__(self):
549 return 2
550
551 def __getitem__(self, index):
552 return (self.left, self.top, self.right, self.bottom)[index]
553
554 def __setitem__(self, index, value):
555 if index == 0:
556 self.left = value
557 elif index == 1:
558 self.top = value
559 elif index == 2:
560 self.right = value
561 elif index == 3:
562 self.bottom = value
563 else:
564 raise IndexError("index out of range")
565
566 @property
567 def _as_parameter_(self):
568 """
569 Compatibility with ctypes.
570 Allows passing transparently a Point object to an API call.
571 """
572 return RECT(self.left, self.top, self.right, self.bottom)
573
574 def __get_width(self):
575 return self.right - self.left
576
577 def __get_height(self):
578 return self.bottom - self.top
579
580 def __set_width(self, value):
581 self.right = value - self.left
582
583 def __set_height(self, value):
584 self.bottom = value - self.top
585
586 width = property(__get_width, __set_width)
587 height = property(__get_height, __set_height)

589 def screen_to_client(self, hWnd):
590 """
591 Translates window screen coordinates to client coordinates.
592
593 @see: L{client_to_screen}, L{translate}
