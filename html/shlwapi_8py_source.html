263# BOOL PathRelativePathTo(
264# _Out_ LPTSTR pszPath,
265# _In_ LPCTSTR pszFrom,

268# _In_ DWORD dwAttrTo

270def PathRelativePathToA(pszFrom=None, dwAttrFrom=FILE_ATTRIBUTE_DIRECTORY, pszTo=None, dwAttrTo=FILE_ATTRIBUTE_DIRECTORY):
271 _PathRelativePathToA = windll.shlwapi.PathRelativePathToA
272 _PathRelativePathToA.argtypes = [LPSTR, LPSTR, DWORD, LPSTR, DWORD]
273 _PathRelativePathToA.restype = bool
274 _PathRelativePathToA.errcheck = RaiseIfZero
275
276 # Make the paths absolute or the function fails.
277 if pszFrom:
278 pszFrom = GetFullPathNameA(pszFrom)[0]
279 else:
280 pszFrom = GetCurrentDirectoryA()
281 if pszTo:
282 pszTo = GetFullPathNameA(pszTo)[0]
283 else:
284 pszTo = GetCurrentDirectoryA()
285
286 # Argh, this function doesn't receive an output buffer size!
287 # We'll try to guess the maximum possible buffer size.
288 dwPath = max((len(pszFrom) + len(pszTo)) * 2 + 1, MAX_PATH + 1)
289 pszPath = ctypes.create_string_buffer("", dwPath)
290
291 # Also, it doesn't set the last error value.
292 # Whoever coded it must have been drunk or tripping on acid. Or both.
293 # The only failure conditions I've seen were invalid paths, paths not
294 # on the same drive, or the path is not absolute.
295 SetLastError(ERROR_INVALID_PARAMETER)
296
297 _PathRelativePathToA(pszPath, pszFrom, dwAttrFrom, pszTo, dwAttrTo)
298 return pszPath.value

419 _PathFindOnPathA.restype = bool
420
421 pszFile = ctypes.create_string_buffer(pszFile, MAX_PATH)

432 if _PathFindOnPathA(pszFile, ppszOtherDirs):
