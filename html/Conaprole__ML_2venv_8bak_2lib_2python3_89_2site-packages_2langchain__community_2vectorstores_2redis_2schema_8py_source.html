167 """Schema for Redis index."""
168
169 # always have a content field for text
170 text: List[TextFieldSchema] = [TextFieldSchema(name="content")]
171 tag: Optional[List[TagFieldSchema]] = None
172 numeric: Optional[List[NumericFieldSchema]] = None

180 def add_content_field(self) -> None:
181 if self.text is None:
182 self.text = []
183 for field in self.text:
184 if field.name == self.content_key:
185 return
186 self.text.append(TextFieldSchema(name=self.content_key))

188 def add_vector_field(self, vector_field: Dict[str, Any]) -> None:
189 # catch case where user inputted no vector field spec
190 # in the index schema
191 if self.vector is None:

194 # ignore types as pydantic is handling type validation and conversion
195 if vector_field["algorithm"] == "FLAT":

200 raise ValueError(

205 def as_dict(self) -> Dict[str, List[Any]]:
206 schemas: Dict[str, List[Any]] = {"text": [], "tag": [], "numeric": []}
207 # iter over all class attributes
208 for attr, attr_value in self.__dict__.items():
209 # only non-empty lists
210 if isinstance(attr_value, list) and len(attr_value) > 0:
211 field_values: List[Dict[str, Any]] = []
212 # iterate over all fields in each category (tag, text, etc)
213 for val in attr_value:
214 value: Dict[str, Any] = {}
215 # iterate over values within each field to extract
216 # settings for that field (i.e. name, weight, etc)
217 for field, field_value in val.__dict__.items():
218 # make enums into strings
219 if isinstance(field_value, Enum):
220 value[field] = field_value.value
221 # don't write null values
222 elif field_value is not None:
223 value[field] = field_value
224 field_values.append(value)
225
226 schemas[attr] = field_values

228 schema: Dict[str, List[Any]] = {}
229 # only write non-empty lists from defaults

287def read_schema(
288 index_schema: Optional[Union[Dict[str, List[Any]], str, os.PathLike]],
289) -> Dict[str, Any]:
290 """Read in the index schema from a dict or yaml file.
291
292 Check if it is a dict and return RedisModel otherwise, check if it's a path and
