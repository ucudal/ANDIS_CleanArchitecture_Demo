39def xor_string(hash1: bytes, hash2: bytes, hash_size: int) -> bytes:
40 """Encrypt/Decrypt function used for password encryption in
41 authentication, using a simple XOR.
42
43 Args:
44 hash1 (str): The first hash.
45 hash2 (str): The second hash.
46
47 Returns:
48 str: A string with the xor applied.
49 """
50 xored = [h1 ^ h2 for (h1, h2) in zip(hash1, hash2)]
51 return struct.pack(f"{hash_size}B", *xored)

97 def auth_data(self, data: bytes) -> str:
98 """Hashing for MySQL 4.1 authentication.
99
100 Args:
101 data (bytes): The authentication data.
102
103 Returns:
104 str: The authentication response.
105 """
106 if self._password:
107 password = (
108 self._password.encode("utf-8")
109 if isinstance(self._password, str)
110 else self._password
111 )
112 hash1 = hashlib.sha1(password).digest()
113 hash2 = hashlib.sha1(hash1).digest()
114 xored = xor_string(hash1, hashlib.sha1(data + hash2).digest(), 20)
115 return f"\0{self._username}\0*{hexlify(xored)}\0"
116 return f"\0{self._username}\0"

166 def auth_data(self, data: bytes) -> str:
167 """Hashing for SHA256_MEMORY authentication.
168
169 The scramble is of the form:
170 SHA256(SHA256(SHA256(PASSWORD)),NONCE) XOR SHA256(PASSWORD)
171
172 Args:
173 data (bytes): The authentication data.
174
175 Returns:
176 str: The authentication response.
177 """
178 password = (
179 self._password.encode("utf-8")
180 if isinstance(self._password, str)
181 else self._password
182 )
183 hash1 = hashlib.sha256(password).digest()
184 hash2 = hashlib.sha256(hashlib.sha256(hash1).digest() + data).digest()
185 xored = xor_string(hash2, hash1, 32)
186 return f"\0{self._username}\0{hexlify(xored)}"
