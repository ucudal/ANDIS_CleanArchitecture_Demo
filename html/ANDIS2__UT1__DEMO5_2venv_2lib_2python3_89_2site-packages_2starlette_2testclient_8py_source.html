117 async def _run(self, *, task_status: anyio.abc.TaskStatus[anyio.CancelScope]) -> None:
118 """
119 The sub-thread in which the websocket session runs.
120 """
121 send: anyio.create_memory_object_stream[Message] = anyio.create_memory_object_stream(math.inf)
122 send_tx, send_rx = send
123 receive: anyio.create_memory_object_stream[Message] = anyio.create_memory_object_stream(math.inf)
124 receive_tx, receive_rx = receive
125 with send_tx, send_rx, receive_tx, receive_rx, anyio.CancelScope() as cs:
126 self._receive_tx = receive_tx
127 self._send_rx = send_rx
128 task_status.started(cs)
129 await self.app(self.scope, receive_rx.receive, send_tx.send)
130
131 # wait for cs.cancel to be called before closing streams
132 await anyio.sleep_forever()

134 def _raise_on_close(self, message: Message) -> None:
135 if message["type"] == "websocket.close":
136 raise WebSocketDisconnect(code=message.get("code", 1000), reason=message.get("reason", ""))
137 elif message["type"] == "websocket.http.response.start":
138 status_code: int = message["status"]
139 headers: list[tuple[bytes, bytes]] = message["headers"]
140 body: list[bytes] = []
141 while True:
142 message = self.receive()
143 assert message["type"] == "websocket.http.response.body"
144 body.append(message["body"])
145 if not message.get("more_body", False):
146 break
147 raise WebSocketDenialResponse(status_code=status_code, headers=headers, content=b"".join(body))

284 async def receive() -> Message:
285 nonlocal request_complete
286
287 if request_complete:
288 if not response_complete.is_set():
289 await response_complete.wait()
290 return {"type": "http.disconnect"}
291
292 body = request.read()
293 if isinstance(body, str):
294 body_bytes: bytes = body.encode("utf-8") # pragma: no cover
295 elif body is None:
296 body_bytes = b"" # pragma: no cover

312 async def send(message: Message) -> None:
313 nonlocal raw_kwargs, response_started, template, context
314
315 if message["type"] == "http.response.start":
316 assert not response_started, 'Received multiple "http.response.start" messages.'
317 raw_kwargs["status_code"] = message["status"]
318 raw_kwargs["headers"] = [(key.decode(), value.decode()) for key, value in message.get("headers", [])]
319 response_started = True
320 elif message["type"] == "http.response.body":
321 assert response_started, 'Received "http.response.body" without "http.response.start".'
322 assert not response_complete.is_set(), 'Received "http.response.body" after response completed.'
323 body = message.get("body", b"")
324 more_body = message.get("more_body", False)
325 if request.method != "HEAD":
326 raw_kwargs["stream"].write(body)
327 if not more_body:
328 raw_kwargs["stream"].seek(0)
329 response_complete.set()
330 elif message["type"] == "http.response.debug":
331 template = message["info"]["template"]
332 context = message["info"]["context"]
333
334 try:
335 with self.portal_factory() as portal:
336 response_complete = portal.call(anyio.Event)
337 portal.call(self.app, scope, receive, send)
338 except BaseException as exc:
339 if self.raise_server_exceptions:
340 raise exc
341
342 if self.raise_server_exceptions:
343 assert response_started, "TestClient did not receive any response."
344 elif not response_started:
345 raw_kwargs = {
346 "status_code": 500,
347 "headers": [],
348 "stream": io.BytesIO(),
349 }
350
351 raw_kwargs["stream"] = httpx.ByteStream(raw_kwargs["stream"].read())
352
353 response = httpx.Response(**raw_kwargs, request=request)
354 if template is not None:
355 response.template = template # type: ignore[attr-defined]
356 response.context = context # type: ignore[attr-defined]
