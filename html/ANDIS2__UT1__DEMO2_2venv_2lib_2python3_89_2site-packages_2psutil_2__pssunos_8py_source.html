123def virtual_memory():
124 """Report virtual memory metrics."""
125 # we could have done this with kstat, but IMHO this is good enough
126 total = os.sysconf('SC_PHYS_PAGES') * PAGE_SIZE
127 # note: there's no difference on Solaris
128 free = avail = os.sysconf('SC_AVPHYS_PAGES') * PAGE_SIZE
129 used = total - free
130 percent = usage_percent(used, total, round_=1)
131 return svmem(total, avail, percent, used, free)
132
133
134def swap_memory():
135 """Report swap memory metrics."""
136 sin, sout = cext.swap_mem()
137 # XXX
138 # we are supposed to get total/free by doing so:
139 # http://cvs.opensolaris.org/source/xref/onnv/onnv-gate/
140 # usr/src/cmd/swap/swap.c
141 # ...nevertheless I can't manage to obtain the same numbers as 'swap'
142 # cmdline utility, so let's parse its output (sigh!)
143 p = subprocess.Popen(
144 [
145 '/usr/bin/env',
146 f"PATH=/usr/sbin:/sbin:{os.environ['PATH']}",
147 'swap',
148 '-l',
149 ],
150 stdout=subprocess.PIPE,
151 )
152 stdout, _ = p.communicate()
153 stdout = stdout.decode(sys.stdout.encoding)
154 if p.returncode != 0:
155 msg = f"'swap -l' failed (retcode={p.returncode})"
156 raise RuntimeError(msg)
157
158 lines = stdout.strip().split('\n')[1:]
159 if not lines:
160 msg = 'no swap device(s) configured'
161 raise RuntimeError(msg)
162 total = free = 0
163 for line in lines:
164 line = line.split()
165 t, f = line[3:5]
166 total += int(int(t) * 512)
167 free += int(int(f) * 512)
168 used = total - free
169 percent = usage_percent(used, total, round_=1)
170 return _common.sswap(
171 total, used, free, percent, sin * PAGE_SIZE, sout * PAGE_SIZE
172 )

259def net_connections(kind, _pid=-1):
260 """Return socket connections. If pid == -1 return system-wide
261 connections (as opposed to connections opened by one process only).
262 Only INET sockets are returned (UNIX are not).
263 """
264 families, types = _common.conn_tmap[kind]
265 rawlist = cext.net_connections(_pid)
266 ret = set()
267 for item in rawlist:
268 fd, fam, type_, laddr, raddr, status, pid = item
269 if fam not in families:
270 continue
271 if type_ not in types:
272 continue
273 # TODO: refactor and use _common.conn_to_ntuple.
274 if fam in {AF_INET, AF_INET6}:
275 if laddr:
276 laddr = _common.addr(*laddr)
277 if raddr:
278 raddr = _common.addr(*raddr)
279 status = TCP_STATUSES[status]
280 fam = sockfam_to_enum(fam)
281 type_ = socktype_to_enum(type_)
282 if _pid == -1:
283 nt = _common.sconn(fd, fam, type_, laddr, raddr, status, pid)
284 else:
285 nt = _common.pconn(fd, fam, type_, laddr, raddr, status)
286 ret.add(nt)
287 return list(ret)

346def wrap_exceptions(fun):
347 """Call callable into a try/except clause and translate ENOENT,
348 EACCES and EPERM in NoSuchProcess or AccessDenied exceptions.
349 """
350
351 @functools.wraps(fun)
352 def wrapper(self, *args, **kwargs):
353 pid, ppid, name = self.pid, self._ppid, self._name
354 try:
355 return fun(self, *args, **kwargs)
356 except (FileNotFoundError, ProcessLookupError) as err:
357 # ENOENT (no such file or directory) gets raised on open().
358 # ESRCH (no such process) can get raised on read() if
359 # process is gone in meantime.
360 if not pid_exists(pid):
361 raise NoSuchProcess(pid, name) from err
362 raise ZombieProcess(pid, name, ppid) from err
363 except PermissionError as err:
364 raise AccessDenied(pid, name) from err
365 except OSError as err:
366 if pid == 0:
367 if 0 in pids():
368 raise AccessDenied(pid, name) from err
369 raise
370 raise
371
372 return wrapper

386 def _assert_alive(self):
387 """Raise NSP if the process disappeared on us."""
388 # For those C function who do not raise NSP, possibly returning
389 # incorrect or incomplete result.
390 os.stat(f"{self._procfs_path}/{self.pid}")

498 @wrap_exceptions
499 def cpu_times(self):

501 times = cext.proc_cpu_times(self.pid, self._procfs_path)

618 def _get_unix_sockets(self, pid):
619 """Get UNIX sockets used by process by parsing 'pfiles' output."""
620 # TODO: rewrite this in C (...but the damn netstat source code
621 # does not include this part! Argh!!)
622 cmd = ["pfiles", str(pid)]
623 p = subprocess.Popen(
624 cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE
625 )
626 stdout, stderr = p.communicate()
627 stdout, stderr = (
628 x.decode(sys.stdout.encoding) for x in (stdout, stderr)
629 )
630 if p.returncode != 0:
631 if 'permission denied' in stderr.lower():
632 raise AccessDenied(self.pid, self._name)
633 if 'no such process' in stderr.lower():
634 raise NoSuchProcess(self.pid, self._name)
635 msg = f"{cmd!r} command error\n{stderr}"
636 raise RuntimeError(msg)
637
638 lines = stdout.split('\n')[2:]
639 for i, line in enumerate(lines):
640 line = line.lstrip()
641 if line.startswith('sockname: AF_UNIX'):
642 path = line.split(' ', 2)[2]
643 type = lines[i - 2].strip()
644 if type == 'SOCK_STREAM':
645 type = socket.SOCK_STREAM
646 elif type == 'SOCK_DGRAM':
647 type = socket.SOCK_DGRAM
648 else:
649 type = -1
650 yield (-1, socket.AF_UNIX, type, path, "", _common.CONN_NONE)

652 @wrap_exceptions
653 def net_connections(self, kind='inet'):
654 ret = net_connections(kind, _pid=self.pid)
655 # The underlying C implementation retrieves all OS connections
656 # and filters them by PID. At this point we can't tell whether
657 # an empty list means there were no connections for process or
658 # process is no longer active so we force NSP in case the PID
659 # is no longer there.
660 if not ret:
661 # will raise NSP if process is gone
662 os.stat(f"{self._procfs_path}/{self.pid}")
663
664 # UNIX sockets
665 if kind in {'all', 'unix'}:
666 ret.extend([
667 _common.pconn(*conn)
668 for conn in self._get_unix_sockets(self.pid)
669 ])
670 return ret

675 @wrap_exceptions
676 def memory_maps(self):
677 def toaddr(start, end):
678 return "{}-{}".format(
679 hex(start)[2:].strip('L'), hex(end)[2:].strip('L')
680 )
681
682 procfs_path = self._procfs_path
683 retlist = []
684 try:
685 rawlist = cext.proc_memory_maps(self.pid, procfs_path)
686 except OSError as err:
687 if err.errno == errno.EOVERFLOW and not IS_64_BIT:
688 # We may get here if we attempt to query a 64bit process
689 # with a 32bit python.
690 # Error originates from read() and also tools like "cat"
691 # fail in the same way (!).
692 # Since there simply is no way to determine CPU times we
693 # return 0.0 as a fallback. See:
694 # https://github.com/giampaolo/psutil/issues/857
695 return []
696 else:
697 raise
698 hit_enoent = False
699 for item in rawlist:
700 addr, addrsize, perm, name, rss, anon, locked = item
701 addr = toaddr(addr, addrsize)
702 if not name.startswith('['):
703 try:
704 name = os.readlink(f"{procfs_path}/{self.pid}/path/{name}")
705 except OSError as err:
706 if err.errno == errno.ENOENT:
707 # sometimes the link may not be resolved by
708 # readlink() even if it exists (ls shows it).
709 # If that's the case we just return the
