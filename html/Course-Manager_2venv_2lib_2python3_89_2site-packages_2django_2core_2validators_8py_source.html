17def _lazy_re_compile(regex, flags=0):
18 """Lazily compile a regex with flags."""
19 def _compile():
20 # Compile the regex if it was not passed pre-compiled.
21 if isinstance(regex, str):
22 return re.compile(regex, flags)
23 else:
24 assert not flags, "flags must be empty if regex is passed pre-compiled"
25 return regex
26 return SimpleLazyObject(_compile)

76 ul = '\u00a1-\uffff' # unicode letters range (must not be a raw string)
77
78 # IP patterns
79 ipv4_re = r'(?:25[0-5]|2[0-4]\d|[0-1]?\d?\d)(?:\.(?:25[0-5]|2[0-4]\d|[0-1]?\d?\d)){3}'
80 ipv6_re = r'\‍[[0-9a-f:\.]+\‍]' # (simple regex, validated later)
81
82 # Host patterns
83 hostname_re = r'[a-z' + ul + r'0-9](?:[a-z' + ul + r'0-9-]{0,61}[a-z' + ul + r'0-9])?'
84 # Max length for domain name labels is 63 characters per RFC 1034 sec. 3.1
85 domain_re = r'(?:\.(?!-)[a-z' + ul + r'0-9-]{1,63}(?<!-))*'
86 tld_re = (
87 r'\.' # dot
88 r'(?!-)' # can't start with a dash
89 r'(?:[a-z' + ul + '-]{2,63}' # domain label
90 r'|xn--[a-z0-9]{1,59})' # or punycode label
91 r'(?<!-)' # can't end with a dash
92 r'\.?' # may have a trailing dot
93 )
94 host_re = '(' + hostname_re + domain_re + tld_re + '|localhost)'
95
96 regex = _lazy_re_compile(
97 r'^(?:[a-z0-9\.\-\+]*)://' # scheme is validated separately
98 r'(?:[^\s:@/]+(?::[^\s:@/]*)?@)?' # user:pass authentication
99 r'(?:' + ipv4_re + '|' + ipv6_re + '|' + host_re + ')'
100 r'(?::\d{2,5})?' # port
101 r'(?:[/?#][^\s]*)?' # resource path
102 r'\Z', re.IGNORECASE)
103 message = _('Enter a valid URL.')
104 schemes = ['http', 'https', 'ftp', 'ftps']
105
106 def __init__(self, schemes=None, **kwargs):
107 super().__init__(**kwargs)
108 if schemes is not None:
109 self.schemes = schemes
110
111 def __call__(self, value):
112 # Check first if the scheme is valid
113 scheme = value.split('://')[0].lower()
114 if scheme not in self.schemes:
115 raise ValidationError(self.message, code=self.code)
116
117 # Then check full URL
118 try:
119 super().__call__(value)
120 except ValidationError as e:
121 # Trivial case failed. Try for possible IDN domain
122 if value:
123 try:
124 scheme, netloc, path, query, fragment = urlsplit(value)
125 except ValueError: # for example, "Invalid IPv6 URL"
126 raise ValidationError(self.message, code=self.code)
127 try:
128 netloc = punycode(netloc) # IDN -> ACE
129 except UnicodeError: # invalid domain part
130 raise e
131 url = urlunsplit((scheme, netloc, path, query, fragment))
132 super().__call__(url)
133 else:
134 raise
135 else:
136 # Now verify IPv6 in the netloc part
137 host_match = re.search(r'^\‍[(.+)\‍](?::\d{2,5})?$', urlsplit(value).netloc)
138 if host_match:
139 potential_ip = host_match.groups()[0]
140 try:
141 validate_ipv6_address(potential_ip)
142 except ValidationError:
143 raise ValidationError(self.message, code=self.code)
144
145 # The maximum length of a full host name is 253 characters per RFC 1034
146 # section 3.1. It's defined to be 255 bytes or less, but this includes
147 # one byte for the length of the name and one byte for the trailing dot
148 # that's used to indicate absolute names in DNS.
149 if len(urlsplit(value).netloc) > 253:
150 raise ValidationError(self.message, code=self.code)

166 message = _('Enter a valid email address.')
167 code = 'invalid'
168 user_regex = _lazy_re_compile(
169 r"(^[-!#$%&'*+/=?^_`{}|~0-9A-Z]+(\.[-!#$%&'*+/=?^_`{}|~0-9A-Z]+)*\Z" # dot-atom
170 r'|^"([\001-\010\013\014\016-\037!#-\‍[\‍]-\177]|\\[\001-\011\013\014\016-\177])*"\Z)', # quoted-string
171 re.IGNORECASE)
172 domain_regex = _lazy_re_compile(
173 # max length for domain name labels is 63 characters per RFC 1034
174 r'((?:[A-Z0-9](?:[A-Z0-9-]{0,61}[A-Z0-9])?\.)+)(?:[A-Z0-9-]{2,63}(?<!-))\Z',
175 re.IGNORECASE)
176 literal_regex = _lazy_re_compile(
177 # literal form, ipv4 or ipv6 address (SMTP 4.1.3)
178 r'\‍[([A-f0-9:\.]+)\‍]\Z',
179 re.IGNORECASE)
180 domain_whitelist = ['localhost']
181
182 def __init__(self, message=None, code=None, whitelist=None):
183 if message is not None:
184 self.message = message
185 if code is not None:
186 self.code = code
187 if whitelist is not None:
188 self.domain_whitelist = whitelist
189
190 def __call__(self, value):
191 if not value or '@' not in value:
192 raise ValidationError(self.message, code=self.code)
193
194 user_part, domain_part = value.rsplit('@', 1)
195
196 if not self.user_regex.match(user_part):
197 raise ValidationError(self.message, code=self.code)
198
199 if (domain_part not in self.domain_whitelist and
200 not self.validate_domain_part(domain_part)):
201 # Try for possible IDN domain-part
202 try:
203 domain_part = punycode(domain_part)
204 except UnicodeError:
205 pass
206 else:
207 if self.validate_domain_part(domain_part):
208 return
209 raise ValidationError(self.message, code=self.code)
210
211 def validate_domain_part(self, domain_part):
212 if self.domain_regex.match(domain_part):
213 return True
214
215 literal_match = self.literal_regex.match(domain_part)
216 if literal_match:
217 ip_address = literal_match.group(1)
218 try:
219 validate_ipv46_address(ip_address)
220 return True
221 except ValidationError:
222 pass
223 return False

389@deconstructible
390class DecimalValidator:
391 """
392 Validate that the input does not exceed the maximum number of digits
393 expected, otherwise raise ValidationError.
394 """
395 messages = {
396 'invalid': _('Enter a number.'),
397 'max_digits': ngettext_lazy(
398 'Ensure that there are no more than %(max)s digit in total.',
399 'Ensure that there are no more than %(max)s digits in total.',
400 'max'
401 ),
402 'max_decimal_places': ngettext_lazy(
403 'Ensure that there are no more than %(max)s decimal place.',
404 'Ensure that there are no more than %(max)s decimal places.',
405 'max'
406 ),
407 'max_whole_digits': ngettext_lazy(
408 'Ensure that there are no more than %(max)s digit before the decimal point.',
409 'Ensure that there are no more than %(max)s digits before the decimal point.',
410 'max'
411 ),
412 }
413
414 def __init__(self, max_digits, decimal_places):
415 self.max_digits = max_digits
416 self.decimal_places = decimal_places
417
418 def __call__(self, value):
419 digit_tuple, exponent = value.as_tuple()[1:]
420 if exponent in {'F', 'n', 'N'}:
421 raise ValidationError(self.messages['invalid'])
422 if exponent >= 0:
423 # A positive exponent adds that many trailing zeros.
424 digits = len(digit_tuple) + exponent
425 decimals = 0
426 else:
427 # If the absolute value of the negative exponent is larger than the
428 # number of digits, then it's the same as the number of digits,
429 # because it'll consume all of the digits in digit_tuple and then
430 # add abs(exponent) - len(digit_tuple) leading zeros after the
431 # decimal point.
432 if abs(exponent) > len(digit_tuple):
433 digits = decimals = abs(exponent)
434 else:
435 digits = len(digit_tuple)
436 decimals = abs(exponent)
437 whole_digits = digits - decimals
438
439 if self.max_digits is not None and digits > self.max_digits:
440 raise ValidationError(
441 self.messages['max_digits'],
442 code='max_digits',
443 params={'max': self.max_digits},
444 )
445 if self.decimal_places is not None and decimals > self.decimal_places:
446 raise ValidationError(
447 self.messages['max_decimal_places'],
448 code='max_decimal_places',
449 params={'max': self.decimal_places},
450 )
451 if (self.max_digits is not None and self.decimal_places is not None and
452 whole_digits > (self.max_digits - self.decimal_places)):
453 raise ValidationError(
454 self.messages['max_whole_digits'],
455 code='max_whole_digits',
456 params={'max': (self.max_digits - self.decimal_places)},
457 )
458
459 def __eq__(self, other):
460 return (
461 isinstance(other, self.__class__) and
462 self.max_digits == other.max_digits and
463 self.decimal_places == other.decimal_places
464 )
