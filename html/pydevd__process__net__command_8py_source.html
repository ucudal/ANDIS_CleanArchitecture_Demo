31 def process_net_command(self, py_db, cmd_id, seq, text):
32 """Processes a command received from the Java side
33
34 @param cmd_id: the id of the command
35 @param seq: the sequence of the command
36 @param text: the text received in the command
37 """
38
39 # We can only proceed if the client is already authenticated or if it's the
40 # command to authenticate.
41 if cmd_id != CMD_AUTHENTICATE and not py_db.authentication.is_authenticated():
42 cmd = py_db.cmd_factory.make_error_message(seq, "Client not authenticated.")
43 py_db.writer.add_command(cmd)
44 return
45
46 meaning = ID_TO_MEANING[str(cmd_id)]
47
48 # print('Handling %s (%s)' % (meaning, text))
49
50 method_name = meaning.lower()
51
52 on_command = getattr(self, method_name.lower(), None)
53 if on_command is None:
54 # I have no idea what this is all about
55 cmd = py_db.cmd_factory.make_error_message(seq, "unexpected command " + str(cmd_id))
56 py_db.writer.add_command(cmd)
57 return
58
59 lock = py_db._main_lock
60 if method_name == "cmd_thread_dump_to_stderr":
61 # We can skip the main debugger locks for cases where we know it's not needed.
62 lock = NULL
63
64 with lock:
65 try:
66 cmd = on_command(py_db, cmd_id, seq, text)
67 if cmd is not None:
68 py_db.writer.add_command(cmd)
69 except:
70 if traceback is not None and sys is not None and pydev_log_exception is not None:
71 pydev_log_exception()
72
73 stream = StringIO()
74 traceback.print_exc(file=stream)
75 cmd = py_db.cmd_factory.make_error_message(
76 seq,
77 "Unexpected exception in process_net_command.\nInitial params: %s. Exception: %s"
78 % (((cmd_id, seq, text), stream.getvalue())),
79 )
80 if cmd is not None:
81 py_db.writer.add_command(cmd)

593 #

596 #
