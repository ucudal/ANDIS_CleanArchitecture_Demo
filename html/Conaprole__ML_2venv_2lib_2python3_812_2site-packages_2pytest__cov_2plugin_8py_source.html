21def validate_report(arg):
22 file_choices = ['annotate', 'html', 'xml', 'json', 'lcov']
23 term_choices = ['term', 'term-missing']
24 term_modifier_choices = ['skip-covered']
25 all_choices = term_choices + file_choices
26 values = arg.split(':', 1)
27 report_type = values[0]
28 if report_type not in [*all_choices, '']:
29 msg = f'invalid choice: "{arg}" (choose from "{all_choices}")'
30 raise argparse.ArgumentTypeError(msg)
31
32 if report_type == 'lcov' and coverage.version_info <= (6, 3):
33 raise argparse.ArgumentTypeError('LCOV output is only supported with coverage.py >= 6.3')
34
35 if len(values) == 1:
36 return report_type, None
37
38 report_modifier = values[1]
39 if report_type in term_choices and report_modifier in term_modifier_choices:
40 return report_type, report_modifier
41
42 if report_type not in file_choices:
43 msg = f'output specifier not supported for: "{arg}" (choose from "{file_choices}")'
44 raise argparse.ArgumentTypeError(msg)
45
46 return values

49def validate_fail_under(num_str):
50 try:
51 value = int(num_str)

57 if value > 100:

59 'Your desire for over-achievement is admirable but misplaced. '
60 'The maximum value is 100. Perhaps write more integration tests?'

167def _prepare_cov_source(cov_source):
168 """
169 Prepare cov_source so that:
170
171 --cov --cov=foobar is equivalent to --cov (cov_source=None)
172 --cov=foo --cov=bar is equivalent to cov_source=['foo', 'bar']

174 return None if True in cov_source else [path for path in cov_source if path is not True]

312 def _should_report(self):
313 needed = self.options.cov_report or self.options.cov_fail_under
314 return needed and not (self.failed and self.options.no_cov_on_fail)

316 # we need to wrap pytest_runtestloop. by the time pytest_sessionfinish
317 # runs, it's too late to set testsfailed
318 @pytest.hookimpl(hookwrapper=True)
319 def pytest_runtestloop(self, session):
320 yield
321
322 if self._disabled:
323 return
324
325 compat_session = compat.SessionWrapper(session)
326
327 self.failed = bool(compat_session.testsfailed)
328 if self.cov_controller is not None:
329 self.cov_controller.finish()
330
331 if not self._is_worker(session) and self._should_report():
332 # import coverage lazily here to avoid importing
333 # it for unit tests that don't need it
334 from coverage.misc import CoverageException
335
336 try:
337 self.cov_total = self.cov_controller.summary(self.cov_report)
338 except CoverageException as exc:
339 message = f'Failed to generate report: {exc}\n'
340 session.config.pluginmanager.getplugin('terminalreporter').write(f'\nWARNING: {message}\n', red=True, bold=True)
341 warnings.warn(CovReportWarning(message), stacklevel=1)
342 self.cov_total = 0
343 assert self.cov_total is not None, 'Test coverage should never be `None`'
344 cov_fail_under = self.options.cov_fail_under
345 cov_precision = self.options.cov_precision
346 if cov_fail_under is None or self.options.collectonly:
347 return
348 if should_fail_under(self.cov_total, cov_fail_under, cov_precision):
349 message = 'Coverage failure: total of {total} is less than fail-under={fail_under:.{p}f}'.format(
350 total=display_covered(self.cov_total, cov_precision),
351 fail_under=cov_fail_under,
352 p=cov_precision,
353 )
354 session.config.pluginmanager.getplugin('terminalreporter').write(f'\nERROR: {message}\n', red=True, bold=True)
355 warnings.warn(CovFailUnderWarning(message), stacklevel=1)
356 # make sure we get the EXIT_TESTSFAILED exit code
357 compat_session.testsfailed += 1

359 def pytest_terminal_summary(self, terminalreporter):
360 if self._disabled:
361 if self.options.no_cov_should_warn:
362 message = 'Coverage disabled via --no-cov switch!'
363 terminalreporter.write(f'WARNING: {message}\n', red=True, bold=True)
364 warnings.warn(CovDisabledWarning(message), stacklevel=1)
365 return
366 if self.cov_controller is None:
367 return
368
369 if self.cov_total is None:
370 # we shouldn't report, or report generation failed (error raised above)
371 return
372
373 report = self.cov_report.getvalue()
374
375 # Avoid undesirable new lines when output is disabled with "--cov-report=".
376 if report:
377 terminalreporter.write('\n' + report + '\n')
378
379 if self.options.cov_fail_under is not None and self.options.cov_fail_under > 0:
380 failed = self.cov_total < self.options.cov_fail_under
381 markup = {'red': True, 'bold': True} if failed else {'green': True}
382 message = '{fail}Required test coverage of {required}% {reached}. ' 'Total coverage: {actual:.2f}%\n'.format(
383 required=self.options.cov_fail_under,
384 actual=self.cov_total,
385 fail='FAIL ' if failed else '',
386 reached='not reached' if failed else 'reached',
387 )
388 terminalreporter.write(message, **markup)
