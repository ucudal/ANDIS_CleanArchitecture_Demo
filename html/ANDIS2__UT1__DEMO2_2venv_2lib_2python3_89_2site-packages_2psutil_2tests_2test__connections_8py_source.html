65 def compare_procsys_connections(self, pid, proc_cons, kind='all'):
66 """Given a process PID and its list of connections compare
67 those against system-wide connections retrieved via
68 psutil.net_connections.
69 """
70 try:
71 sys_cons = psutil.net_connections(kind=kind)
72 except psutil.AccessDenied:
73 # On MACOS, system-wide connections are retrieved by iterating
74 # over all processes
75 if MACOS:
76 return
77 else:
78 raise
79 # Filter for this proc PID and exlucde PIDs from the tuple.
80 sys_cons = [c[:-1] for c in sys_cons if c.pid == pid]
81 sys_cons.sort()
82 proc_cons.sort()
83 assert proc_cons == sys_cons

109 def get_conn_from_sock(self, sock):
110 cons = this_proc_net_connections(kind='all')
111 smap = {c.fd: c for c in cons}
112 if NETBSD or FREEBSD:
113 # NetBSD opens a UNIX socket to /var/log/run
114 # so there may be more connections.
115 return smap[sock.fileno()]
116 else:
117 assert len(cons) == 1
118 if cons[0].fd != -1:
119 assert smap[sock.fileno()].fd == sock.fileno()
120 return cons[0]

122 def check_socket(self, sock):
123 """Given a socket, makes sure it matches the one obtained
124 via psutil. It assumes this process created one connection
125 only (the one supposed to be checked).
126 """
127 conn = self.get_conn_from_sock(sock)
128 check_connection_ntuple(conn)
129
130 # fd, family, type
131 if conn.fd != -1:
132 assert conn.fd == sock.fileno()
133 assert conn.family == sock.family
134 # see: http://bugs.python.org/issue30204
135 assert conn.type == sock.getsockopt(socket.SOL_SOCKET, socket.SO_TYPE)
136
137 # local address
138 laddr = sock.getsockname()
139 if not laddr and isinstance(laddr, bytes):
140 # See: http://bugs.python.org/issue30205
141 laddr = laddr.decode()
142 if sock.family == AF_INET6:
143 laddr = laddr[:2]
144 assert conn.laddr == laddr
145
146 # XXX Solaris can't retrieve system-wide UNIX sockets
147 if sock.family == AF_UNIX and HAS_NET_CONNECTIONS_UNIX:
148 cons = this_proc_net_connections(kind='all')
149 self.compare_procsys_connections(os.getpid(), cons, kind='all')
150 return conn
