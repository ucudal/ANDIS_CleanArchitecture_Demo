13class EventData(TypedDict, total=False):
14 """Data associated with a streaming event."""
15
16 input: Any
17 """The input passed to the Runnable that generated the event.
18
19 Inputs will sometimes be available at the *START* of the Runnable, and
20 sometimes at the *END* of the Runnable.
21
22 If a Runnable is able to stream its inputs, then its input by definition
23 won't be known until the *END* of the Runnable when it has finished streaming
24 its inputs.
25 """
26 output: Any
27 """The output of the Runnable that generated the event.
28
29 Outputs will only be available at the *END* of the Runnable.
30
31 For most Runnables, this field can be inferred from the `chunk` field,
32 though there might be some exceptions for special cased Runnables (e.g., like
33 chat models), which may return more information.
34 """
35 chunk: Any
36 """A streaming chunk from the output that generated the event.
37
38 chunks support addition in general, and adding them up should result
39 in the output of the Runnable that generated the event.
40 """

43class BaseStreamEvent(TypedDict):
44 """Streaming event.
45
46 Schema of a streaming event which is produced from the astream_events method.
47
48 Example:
49
50 .. code-block:: python
51
52 from langchain_core.runnables import RunnableLambda
53
54 async def reverse(s: str) -> str:
55 return s[::-1]
56
57 chain = RunnableLambda(func=reverse)
58
59 events = [event async for event in chain.astream_events("hello")]
60
61 # will produce the following events
62 # (where some fields have been omitted for brevity):
63 [
64 {
65 "data": {"input": "hello"},
66 "event": "on_chain_start",
67 "metadata": {},
68 "name": "reverse",
69 "tags": [],
70 },
71 {
72 "data": {"chunk": "olleh"},
73 "event": "on_chain_stream",
74 "metadata": {},
75 "name": "reverse",
76 "tags": [],
77 },
78 {
79 "data": {"output": "olleh"},
80 "event": "on_chain_end",
81 "metadata": {},
82 "name": "reverse",
83 "tags": [],
84 },
85 ]
86 """
87
88 event: str
89 """Event names are of the format: on_[runnable_type]_(start|stream|end).
90
91 Runnable types are one of:
92
93 - **llm** - used by non chat models
94 - **chat_model** - used by chat models
95 - **prompt** -- e.g., ChatPromptTemplate
96 - **tool** -- from tools defined via @tool decorator or inheriting
97 from Tool/BaseTool
98 - **chain** - most Runnables are of this type
99
100 Further, the events are categorized as one of:
101
102 - **start** - when the Runnable starts
103 - **stream** - when the Runnable is streaming
104 - **end* - when the Runnable ends
105
106 start, stream and end are associated with slightly different `data` payload.
107
108 Please see the documentation for `EventData` for more details.
109 """
110 run_id: str
111 """An randomly generated ID to keep track of the execution of the given Runnable.
112
113 Each child Runnable that gets invoked as part of the execution of a parent Runnable
114 is assigned its own unique ID.
115 """
116 tags: NotRequired[list[str]]
117 """Tags associated with the Runnable that generated this event.
118
119 Tags are always inherited from parent Runnables.
120
121 Tags can either be bound to a Runnable using `.with_config({"tags": ["hello"]})`
122 or passed at run time using `.astream_events(..., {"tags": ["hello"]})`.
123 """
124 metadata: NotRequired[dict[str, Any]]
125 """Metadata associated with the Runnable that generated this event.
126
127 Metadata can either be bound to a Runnable using
128
129 `.with_config({"metadata": { "foo": "bar" }})`
130
131 or passed at run time using
132
133 `.astream_events(..., {"metadata": {"foo": "bar"}})`.
134 """
135
136 parent_ids: Sequence[str]
137 """A list of the parent IDs associated with this event.
138
139 Root Events will have an empty list.
140
141 For example, if a Runnable A calls Runnable B, then the event generated by Runnable
142 B will have Runnable A's ID in the parent_ids field.
143
144 The order of the parent IDs is from the root parent to the immediate parent.
145
146 Only supported as of v2 of the astream events API. v1 will return an empty list.
147 """
