216 @property
217 def next_paragraph_style(self):
218 """
219 |_ParagraphStyle| object representing the style to be applied
220 automatically to a new paragraph inserted after a paragraph of this
221 style. Returns self if no next paragraph style is defined. Assigning
222 |None| or *self* removes the setting such that new paragraphs are
223 created using this same style.
224 """
225 next_style_elm = self._element.next_style
226 if next_style_elm is None:
227 return self
228 if next_style_elm.type != WD_STYLE_TYPE.PARAGRAPH:
229 return self

232 @next_paragraph_style.setter
233 def next_paragraph_style(self, style):
234 if style is None or style.style_id == self.style_id:
235 self._element._remove_next()
236 else:
237 self._element.get_or_add_next().val = style.style_id
