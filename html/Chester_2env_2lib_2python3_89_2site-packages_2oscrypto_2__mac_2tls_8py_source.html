102def _read_callback(connection_id, data_buffer, data_length_pointer):
103 """
104 Callback called by Secure Transport to actually read the socket
105
106 :param connection_id:
107 An integer identifying the connection
108
109 :param data_buffer:
110 A char pointer FFI type to write the data to
111
112 :param data_length_pointer:
113 A size_t pointer FFI type of the amount of data to read. Will be
114 overwritten with the amount of data read on return.
115
116 :return:
117 An integer status code of the result - 0 for success
118 """
119
120 self = None
121 try:
122 self = _connection_refs.get(connection_id)
123 if not self:
124 socket = _socket_refs.get(connection_id)
125 else:
126 socket = self._socket
127
128 if not self and not socket:
129 return 0
130
131 bytes_requested = deref(data_length_pointer)
132
133 timeout = socket.gettimeout()
134 error = None
135 data = b''
136 try:
137 while len(data) < bytes_requested:
138 # Python 2 on Travis CI seems to have issues with blocking on
139 # recv() for longer than the socket timeout value, so we select
140 if timeout is not None and timeout > 0.0:
141 read_ready, _, _ = select.select([socket], [], [], timeout)
142 if len(read_ready) == 0:
143 raise socket_.error(errno.EAGAIN, 'timed out')
144 chunk = socket.recv(bytes_requested - len(data))
145 data += chunk
146 if chunk == b'':
147 if len(data) == 0:
148 if timeout is None:
149 return SecurityConst.errSSLClosedNoNotify
150 return SecurityConst.errSSLClosedAbort
151 break
152 except (socket_.error) as e:
153 error = e.errno
154
155 if error is not None and error != errno.EAGAIN:
156 if error == errno.ECONNRESET or error == errno.EPIPE:
157 return SecurityConst.errSSLClosedNoNotify
158 return SecurityConst.errSSLClosedAbort
159
160 if self and not self._done_handshake:
161 # SecureTransport doesn't bother to check if the TLS record header
162 # is valid before asking to read more data, which can result in
163 # connection hangs. Here we do basic checks to get around the issue.
164 if len(data) >= 3 and len(self._server_hello) == 0:
165 # Check to ensure it is an alert or handshake first
166 valid_record_type = data[0:1] in set([b'\x15', b'\x16'])
167 # Check if the protocol version is SSL 3.0 or TLS 1.0-1.3
168 valid_protocol_version = data[1:3] in set([
169 b'\x03\x00',
170 b'\x03\x01',
171 b'\x03\x02',
172 b'\x03\x03',
173 b'\x03\x04'
174 ])
175 if not valid_record_type or not valid_protocol_version:
176 self._server_hello += data + _read_remaining(socket)
177 return SecurityConst.errSSLProtocol
178 self._server_hello += data
179
180 write_to_buffer(data_buffer, data)
181 pointer_set(data_length_pointer, len(data))
182
183 if len(data) != bytes_requested:
184 return SecurityConst.errSSLWouldBlock
185
186 return 0
187 except (KeyboardInterrupt) as e:
188 if self:
189 self._exception = e
190 return SecurityConst.errSSLClosedAbort

193def _read_remaining(socket):
194 """
195 Reads everything available from the socket - used for debugging when there
196 is a protocol error
197
198 :param socket:
199 The socket to read from
200
201 :return:
202 A byte string of the remaining data
203 """
204
205 output = b''
206 old_timeout = socket.gettimeout()
207 try:
208 socket.settimeout(0.0)
209 output += socket.recv(8192)
210 except (socket_.error):
211 pass
212 finally:
213 socket.settimeout(old_timeout)
214 return output

217def _write_callback(connection_id, data_buffer, data_length_pointer):
218 """
219 Callback called by Secure Transport to actually write to the socket
220
221 :param connection_id:
222 An integer identifying the connection
223
224 :param data_buffer:
225 A char pointer FFI type containing the data to write
226
227 :param data_length_pointer:
228 A size_t pointer FFI type of the amount of data to write. Will be
229 overwritten with the amount of data actually written on return.
230
231 :return:
232 An integer status code of the result - 0 for success
233 """
234
235 try:
236 self = _connection_refs.get(connection_id)
237 if not self:
238 socket = _socket_refs.get(connection_id)
239 else:
240 socket = self._socket
241
242 if not self and not socket:
243 return 0
244
245 data_length = deref(data_length_pointer)
246 data = bytes_from_buffer(data_buffer, data_length)
247
248 if self and not self._done_handshake:
249 self._client_hello += data
250
251 error = None
252 try:
253 sent = socket.send(data)
254 except (socket_.error) as e:
255 error = e.errno
256
257 if error is not None and error != errno.EAGAIN:
258 if error == errno.ECONNRESET or error == errno.EPIPE:
259 return SecurityConst.errSSLClosedNoNotify
260 return SecurityConst.errSSLClosedAbort
261
262 if sent != data_length:
263 pointer_set(data_length_pointer, sent)
264 return SecurityConst.errSSLWouldBlock
265
266 return 0
267 except (KeyboardInterrupt) as e:
268 self._exception = e
269 return SecurityConst.errSSLPeerUserCancelled

288 def __init__(self, protocol=None, manual_validation=False, extra_trust_roots=None):
289 """
290 :param protocol:
291 A unicode string or set of unicode strings representing allowable
292 protocols to negotiate with the server:
293
294 - "TLSv1.2"
295 - "TLSv1.1"
296 - "TLSv1"
297 - "SSLv3"
298
299 Default is: {"TLSv1", "TLSv1.1", "TLSv1.2"}
300
301 :param manual_validation:
302 If certificate and certificate path validation should be skipped
303 and left to the developer to implement
304
305 :param extra_trust_roots:
306 A list containing one or more certificates to be treated as trust
307 roots, in one of the following formats:
308 - A byte string of the DER encoded certificate
309 - A unicode string of the certificate filename
310 - An asn1crypto.x509.Certificate object
311 - An oscrypto.asymmetric.Certificate object
312
313 :raises:
314 ValueError - when any of the parameters contain an invalid value
315 TypeError - when any of the parameters are of the wrong type
316 OSError - when an error is returned by the OS crypto library
317 """
318
319 if not isinstance(manual_validation, bool):
320 raise TypeError(pretty_message(
321 '''
322 manual_validation must be a boolean, not %s
323 ''',
324 type_name(manual_validation)
325 ))
326
327 self._manual_validation = manual_validation
328
329 if protocol is None:
330 protocol = set(['TLSv1', 'TLSv1.1', 'TLSv1.2'])
331
332 if isinstance(protocol, str_cls):
333 protocol = set([protocol])
334 elif not isinstance(protocol, set):
335 raise TypeError(pretty_message(
336 '''
337 protocol must be a unicode string or set of unicode strings,
338 not %s
339 ''',
340 type_name(protocol)
341 ))
342
343 unsupported_protocols = protocol - set(['SSLv3', 'TLSv1', 'TLSv1.1', 'TLSv1.2'])
344 if unsupported_protocols:
345 raise ValueError(pretty_message(
346 '''
347 protocol must contain only the unicode strings "SSLv3", "TLSv1",
348 "TLSv1.1", "TLSv1.2", not %s
349 ''',
350 repr(unsupported_protocols)
351 ))
352
353 self._protocols = protocol
354
355 self._extra_trust_roots = []
356 if extra_trust_roots:
357 for extra_trust_root in extra_trust_roots:
358 if isinstance(extra_trust_root, Certificate):
359 extra_trust_root = extra_trust_root.asn1
360 elif isinstance(extra_trust_root, byte_cls):
361 extra_trust_root = parse_certificate(extra_trust_root)
362 elif isinstance(extra_trust_root, str_cls):
363 with open(extra_trust_root, 'rb') as f:
364 extra_trust_root = parse_certificate(f.read())
365 elif not isinstance(extra_trust_root, Asn1Certificate):
366 raise TypeError(pretty_message(
367 '''
368 extra_trust_roots must be a list of byte strings, unicode
369 strings, asn1crypto.x509.Certificate objects or
370 oscrypto.asymmetric.Certificate objects, not %s
371 ''',
372 type_name(extra_trust_root)
373 ))
374 self._extra_trust_roots.append(extra_trust_root)
375
376 self._peer_id = rand_bytes(8)

534 def _handshake(self):
535 """
536 Perform an initial TLS handshake
537 """
538
539 session_context = None
540 ssl_policy_ref = None
541 crl_search_ref = None
542 crl_policy_ref = None
543 ocsp_search_ref = None
544 ocsp_policy_ref = None
545 policy_array_ref = None
546 trust_ref = None
547
548 try:
549 if osx_version_info < (10, 8):
550 session_context_pointer = new(Security, 'SSLContextRef *')
551 result = Security.SSLNewContext(False, session_context_pointer)
552 handle_sec_error(result)
553 session_context = unwrap(session_context_pointer)
554
555 else:
556 session_context = Security.SSLCreateContext(
557 null(),
558 SecurityConst.kSSLClientSide,
559 SecurityConst.kSSLStreamType
560 )
561
562 result = Security.SSLSetIOFuncs(
563 session_context,
564 _read_callback_pointer,
565 _write_callback_pointer
566 )
567 handle_sec_error(result)
568
569 self._connection_id = id(self) % 2147483647
570 _connection_refs[self._connection_id] = self
571 _socket_refs[self._connection_id] = self._socket
572 result = Security.SSLSetConnection(session_context, self._connection_id)
573 handle_sec_error(result)
574
575 utf8_domain = self._hostname.encode('utf-8')
576 result = Security.SSLSetPeerDomainName(
577 session_context,
578 utf8_domain,
579 len(utf8_domain)
580 )
581 handle_sec_error(result)
582
583 if osx_version_info >= (10, 10):
584 disable_auto_validation = self._session._manual_validation or self._session._extra_trust_roots
585 explicit_validation = (not self._session._manual_validation) and self._session._extra_trust_roots
586 else:
587 disable_auto_validation = True
588 explicit_validation = not self._session._manual_validation
589
590 # Ensure requested protocol support is set for the session
591 if osx_version_info < (10, 8):
592 for protocol in ['SSLv2', 'SSLv3', 'TLSv1']:
593 protocol_const = _PROTOCOL_STRING_CONST_MAP[protocol]
594 enabled = protocol in self._session._protocols
595 result = Security.SSLSetProtocolVersionEnabled(
596 session_context,
597 protocol_const,
598 enabled
599 )
600 handle_sec_error(result)
601
602 if disable_auto_validation:
603 result = Security.SSLSetEnableCertVerify(session_context, False)
604 handle_sec_error(result)
605
606 else:
607 protocol_consts = [_PROTOCOL_STRING_CONST_MAP[protocol] for protocol in self._session._protocols]
608 min_protocol = min(protocol_consts)
609 max_protocol = max(protocol_consts)
610 result = Security.SSLSetProtocolVersionMin(
611 session_context,
612 min_protocol
613 )
614 handle_sec_error(result)
615 result = Security.SSLSetProtocolVersionMax(
616 session_context,
617 max_protocol
618 )
619 handle_sec_error(result)
620
621 if disable_auto_validation:
622 result = Security.SSLSetSessionOption(
623 session_context,
624 SecurityConst.kSSLSessionOptionBreakOnServerAuth,
625 True
626 )
627 handle_sec_error(result)
628
629 # Disable all sorts of bad cipher suites
630 supported_ciphers_pointer = new(Security, 'size_t *')
631 result = Security.SSLGetNumberSupportedCiphers(session_context, supported_ciphers_pointer)
632 handle_sec_error(result)
633
634 supported_ciphers = deref(supported_ciphers_pointer)
635
636 cipher_buffer = buffer_from_bytes(supported_ciphers * 4)
637 supported_cipher_suites_pointer = cast(Security, 'uint32_t *', cipher_buffer)
638 result = Security.SSLGetSupportedCiphers(
639 session_context,
640 supported_cipher_suites_pointer,
641 supported_ciphers_pointer
642 )
643 handle_sec_error(result)
644
645 supported_ciphers = deref(supported_ciphers_pointer)
646 supported_cipher_suites = array_from_pointer(
647 Security,
648 'uint32_t',
649 supported_cipher_suites_pointer,
650 supported_ciphers
651 )
652 good_ciphers = []
653 for supported_cipher_suite in supported_cipher_suites:
654 cipher_suite = int_to_bytes(supported_cipher_suite, width=2)
655 cipher_suite_name = CIPHER_SUITE_MAP.get(cipher_suite, cipher_suite)
656 good_cipher = _cipher_blacklist_regex.search(cipher_suite_name) is None
657 if good_cipher:
658 good_ciphers.append(supported_cipher_suite)
659
660 num_good_ciphers = len(good_ciphers)
661 good_ciphers_array = new(Security, 'uint32_t[]', num_good_ciphers)
662 array_set(good_ciphers_array, good_ciphers)
663 good_ciphers_pointer = cast(Security, 'uint32_t *', good_ciphers_array)
664 result = Security.SSLSetEnabledCiphers(
665 session_context,
666 good_ciphers_pointer,
667 num_good_ciphers
668 )
669 handle_sec_error(result)
670
671 # Set a peer id from the session to allow for session reuse, the hostname
672 # is appended to prevent a bug on OS X 10.7 where it tries to reuse a
673 # connection even if the hostnames are different.
674 peer_id = self._session._peer_id + self._hostname.encode('utf-8')
675 result = Security.SSLSetPeerID(session_context, peer_id, len(peer_id))
676 handle_sec_error(result)
677
678 handshake_result = Security.SSLHandshake(session_context)
679 if self._exception is not None:
680 exception = self._exception
681 self._exception = None
682 raise exception
683 while handshake_result == SecurityConst.errSSLWouldBlock:
684 handshake_result = Security.SSLHandshake(session_context)
685 if self._exception is not None:
686 exception = self._exception
687 self._exception = None
688 raise exception

690 if osx_version_info < (10, 8) and osx_version_info >= (10, 7):
691 do_validation = explicit_validation and handshake_result == 0
692 else:
693 do_validation = explicit_validation and handshake_result == SecurityConst.errSSLServerAuthCompleted
694
695 if do_validation:
696 trust_ref_pointer = new(Security, 'SecTrustRef *')
697 result = Security.SSLCopyPeerTrust(
698 session_context,
699 trust_ref_pointer
700 )
701 handle_sec_error(result)
702 trust_ref = unwrap(trust_ref_pointer)
703
704 cf_string_hostname = CFHelpers.cf_string_from_unicode(self._hostname)
705 ssl_policy_ref = Security.SecPolicyCreateSSL(True, cf_string_hostname)
706 result = CoreFoundation.CFRelease(cf_string_hostname)
707 handle_cf_error(result)
708

1022 def read(self, max_length):
1023 """
1024 Reads data from the TLS-wrapped socket
1025
1026 :param max_length:
1027 The number of bytes to read - output may be less than this

1047 type_name(max_length)

1052 # buffered data to respond to read requests

1125 def read_until(self, marker):
1126 """
1127 Reads data from the socket until a marker is found. Data read includes
1128 the marker.
1129
1130 :param marker:
1131 A byte string or regex object from re.compile(). Used to determine
1132 when to stop reading. Regex objects are more inefficient since
1133 they must scan the entire byte string of read data each time data
1134 is read off the socket.
1135
1136 :return:
1137 A byte string of the data read, including the marker
1138 """
1139
1140 if not isinstance(marker, byte_cls) and not isinstance(marker, Pattern):
1141 raise TypeError(pretty_message(
1142 '''
1143 marker must be a byte string or compiled regex object, not %s
1144 ''',
1145 type_name(marker)
1146 ))
1147
1148 output = b''
1149
1150 is_regex = isinstance(marker, Pattern)
1151
1152 while True:
1153 if len(self._decrypted_bytes) > 0:
1154 chunk = self._decrypted_bytes
1155 self._decrypted_bytes = b''
1156 else:
1157 to_read = self._os_buffered_size() or 8192
1158 chunk = self.read(to_read)
1159
1160 offset = len(output)
1161 output += chunk
1162
1163 if is_regex:
1164 match = marker.search(output)
1165 if match is not None:
1166 end = match.end()
1167 break
1168 else:
1169 # If the marker was not found last time, we have to start
1170 # at a position where the marker would have its final char
1171 # in the newly read chunk
1172 start = max(0, offset - len(marker) - 1)
1173 match = output.find(marker, start)
1174 if match != -1:
1175 end = match + len(marker)
1176 break
1177
1178 self._decrypted_bytes = output[end:] + self._decrypted_bytes
1179 return output[0:end]
