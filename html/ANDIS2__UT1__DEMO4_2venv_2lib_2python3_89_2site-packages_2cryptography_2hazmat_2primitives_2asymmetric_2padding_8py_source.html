103def calculate_max_pss_salt_length(

106) -> int:
107 if not isinstance(key, (rsa.RSAPrivateKey, rsa.RSAPublicKey)):
108 raise TypeError("key must be an RSA public or private key")
109 # bit length - 1 per RFC 3447
110 emlen = (key.key_size + 6) // 8
