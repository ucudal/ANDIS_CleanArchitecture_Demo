76def process_template_file(contents):
77 ret = []
78 template_lines = []
79
80 append_to = ret
81 for line in contents.splitlines(keepends=False):
82 if line.strip() == "### TEMPLATE_START":
83 append_to = template_lines
84 elif line.strip() == "### TEMPLATE_END":
85 append_to = ret
86 for line in process_template_lines(template_lines):
87 ret.append(line)
88 else:
89 append_to.append(line)
90
