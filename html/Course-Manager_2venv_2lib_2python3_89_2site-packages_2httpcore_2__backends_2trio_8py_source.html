85 def is_connection_dropped(self) -> bool:
86 # Adapted from: https://github.com/encode/httpx/pull/143#issuecomment-515202982
87 stream = self.stream
88
89 # Peek through any SSLStream wrappers to get the underlying SocketStream.
90 while isinstance(stream, trio.SSLStream):
91 stream = stream.transport_stream
92 assert isinstance(stream, trio.SocketStream)

94 # Counter-intuitively, what we really want to know here is whether the socket is
