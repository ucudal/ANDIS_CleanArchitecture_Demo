17def gnu_getopt(args, shortopts, longopts=[]):
18 """getopt(args, options[, long_options]) -> opts, args
19
20 This function works like getopt(), except that GNU style scanning
21 mode is used by default. This means that option and non-option
22 arguments may be intermixed. The getopt() function stops
23 processing options as soon as a non-option argument is
24 encountered.
25
26 If the first character of the option string is `+', or if the
27 environment variable POSIXLY_CORRECT is set, then option
28 processing stops as soon as a non-option argument is encountered.
29 """
30
31 opts = []
32 prog_args = []
33 if type("") == type(longopts):
34 longopts = [longopts]
35 else:
36 longopts = list(longopts)
37
38 # Allow options after non-option arguments?
39 all_options_first = False
40 if shortopts.startswith("+"):
41 shortopts = shortopts[1:]
42 all_options_first = True
43
44 while args:
45 if args[0] == "--":
46 prog_args += args[1:]
47 break
48
49 if args[0][:2] == "--":
50 opts, args = do_longs(opts, args[0][2:], longopts, args[1:])
51 elif args[0][:1] == "-":
52 opts, args = do_shorts(opts, args[0][1:], shortopts, args[1:])
53 else:
54 if all_options_first:
55 prog_args += args
56 break
57 else:
58 prog_args.append(args[0])
59 args = args[1:]
60
61 return opts, prog_args

64def do_longs(opts, opt, longopts, args):
65 try:
66 i = opt.index("=")
67 except ValueError:
68 optarg = None
69 else:
70 opt, optarg = opt[:i], opt[i + 1 :]
71
72 has_arg, opt = long_has_args(opt, longopts)
73 if has_arg:
74 if optarg is None:
75 if not args:
76 raise GetoptError("option --%s requires argument" % opt, opt)
77 optarg, args = args[0], args[1:]
78 elif optarg:
79 raise GetoptError("option --%s must not have an argument" % opt, opt)
80 opts.append(("--" + opt, optarg or ""))
81 return opts, args

87def long_has_args(opt, longopts):
88 possibilities = [o for o in longopts if o.startswith(opt)]
89 if not possibilities:
90 raise GetoptError("option --%s not recognized" % opt, opt)
91 # Is there an exact match?
92 if opt in possibilities:
93 return False, opt
94 elif opt + "=" in possibilities:
95 return True, opt
96 # No exact match, so better be unique.
97 if len(possibilities) > 1:
98 # XXX since possibilities contains all valid continuations, might be
99 # nice to work them into the error msg
100 raise GetoptError("option --%s not a unique prefix" % opt, opt)
101 assert len(possibilities) == 1
102 unique_match = possibilities[0]
103 has_arg = unique_match.endswith("=")
104 if has_arg:
105 unique_match = unique_match[:-1]
106 return has_arg, unique_match

109def do_shorts(opts, optstring, shortopts, args):
110 while optstring != "":
111 opt, optstring = optstring[0], optstring[1:]
112 if short_has_arg(opt, shortopts):
113 if optstring == "":
114 if not args:
115 raise GetoptError("option -%s requires argument" % opt, opt)
116 optstring, args = args[0], args[1:]
117 optarg, optstring = optstring, ""
118 else:
119 optarg = ""
120 opts.append(("-" + opt, optarg))
121 return opts, args

124def short_has_arg(opt, shortopts):
125 for i in range(len(shortopts)):
126 if opt == shortopts[i] != ":":
