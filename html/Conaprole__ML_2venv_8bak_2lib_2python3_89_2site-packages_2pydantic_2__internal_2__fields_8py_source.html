74def collect_model_fields( # noqa: C901
75 cls: type[BaseModel],
76 bases: tuple[type[Any], ...],
77 config_wrapper: ConfigWrapper,
78 ns_resolver: NsResolver | None,
79 *,
80 typevars_map: dict[Any, Any] | None = None,
81) -> tuple[dict[str, FieldInfo], set[str]]:
82 """Collect the fields of a nascent pydantic model.
83
84 Also collect the names of any ClassVars present in the type hints.
85
86 The returned value is a tuple of two items: the fields dict, and the set of ClassVar names.
87
88 Args:
89 cls: BaseModel or dataclass.
90 bases: Parents of the class, generally `cls.__bases__`.
91 config_wrapper: The config wrapper instance.
92 ns_resolver: Namespace resolver to use when getting model annotations.
93 typevars_map: A dictionary mapping type variables to their concrete types.
94
95 Returns:
96 A tuple contains fields and class variables.
97
98 Raises:
99 NameError:
100 - If there is a conflict between a field name and protected namespaces.
101 - If there is a field other than `root` in `RootModel`.
102 - If a field shadows an attribute in the parent model.
103 """
104 BaseModel = import_cached_base_model()
105 FieldInfo_ = import_cached_field_info()
106
107 parent_fields_lookup: dict[str, FieldInfo] = {}
108 for base in reversed(bases):
109 if model_fields := getattr(base, '__pydantic_fields__', None):
110 parent_fields_lookup.update(model_fields)
111
112 type_hints = _typing_extra.get_model_type_hints(cls, ns_resolver=ns_resolver)
113
114 # https://docs.python.org/3/howto/annotations.html#accessing-the-annotations-dict-of-an-object-in-python-3-9-and-older
115 # annotations is only used for finding fields in parent classes
116 annotations = cls.__dict__.get('__annotations__', {})
117 fields: dict[str, FieldInfo] = {}
118
119 class_vars: set[str] = set()
120 for ann_name, (ann_type, evaluated) in type_hints.items():
121 if ann_name == 'model_config':
122 # We never want to treat `model_config` as a field
123 # Note: we may need to change this logic if/when we introduce a `BareModel` class with no
124 # protected namespaces (where `model_config` might be allowed as a field name)
125 continue
126
127 for protected_namespace in config_wrapper.protected_namespaces:
128 ns_violation: bool = False
129 if isinstance(protected_namespace, Pattern):
130 ns_violation = protected_namespace.match(ann_name) is not None
131 elif isinstance(protected_namespace, str):
132 ns_violation = ann_name.startswith(protected_namespace)
133
134 if ns_violation:
135 for b in bases:
136 if hasattr(b, ann_name):
137 if not (issubclass(b, BaseModel) and ann_name in getattr(b, '__pydantic_fields__', {})):
138 raise NameError(
139 f'Field "{ann_name}" conflicts with member {getattr(b, ann_name)}'
140 f' of protected namespace "{protected_namespace}".'
141 )
142 else:
143 valid_namespaces = ()
144 for pn in config_wrapper.protected_namespaces:
145 if isinstance(pn, Pattern):
146 if not pn.match(ann_name):
147 valid_namespaces += (f're.compile({pn.pattern})',)
148 else:
149 if not ann_name.startswith(pn):
150 valid_namespaces += (pn,)
151
152 warnings.warn(
153 f'Field "{ann_name}" in {cls.__name__} has conflict with protected namespace "{protected_namespace}".'
154 '\n\nYou may be able to resolve this warning by setting'
155 f" `model_config['protected_namespaces'] = {valid_namespaces}`.",
156 UserWarning,
157 )
158 if _typing_extra.is_classvar_annotation(ann_type):
159 class_vars.add(ann_name)
160 continue
161 if _is_finalvar_with_default_val(ann_type, getattr(cls, ann_name, PydanticUndefined)):
162 class_vars.add(ann_name)
163 continue
164 if not is_valid_field_name(ann_name):
165 continue
166 if cls.__pydantic_root_model__ and ann_name != 'root':
167 raise NameError(
168 f"Unexpected field with name {ann_name!r}; only 'root' is allowed as a field of a `RootModel`"
169 )
170
171 # when building a generic model with `MyModel[int]`, the generic_origin check makes sure we don't get
172 # "... shadows an attribute" warnings
173 generic_origin = getattr(cls, '__pydantic_generic_metadata__', {}).get('origin')
174 for base in bases:
175 dataclass_fields = {
176 field.name for field in (dataclasses.fields(base) if dataclasses.is_dataclass(base) else ())
177 }
178 if hasattr(base, ann_name):
179 if base is generic_origin:
180 # Don't warn when "shadowing" of attributes in parametrized generics
181 continue
182
183 if ann_name in dataclass_fields:
184 # Don't warn when inheriting stdlib dataclasses whose fields are "shadowed" by defaults being set
185 # on the class instance.
186 continue
187
188 if ann_name not in annotations:
189 # Don't warn when a field exists in a parent class but has not been defined in the current class
190 continue
191
192 warnings.warn(
193 f'Field name "{ann_name}" in "{cls.__qualname__}" shadows an attribute in parent '
194 f'"{base.__qualname__}"',
195 UserWarning,
196 )
197
198 try:
199 default = getattr(cls, ann_name, PydanticUndefined)
200 if default is PydanticUndefined:
201 raise AttributeError
202 except AttributeError:
203 if ann_name in annotations:
204 field_info = FieldInfo_.from_annotation(ann_type)
205 field_info.evaluated = evaluated
206 else:
207 # if field has no default value and is not in __annotations__ this means that it is
208 # defined in a base class and we can take it from there
209 if ann_name in parent_fields_lookup:
210 # The field was present on one of the (possibly multiple) base classes
211 # copy the field to make sure typevar substitutions don't cause issues with the base classes
212 field_info = copy(parent_fields_lookup[ann_name])
213 else:
214 # The field was not found on any base classes; this seems to be caused by fields not getting
215 # generated thanks to models not being fully defined while initializing recursive models.
216 # Nothing stops us from just creating a new FieldInfo for this type hint, so we do this.
217 field_info = FieldInfo_.from_annotation(ann_type)
218 field_info.evaluated = evaluated
219 else:
220 _warn_on_nested_alias_in_annotation(ann_type, ann_name)
221 if isinstance(default, FieldInfo_) and ismethoddescriptor(default.default):
222 # the `getattr` call above triggers a call to `__get__` for descriptors, so we do
223 # the same if the `= field(default=...)` form is used. Note that we only do this
224 # for method descriptors for now, we might want to extend this to any descriptor
225 # in the future (by simply checking for `hasattr(default.default, '__get__')`).
226 default.default = default.default.__get__(None, cls)
227
228 field_info = FieldInfo_.from_annotated_attribute(ann_type, default)
229 field_info.evaluated = evaluated
230 # attributes which are fields are removed from the class namespace:
231 # 1. To match the behaviour of annotation-only fields
232 # 2. To avoid false positives in the NameError check above
233 try:
234 delattr(cls, ann_name)
235 except AttributeError:
236 pass # indicates the attribute was on a parent class
237
238 # Use cls.__dict__['__pydantic_decorators__'] instead of cls.__pydantic_decorators__
239 # to make sure the decorators have already been built for this exact class
240 decorators: DecoratorInfos = cls.__dict__['__pydantic_decorators__']
241 if ann_name in decorators.computed_fields:
242 raise ValueError("you can't override a field with a computed field")
243 fields[ann_name] = field_info
244
245 if typevars_map:
246 for field in fields.values():
247 field.apply_typevars_map(typevars_map)
248
249 _update_fields_from_docstrings(cls, fields, config_wrapper)
250 return fields, class_vars

253def _warn_on_nested_alias_in_annotation(ann_type: type[Any], ann_name: str) -> None:
254 FieldInfo = import_cached_field_info()
255
256 args = getattr(ann_type, '__args__', None)
257 if args:
258 for anno_arg in args:
259 if _typing_extra.is_annotated(anno_arg):
260 for anno_type_arg in _typing_extra.get_args(anno_arg):
261 if isinstance(anno_type_arg, FieldInfo) and anno_type_arg.alias is not None:
262 warnings.warn(
263 f'`alias` specification on field "{ann_name}" must be set on outermost annotation to take effect.',
264 UserWarning,
265 )
266 return

282def collect_dataclass_fields(
283 cls: type[StandardDataclass],
284 *,
285 ns_resolver: NsResolver | None = None,
286 typevars_map: dict[Any, Any] | None = None,
287 config_wrapper: ConfigWrapper | None = None,
288) -> dict[str, FieldInfo]:
289 """Collect the fields of a dataclass.
290
291 Args:
292 cls: dataclass.
293 ns_resolver: Namespace resolver to use when getting dataclass annotations.
294 Defaults to an empty instance.
295 typevars_map: A dictionary mapping type variables to their concrete types.
296 config_wrapper: The config wrapper instance.
297
298 Returns:
299 The dataclass fields.
300 """
301 FieldInfo_ = import_cached_field_info()
302
303 fields: dict[str, FieldInfo] = {}
304 ns_resolver = ns_resolver or NsResolver()
305 dataclass_fields = cls.__dataclass_fields__
306
307 # The logic here is similar to `_typing_extra.get_cls_type_hints`,
308 # although we do it manually as stdlib dataclasses already have annotations
309 # collected in each class:
310 for base in reversed(cls.__mro__):
311 if not dataclasses.is_dataclass(base):
312 continue
313
314 with ns_resolver.push(base):
315 for ann_name, dataclass_field in dataclass_fields.items():
316 if ann_name not in base.__dict__.get('__annotations__', {}):
317 # `__dataclass_fields__`contains every field, even the ones from base classes.
318 # Only collect the ones defined on `base`.
319 continue
320
321 globalns, localns = ns_resolver.types_namespace
322 ann_type, _ = _typing_extra.try_eval_type(dataclass_field.type, globalns, localns)
323
324 if _typing_extra.is_classvar_annotation(ann_type):
325 continue
326
327 if (
328 not dataclass_field.init
329 and dataclass_field.default is dataclasses.MISSING
330 and dataclass_field.default_factory is dataclasses.MISSING
331 ):
332 # TODO: We should probably do something with this so that validate_assignment behaves properly
333 # Issue: https://github.com/pydantic/pydantic/issues/5470
334 continue
335
336 if isinstance(dataclass_field.default, FieldInfo_):
337 if dataclass_field.default.init_var:
338 if dataclass_field.default.init is False:
339 raise PydanticUserError(
340 f'Dataclass field {ann_name} has init=False and init_var=True, but these are mutually exclusive.',
341 code='clashing-init-and-init-var',
342 )
343
344 # TODO: same note as above re validate_assignment
345 continue
346 field_info = FieldInfo_.from_annotated_attribute(ann_type, dataclass_field.default)
347 else:
348 field_info = FieldInfo_.from_annotated_attribute(ann_type, dataclass_field)
349
350 fields[ann_name] = field_info
351
352 if field_info.default is not PydanticUndefined and isinstance(
353 getattr(cls, ann_name, field_info), FieldInfo_
354 ):
355 # We need this to fix the default when the "default" from __dataclass_fields__ is a pydantic.FieldInfo
356 setattr(cls, ann_name, field_info.default)
357
358 if typevars_map:
359 for field in fields.values():
360 # We don't pass any ns, as `field.annotation`
361 # was already evaluated. TODO: is this method relevant?
362 # Can't we juste use `_generics.replace_types`?
363 field.apply_typevars_map(typevars_map)
364
365 if config_wrapper is not None:
366 _update_fields_from_docstrings(cls, fields, config_wrapper)
367
368 return fields

379def takes_validated_data_argument(
380 default_factory: Callable[[], Any] | Callable[[dict[str, Any]], Any],
381) -> TypeIs[Callable[[dict[str, Any]], Any]]:
382 """Whether the provided default factory callable has a validated data parameter."""
383 try:
384 sig = signature(default_factory)
385 except (ValueError, TypeError):
386 # `inspect.signature` might not be able to infer a signature, e.g. with C objects.
387 # In this case, we assume no data argument is present:
388 return False
389
390 parameters = list(sig.parameters.values())
391
392 return len(parameters) == 1 and can_be_positional(parameters[0]) and parameters[0].default is Parameter.empty
