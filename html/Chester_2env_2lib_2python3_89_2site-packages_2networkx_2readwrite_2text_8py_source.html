59def generate_network_text(
60 graph, with_labels=True, sources=None, max_depth=None, ascii_only=False
61):
62 """Generate lines in the "network text" format
63
64 This works via a depth-first traversal of the graph and writing a line for
65 each unique node encountered. Non-tree edges are written to the right of
66 each node, and connection to a non-tree edge is indicated with an ellipsis.
67 This representation works best when the input graph is a forest, but any
68 graph can be represented.
69
70 This notation is original to networkx, although it is simple enough that it
71 may be known in existing literature. See #5602 for details. The procedure
72 is summarized as follows:
73
74 1. Given a set of source nodes (which can be specified, or automatically
75 discovered via finding the (strongly) connected components and choosing one
76 node with minimum degree from each), we traverse the graph in depth first
77 order.
78
79 2. Each reachable node will be printed exactly once on it's own line.
80
81 3. Edges are indicated in one of three ways:
82
83 a. a parent "L-style" connection on the upper left. This corresponds to
84 a traversal in the directed DFS tree.
85
86 b. a backref "<-style" connection shown directly on the right. For
87 directed graphs, these are drawn for any incoming edges to a node that
88 is not a parent edge. For undirected graphs, these are drawn for only
89 the non-parent edges that have already been represented (The edges that
90 have not been represented will be handled in the recursive case).
91
92 c. a child "L-style" connection on the lower right. Drawing of the
93 children are handled recursively.
94
95 4. The children of each node (wrt the directed DFS tree) are drawn
96 underneath and to the right of it. In the case that a child node has already
97 been drawn the connection is replaced with an ellipsis ("...") to indicate
98 that there is one or more connections represented elsewhere.
99
100 5. If a maximum depth is specified, an edge to nodes past this maximum
101 depth will be represented by an ellipsis.
102
103 Parameters
104 ----------
105 graph : nx.DiGraph | nx.Graph
106 Graph to represent
107
108 with_labels : bool | str
109 If True will use the "label" attribute of a node to display if it
110 exists otherwise it will use the node value itself. If given as a
111 string, then that attribute name will be used instead of "label".
112 Defaults to True.
113
114 sources : List
115 Specifies which nodes to start traversal from. Note: nodes that are not
116 reachable from one of these sources may not be shown. If unspecified,
117 the minimal set of nodes needed to reach all others will be used.
118
119 max_depth : int | None
120 The maximum depth to traverse before stopping. Defaults to None.
121
122 ascii_only : Boolean
123 If True only ASCII characters are used to construct the visualization
124
125 Yields
126 ------
127 str : a line of generated text
128 """
129 is_directed = graph.is_directed()
130
131 if is_directed:
132 glyphs = AsciiDirectedGlyphs if ascii_only else UtfDirectedGlyphs
133 succ = graph.succ
134 pred = graph.pred
135 else:
136 glyphs = AsciiUndirectedGlyphs if ascii_only else UtfUndirectedGlyphs
137 succ = graph.adj
138 pred = graph.adj
139
140 if isinstance(with_labels, str):
141 label_attr = with_labels
142 elif with_labels:
143 label_attr = "label"
144 else:
145 label_attr = None
146
147 if max_depth == 0:
148 yield glyphs.empty + " ..."
149 elif len(graph.nodes) == 0:
150 yield glyphs.empty
151 else:
152 # If the nodes to traverse are unspecified, find the minimal set of
153 # nodes that will reach the entire graph
154 if sources is None:
155 sources = _find_sources(graph)
156
157 # Populate the stack with each:
158 # 1. parent node in the DFS tree (or None for root nodes),
159 # 2. the current node in the DFS tree
160 # 2. a list of indentations indicating depth
161 # 3. a flag indicating if the node is the final one to be written.
162 # Reverse the stack so sources are popped in the correct order.
163 last_idx = len(sources) - 1
164 stack = [
165 (None, node, [], (idx == last_idx)) for idx, node in enumerate(sources)
166 ][::-1]
167
168 num_skipped_children = defaultdict(lambda: 0)
169 seen_nodes = set()
170 while stack:
171 parent, node, indents, this_islast = stack.pop()
172
173 if node is not Ellipsis:
174 skip = node in seen_nodes
175 if skip:
176 # Mark that we skipped a parent's child
177 num_skipped_children[parent] += 1
178
179 if this_islast:
180 # If we reached the last child of a parent, and we skipped
181 # any of that parents children, then we should emit an
182 # ellipsis at the end after this.
183 if num_skipped_children[parent] and parent is not None:
184 # Append the ellipsis to be emitted last
185 next_islast = True
186 try_frame = (node, Ellipsis, indents, next_islast)
187 stack.append(try_frame)
188
189 # Redo this frame, but not as a last object
190 next_islast = False
191 try_frame = (parent, node, indents, next_islast)
192 stack.append(try_frame)
193 continue
194
195 if skip:
196 continue
197 seen_nodes.add(node)
198
199 if not indents:
200 # Top level items (i.e. trees in the forest) get different
201 # glyphs to indicate they are not actually connected
202 if this_islast:
203 this_prefix = indents + [glyphs.newtree_last]
204 next_prefix = indents + [glyphs.endof_forest]
205 else:
206 this_prefix = indents + [glyphs.newtree_mid]
207 next_prefix = indents + [glyphs.within_forest]
208
209 else:
210 # For individual tree edges distinguish between directed and
211 # undirected cases
212 if this_islast:
213 this_prefix = indents + [glyphs.last]
214 next_prefix = indents + [glyphs.endof_forest]
215 else:
216 this_prefix = indents + [glyphs.mid]
217 next_prefix = indents + [glyphs.within_tree]
218
219 if node is Ellipsis:
220 label = " ..."
221 suffix = ""
222 children = []
223 else:
224 if label_attr is not None:
225 label = str(graph.nodes[node].get(label_attr, node))
226 else:
227 label = str(node)
228
229 # Determine:
230 # (1) children to traverse into after showing this node.
231 # (2) parents to immediately show to the right of this node.
232 if is_directed:
233 # In the directed case we must show every successor node
234 # note: it may be skipped later, but we don't have that
235 # information here.
236 children = list(succ[node])
237 # In the directed case we must show every predecessor
238 # except for parent we directly traversed from.
239 handled_parents = {parent}
240 else:
241 # Showing only the unseen children results in a more
242 # concise representation for the undirected case.
243 children = [
244 child for child in succ[node] if child not in seen_nodes
245 ]
246
247 # In the undirected case, parents are also children, so we
248 # only need to immediately show the ones we can no longer
249 # traverse
250 handled_parents = {*children, parent}
251
252 if max_depth is not None and len(indents) == max_depth - 1:
253 # Use ellipsis to indicate we have reached maximum depth
254 if children:
255 children = [Ellipsis]
256 handled_parents = {parent}
257
258 # The other parents are other predecessors of this node that
259 # are not handled elsewhere.
260 other_parents = [p for p in pred[node] if p not in handled_parents]
261 if other_parents:
262 if label_attr is not None:
263 other_parents_labels = ", ".join(
264 [
265 str(graph.nodes[p].get(label_attr, p))
266 for p in other_parents
267 ]
268 )
269 else:
270 other_parents_labels = ", ".join(
271 [str(p) for p in other_parents]
272 )
273 suffix = " ".join(["", glyphs.backedge, other_parents_labels])
274 else:
275 suffix = ""
276
277 # Emit the line for this node, this will be called for each node
278 # exactly once.
279 yield "".join(this_prefix + [label, suffix])
280
281 # Push children on the stack in reverse order so they are popped in
282 # the original order.
283 for idx, child in enumerate(children[::-1]):
284 next_islast = idx == 0
285 try_frame = (node, child, next_prefix, next_islast)
286 stack.append(try_frame)

429def _find_sources(graph):
430 """
431 Determine a minimal set of nodes such that the entire graph is reachable
432 """
433 # For each connected part of the graph, choose at least
434 # one node as a starting point, preferably without a parent
435 if graph.is_directed():
436 # Choose one node from each SCC with minimum in_degree
437 sccs = list(nx.strongly_connected_components(graph))
438 # condensing the SCCs forms a dag, the nodes in this graph with
439 # 0 in-degree correspond to the SCCs from which the minimum set
440 # of nodes from which all other nodes can be reached.
441 scc_graph = nx.condensation(graph, sccs)
442 supernode_to_nodes = {sn: [] for sn in scc_graph.nodes()}
443 # Note: the order of mapping differs between pypy and cpython
444 # so we have to loop over graph nodes for consistency
445 mapping = scc_graph.graph["mapping"]
446 for n in graph.nodes:
447 sn = mapping[n]
448 supernode_to_nodes[sn].append(n)
449 sources = []
450 for sn in scc_graph.nodes():
451 if scc_graph.in_degree[sn] == 0:
452 scc = supernode_to_nodes[sn]
453 node = min(scc, key=lambda n: graph.in_degree[n])
454 sources.append(node)
455 else:
456 # For undirected graph, the entire graph will be reachable as
457 # long as we consider one node from every connected component
458 sources = [
459 min(cc, key=lambda n: graph.degree[n])
460 for cc in nx.connected_components(graph)
461 ]
462 sources = sorted(sources, key=lambda n: graph.degree[n])
463 return sources
