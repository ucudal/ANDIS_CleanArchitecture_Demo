69def get_type_ref(type_: type[Any], args_override: tuple[type[Any], ...] | None = None) -> str:
70 """Produces the ref to be used for this type by pydantic_core's core schemas.
71
72 This `args_override` argument was added for the purpose of creating valid recursive references
73 when creating generic models without needing to create a concrete class.
74 """
75 origin = get_origin(type_) or type_
76
77 args = get_args(type_) if is_generic_alias(type_) else (args_override or ())
78 generic_metadata = getattr(type_, '__pydantic_generic_metadata__', None)
79 if generic_metadata:
80 origin = generic_metadata['origin'] or origin
81 args = generic_metadata['args'] or args
82
83 module_name = getattr(origin, '__module__', '<No __module__>')
84 if is_type_alias_type(origin):
85 type_ref = f'{module_name}.{origin.__name__}:{id(origin)}'
86 else:
87 try:
88 qualname = getattr(origin, '__qualname__', f'<No __qualname__: {origin}>')
89 except Exception:
90 qualname = getattr(origin, '__qualname__', '<No __qualname__>')
91 type_ref = f'{module_name}.{qualname}:{id(origin)}'
92
93 arg_refs: list[str] = []
94 for arg in args:
95 if isinstance(arg, str):
96 # Handle string literals as a special case; we may be able to remove this special handling if we
97 # wrap them in a ForwardRef at some point.
98 arg_ref = f'{arg}:str-{id(arg)}'
99 else:
100 arg_ref = f'{_repr.display_as_type(arg)}:{id(arg)}'
101 arg_refs.append(arg_ref)
102 if arg_refs:
103 type_ref = f'{type_ref}[{",".join(arg_refs)}]'
104 return type_ref

128def define_expected_missing_refs(
129 schema: core_schema.CoreSchema, allowed_missing_refs: set[str]
130) -> core_schema.CoreSchema | None:
131 if not allowed_missing_refs:
132 # in this case, there are no missing refs to potentially substitute, so there's no need to walk the schema
133 # this is a common case (will be hit for all non-generic models), so it's worth optimizing for

183 def _build_schema_type_to_method(self) -> dict[core_schema.CoreSchemaType, Recurse]:
184 mapping: dict[core_schema.CoreSchemaType, Recurse] = {}
185 key: core_schema.CoreSchemaType

194 def _walk(self, schema: core_schema.CoreSchema, f: Walk) -> core_schema.CoreSchema:
195 schema = self._schema_type_to_method[schema['type']](self._copy_schema(schema), f)
196 ser_schema: core_schema.SerSchema | None = schema.get('serialization') # type: ignore
197 if ser_schema:

199 return schema

202 sub_schema = schema.get('schema', None)

207 def _handle_ser_schemas(self, ser_schema: core_schema.SerSchema, f: Walk) -> core_schema.SerSchema:
208 schema: core_schema.CoreSchema | None = ser_schema.get('schema', None)
209 return_schema: core_schema.CoreSchema | None = ser_schema.get('return_schema', None)
210 if schema is not None or return_schema is not None:
211 ser_schema = self._copy_schema(ser_schema)
212 if schema is not None:
213 ser_schema['schema'] = self.walk(schema, f) # type: ignore
214 if return_schema is not None:

218 def handle_definitions_schema(self, schema: core_schema.DefinitionsSchema, f: Walk) -> core_schema.CoreSchema:
219 new_definitions: list[core_schema.CoreSchema] = []
220 for definition in schema['definitions']:
221 if 'schema_ref' in definition and 'ref' in definition:
222 # This indicates a purposely indirect reference
223 # We want to keep such references around for implications related to JSON schema, etc.:
224 new_definitions.append(definition)
225 # However, we still need to walk the referenced definition:
226 self.walk(definition, f)
227 continue
228
229 updated_definition = self.walk(definition, f)
230 if 'ref' in updated_definition:
231 # If the updated definition schema doesn't have a 'ref', it shouldn't go in the definitions
232 # This is most likely to happen due to replacing something with a definition reference, in
233 # which case it should certainly not go in the definitions list
234 new_definitions.append(updated_definition)
235 new_inner_schema = self.walk(schema['schema'], f)
236
237 if not new_definitions and len(schema) == 3:
238 # This means we'd be returning a "trivial" definitions schema that just wrapped the inner schema
239 return new_inner_schema
240
241 new_schema = self._copy_schema(schema)
242 new_schema['schema'] = new_inner_schema
243 new_schema['definitions'] = new_definitions
244 return new_schema

424def walk_core_schema(schema: core_schema.CoreSchema, f: Walk, *, copy: bool = True) -> core_schema.CoreSchema:
425 """Recursively traverse a CoreSchema.
426
427 Args:
428 schema (core_schema.CoreSchema): The CoreSchema to process, it will not be modified.
429 f (Walk): A function to apply. This function takes two arguments:
430 1. The current CoreSchema that is being processed
431 (not the same one you passed into this function, one level down).
432 2. The "next" `f` to call. This lets you for example use `f=functools.partial(some_method, some_context)`
433 to pass data down the recursive calls without using globals or other mutable state.
434 copy: Whether schema should be recursively copied.
435
436 Returns:
437 core_schema.CoreSchema: A processed CoreSchema.
438 """
439 return f(schema.copy() if copy else schema, _dispatch if copy else _dispatch_no_copy)

442def simplify_schema_references(schema: core_schema.CoreSchema) -> core_schema.CoreSchema: # noqa: C901
443 definitions: dict[str, core_schema.CoreSchema] = {}
444 ref_counts: dict[str, int] = defaultdict(int)
445 involved_in_recursion: dict[str, bool] = {}
446 current_recursion_ref_count: dict[str, int] = defaultdict(int)
447
448 def collect_refs(s: core_schema.CoreSchema, recurse: Recurse) -> core_schema.CoreSchema:
449 if s['type'] == 'definitions':
450 for definition in s['definitions']:
451 ref = get_ref(definition)
452 assert ref is not None
453 if ref not in definitions:
454 definitions[ref] = definition
455 recurse(definition, collect_refs)
456 return recurse(s['schema'], collect_refs)
457 else:
458 ref = get_ref(s)
459 if ref is not None:
460 new = recurse(s, collect_refs)
461 new_ref = get_ref(new)
462 if new_ref:
463 definitions[new_ref] = new
464 return core_schema.definition_reference_schema(schema_ref=ref)
465 else:
466 return recurse(s, collect_refs)
467
468 schema = walk_core_schema(schema, collect_refs)
469
470 def count_refs(s: core_schema.CoreSchema, recurse: Recurse) -> core_schema.CoreSchema:
471 if s['type'] != 'definition-ref':
472 return recurse(s, count_refs)
473 ref = s['schema_ref']
474 ref_counts[ref] += 1
475
476 if ref_counts[ref] >= 2:
477 # If this model is involved in a recursion this should be detected
478 # on its second encounter, we can safely stop the walk here.
479 if current_recursion_ref_count[ref] != 0:
480 involved_in_recursion[ref] = True
481 return s
482
483 current_recursion_ref_count[ref] += 1
484 if 'serialization' in s:
485 # Even though this is a `'definition-ref'` schema, there might
486 # be more references inside the serialization schema:
487 recurse(s, count_refs)
488
489 next_s = definitions[ref]
490 visited: set[str] = set()
491 while next_s['type'] == 'definition-ref':
492 if next_s['schema_ref'] in visited:
493 raise PydanticUserError(
494 f'{ref} contains a circular reference to itself.', code='circular-reference-schema'
495 )
496
497 visited.add(next_s['schema_ref'])
498 ref_counts[next_s['schema_ref']] += 1
499 next_s = definitions[next_s['schema_ref']]
500
501 recurse(next_s, count_refs)
502 current_recursion_ref_count[ref] -= 1
503 return s
504
505 schema = walk_core_schema(schema, count_refs, copy=False)
506
507 assert all(c == 0 for c in current_recursion_ref_count.values()), 'this is a bug! please report it'
508
509 def can_be_inlined(s: core_schema.DefinitionReferenceSchema, ref: str) -> bool:
510 if ref_counts[ref] > 1:
511 return False
512 if involved_in_recursion.get(ref, False):
513 return False
514 if 'serialization' in s:
515 return False
516 if 'metadata' in s:
517 metadata = s['metadata']
518 for k in [
519 *CoreMetadata.__annotations__.keys(),
520 'pydantic.internal.union_discriminator',
521 'pydantic.internal.tagged_union_tag',
522 ]:
523 if k in metadata:
524 # we need to keep this as a ref
525 return False
526 return True

528 def inline_refs(s: core_schema.CoreSchema, recurse: Recurse) -> core_schema.CoreSchema:
529 # Assume there are no infinite loops, because we already checked for that in `count_refs`
530 while s['type'] == 'definition-ref':
531 ref = s['schema_ref']
532
533 # Check if the reference is only used once, not involved in recursion and does not have
534 # any extra keys (like 'serialization')
535 if can_be_inlined(s, ref):
536 # Inline the reference by replacing the reference with the actual schema
537 new = definitions.pop(ref)
538 ref_counts[ref] -= 1 # because we just replaced it!
539 # put all other keys that were on the def-ref schema into the inlined version
540 # in particular this is needed for `serialization`
541 if 'serialization' in s:
542 new['serialization'] = s['serialization']
543 s = new
544 else:
545 break
546 return recurse(s, inline_refs)
547
548 schema = walk_core_schema(schema, inline_refs, copy=False)
549
550 def_values = [v for v in definitions.values() if ref_counts[v['ref']] > 0] # type: ignore
551
552 if def_values:
553 schema = core_schema.definitions_schema(schema=schema, definitions=def_values)
554 return schema

557def _strip_metadata(schema: CoreSchema) -> CoreSchema:
558 def strip_metadata(s: CoreSchema, recurse: Recurse) -> CoreSchema:
559 s = s.copy()
560 s.pop('metadata', None)
561 if s['type'] == 'model-fields':
562 s = s.copy()
563 s['fields'] = {k: v.copy() for k, v in s['fields'].items()}
564 for field_name, field_schema in s['fields'].items():
565 field_schema.pop('metadata', None)
566 s['fields'][field_name] = field_schema
567 computed_fields = s.get('computed_fields', None)
568 if computed_fields:
569 s['computed_fields'] = [cf.copy() for cf in computed_fields]
570 for cf in computed_fields:
571 cf.pop('metadata', None)
572 else:
573 s.pop('computed_fields', None)
574 elif s['type'] == 'model':
575 # remove some defaults
576 if s.get('custom_init', True) is False:
577 s.pop('custom_init')
578 if s.get('root_model', True) is False:
579 s.pop('root_model')
580 if {'title'}.issuperset(s.get('config', {}).keys()):
581 s.pop('config', None)
582
583 return recurse(s, strip_metadata)
584
585 return walk_core_schema(schema, strip_metadata)
