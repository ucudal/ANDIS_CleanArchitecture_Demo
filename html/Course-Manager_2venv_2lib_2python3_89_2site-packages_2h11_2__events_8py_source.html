31 def __init__(self, **kwargs):
32 _parsed = kwargs.pop("_parsed", False)
33 allowed = set(self._fields)
34 for kwarg in kwargs:
35 if kwarg not in allowed:
36 raise TypeError(
37 "unrecognized kwarg {} for {}".format(
38 kwarg, self.__class__.__name__
39 )
40 )
41 required = allowed.difference(self._defaults)
42 for field in required:
43 if field not in kwargs:
44 raise TypeError(
45 "missing required kwarg {} for {}".format(
46 field, self.__class__.__name__
47 )
48 )
49 self.__dict__.update(self._defaults)
50 self.__dict__.update(kwargs)
51
52 # Special handling for some fields
53
54 if "headers" in self.__dict__:
55 self.headers = _headers.normalize_and_validate(
56 self.headers, _parsed=_parsed
57 )
58
59 if not _parsed:
60 for field in ["method", "target", "http_version", "reason"]:
61 if field in self.__dict__:
62 self.__dict__[field] = bytesify(self.__dict__[field])

66 raise LocalProtocolError("status code must be integer")

95class Request(_EventBundle):
96 """The beginning of an HTTP request.
97
98 Fields:
99
100 .. attribute:: method
101
102 An HTTP method, e.g. ``b"GET"`` or ``b"POST"``. Always a byte
103 string. :term:`Bytes-like objects <bytes-like object>` and native
104 strings containing only ascii characters will be automatically
105 converted to byte strings.
106
107 .. attribute:: target
108
109 The target of an HTTP request, e.g. ``b"/index.html"``, or one of the
110 more exotic formats described in `RFC 7320, section 5.3
111 <https://tools.ietf.org/html/rfc7230#section-5.3>`_. Always a byte
112 string. :term:`Bytes-like objects <bytes-like object>` and native
113 strings containing only ascii characters will be automatically
114 converted to byte strings.
115
116 .. attribute:: headers
117
118 Request headers, represented as a list of (name, value) pairs. See
119 :ref:`the header normalization rules <headers-format>` for details.
120
121 .. attribute:: http_version
122
123 The HTTP protocol version, represented as a byte string like
124 ``b"1.1"``. See :ref:`the HTTP version normalization rules
125 <http_version-format>` for details.
126
127 """
128
129 _fields = ["method", "target", "headers", "http_version"]
130 _defaults = {"http_version": b"1.1"}
131
132 def _validate(self):
133 # "A server MUST respond with a 400 (Bad Request) status code to any
134 # HTTP/1.1 request message that lacks a Host header field and to any
135 # request message that contains more than one Host header field or a
136 # Host header field with an invalid field-value."
137 # -- https://tools.ietf.org/html/rfc7230#section-5.4
138 host_count = 0
139 for name, value in self.headers:
140 if name == b"host":
141 host_count += 1
142 if self.http_version == b"1.1" and host_count == 0:
143 raise LocalProtocolError("Missing mandatory Host: header")
144 if host_count > 1:
145 raise LocalProtocolError("Found multiple Host: headers")
146
147 validate(request_target_re, self.target, "Illegal target characters")

177
