59@contextlib.contextmanager
60def override_config(cov: Coverage, **kwargs: TConfigValueIn) -> Iterator[None]:
61 """Temporarily tweak the configuration of `cov`.
62
63 The arguments are applied to `cov.config` with the `from_args` method.
64 At the end of the with-statement, the old configuration is restored.
65 """
66 original_config = cov.config
67 cov.config = cov.config.copy()
68 try:
69 cov.config.from_args(**kwargs)
70 yield
71 finally:
72 cov.config = original_config

78class Coverage(TConfigurable):
79 """Programmatic access to coverage.py.
80
81 To use::
82
83 from coverage import Coverage
84
85 cov = Coverage()
86 cov.start()
87 #.. call your code ..
88 cov.stop()
89 cov.html_report(directory="covhtml")
90
91 A context manager is available to do the same thing::
92
93 cov = Coverage()
94 with cov.collect():
95 #.. call your code ..
96 cov.html_report(directory="covhtml")
97
98 Note: in keeping with Python custom, names starting with underscore are
99 not part of the public API. They might stop working at any point. Please
100 limit yourself to documented methods to avoid problems.
101
102 Methods can raise any of the exceptions described in :ref:`api_exceptions`.
103
104 """

123 def __init__( # pylint: disable=too-many-arguments
124 self,
125 data_file: FilePath | DefaultValue | None = DEFAULT_DATAFILE,
126 data_suffix: str | bool | None = None,
127 cover_pylib: bool | None = None,
128 auto_data: bool = False,
129 timid: bool | None = None,
130 branch: bool | None = None,
131 config_file: FilePath | bool = True,
132 source: Iterable[str] | None = None,
133 source_pkgs: Iterable[str] | None = None,
134 omit: str | Iterable[str] | None = None,
135 include: str | Iterable[str] | None = None,
136 debug: Iterable[str] | None = None,
137 concurrency: str | Iterable[str] | None = None,
138 check_preimported: bool = False,
139 context: str | None = None,
140 messages: bool = False,
141 plugins: Iterable[Callable[..., None]] | None = None,
142 ) -> None:
143 """
144 Many of these arguments duplicate and override values that can be
145 provided in a configuration file. Parameters that are missing here
146 will use values from the config file.
147
148 `data_file` is the base name of the data file to use. The config value
149 defaults to ".coverage". None can be provided to prevent writing a data
150 file. `data_suffix` is appended (with a dot) to `data_file` to create
151 the final file name. If `data_suffix` is simply True, then a suffix is
152 created with the machine and process identity included.
153
154 `cover_pylib` is a boolean determining whether Python code installed
155 with the Python interpreter is measured. This includes the Python
156 standard library and any packages installed with the interpreter.
157
158 If `auto_data` is true, then any existing data file will be read when
159 coverage measurement starts, and data will be saved automatically when
160 measurement stops.
161
162 If `timid` is true, then a slower and simpler trace function will be
163 used. This is important for some environments where manipulation of
164 tracing functions breaks the faster trace function.
165
166 If `branch` is true, then branch coverage will be measured in addition
167 to the usual statement coverage.
168
169 `config_file` determines what configuration file to read:
170
171 * If it is ".coveragerc", it is interpreted as if it were True,
172 for backward compatibility.
173
174 * If it is a string, it is the name of the file to read. If the
175 file can't be read, it is an error.
176
177 * If it is True, then a few standard files names are tried
178 (".coveragerc", "setup.cfg", "tox.ini"). It is not an error for
179 these files to not be found.
180
181 * If it is False, then no configuration file is read.
182
183 `source` is a list of file paths or package names. Only code located
184 in the trees indicated by the file paths or package names will be
185 measured.
186
187 `source_pkgs` is a list of package names. It works the same as
188 `source`, but can be used to name packages where the name can also be
189 interpreted as a file path.
190
191 `include` and `omit` are lists of file name patterns. Files that match
192 `include` will be measured, files that match `omit` will not. Each
193 will also accept a single string argument.
194
195 `debug` is a list of strings indicating what debugging information is
196 desired.
197
198 `concurrency` is a string indicating the concurrency library being used
199 in the measured code. Without this, coverage.py will get incorrect
200 results if these libraries are in use. Valid strings are "greenlet",
201 "eventlet", "gevent", "multiprocessing", or "thread" (the default).
202 This can also be a list of these strings.
203
204 If `check_preimported` is true, then when coverage is started, the
205 already-imported files will be checked to see if they should be
206 measured by coverage. Importing measured files before coverage is
207 started can mean that code is missed.
208
209 `context` is a string to use as the :ref:`static context
210 <static_contexts>` label for collected data.
211
212 If `messages` is true, some messages will be printed to stdout
213 indicating what is happening.
214
215 If `plugins` are passed, they are an iterable of function objects
216 accepting a `reg` object to register plugins, as described in
217 :ref:`api_plugin`. When they are provided, they will override the
218 plugins found in the coverage configuration file.
219
220 .. versionadded:: 4.0
221 The `concurrency` parameter.
222
223 .. versionadded:: 4.2
224 The `concurrency` parameter can now be a list of strings.
225
226 .. versionadded:: 5.0
227 The `check_preimported` and `context` parameters.
228
229 .. versionadded:: 5.3
230 The `source_pkgs` parameter.
231
232 .. versionadded:: 6.0
233 The `messages` parameter.
234
235 .. versionadded:: 7.7
236 The `plugins` parameter.
237
238 """
239 # Start self.config as a usable default configuration. It will soon be
240 # replaced with the real configuration.
241 self.config = CoverageConfig()
242
243 # data_file=None means no disk file at all. data_file missing means
244 # use the value from the config file.
245 self._no_disk = data_file is None
246 if isinstance(data_file, DefaultValue):
247 data_file = None
248 if data_file is not None:
249 data_file = os.fspath(data_file)
250
251 # This is injectable by tests.
252 self._debug_file: IO[str] | None = None
253
254 self._auto_load = self._auto_save = auto_data
255 self._data_suffix_specified = data_suffix
256
257 # Is it ok for no data to be collected?
258 self._warn_no_data = True
259 self._warn_unimported_source = True
260 self._warn_preimported_source = check_preimported
261 self._no_warn_slugs: set[str] = set()
262 self._messages = messages
263
264 # A record of all the warnings that have been issued.
265 self._warnings: list[str] = []
266
267 # Other instance attributes, set with placebos or placeholders.
268 # More useful objects will be created later.
269 self._debug: DebugControl = NoDebugging()
270 self._inorout: InOrOut | None = None
271 self._plugins: Plugins = Plugins()
272 self._plugin_override = cast(Union[Iterable[TCoverageInit], None], plugins)
273 self._data: CoverageData | None = None
274 self._core: Core | None = None
275 self._collector: Collector | None = None
276 self._metacov = False
277
278 self._file_mapper: Callable[[str], str] = abs_file
279 self._data_suffix = self._run_suffix = None
280 self._exclude_re: dict[str, str] = {}
281 self._old_sigterm: Callable[[int, FrameType | None], Any] | None = None
282
283 # State machine variables:
284 # Have we initialized everything?
285 self._inited = False
286 self._inited_for_start = False
287 # Have we started collecting and not stopped it?
288 self._started = False
289 # Should we write the debug output?
290 self._should_write_debug = True
291
292 # Build our configuration from a number of sources.
293 if not isinstance(config_file, bool):
294 config_file = os.fspath(config_file)
295 self.config = read_coverage_config(
296 config_file=config_file,
297 warn=self._warn,
298 data_file=data_file,
299 cover_pylib=cover_pylib,
300 timid=timid,
301 branch=branch,
302 parallel=bool_or_none(data_suffix),
303 source=source,
304 source_pkgs=source_pkgs,
305 run_omit=omit,
306 run_include=include,
307 debug=debug,
308 report_omit=omit,
309 report_include=include,
310 concurrency=concurrency,
311 context=context,
312 )
313
314 # If we have subprocess measurement happening automatically, then we
315 # want any explicit creation of a Coverage object to mean, this process
316 # is already coverage-aware, so don't auto-measure it. By now, the
317 # auto-creation of a Coverage object has already happened. But we can
318 # find it and tell it not to save its data.
319 if not env.METACOV:
320 _prevent_sub_process_measurement()

322 def _init(self) -> None:
323 """Set all the initial state.
324
325 This is called by the public methods to initialize state. This lets us
326 construct a :class:`Coverage` object, then tweak its state before this
327 function is called.
328
329 """
330 if self._inited:
331 return
332
333 self._inited = True
334
335 # Create and configure the debugging controller.
336 self._debug = DebugControl(self.config.debug, self._debug_file, self.config.debug_file)
337 if self._debug.should("process"):
338 self._debug.write("Coverage._init")
339
340 if "multiprocessing" in (self.config.concurrency or ()):
341 # Multi-processing uses parallel for the subprocesses, so also use
342 # it for the main process.
343 self.config.parallel = True
344
345 # _exclude_re is a dict that maps exclusion list names to compiled regexes.
346 self._exclude_re = {}
347
348 set_relative_directory()
349 if self.config.relative_files:
350 self._file_mapper = relative_filename
351
352 # Load plugins
353 self._plugins = Plugins(self._debug)
354 if self._plugin_override:
355 self._plugins.load_from_callables(self._plugin_override)
356 else:
357 self._plugins.load_from_config(self.config.plugins, self.config)
358
359 # Run configuring plugins.
360 for plugin in self._plugins.configurers:
361 # We need an object with set_option and get_option. Either self or
362 # self.config will do. Choosing randomly stops people from doing
363 # other things with those objects, against the public API. Yes,
364 # this is a bit childish. :)
365 plugin.configure([self, self.config][int(time.time()) % 2])

367 def _post_init(self) -> None:
368 """Stuff to do after everything is initialized."""
369 if self._should_write_debug:
370 self._should_write_debug = False
371 self._write_startup_debug()
372
373 # "[run] _crash" will raise an exception if the value is close by in
374 # the call stack, for testing error handling.
375 if self.config._crash and self.config._crash in short_stack():
376 raise RuntimeError(f"Crashing because called by {self.config._crash}")

378 def _write_startup_debug(self) -> None:
379 """Write out debug info at startup if needed."""
380 wrote_any = False
381 with self._debug.without_callers():
382 if self._debug.should("config"):
383 config_info = self.config.debug_info()
384 write_formatted_info(self._debug.write, "config", config_info)
385 wrote_any = True
386
387 if self._debug.should("sys"):
388 write_formatted_info(self._debug.write, "sys", self.sys_info())
389 for plugin in self._plugins:
390 header = "sys: " + plugin._coverage_plugin_name
391 info = plugin.sys_info()
392 write_formatted_info(self._debug.write, header, info)
393 wrote_any = True
394
395 if self._debug.should("pybehave"):
396 write_formatted_info(self._debug.write, "pybehave", env.debug_info())
397 wrote_any = True
398
399 if wrote_any:
400 write_formatted_info(self._debug.write, "end", ())

414 def _check_include_omit_etc(self, filename: str, frame: FrameType) -> bool:
415 """Check a file name against the include/omit/etc, rules, verbosely.
416
417 Returns a boolean: True if the file should be traced, False if not.
418
419 """
420 assert self._inorout is not None
421 reason = self._inorout.check_include_omit_etc(filename, frame)
422 if self._debug.should("trace"):
423 if not reason:
424 msg = f"Including {filename!r}"
425 else:
426 msg = f"Not including {filename!r}: {reason}"
427 self._debug.write(msg)
428
429 return not reason

431 def _warn(self, msg: str, slug: str | None = None, once: bool = False) -> None:
432 """Use `msg` as a warning.
433
434 For warning suppression, use `slug` as the shorthand.
435
436 If `once` is true, only show this warning once (determined by the
437 slug.)
438
439 """
440 if not self._no_warn_slugs:
441 self._no_warn_slugs = set(self.config.disable_warnings)
442
443 if slug in self._no_warn_slugs:
444 # Don't issue the warning
445 return
446
447 self._warnings.append(msg)
448 if slug:
449 msg = f"{msg} ({slug})"
450 if self._debug.should("pid"):
451 msg = f"[{os.getpid()}] {msg}"
452 warnings.warn(msg, category=CoverageWarning, stacklevel=2)
453
454 if once:
455 assert slug is not None
456 self._no_warn_slugs.add(slug)

482 """Set an option in the configuration.

513 def load(self) -> None:
514 """Load previously-collected coverage data from the data file."""
515 self._init()
516 if self._collector is not None:
517 self._collector.reset()
518 should_skip = self.config.parallel and not os.path.exists(self.config.data_file)
519 if not should_skip:
520 self._init_data(suffix=None)
521 self._post_init()
522 if not should_skip:
523 assert self._data is not None
524 self._data.read()

526 def _init_for_start(self) -> None:
527 """Initialization for start()"""
528 # Construct the collector.
529 concurrency: list[str] = self.config.concurrency or []
530 if "multiprocessing" in concurrency:
531 if self.config.config_file is None:
532 raise ConfigError("multiprocessing requires a configuration file")
533 patch_multiprocessing(rcfile=self.config.config_file)
534
535 dycon = self.config.dynamic_context
536 if not dycon or dycon == "none":
537 context_switchers = []
538 elif dycon == "test_function":
539 context_switchers = [should_start_context_test_function]
540 else:
541 raise ConfigError(f"Don't understand dynamic_context setting: {dycon!r}")
542
543 context_switchers.extend(
544 plugin.dynamic_context for plugin in self._plugins.context_switchers
545 )
546
547 should_start_context = combine_context_switchers(context_switchers)
548
549 self._core = Core(
550 warn=self._warn,
551 config=self.config,
552 dynamic_contexts=(should_start_context is not None),
553 metacov=self._metacov,
554 )
555 self._collector = Collector(
556 core=self._core,
557 should_trace=self._should_trace,
558 check_include=self._check_include_omit_etc,
559 should_start_context=should_start_context,
560 file_mapper=self._file_mapper,
561 branch=self.config.branch,
562 warn=self._warn,
563 concurrency=concurrency,
564 )
565
566 suffix = self._data_suffix_specified
567 if suffix:
568 if not isinstance(suffix, str):
569 # if data_suffix=True, use .machinename.pid.random
570 suffix = True
571 elif self.config.parallel:
572 if suffix is None:
573 suffix = True
574 elif not isinstance(suffix, str):
575 suffix = bool(suffix)
576 else:
577 suffix = None
578
579 self._init_data(suffix)
580
581 assert self._data is not None
582 self._collector.use_data(self._data, self.config.context)
583
584 # Early warning if we aren't going to be able to support plugins.
585 if self._plugins.file_tracers and not self._core.supports_plugins:
586 self._warn(
587 "Plugin file tracers ({}) aren't supported with {}".format(
588 ", ".join(
589 plugin._coverage_plugin_name
590 for plugin in self._plugins.file_tracers
591 ),
592 self._collector.tracer_name(),
593 ),
594 )
595 for plugin in self._plugins.file_tracers:
596 plugin._coverage_enabled = False
597
598 # Create the file classifying substructure.
599 self._inorout = InOrOut(
600 config=self.config,
601 warn=self._warn,
602 debug=(self._debug if self._debug.should("trace") else None),
603 include_namespace_packages=self.config.include_namespace_packages,
604 )
605 self._inorout.plugins = self._plugins
606 self._inorout.disp_class = self._core.file_disposition_class
607
608 # It's useful to write debug info after initing for start.
609 self._should_write_debug = True
610
611 # Register our clean-up handlers.
612 atexit.register(self._atexit)
613 if self.config.sigterm:
614 is_main = (threading.current_thread() == threading.main_thread())
615 if is_main and not env.WINDOWS:
616 # The Python docs seem to imply that SIGTERM works uniformly even
617 # on Windows, but that's not my experience, and this agrees:
618 # https://stackoverflow.com/questions/35772001/x/35792192#35792192
619 self._old_sigterm = signal.signal( # type: ignore[assignment]
620 signal.SIGTERM, self._on_sigterm,
621 )
622
623 def _init_data(self, suffix: str | bool | None) -> None:
624 """Create a data file if we don't have one yet."""
625 if self._data is None:
626 # Create the data file. We do this at construction time so that the
627 # data file will be written into the directory where the process
628 # started rather than wherever the process eventually chdir'd to.
629 ensure_dir_for_file(self.config.data_file)
630 self._data = CoverageData(
631 basename=self.config.data_file,
632 suffix=suffix,
633 warn=self._warn,
634 debug=self._debug,
635 no_disk=self._no_disk,
636 )

638 def start(self) -> None:
639 """Start measuring code coverage.
640
641 Coverage measurement is only collected in functions called after
642 :meth:`start` is invoked. Statements in the same scope as
643 :meth:`start` won't be measured.
644
645 Once you invoke :meth:`start`, you must also call :meth:`stop`
646 eventually, or your process might not shut down cleanly.
647
648 The :meth:`collect` method is a context manager to handle both
649 starting and stopping collection.
650
651 """
652 self._init()
653 if not self._inited_for_start:
654 self._inited_for_start = True
655 self._init_for_start()
656 self._post_init()
657
658 assert self._collector is not None
659 assert self._inorout is not None
660
661 # Issue warnings for possible problems.
662 self._inorout.warn_conflicting_settings()
663
664 # See if we think some code that would eventually be measured has
665 # already been imported.
666 if self._warn_preimported_source:
667 self._inorout.warn_already_imported_files()
668
669 if self._auto_load:
670 self.load()
671
672 self._collector.start()
673 self._started = True
674 self._instances.append(self)
675
676 def stop(self) -> None:
677 """Stop measuring code coverage."""
678 if self._instances:
679 if self._instances[-1] is self:
680 self._instances.pop()
681 if self._started:
682 assert self._collector is not None
683 self._collector.stop()
684 self._started = False

686 @contextlib.contextmanager
687 def collect(self) -> Iterator[None]:
688 """A context manager to start/stop coverage measurement collection.
689
690 .. versionadded:: 7.3
691
692 """
693 self.start()
694 try:
695 yield
696 finally:
697 self.stop() # pragma: nested

699 def _atexit(self, event: str = "atexit") -> None:
700 """Clean up on process shutdown."""
701 if self._debug.should("process"):
702 self._debug.write(f"{event}: pid: {os.getpid()}, instance: {self!r}")
703 if self._started:
704 self.stop()
705 if self._auto_save or event == "sigterm":
706 self.save()

708 def _on_sigterm(self, signum_unused: int, frame_unused: FrameType | None) -> None:
709 """A handler for signal.SIGTERM."""
710 self._atexit("sigterm")
711 # Statements after here won't be seen by metacov because we just wrote
712 # the data, and are about to kill the process.
713 signal.signal(signal.SIGTERM, self._old_sigterm) # pragma: not covered
714 os.kill(os.getpid(), signal.SIGTERM) # pragma: not covered

716 def erase(self) -> None:
717 """Erase previously collected coverage data.
718
719 This removes the in-memory data collected in this session as well as
720 discarding the data file.
721
722 """
723 self._init()
724 self._post_init()
725 if self._collector is not None:
726 self._collector.reset()

1375def process_startup() -> Coverage | None:
1376 """Call this at Python start-up to perhaps measure coverage.
1377
1378 If the environment variable COVERAGE_PROCESS_START is defined, coverage
1379 measurement is started. The value of the variable is the config file
1380 to use.
1381
1382 There are two ways to configure your Python installation to invoke this
1383 function when Python starts:
1384
1385 #. Create or append to sitecustomize.py to add these lines::
1386
1387 import coverage
1388 coverage.process_startup()
1389
1390 #. Create a .pth file in your Python installation containing::
1391
1392 import coverage; coverage.process_startup()
1393
1394 Returns the :class:`Coverage` instance that was started, or None if it was
1395 not started by this call.
1396
1397 """
1398 cps = os.getenv("COVERAGE_PROCESS_START")
1399 if not cps:
1400 # No request for coverage, nothing to do.
1401 return None
1402
1403 # This function can be called more than once in a process. This happens
1404 # because some virtualenv configurations make the same directory visible
1405 # twice in sys.path. This means that the .pth file will be found twice,
1406 # and executed twice, executing this function twice. We set a global
1407 # flag (an attribute on this function) to indicate that coverage.py has
1408 # already been started, so we can avoid doing it twice.
1409 #
1410 # https://github.com/nedbat/coveragepy/issues/340 has more details.
1411
1412 if hasattr(process_startup, "coverage"):
1413 # We've annotated this function before, so we must have already
1414 # started coverage.py in this process. Nothing to do.
1415 return None
1416
1417 cov = Coverage(config_file=cps)
1418 process_startup.coverage = cov # type: ignore[attr-defined]
1419 cov._warn_no_data = False
