55 def _create_like_index_sql(self, model, field):
56 """
57 Return the statement to create an index with varchar operator pattern
58 when the column type is 'varchar' or 'text', otherwise return None.
59 """
60 db_type = field.db_type(connection=self.connection)
61 if db_type is not None and (field.db_index or field.unique):
62 # Fields with database column types of `varchar` and `text` need
63 # a second index that specifies their operator class, which is
64 # needed when performing correct LIKE queries outside the
65 # C locale. See #12234.
66 #
67 # The same doesn't apply to array fields such as varchar[size]
68 # and text[size], so skip them.
69 if '[' in db_type:
70 return None
71 if db_type.startswith('varchar'):
72 return self._create_index_sql(model, [field], suffix='_like', opclasses=['varchar_pattern_ops'])
73 elif db_type.startswith('text'):
74 return self._create_index_sql(model, [field], suffix='_like', opclasses=['text_pattern_ops'])
75 return None

140 def _alter_field(self, model, old_field, new_field, old_type, new_type,
141 old_db_params, new_db_params, strict=False):
142 # Drop indexes on varchar/text/citext columns that are changing to a
143 # different type.
144 if (old_field.db_index or old_field.unique) and (
145 (old_type.startswith('varchar') and not new_type.startswith('varchar')) or
146 (old_type.startswith('text') and not new_type.startswith('text')) or
147 (old_type.startswith('citext') and not new_type.startswith('citext'))
148 ):
149 index_name = self._create_index_name(model._meta.db_table, [old_field.column], suffix='_like')
150 self.execute(self._delete_index_sql(model, index_name))
151
152 super()._alter_field(
153 model, old_field, new_field, old_type, new_type, old_db_params,
154 new_db_params, strict,
155 )
156 # Added an index? Create any PostgreSQL-specific indexes.
157 if ((not (old_field.db_index or old_field.unique) and new_field.db_index) or
158 (not old_field.unique and new_field.unique)):
159 like_index_statement = self._create_like_index_sql(model, new_field)
160 if like_index_statement is not None:
161 self.execute(like_index_statement)
162
163 # Removed an index? Drop any PostgreSQL-specific indexes.
164 if old_field.unique and not (new_field.db_index or new_field.unique):
165 index_to_remove = self._create_index_name(model._meta.db_table, [old_field.column], suffix='_like')
166 self.execute(self._delete_index_sql(model, index_to_remove))
