167 def _get_connection_from_pool(
168 self, origin: Origin
169 ) -> Optional[SyncHTTPConnection]:
170 # Determine expired keep alive connections on this origin.
171 seen_http11 = False
172 pending_connection = None
173 reuse_connection = None
174 connections_to_close = set()

176 for connection in self._connections_for_origin(origin):
177 if connection.is_http11:
178 seen_http11 = True

180 if connection.state == ConnectionState.IDLE:
181 if connection.is_connection_dropped():
182 # IDLE connections that have been dropped should be
183 # removed from the pool.
184 connections_to_close.add(connection)
185 self._remove_from_pool(connection)
186 else:
187 # IDLE connections that are still maintained may
188 # be reused.
189 reuse_connection = connection
190 elif connection.state == ConnectionState.ACTIVE and connection.is_http2:
191 # HTTP/2 connections may be reused.
192 reuse_connection = connection
193 elif connection.state == ConnectionState.PENDING:
194 # Pending connections may potentially be reused.
195 pending_connection = connection
196
197 if reuse_connection is not None:
198 # Mark the connection as READY before we return it, to indicate
199 # that if it is HTTP/1.1 then it should not be re-acquired.
200 reuse_connection.mark_as_ready()
