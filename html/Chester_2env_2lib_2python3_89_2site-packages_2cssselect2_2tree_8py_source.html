29 """
30 A wrapper for an ElementTree :class:`xml.etree.ElementTree.Element`
31 for Selector matching.
32
33 This class should not be instanciated directly.
34 :meth:`from_xml_root` or :meth:`from_html_root` should be used
35 for the root element of a document,
36 and other elements should be accessed (and wrappers generated)
37 using methods such as :meth:`iter_children` and :meth:`iter_subtree`.
38
39 :class:`ElementWrapper` objects compare equal
40 if their underlying :class:`xml.etree.ElementTree.Element` do.
41

207 def iter_subtree(self):
208 """Return an iterator of newly-created :class:`ElementWrapper` objects
209 for the entire subtree rooted at this element,
210 in tree order.
211
212 Unlike in other methods, the element itself *is* included.
213
214 This loops over an entire document:
215
216 .. code-block:: python
217
218 for element in ElementWrapper.from_root(root_etree).iter_subtree():
219 ...
220
221 """
222 stack = [iter([self])]
223 while stack:
224 element = next(stack[-1], None)
225 if element is None:
226 stack.pop()
227 else:
228 yield element
229 stack.append(element.iter_children())

254 def query_all(self, *selectors):
255 """
256 Return elements, in tree order, that match any of the given selectors.
257
258 Selectors are `scoped`_ to the subtree rooted at this element.
259
260 .. _scoped: https://drafts.csswg.org/selectors-4/#scoping
261
262 :param selectors:
263 Each given selector is either a :class:`compiler.CompiledSelector`,
264 or an argument to :func:`compile_selector_list`.
265 :returns:
266 An iterator of newly-created :class:`ElementWrapper` objects.
267
268 """
269 tests = self._compile(selectors)
270 if len(tests) == 1:
271 return filter(tests[0], self.iter_subtree())
272 elif selectors:
273 return (
274 element
275 for element in self.iter_subtree()
276 if any(test(element) for test in tests)
277 )
278 else:
279 return iter(())
