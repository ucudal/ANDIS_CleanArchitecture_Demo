171 def showtraceback(self, exc_tuple=None, *args, **kwargs):
172 # IPython does a lot of clever stuff with Exceptions. However mostly
173 # it is related to IPython running in a terminal instead of an IDE.
174 # (e.g. it prints out snippets of code around the stack trace)
175 # PyDev does a lot of clever stuff too, so leave exception handling
176 # with default print_exc that PyDev can parse and do its clever stuff
177 # with (e.g. it puts links back to the original source code)
178 try:
179 if exc_tuple is None:
180 etype, value, tb = sys.exc_info()
181 else:
182 etype, value, tb = exc_tuple
183 except ValueError:
184 return
185
186 if tb is not None:
187 traceback.print_exception(etype, value, tb)

376 def is_complete(self, string):
377 # Based on IPython 0.10.1
378
379 if string in ("", "\n"):
380 # Prefiltering, eg through ipython0, may return an empty
381 # string although some operations have been accomplished. We
382 # thus want to consider an empty string as a complete
383 # statement.
384 return True
385 else:
386 try:
387 # Add line returns here, to make sure that the statement is
388 # complete (except if '\' was used).
389 # This should probably be done in a different place (like
390 # maybe 'prefilter_input' method? For now, this works.
391 clean_string = string.rstrip("\n")
392 if not clean_string.endswith("\\"):
393 clean_string += "\n\n"
394
395 is_complete = codeop.compile_command(clean_string, "<string>", "exec")
396 except Exception:
397 # XXX: Hack: return True so that the
398 # code gets executed and the error captured.
399 is_complete = True
