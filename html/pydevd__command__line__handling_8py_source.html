129def process_command_line(argv):
130 """parses the arguments.
131 removes our arguments from the command line"""
132 setup = {}
133 for handler in ACCEPTED_ARG_HANDLERS:
134 setup[handler.arg_name] = handler.default_val
135 setup["file"] = ""
136 setup["qt-support"] = ""
137
138 initial_argv = tuple(argv)
139
140 i = 0
141 del argv[0]
142 while i < len(argv):
143 handler = ARGV_REP_TO_HANDLER.get(argv[i])
144 if handler is not None:
145 handler.handle_argv(argv, i, setup)
146
147 elif argv[i].startswith("--qt-support"):
148 # The --qt-support is special because we want to keep backward compatibility:
149 # Previously, just passing '--qt-support' meant that we should use the auto-discovery mode
150 # whereas now, if --qt-support is passed, it should be passed as --qt-support=<mode>, where
151 # mode can be one of 'auto', 'none', 'pyqt5', 'pyqt4', 'pyside', 'pyside2'.
152 if argv[i] == "--qt-support":
153 setup["qt-support"] = "auto"
154
155 elif argv[i].startswith("--qt-support="):
156 qt_support = argv[i][len("--qt-support=") :]
157 valid_modes = ("none", "auto", "pyqt5", "pyqt4", "pyside", "pyside2")
158 if qt_support not in valid_modes:
159 raise ValueError("qt-support mode invalid: " + qt_support)
160 if qt_support == "none":
161 # On none, actually set an empty string to evaluate to False.
162 setup["qt-support"] = ""
163 else:
164 setup["qt-support"] = qt_support
165 else:
166 raise ValueError("Unexpected definition for qt-support flag: " + argv[i])
167
168 del argv[i]
169
170 elif argv[i] == "--file":
171 # --file is special because it's the last one (so, no handler for it).
172 del argv[i]
173 setup["file"] = argv[i]
174 i = len(argv) # pop out, file is our last argument
175
176 elif argv[i] == "--DEBUG":
177 sys.stderr.write("pydevd: --DEBUG parameter deprecated. Use `--debug-level=3` instead.\n")
178
179 else:
180 raise ValueError("Unexpected option: %s when processing: %s" % (argv[i], initial_argv))
181 return setup
