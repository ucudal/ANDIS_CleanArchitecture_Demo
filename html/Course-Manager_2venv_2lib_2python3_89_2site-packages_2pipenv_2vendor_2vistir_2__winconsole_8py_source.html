183class _WindowsConsoleReader(_WindowsConsoleRawIOBase):
184 def readable(self):
185 return True

187 def readinto(self, b):
188 bytes_to_be_read = len(b)
189 if not bytes_to_be_read:
190 return 0
191 elif bytes_to_be_read % 2:
192 raise ValueError(
193 "cannot read odd number of bytes from " "UTF-16-LE encoded console"
194 )
195
196 buffer = get_buffer(b, writable=True)
197 code_units_to_be_read = bytes_to_be_read // 2
198 code_units_read = c_ulong()
199
200 rv = ReadConsoleW(
201 self.handle, buffer, code_units_to_be_read, byref(code_units_read), None
202 )
203 if GetLastError() == ERROR_OPERATION_ABORTED:
204 # wait for KeyboardInterrupt
205 time.sleep(0.1)
206 if not rv:
207 raise OSError("Windows error: %s" % GetLastError())
208
209 if buffer[0] == EOF:
210 return 0
211 return 2 * code_units_read.value

227 bytes_to_be_written = len(b)
228 buf = get_buffer(b)
229 code_units_to_be_written = min(bytes_to_be_written, MAX_BYTES_WRITTEN) // 2

281class WindowsChunkedWriter(object):
282 """
283 Wraps a stream (such as stdout), acting as a transparent proxy for all
284 attribute access apart from method 'write()' which we wrap to write in
285 limited chunks due to a Windows limitation on binary console streams.
286 """
287
288 def __init__(self, wrapped):
289 # double-underscore everything to prevent clashes with names of
290 # attributes on the wrapped stream object.
291 self.__wrapped = wrapped
292
293 def __getattr__(self, name):
294 return getattr(self.__wrapped, name)
295
296 def write(self, text):
297 total_to_write = len(text)
298 written = 0
299
300 while written < total_to_write:
301 to_write = min(total_to_write - written, MAX_BYTES_WRITTEN)
302 self.__wrapped.write(text[written : written + to_write])
303 written += to_write
