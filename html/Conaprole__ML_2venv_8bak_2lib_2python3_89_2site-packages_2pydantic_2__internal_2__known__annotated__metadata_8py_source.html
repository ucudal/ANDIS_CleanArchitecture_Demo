105def expand_grouped_metadata(annotations: Iterable[Any]) -> Iterable[Any]:
106 """Expand the annotations.
107
108 Args:
109 annotations: An iterable of annotations.
110
111 Returns:
112 An iterable of expanded annotations.
113
114 Example:
115 ```python
116 from annotated_types import Ge, Len
117
118 from pydantic._internal._known_annotated_metadata import expand_grouped_metadata
119
120 print(list(expand_grouped_metadata([Ge(4), Len(5)])))
121 #> [Ge(ge=4), MinLen(min_length=5)]
122 ```
123 """
124 import annotated_types as at
125
126 FieldInfo = import_cached_field_info()
127
128 for annotation in annotations:
129 if isinstance(annotation, at.GroupedMetadata):
130 yield from annotation
131 elif isinstance(annotation, FieldInfo):
132 yield from annotation.metadata
133 # this is a bit problematic in that it results in duplicate metadata
134 # all of our "consumers" can handle it, but it is not ideal
135 # we probably should split up FieldInfo into:
136 # - annotated types metadata
137 # - individual metadata known only to Pydantic
138 annotation = copy(annotation)
139 annotation.metadata = []
140 yield annotation
141 else:
142 yield annotation

146def _get_at_to_constraint_map() -> dict[type, str]:
147 """Return a mapping of annotated types to constraints.
148
149 Normally, we would define a mapping like this in the module scope, but we can't do that
150 because we don't permit module level imports of `annotated_types`, in an attempt to speed up
151 the import time of `pydantic`. We still only want to have this dictionary defined in one place,
152 so we use this function to cache the result.
153 """
154 import annotated_types as at
155
156 return {
157 at.Gt: 'gt',
158 at.Ge: 'ge',
159 at.Lt: 'lt',
160 at.Le: 'le',
161 at.MultipleOf: 'multiple_of',
162 at.MinLen: 'min_length',
163 at.MaxLen: 'max_length',
164 }

167def apply_known_metadata(annotation: Any, schema: CoreSchema) -> CoreSchema | None: # noqa: C901
168 """Apply `annotation` to `schema` if it is an annotation we know about (Gt, Le, etc.).
169 Otherwise return `None`.
170
171 This does not handle all known annotations. If / when it does, it can always
172 return a CoreSchema and return the unmodified schema if the annotation should be ignored.
173
174 Assumes that GroupedMetadata has already been expanded via `expand_grouped_metadata`.
175
176 Args:
177 annotation: The annotation.
178 schema: The schema.
179
180 Returns:
181 An updated schema with annotation if it is an annotation we know about, `None` otherwise.
182
183 Raises:
184 PydanticCustomError: If `Predicate` fails.
185 """
186 import annotated_types as at
187
188 from ._validators import NUMERIC_VALIDATOR_LOOKUP, forbid_inf_nan_check
189
190 schema = schema.copy()
191 schema_update, other_metadata = collect_known_metadata([annotation])
192 schema_type = schema['type']
193
194 chain_schema_constraints: set[str] = {
195 'pattern',
196 'strip_whitespace',
197 'to_lower',
198 'to_upper',
199 'coerce_numbers_to_str',
200 }
201 chain_schema_steps: list[CoreSchema] = []
202
203 for constraint, value in schema_update.items():
204 if constraint not in CONSTRAINTS_TO_ALLOWED_SCHEMAS:
205 raise ValueError(f'Unknown constraint {constraint}')
206 allowed_schemas = CONSTRAINTS_TO_ALLOWED_SCHEMAS[constraint]
207
208 # if it becomes necessary to handle more than one constraint
209 # in this recursive case with function-after or function-wrap, we should refactor
210 # this is a bit challenging because we sometimes want to apply constraints to the inner schema,
211 # whereas other times we want to wrap the existing schema with a new one that enforces a new constraint.
212 if schema_type in {'function-before', 'function-wrap', 'function-after'} and constraint == 'strict':
213 schema['schema'] = apply_known_metadata(annotation, schema['schema']) # type: ignore # schema is function schema
214 return schema
215
216 # if we're allowed to apply constraint directly to the schema, like le to int, do that
217 if schema_type in allowed_schemas:
218 if constraint == 'union_mode' and schema_type == 'union':
219 schema['mode'] = value # type: ignore # schema is UnionSchema
220 else:
221 schema[constraint] = value
222 continue
223
224 # else, apply a function after validator to the schema to enforce the corresponding constraint
225 if constraint in chain_schema_constraints:
226
227 def _apply_constraint_with_incompatibility_info(
228 value: Any, handler: cs.ValidatorFunctionWrapHandler
229 ) -> Any:
230 try:
231 x = handler(value)
232 except ValidationError as ve:
233 # if the error is about the type, it's likely that the constraint is incompatible the type of the field
234 # for example, the following invalid schema wouldn't be caught during schema build, but rather at this point
235 # with a cryptic 'string_type' error coming from the string validator,
236 # that we'd rather express as a constraint incompatibility error (TypeError)
237 # Annotated[list[int], Field(pattern='abc')]
238 if 'type' in ve.errors()[0]['type']:
239 raise TypeError(
240 f"Unable to apply constraint '{constraint}' to supplied value {value} for schema of type '{schema_type}'" # noqa: B023
241 )
242 raise ve
243 return x
244
245 chain_schema_steps.append(
246 cs.no_info_wrap_validator_function(
247 _apply_constraint_with_incompatibility_info, cs.str_schema(**{constraint: value})
248 )
249 )
250 elif constraint in NUMERIC_VALIDATOR_LOOKUP:
251 if constraint in LENGTH_CONSTRAINTS:
252 inner_schema = schema
253 while inner_schema['type'] in {'function-before', 'function-wrap', 'function-after'}:
254 inner_schema = inner_schema['schema'] # type: ignore
255 inner_schema_type = inner_schema['type']
256 if inner_schema_type == 'list' or (
257 inner_schema_type == 'json-or-python' and inner_schema['json_schema']['type'] == 'list' # type: ignore
258 ):
259 js_constraint_key = 'minItems' if constraint == 'min_length' else 'maxItems'
260 else:
261 js_constraint_key = 'minLength' if constraint == 'min_length' else 'maxLength'
262 else:
263 js_constraint_key = constraint
264
265 schema = cs.no_info_after_validator_function(
266 partial(NUMERIC_VALIDATOR_LOOKUP[constraint], **{constraint: value}), schema
267 )
268 metadata = schema.get('metadata', {})
269 if (existing_json_schema_updates := metadata.get('pydantic_js_updates')) is not None:
270 metadata['pydantic_js_updates'] = {
271 **existing_json_schema_updates,
272 **{js_constraint_key: as_jsonable_value(value)},
273 }
274 else:
275 metadata['pydantic_js_updates'] = {js_constraint_key: as_jsonable_value(value)}
276 schema['metadata'] = metadata
277 elif constraint == 'allow_inf_nan' and value is False:
278 schema = cs.no_info_after_validator_function(
279 forbid_inf_nan_check,
280 schema,
281 )
282 else:
283 # It's rare that we'd get here, but it's possible if we add a new constraint and forget to handle it
284 # Most constraint errors are caught at runtime during attempted application
285 raise RuntimeError(f"Unable to apply constraint '{constraint}' to schema of type '{schema_type}'")
286
287 for annotation in other_metadata:
288 if (annotation_type := type(annotation)) in (at_to_constraint_map := _get_at_to_constraint_map()):
289 constraint = at_to_constraint_map[annotation_type]
290 validator = NUMERIC_VALIDATOR_LOOKUP.get(constraint)
291 if validator is None:
292 raise ValueError(f'Unknown constraint {constraint}')
293 schema = cs.no_info_after_validator_function(
294 partial(validator, {constraint: getattr(annotation, constraint)}), schema
295 )
296 continue
297 elif isinstance(annotation, (at.Predicate, at.Not)):
298 predicate_name = f'{annotation.func.__qualname__}' if hasattr(annotation.func, '__qualname__') else ''
299
300 def val_func(v: Any) -> Any:
301 predicate_satisfied = annotation.func(v) # noqa: B023
302
303 # annotation.func may also raise an exception, let it pass through
304 if isinstance(annotation, at.Predicate): # noqa: B023
305 if not predicate_satisfied:
306 raise PydanticCustomError(
307 'predicate_failed',
308 f'Predicate {predicate_name} failed', # type: ignore # noqa: B023
309 )
310 else:
311 if predicate_satisfied:
312 raise PydanticCustomError(
313 'not_operation_failed',
314 f'Not of {predicate_name} failed', # type: ignore # noqa: B023
315 )
316
317 return v
318
319 schema = cs.no_info_after_validator_function(val_func, schema)
320 else:
321 # ignore any other unknown metadata
322 return None
323
324 if chain_schema_steps:
325 chain_schema_steps = [schema] + chain_schema_steps
326 return cs.chain_schema(chain_schema_steps)
327
328 return schema

331def collect_known_metadata(annotations: Iterable[Any]) -> tuple[dict[str, Any], list[Any]]:
332 """Split `annotations` into known metadata and unknown annotations.
333
334 Args:
335 annotations: An iterable of annotations.
336
337 Returns:
338 A tuple contains a dict of known metadata and a list of unknown annotations.
339
340 Example:
341 ```python
342 from annotated_types import Gt, Len
343
344 from pydantic._internal._known_annotated_metadata import collect_known_metadata
345
346 print(collect_known_metadata([Gt(1), Len(42), ...]))
347 #> ({'gt': 1, 'min_length': 42}, [Ellipsis])
348 ```
349 """
350 annotations = expand_grouped_metadata(annotations)
351
352 res: dict[str, Any] = {}
353 remaining: list[Any] = []
354
355 for annotation in annotations:
356 # isinstance(annotation, PydanticMetadata) also covers ._fields:_PydanticGeneralMetadata
357 if isinstance(annotation, PydanticMetadata):
358 res.update(annotation.__dict__)
359 # we don't use dataclasses.asdict because that recursively calls asdict on the field values
360 elif (annotation_type := type(annotation)) in (at_to_constraint_map := _get_at_to_constraint_map()):
361 constraint = at_to_constraint_map[annotation_type]
362 res[constraint] = getattr(annotation, constraint)
363 elif isinstance(annotation, type) and issubclass(annotation, PydanticMetadata):
364 # also support PydanticMetadata classes being used without initialisation,
365 # e.g. `Annotated[int, Strict]` as well as `Annotated[int, Strict()]`
366 res.update({k: v for k, v in vars(annotation).items() if not k.startswith('_')})
367 else:
368 remaining.append(annotation)
369 # Nones can sneak in but pydantic-core will reject them
370 # it'd be nice to clean things up so we don't put in None (we probably don't _need_ to, it was just easier)
371 # but this is simple enough to kick that can down the road
372 res = {k: v for k, v in res.items() if v is not None}
373 return res, remaining

376def check_metadata(metadata: dict[str, Any], allowed: Iterable[str], source_type: Any) -> None:
377 """A small utility function to validate that the given metadata can be applied to the target.
378 More than saving lines of code, this gives us a consistent error message for all of our internal implementations.
379
380 Args:
381 metadata: A dict of metadata.
382 allowed: An iterable of allowed metadata.
383 source_type: The source type.
384
385 Raises:
386 TypeError: If there is metadatas that can't be applied on source type.
387 """
388 unknown = metadata.keys() - set(allowed)
389 if unknown:
390 raise TypeError(
391 f'The following constraints cannot be applied to {source_type!r}: {", ".join([f"{k!r}" for k in unknown])}'
392 )
