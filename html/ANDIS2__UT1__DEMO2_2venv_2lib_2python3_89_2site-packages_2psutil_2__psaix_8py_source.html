204def net_connections(kind, _pid=-1):
205 """Return socket connections. If pid == -1 return system-wide
206 connections (as opposed to connections opened by one process only).
207 """
208 families, types = _common.conn_tmap[kind]
209 rawlist = cext.net_connections(_pid)

212 fd, fam, type_, laddr, raddr, status, pid = item

216 continue

218 fd,
219 fam,

222 raddr,
223 status,
224 TCP_STATUSES,
225 pid=pid if _pid == -1 else None,
226 )
227 ret.append(nt)

231def net_if_stats():
232 """Get NIC stats (isup, duplex, speed, mtu)."""
233 duplex_map = {"Full": NIC_DUPLEX_FULL, "Half": NIC_DUPLEX_HALF}
234 names = {x[0] for x in net_if_addrs()}
235 ret = {}
236 for name in names:
237 mtu = cext_posix.net_if_mtu(name)
238 flags = cext_posix.net_if_flags(name)
239
240 # try to get speed and duplex
241 # TODO: rewrite this in C (entstat forks, so use truss -f to follow.
242 # looks like it is using an undocumented ioctl?)
243 duplex = ""
244 speed = 0
245 p = subprocess.Popen(
246 ["/usr/bin/entstat", "-d", name],
247 stdout=subprocess.PIPE,
248 stderr=subprocess.PIPE,
249 )
250 stdout, stderr = p.communicate()
251 stdout, stderr = (
252 x.decode(sys.stdout.encoding) for x in (stdout, stderr)
253 )
254 if p.returncode == 0:
255 re_result = re.search(
256 r"Running: (\d+) Mbps.*?(\w+) Duplex", stdout
257 )
258 if re_result is not None:
259 speed = int(re_result.group(1))
260 duplex = re_result.group(2)
261
262 output_flags = ','.join(flags)
263 isup = 'running' in flags
264 duplex = duplex_map.get(duplex, NIC_DUPLEX_UNKNOWN)
265 ret[name] = _common.snicstats(isup, duplex, speed, mtu, output_flags)
266 return ret

313def wrap_exceptions(fun):
314 """Call callable into a try/except clause and translate ENOENT,
315 EACCES and EPERM in NoSuchProcess or AccessDenied exceptions.
316 """
317
318 @functools.wraps(fun)
319 def wrapper(self, *args, **kwargs):
320 pid, ppid, name = self.pid, self._ppid, self._name
321 try:
322 return fun(self, *args, **kwargs)
323 except (FileNotFoundError, ProcessLookupError) as err:
324 # ENOENT (no such file or directory) gets raised on open().
325 # ESRCH (no such process) can get raised on read() if
326 # process is gone in meantime.
327 if not pid_exists(pid):
328 raise NoSuchProcess(pid, name) from err
329 raise ZombieProcess(pid, name, ppid) from err
330 except PermissionError as err:
331 raise AccessDenied(pid, name) from err
332
333 return wrapper

372 @wrap_exceptions
373 def exe(self):
374 # there is no way to get executable path in AIX other than to guess,
375 # and guessing is more complex than what's in the wrapping class
376 cmdline = self.cmdline()
377 if not cmdline:
378 return ''
379 exe = cmdline[0]
380 if os.path.sep in exe:
381 # relative or absolute path
382 if not os.path.isabs(exe):
383 # if cwd has changed, we're out of luck - this may be wrong!
384 exe = os.path.abspath(os.path.join(self.cwd(), exe))
385 if (
386 os.path.isabs(exe)
387 and os.path.isfile(exe)
388 and os.access(exe, os.X_OK)
389 ):
390 return exe
391 # not found, move to search in PATH using basename only
392 exe = os.path.basename(exe)
393 # search for exe name PATH
394 for path in os.environ["PATH"].split(":"):
395 possible_exe = os.path.abspath(os.path.join(path, exe))
396 if os.path.isfile(possible_exe) and os.access(
397 possible_exe, os.X_OK
398 ):
399 return possible_exe
400 return ''
