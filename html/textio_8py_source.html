78 def integer(token):
79 """
80 Convert numeric strings into integers.
81
82 @type token: str
83 @param token: String to parse.
84
85 @rtype: int
86 @return: Parsed integer value.
87 """
88 token = token.strip()
89 neg = False
90 if token.startswith(compat.b("-")):
91 token = token[1:]
92 neg = True
93 if token.startswith(compat.b("0x")):
94 result = int(token, 16) # hexadecimal
95 elif token.startswith(compat.b("0b")):
96 result = int(token[2:], 2) # binary
97 elif token.startswith(compat.b("0o")):
98 result = int(token, 8) # octal
99 else:
100 try:
101 result = int(token) # decimal
102 except ValueError:
103 result = int(token, 16) # hexadecimal (no "0x" prefix)
104 if neg:
105 result = -result
106 return result

143 @staticmethod
144 def pattern(token):
145 """
146 Convert an hexadecimal search pattern into a POSIX regular expression.
147
148 For example, the following pattern::
149
150 "B8 0? ?0 ?? ??"
151
152 Would match the following data::
153
154 "B8 0D F0 AD BA" # mov eax, 0xBAADF00D
155
156 @type token: str
157 @param token: String to parse.
158
159 @rtype: str
160 @return: Parsed string value.
161 """
162 token = "".join([c for c in token if c == "?" or c.isalnum()])
163 if len(token) % 2 != 0:
164 raise ValueError("Missing characters in hex data")
165 regexp = ""
166 for i in compat.xrange(0, len(token), 2):
167 x = token[i : i + 2]
168 if x == "??":
169 regexp += "."
170 elif x[0] == "?":
171 f = "\\x%%.1x%s" % x[1]
172 x = "".join([f % c for c in compat.xrange(0, 0x10)])
173 regexp = "%s[%s]" % (regexp, x)
174 elif x[1] == "?":
175 f = "\\x%s%%.1x" % x[0]
176 x = "".join([f % c for c in compat.xrange(0, 0x10)])
177 regexp = "%s[%s]" % (regexp, x)
178 else:
179 regexp = "%s\\x%s" % (regexp, x)
180 return regexp

199 """
200 Read a list of integers from a file.
201
202 The file format is:
203
204 - # anywhere in the line begins a comment
205 - leading and trailing spaces are ignored
206 - empty lines are ignored
207 - integers can be specified as:
208 - decimal numbers ("100" is 100)
209 - hexadecimal numbers ("0x100" is 256)
210 - binary numbers ("0b100" is 4)
211 - octal numbers ("0100" is 64)
212
213 @type filename: str
214 @param filename: Name of the file to read.
215
216 @rtype: list( int )
217 @return: List of integers read from the file.
218 """

220 result = list()
221 fd = open(filename, "r")
222 for line in fd:
223 count = count + 1
224 if "#" in line:
225 line = line[: line.find("#")]
226 line = line.strip()
227 if line:
228 try:
229 value = cls.integer(line)
230 except ValueError:
231 e = sys.exc_info()[1]
232 msg = "Error in line %d of %s: %s"
233 msg = msg % (count, filename, str(e))
234 raise ValueError(msg)
235 result.append(value)
236 return result

1072class Table(object):
1073 """
1074 Text based table. The number of columns and the width of each column
1075 is automatically calculated.
1076 """
1077
1078 def __init__(self, sep=" "):
1079 """
1080 @type sep: str
1081 @param sep: Separator between cells in each row.
1082 """
1083 self.__cols = list()
1084 self.__width = list()
1085 self.__sep = sep
1086
1087 def addRow(self, *row):
1088 """
1089 Add a row to the table. All items are converted to strings.
1090
1091 @type row: tuple
1092 @keyword row: Each argument is a cell in the table.
1093 """
1094 row = [str(item) for item in row]
1095 len_row = [len(item) for item in row]
1096 width = self.__width
1097 len_old = len(width)
1098 len_new = len(row)
1099 known = min(len_old, len_new)
1100 missing = len_new - len_old
1101 if missing > 0:
1102 width.extend(len_row[-missing:])
1103 elif missing < 0:
1104 len_row.extend([0] * (-missing))
1105 self.__width = [max(width[i], len_row[i]) for i in compat.xrange(len(len_row))]
1106 self.__cols.append(row)
1107
1108 def justify(self, column, direction):
1109 """
1110 Make the text in a column left or right justified.
1111
1112 @type column: int
1113 @param column: Index of the column.
1114
1115 @type direction: int
1116 @param direction:
1117 C{-1} to justify left,
1118 C{1} to justify right.
1119
1120 @raise IndexError: Bad column index.
1121 @raise ValueError: Bad direction value.
1122 """
1123 if direction == -1:
1124 self.__width[column] = abs(self.__width[column])
1125 elif direction == 1:
1126 self.__width[column] = -abs(self.__width[column])
1127 else:
1128 raise ValueError("Bad direction value.")
1129
1130 def getWidth(self):
1131 """
1132 Get the width of the text output for the table.
1133
1134 @rtype: int
1135 @return: Width in characters for the text output,
1136 including the newline character.
1137 """
1138 width = 0
1139 if self.__width:
1140 width = sum(abs(x) for x in self.__width)
1141 width = width + len(self.__width) * len(self.__sep) + 1
1142 return width
1143
1144 def getOutput(self):
1145 """
1146 Get the text output for the table.
1147
1148 @rtype: str
1149 @return: Text output.
1150 """
1151 return "%s\n" % "\n".join(self.yieldOutput())
1152
1153 def yieldOutput(self):
1154 """
1155 Generate the text output for the table.
1156
1157 @rtype: generator of str
1158 @return: Text output.
1159 """
1160 width = self.__width
1161 if width:
1162 num_cols = len(width)
1163 fmt = ["%%%ds" % -w for w in width]
1164 if width[-1] > 0:
1165 fmt[-1] = "%s"
1166 fmt = self.__sep.join(fmt)
1167 for row in self.__cols:
1168 row.extend([""] * (num_cols - len(row)))
1169 yield fmt % tuple(row)
1170
1171 def show(self):
1172 """
1173 Print the text output for the table.
1174 """
1175 print(self.getOutput())
