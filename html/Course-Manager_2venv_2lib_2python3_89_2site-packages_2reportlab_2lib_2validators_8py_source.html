194class OneOf(Validator):
195 """Make validator functions for list of choices.
196
197 Usage:
198 f = reportlab.lib.validators.OneOf('happy','sad')
199 or
200 f = reportlab.lib.validators.OneOf(('happy','sad'))
201 f('sad'),f('happy'), f('grumpy')
202 (1,1,0)
203 """
204 def __init__(self, enum,*args):
205 if isSeq(enum):
206 if args!=():
207 raise ValueError("Either all singleton args or a single sequence argument")
208 self._enum = tuple(enum)+args
209 else:
210 self._enum = (enum,)+args
211 self._patterns = tuple((_ for _ in self._enum if isinstance(_,_re_Pattern)))
212 if self._patterns:
213 self._enum = tuple((_ for _ in self._enum if not isinstance(_,_re_Pattern)))
214 self.test = self._test_patterns
215
216 def test(self, x):
217 return x in self._enum
218
219 def _test_patterns(self, x):
220 v = x in self._enum
221 #print(f'{x=} {self._enum=!r} {self._patterns=!r} {v=}')
222 if v: return True
223 for p in self._patterns:
224 v = p.match(x)
225 if v: return True
226 return False

228class SequenceOf(Validator):
229 def __init__(self,elemTest,name=None,emptyOK=1, NoneOK=0, lo=0,hi=0x7fffffff):
230 self._elemTest = elemTest
231 self._emptyOK = emptyOK
232 self._NoneOK = NoneOK
233 self._lo, self._hi = lo, hi
234 if name: self._str = name
235
236 def test(self, x):
237 if not isSeq(x):
238 if x is None: return self._NoneOK
239 return False
240 if x==[] or x==():
241 return self._emptyOK
242 elif not self._lo<=len(x)<=self._hi: return False
243 for e in x:
244 if not self._elemTest(e): return False
245 return True
