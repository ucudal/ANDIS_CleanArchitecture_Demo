19class WSGITransport(httpcore.SyncHTTPTransport):
20 """
21 A custom transport that handles sending requests directly to an WSGI app.
22 The simplest way to use this functionality is to use the `app` argument.
23
24 ```
25 client = httpx.Client(app=app)
26 ```
27
28 Alternatively, you can setup the dispatch instance explicitly.
29 This allows you to include any additional configuration arguments specific
30 to the WSGITransport class:
31
32 ```
33 dispatch = httpx.WSGITransport(
34 app=app,
35 script_name="/submount",
36 remote_addr="1.2.3.4"
37 )
38 client = httpx.Client(dispatch=dispatch)
39 ```
40
41 Arguments:
42
43 * `app` - The ASGI application.
44 * `raise_app_exceptions` - Boolean indicating if exceptions in the application
45 should be raised. Default to `True`. Can be set to `False` for use cases
46 such as testing the content of a client 500 response.
47 * `script_name` - The root path on which the ASGI application should be mounted.
48 * `remote_addr` - A string indicating the client IP of incoming requests.
49 ```
50 """
