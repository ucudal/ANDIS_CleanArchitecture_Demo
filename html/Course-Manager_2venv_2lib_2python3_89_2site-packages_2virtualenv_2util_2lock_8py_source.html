107 def _lock_file(self, lock, no_block=False):
108 # multiple processes might be trying to get a first lock... so we cannot check if this directory exist without
109 # a lock, but that lock might then become expensive, and it's not clear where that lock should live.
110 # Instead here we just ignore if we fail to create the directory.
111 try:
112 os.makedirs(str(self.path))
113 except OSError:
114 pass
