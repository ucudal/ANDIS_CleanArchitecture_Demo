103def urlencode(query, doseq=False):
104 """
105 A version of Python's urllib.parse.urlencode() function that can operate on
106 MultiValueDict and non-string values.
107 """
108 if isinstance(query, MultiValueDict):
109 query = query.lists()
110 elif hasattr(query, 'items'):
111 query = query.items()
112 query_params = []
113 for key, value in query:
114 if value is None:
115 raise TypeError(
116 "Cannot encode None for key '%s' in a query string. Did you "
117 "mean to pass an empty string or omit the value?" % key
118 )
119 elif not doseq or isinstance(value, (str, bytes)):
120 query_val = value
121 else:
122 try:
123 itr = iter(value)
124 except TypeError:
125 query_val = value
126 else:
127 # Consume generators and iterators, when doseq=True, to
128 # work around https://bugs.python.org/issue31706.
129 query_val = []
130 for item in itr:
131 if item is None:
132 raise TypeError(
133 "Cannot encode None for key '%s' in a query "
134 "string. Did you mean to pass an empty string or "
135 "omit the value?" % key
136 )
137 elif not isinstance(item, bytes):
138 item = str(item)
139 query_val.append(item)
140 query_params.append((key, query_val))
141 return original_urlencode(query_params, doseq)

158def parse_http_date(date):
159 """
160 Parse a date format as specified by HTTP RFC7231 section 7.1.1.1.
161
162 The three formats allowed by the RFC are accepted, even if only the first
163 one is still in widespread use.
164
165 Return an integer expressed in seconds since the epoch, in UTC.
166 """
167 # email.utils.parsedate() does the job for RFC1123 dates; unfortunately
168 # RFC7231 makes it mandatory to support RFC850 dates too. So we roll
169 # our own RFC-compliant parsing.
170 for regex in RFC1123_DATE, RFC850_DATE, ASCTIME_DATE:
171 m = regex.match(date)
172 if m is not None:
173 break
174 else:
175 raise ValueError("%r is not in a valid HTTP date format" % date)
176 try:
177 year = int(m.group('year'))
178 if year < 100:
179 current_year = datetime.datetime.utcnow().year
180 current_century = current_year - (current_year % 100)
181 if year - (current_year % 100) > 50:
182 # year that appears to be more than 50 years in the future are
183 # interpreted as representing the past.
184 year += current_century - 100
185 else:
186 year += current_century
187 month = MONTHS.index(m.group('mon').lower()) + 1
188 day = int(m.group('day'))
189 hour = int(m.group('hour'))
190 min = int(m.group('min'))
191 sec = int(m.group('sec'))
192 result = datetime.datetime(year, month, day, hour, min, sec)
193 return calendar.timegm(result.utctimetuple())
194 except Exception as exc:
195 raise ValueError("%r is not a valid date" % date) from exc

210def base36_to_int(s):
211 """
212 Convert a base 36 string to an int. Raise ValueError if the input won't fit
213 into an int.
214 """
215 # To prevent overconsumption of server resources, reject any
216 # base36 string that is longer than 13 base36 digits (13 digits
217 # is sufficient to base36-encode any 64-bit integer)
218 if len(s) > 13:
219 raise ValueError("Base36 input too large")
220 return int(s, 36)

223def int_to_base36(i):
224 """Convert an integer to a base36 string."""
225 char_set = '0123456789abcdefghijklmnopqrstuvwxyz'
226 if i < 0:
227 raise ValueError("Negative base36 conversion input.")
228 if i < 36:
229 return char_set[i]
230 b36 = ''
231 while i != 0:
232 i, n = divmod(i, 36)
233 b36 = char_set[n] + b36
234 return b36

257def parse_etags(etag_str):
258 """
259 Parse a string of ETags given in an If-None-Match or If-Match header as
260 defined by RFC 7232. Return a list of quoted ETags, or ['*'] if all ETags
261 should be matched.
262 """
263 if etag_str.strip() == '*':
264 return ['*']
265 else:
266 # Parse each ETag individually, and return any that are valid.
267 etag_matches = (ETAG_MATCH.match(etag.strip()) for etag in etag_str.split(','))
268 return [match.group(1) for match in etag_matches if match]

283 """

287 Any pattern beginning with a period matches a domain and all of its

301def url_has_allowed_host_and_scheme(url, allowed_hosts, require_https=False):
302 """
303 Return ``True`` if the url uses an allowed host and a safe scheme.
304
305 Always return ``False`` on an empty url.
306
307 If ``require_https`` is ``True``, only 'https' will be considered a valid
308 scheme, as opposed to 'http' and 'https' with the default, ``False``.
309
310 Note: "True" doesn't entail that a URL is "safe". It may still be e.g.
311 quoted incorrectly. Ensure to also use django.utils.encoding.iri_to_uri()
312 on the path component of untrusted URLs.
313 """
314 if url is not None:
315 url = url.strip()
316 if not url:
317 return False
318 if allowed_hosts is None:
319 allowed_hosts = set()
320 elif isinstance(allowed_hosts, str):
321 allowed_hosts = {allowed_hosts}
322 # Chrome treats \ completely as / in paths but it could be part of some
323 # basic auth credentials so we need to check both URLs.
324 return (
325 _url_has_allowed_host_and_scheme(url, allowed_hosts, require_https=require_https) and
326 _url_has_allowed_host_and_scheme(url.replace('\\', '/'), allowed_hosts, require_https=require_https)
327 )

340def _urlparse(url, scheme='', allow_fragments=True):
341 """Parse a URL into 6 components:
342 <scheme>://<netloc>/<path>;<params>?<query>#<fragment>
343 Return a 6-tuple: (scheme, netloc, path, params, query, fragment).
344 Note that we don't break the components up in smaller bits
345 (e.g. netloc is a single string) and we don't expand % escapes."""
346 url, scheme, _coerce_result = _coerce_args(url, scheme)
347 splitresult = _urlsplit(url, scheme, allow_fragments)
348 scheme, netloc, url, query, fragment = splitresult
349 if scheme in uses_params and ';' in url:
350 url, params = _splitparams(url)
351 else:
352 params = ''
353 result = ParseResult(scheme, netloc, url, params, query, fragment)
354 return _coerce_result(result)

357# Copied from urllib.parse.urlsplit() with
358# https://github.com/python/cpython/pull/661 applied.
359def _urlsplit(url, scheme='', allow_fragments=True):
360 """Parse a URL into 5 components:
361 <scheme>://<netloc>/<path>?<query>#<fragment>
362 Return a 5-tuple: (scheme, netloc, path, query, fragment).
363 Note that we don't break the components up in smaller bits
364 (e.g. netloc is a single string) and we don't expand % escapes."""
365 url, scheme, _coerce_result = _coerce_args(url, scheme)
366 netloc = query = fragment = ''
367 i = url.find(':')
368 if i > 0:
369 for c in url[:i]:
370 if c not in scheme_chars:
371 break
372 else:
373 scheme, url = url[:i].lower(), url[i + 1:]
374
375 if url[:2] == '//':
376 netloc, url = _splitnetloc(url, 2)
377 if (('[' in netloc and ']' not in netloc) or
378 (']' in netloc and '[' not in netloc)):
379 raise ValueError("Invalid IPv6 URL")
380 if allow_fragments and '#' in url:
381 url, fragment = url.split('#', 1)
382 if '?' in url:
383 url, query = url.split('?', 1)
384 v = SplitResult(scheme, netloc, url, query, fragment)
385 return _coerce_result(v)

388def _url_has_allowed_host_and_scheme(url, allowed_hosts, require_https=False):
389 # Chrome considers any URL with more than two slashes to be absolute, but
390 # urlparse is not so flexible. Treat any url with three slashes as unsafe.
391 if url.startswith('///'):
392 return False
393 try:
394 url_info = _urlparse(url)
395 except ValueError: # e.g. invalid IPv6 addresses
396 return False
397 # Forbid URLs like http:///example.com - with a scheme, but without a hostname.
398 # In that URL, example.com is not the hostname but, a path component. However,
399 # Chrome will still consider example.com to be the hostname, so we must not
400 # allow this syntax.
401 if not url_info.netloc and url_info.scheme:
402 return False
403 # Forbid URLs that start with control characters. Some browsers (like
404 # Chrome) ignore quite a few control characters at the start of a
405 # URL and might consider the URL as scheme relative.
406 if unicodedata.category(url[0])[0] == 'C':
407 return False
408 scheme = url_info.scheme
409 # Consider URLs without a scheme (e.g. //example.com/p) to be http.
410 if not url_info.scheme and url_info.netloc:
411 scheme = 'http'
412 valid_schemes = ['https'] if require_https else ['http', 'https']
413 return ((not url_info.netloc or url_info.netloc in allowed_hosts) and
414 (not scheme or scheme in valid_schemes))

417def limited_parse_qsl(qs, keep_blank_values=False, encoding='utf-8',
418 errors='replace', fields_limit=None):
419 """
420 Return a list of key/value tuples parsed from query string.
421
422 Copied from urlparse with an additional "fields_limit" argument.
423 Copyright (C) 2013 Python Software Foundation (see LICENSE.python).
424
425 Arguments:
426
427 qs: percent-encoded query string to be parsed
428
429 keep_blank_values: flag indicating whether blank values in
430 percent-encoded queries should be treated as blank strings. A
431 true value indicates that blanks should be retained as blank
432 strings. The default false value indicates that blank values
433 are to be ignored and treated as if they were not included.
434
435 encoding and errors: specify how to decode percent-encoded sequences
436 into Unicode characters, as accepted by the bytes.decode() method.
437
438 fields_limit: maximum number of fields parsed or an exception
439 is raised. None means no limit and is the default.
440 """
441 if fields_limit:
442 pairs = FIELDS_MATCH.split(qs, fields_limit)
443 if len(pairs) > fields_limit:
444 raise TooManyFieldsSent(
445 'The number of GET/POST parameters exceeded '
446 'settings.DATA_UPLOAD_MAX_NUMBER_FIELDS.'
447 )
448 else:
449 pairs = FIELDS_MATCH.split(qs)
450 r = []
451 for name_value in pairs:
452 if not name_value:
453 continue
454 nv = name_value.split('=', 1)
455 if len(nv) != 2:
456 # Handle case of a control-name with no equal sign
457 if keep_blank_values:
458 nv.append('')
459 else:
460 continue
461 if nv[1] or keep_blank_values:
462 name = nv[0].replace('+', ' ')
463 name = unquote(name, encoding=encoding, errors=errors)
464 value = nv[1].replace('+', ' ')
465 value = unquote(value, encoding=encoding, errors=errors)
466 r.append((name, value))
467 return r
