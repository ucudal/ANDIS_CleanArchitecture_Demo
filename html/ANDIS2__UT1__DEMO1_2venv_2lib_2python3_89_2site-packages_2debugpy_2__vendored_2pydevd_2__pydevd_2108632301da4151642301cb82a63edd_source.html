87def _compute_stack_size(block, size, maxsize, *, check_pre_and_post=True):
88 """Generator used to reduce the use of function stacks.
89
90 This allows to avoid nested recursion and allow to treat more cases.
91
92 HOW-TO:
93 Following the methods of Trampoline
94 (see https://en.wikipedia.org/wiki/Trampoline_(computing)),
95
96 We yield either:
97
98 - the arguments that would be used in the recursive calls, i.e,
99 'yield block, size, maxsize' instead of making a recursive call
100 '_compute_stack_size(block, size, maxsize)', if we encounter an
101 instruction jumping to another block or if the block is linked to
102 another one (ie `next_block` is set)
103 - the required stack from the stack if we went through all the instructions
104 or encountered an unconditional jump.
105
106 In the first case, the calling function is then responsible for creating a
107 new generator with those arguments, iterating over it till exhaustion to
108 determine the stacksize required by the block and resuming this function
109 with the determined stacksize.
110
111 """
112 # If the block is currently being visited (seen = True) or if it was visited
113 # previously by using a larger starting size than the one in use, return the
114 # maxsize.
115 if block.seen or block.startsize >= size:
116 yield maxsize
117
118 def update_size(pre_delta, post_delta, size, maxsize):
119 size += pre_delta
120 if size < 0:
121 msg = "Failed to compute stacksize, got negative size"
122 raise RuntimeError(msg)
123 size += post_delta
124 maxsize = max(maxsize, size)
125 return size, maxsize
126
127 # Prevent recursive visit of block if two blocks are nested (jump from one
128 # to the other).
129 block.seen = True
130 block.startsize = size
131
132 for instr in block:
133 # Ignore SetLineno
134 if isinstance(instr, SetLineno):
135 continue
136
137 # For instructions with a jump first compute the stacksize required when the
138 # jump is taken.
139 if instr.has_jump():
140 effect = instr.pre_and_post_stack_effect(jump=True) if check_pre_and_post else (instr.stack_effect(jump=True), 0)
141 taken_size, maxsize = update_size(*effect, size, maxsize)
142 # Yield the parameters required to compute the stacksize required
143 # by the block to which the jumnp points to and resume when we now
144 # the maxsize.
145 maxsize = yield instr.arg, taken_size, maxsize
146
147 # For unconditional jumps abort early since the other instruction will
148 # never be seen.
149 if instr.is_uncond_jump():
150 block.seen = False
151 yield maxsize
152
153 # jump=False: non-taken path of jumps, or any non-jump
154 effect = instr.pre_and_post_stack_effect(jump=False) if check_pre_and_post else (instr.stack_effect(jump=False), 0)
155 size, maxsize = update_size(*effect, size, maxsize)
156
157 if block.next_block:
158 maxsize = yield block.next_block, size, maxsize
159
160 block.seen = False
161 yield maxsize

195 def compute_stacksize(self, *, check_pre_and_post=True):
196 """Compute the stack size by iterating through the blocks
197
198 The implementation make use of a generator function to avoid issue with
199 deeply nested recursions.
200
201 """
202 # In the absence of any block return 0
203 if not self:
204 return 0
205
206 # Ensure that previous calculation do not impact this one.
207 for block in self:
208 block.seen = False
209 block.startsize = -32768 # INT_MIN
210
211 # Starting with Python 3.10, generator and coroutines start with one object
212 # on the stack (None, anything is an error).
213 initial_stack_size = 0
214 if sys.version_info >= (3, 10) and self.flags & (CompilerFlags.GENERATOR | CompilerFlags.COROUTINE | CompilerFlags.ASYNC_GENERATOR):
215 initial_stack_size = 1
216
217 # Create a generator/coroutine responsible of dealing with the first block
218 coro = _compute_stack_size(self[0], initial_stack_size, 0, check_pre_and_post=check_pre_and_post)
219
220 # Create a list of generator that have not yet been exhausted
221 coroutines = []
222
223 push_coroutine = coroutines.append
224 pop_coroutine = coroutines.pop
225 args = None
226
227 try:
228 while True:
229 args = coro.send(None)
230
231 # Consume the stored generators as long as they return a simple
232 # interger that is to be used to resume the last stored generator.
233 while isinstance(args, int):
234 coro = pop_coroutine()
235 args = coro.send(args)
236
237 # Otherwise we enter a new block and we store the generator under
238 # use and create a new one to process the new block
239 push_coroutine(coro)
240 coro = _compute_stack_size(*args, check_pre_and_post=check_pre_and_post)
241
242 except IndexError:
243 # The exception occurs when all the generators have been exhausted
244 # in which case teh last yielded value is the stacksize.
245 assert args is not None
246 return args
