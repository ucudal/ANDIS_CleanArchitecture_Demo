18class Component(util.Observable):
19 """A component managed by a debug adapter: client, launcher, or debug server.
20
21 Every component belongs to a Session, which is used for synchronization and
22 shared data.
23
24 Every component has its own message channel, and provides message handlers for
25 that channel. All handlers should be decorated with @Component.message_handler,
26 which ensures that Session is locked for the duration of the handler. Thus, only
27 one handler is running at any given time across all components, unless the lock
28 is released explicitly or via Session.wait_for().
29
30 Components report changes to their attributes to Session, allowing one component
31 to wait_for() a change caused by another component.
32 """
33
34 def __init__(self, session, stream=None, channel=None):
35 assert (stream is None) ^ (channel is None)
36
37 try:
38 lock_held = session.lock.acquire(blocking=False)
39 assert lock_held, "__init__ of a Component subclass must lock its Session"
40 finally:
41 session.lock.release()
42
43 super().__init__()
44
45 self.session = session
46
47 if channel is None:
48 stream.name = str(self)
49 channel = messaging.JsonMessageChannel(stream, self)
50 channel.start()
51 else:
52 channel.name = channel.stream.name = str(self)
53 channel.handlers = self
54 self.channel = channel
55 self.is_connected = True
56
57 # Do this last to avoid triggering useless notifications for assignments above.
58 self.observers += [lambda *_: self.session.notify_changed()]
59
60 def __str__(self):
61 return f"{type(self).__name__}[{self.session.id}]"

78 @staticmethod
79 def message_handler(f):
80 """Applied to a message handler to automatically lock and unlock the session
81 for its duration, and to validate the session state.
82
83 If the handler raises ComponentNotAvailable or JsonIOError, converts it to
84 Message.cant_handle().
85 """
86
87 @functools.wraps(f)
88 def lock_and_handle(self, message):
89 try:
90 with self.session:
91 return f(self, message)
92 except ComponentNotAvailable as exc:
93 raise message.cant_handle("{0}", exc, silent=True)
94 except messaging.MessageHandlingError as exc:
95 if exc.cause is message:
96 raise
97 else:
98 exc.propagate(message)
99 except messaging.JsonIOError as exc:
100 raise message.cant_handle(
101 "{0} disconnected unexpectedly", exc.stream.name, silent=True
102 )
103
104 return lock_and_handle

106 def disconnect(self):
107 with self.session:
108 self.is_connected = False
109 self.session.finalize("{0} has disconnected".format(self))
