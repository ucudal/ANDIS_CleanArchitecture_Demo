6# Facts
7# -----
8#
9# Headers are:
10# keys: case-insensitive ascii
11# values: mixture of ascii and raw bytes
12#
13# "Historically, HTTP has allowed field content with text in the ISO-8859-1
14# charset [ISO-8859-1], supporting other charsets only through use of
15# [RFC2047] encoding. In practice, most HTTP header field values use only a
16# subset of the US-ASCII charset [USASCII]. Newly defined header fields SHOULD
17# limit their field values to US-ASCII octets. A recipient SHOULD treat other
18# octets in field content (obs-text) as opaque data."
19# And it deprecates all non-ascii values
20#
21# Leading/trailing whitespace in header names is forbidden
22#
23# Values get leading/trailing whitespace stripped
24#
25# Content-Disposition actually needs to contain unicode semantically; to
26# accomplish this it has a terrifically weird way of encoding the filename
27# itself as ascii (and even this still has lots of cross-browser
28# incompatibilities)
29#
30# Order is important:
31# "a proxy MUST NOT change the order of these field values when forwarding a
32# message"
33# (and there are several headers where the order indicates a preference)
34#
35# Multiple occurences of the same header:
36# "A sender MUST NOT generate multiple header fields with the same field name
37# in a message unless either the entire field value for that header field is
38# defined as a comma-separated list [or the header is Set-Cookie which gets a
39# special exception]" - RFC 7230. (cookies are in RFC 6265)
40#
41# So every header aside from Set-Cookie can be merged by b", ".join if it
42# occurs repeatedly. But, of course, they can't necessarily be split by
43# .split(b","), because quoting.
44#
45# Given all this mess (case insensitive, duplicates allowed, order is
46# important, ...), there doesn't appear to be any standard way to handle
47# headers in Python -- they're almost like dicts, but... actually just
48# aren't. For now we punt and just use a super simple representation: headers
49# are a list of pairs
50#
51# [(name1, value1), (name2, value2), ...]
52#
53# where all entries are bytestrings, names are lowercase and have no
54# leading/trailing whitespace, and values are bytestrings with no
55# leading/trailing whitespace. Searching and updating are done via naive O(n)
56# methods.
57#
58# Maybe a dict-of-lists would be better?

65def normalize_and_validate(headers, _parsed=False):
66 new_headers = []
67 saw_content_length = False
68 saw_transfer_encoding = False
69 for name, value in headers:
70 # For headers coming out of the parser, we can safely skip some steps,
71 # because it always returns bytes and has already run these regexes
72 # over the data:
73 if not _parsed:
74 name = bytesify(name)
75 value = bytesify(value)
76 validate(_field_name_re, name, "Illegal header name {!r}", name)
77 validate(_field_value_re, value, "Illegal header value {!r}", value)
78 name = name.lower()
79 if name == b"content-length":
80 if saw_content_length:
81 raise LocalProtocolError("multiple Content-Length headers")
82 validate(_content_length_re, value, "bad Content-Length")
83 saw_content_length = True
84 if name == b"transfer-encoding":
85 # "A server that receives a request message with a transfer coding
86 # it does not understand SHOULD respond with 501 (Not
87 # Implemented)."
88 # https://tools.ietf.org/html/rfc7230#section-3.3.1
89 if saw_transfer_encoding:
90 raise LocalProtocolError(
91 "multiple Transfer-Encoding headers", error_status_hint=501
92 )
93 # "All transfer-coding names are case-insensitive"
94 # -- https://tools.ietf.org/html/rfc7230#section-4
95 value = value.lower()
96 if value != b"chunked":
97 raise LocalProtocolError(
98 "Only Transfer-Encoding: chunked is supported",
99 error_status_hint=501,
100 )
101 saw_transfer_encoding = True
102 new_headers.append((name, value))
103 return new_headers

106def get_comma_header(headers, name):
107 # Should only be used for headers whose value is a list of
108 # comma-separated, case-insensitive values.
109 #
110 # The header name `name` is expected to be lower-case bytes.
111 #
112 # Connection: meets these criteria (including cast insensitivity).
113 #
114 # Content-Length: technically is just a single value (1*DIGIT), but the
115 # standard makes reference to implementations that do multiple values, and
116 # using this doesn't hurt. Ditto, case insensitivity doesn't things either
117 # way.
118 #
119 # Transfer-Encoding: is more complex (allows for quoted strings), so
120 # splitting on , is actually wrong. For example, this is legal:
121 #
122 # Transfer-Encoding: foo; options="1,2", chunked
123 #
124 # and should be parsed as
125 #
126 # foo; options="1,2"
127 # chunked
128 #
129 # but this naive function will parse it as
130 #
131 # foo; options="1
132 # 2"
133 # chunked
134 #
135 # However, this is okay because the only thing we are going to do with
136 # any Transfer-Encoding is reject ones that aren't just "chunked", so
137 # both of these will be treated the same anyway.
138 #
139 # Expect: the only legal value is the literal string
140 # "100-continue". Splitting on commas is harmless. Case insensitive.
141 #
142 out = []
143 for found_name, found_raw_value in headers:
144 if found_name == name:
145 found_raw_value = found_raw_value.lower()
146 for found_split_value in found_raw_value.split(b","):
147 found_split_value = found_split_value.strip()
148 if found_split_value:
149 out.append(found_split_value)
150 return out

153def set_comma_header(headers, name, new_values):
154 # The header name `name` is expected to be lower-case bytes.
155 new_headers = []
156 for found_name, found_raw_value in headers:
157 if found_name != name:
158 new_headers.append((found_name, found_raw_value))
159 for new_value in new_values:
160 new_headers.append((name, new_value))
161 headers[:] = normalize_and_validate(new_headers)

164def has_expect_100_continue(request):
165 # https://tools.ietf.org/html/rfc7231#section-5.1.1
166 # "A server that receives a 100-continue expectation in an HTTP/1.0 request
167 # MUST ignore that expectation."
168 if request.http_version < b"1.1":
169 return False
170 expect = get_comma_header(request.headers, b"expect")
171 return b"100-continue" in expect
