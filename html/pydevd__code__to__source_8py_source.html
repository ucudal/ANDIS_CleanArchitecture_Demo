36class _Token(object):
37 def __init__(self, i_line, instruction=None, tok=_SENTINEL, priority=0, after=None, end_of_line=False):
38 """
39 :param i_line:
40 :param instruction:
41 :param tok:
42 :param priority:
43 :param after:
44 :param end_of_line:
45 Marker to signal only after all the other tokens have been written.
46 """
47 self.i_line = i_line
48 if tok is not _SENTINEL:
49 self.tok = tok
50 else:
51 if instruction is not None:
52 if inspect.iscode(instruction.argval):
53 self.tok = ""
54 else:
55 self.tok = str(instruction.argval)
56 else:
57 raise AssertionError("Either the tok or the instruction is needed.")
58 self.instruction = instruction
59 self.priority = priority
60 self.end_of_line = end_of_line
61 self._after_tokens = set()
62 self._after_handler_tokens = set()
63 if after:
64 self.mark_after(after)
65
66 def mark_after(self, v):
67 if isinstance(v, _Token):
68 self._after_tokens.add(v)
69 elif isinstance(v, _BaseHandler):
70 self._after_handler_tokens.add(v)
71
72 else:
73 raise AssertionError("Unhandled: %s" % (v,))
74
75 def get_after_tokens(self):
76 ret = self._after_tokens.copy()
77 for handler in self._after_handler_tokens:
78 ret.update(handler.tokens)
79 return ret
80
81 def __repr__(self):
82 return "Token(%s, after: %s)" % (self.tok, self.get_after_tokens())

295class _CallFunction(_BaseHandler):
296 """
297
298 CALL_FUNCTION(argc)
299
300 Calls a callable object with positional arguments. argc indicates the number of positional
301 arguments. The top of the stack contains positional arguments, with the right-most argument
302 on top. Below the arguments is a callable object to call. CALL_FUNCTION pops all arguments
303 and the callable object off the stack, calls the callable object with those arguments, and
304 pushes the return value returned by the callable object.
305
306 Changed in version 3.6: This opcode is used only for calls with positional arguments.
307
308 """
309
310 opname = "CALL_FUNCTION"
311
312 def _handle(self):
313 args = []
314 for _i in range(self.instruction.argval + 1):
315 arg = self.stack.pop()
316 args.append(arg)
317 it = reversed(args)
318 name = next(it)
319 max_line = name.i_line
320 for t in name.tokens:
321 self.tokens.append(t)
322
323 tok_open_parens = _Token(name.i_line, None, "(", after=name)
324 self.tokens.append(tok_open_parens)
325
326 prev = tok_open_parens
327 for i, arg in enumerate(it):
328 for t in arg.tokens:
329 t.mark_after(name)
330 t.mark_after(prev)
331 max_line = max(max_line, t.i_line)
332 self.tokens.append(t)
333 prev = arg
334
335 if i > 0:
336 comma_token = _Token(prev.i_line, None, ",", after=prev)
337 self.tokens.append(comma_token)
338 prev = comma_token
339
340 tok_close_parens = _Token(max_line, None, ")", after=prev)
341 self.tokens.append(tok_close_parens)
342
343 self._write_tokens()

349class _MakeFunctionPy3(_BaseHandler):
350 """
351 Pushes a new function object on the stack. From bottom to top, the consumed stack must consist
352 of values if the argument carries a specified flag value
353
354 0x01 a tuple of default values for positional-only and positional-or-keyword parameters in positional order
355
356 0x02 a dictionary of keyword-only parameters' default values
357
358 0x04 an annotation dictionary
359
360 0x08 a tuple containing cells for free variables, making a closure
361
362 the code associated with the function (at TOS1)
363
364 the qualified name of the function (at TOS)
365 """
366
367 opname = "MAKE_FUNCTION"
368 is_lambda = False
369
370 def _handle(self):
371 stack = self.stack
372 self.qualified_name = stack.pop()
373 self.code = stack.pop()
374
375 default_node = None
376 if self.instruction.argval & 0x01:
377 default_node = stack.pop()
378
379 is_lambda = self.is_lambda = "<lambda>" in [x.tok for x in self.qualified_name.tokens]
380
381 if not is_lambda:
382 def_token = _Token(self.i_line, None, "def ")
383 self.tokens.append(def_token)
384
385 for token in self.qualified_name.tokens:
386 self.tokens.append(token)
387 if not is_lambda:
388 token.mark_after(def_token)
389 prev = token
390
391 open_parens_token = _Token(self.i_line, None, "(", after=prev)
392 self.tokens.append(open_parens_token)
393 prev = open_parens_token
394
395 code = self.code.instruction.argval
396
397 if default_node:
398 defaults = ([_SENTINEL] * (len(code.co_varnames) - len(default_node.instruction.argval))) + list(
399 default_node.instruction.argval
400 )
401 else:
402 defaults = [_SENTINEL] * len(code.co_varnames)
403
404 for i, arg in enumerate(code.co_varnames):
405 if i > 0:
406 comma_token = _Token(prev.i_line, None, ", ", after=prev)
407 self.tokens.append(comma_token)
408 prev = comma_token
409
410 arg_token = _Token(self.i_line, None, arg, after=prev)
411 self.tokens.append(arg_token)
412
413 default = defaults[i]
414 if default is not _SENTINEL:
415 eq_token = _Token(default_node.i_line, None, "=", after=prev)
416 self.tokens.append(eq_token)
417 prev = eq_token
418
419 default_token = _Token(default_node.i_line, None, str(default), after=prev)
420 self.tokens.append(default_token)
421 prev = default_token
422
423 tok_close_parens = _Token(prev.i_line, None, "):", after=prev)
424 self.tokens.append(tok_close_parens)
425
426 self._write_tokens()
427
428 stack.push(self)
429 self.writer.indent(prev.i_line + 1)
430 self.writer.dedent(max(self.disassembler.merge_code(code)))

448def _compose_line_contents(line_contents, previous_line_tokens):
449 lst = []
450 handled = set()
451
452 add_to_end_of_line = []
453 delete_indexes = []
454 for i, token in enumerate(line_contents):
455 if token.end_of_line:
456 add_to_end_of_line.append(token)
457 delete_indexes.append(i)
458 for i in reversed(delete_indexes):
459 del line_contents[i]
460 del delete_indexes
461
462 while line_contents:
463 added = False
464 delete_indexes = []
465
466 for i, token in enumerate(line_contents):
467 after_tokens = token.get_after_tokens()
468 for after in after_tokens:
469 if after not in handled and after not in previous_line_tokens:
470 break
471 else:
472 added = True
473 previous_line_tokens.add(token)
474 handled.add(token)
475 lst.append(token.tok)
476 delete_indexes.append(i)
477
478 for i in reversed(delete_indexes):
479 del line_contents[i]
480
481 if not added:
482 if add_to_end_of_line:
483 line_contents.extend(add_to_end_of_line)
484 del add_to_end_of_line[:]
485 continue
486
487 # Something is off, let's just add as is.
488 for token in line_contents:
489 if token not in handled:
490 lst.append(token.tok)
491
492 stream = StringIO()
493 _print_after_info(line_contents, stream)
494 pydev_log.critical("Error. After markers are not correct:\n%s", stream.getvalue())
495 break
496 return "".join(lst)

499class _PyCodeToSource(object):
500 def __init__(self, co, memo=None):
501 if memo is None:
502 memo = {}
503 self.memo = memo
504 self.co = co
505 self.instructions = list(iter_instructions(co))
506 self.stack = _Stack()
507 self.writer = _Writer()
508
509 def _process_next(self, i_line):
510 instruction = self.instructions.pop(0)
511 handler_class = _op_name_to_handler.get(instruction.opname)
512 if handler_class is not None:
513 s = handler_class(i_line, instruction, self.stack, self.writer, self)
514 if DEBUG:
515 print(s)
516
517 else:
518 if DEBUG:
519 print("UNHANDLED", instruction)
520
521 def build_line_to_contents(self):
522 co = self.co
523
524 op_offset_to_line = dict(dis.findlinestarts(co))
525 curr_line_index = 0
526
527 instructions = self.instructions
528 while instructions:
529 instruction = instructions[0]
530 new_line_index = op_offset_to_line.get(instruction.offset)
531 if new_line_index is not None:
532 curr_line_index = new_line_index
533
534 self._process_next(curr_line_index)
535 return self.writer.line_to_contents
536
537 def merge_code(self, code):
538 if DEBUG:
539 print("merge code ----")
540 # for d in dir(code):
541 # if not d.startswith('_'):
542 # print(d, getattr(code, d))
543 line_to_contents = _PyCodeToSource(code, self.memo).build_line_to_contents()
544 lines = []
545 for line, contents in sorted(line_to_contents.items()):
546 lines.append(line)
547 self.writer.get_line(line).extend(contents)
548 if DEBUG:
549 print("end merge code ----")
550 return lines
551
552 def disassemble(self):
553 show_lines = False
554 line_to_contents = self.build_line_to_contents()
555 stream = StringIO()
556 last_line = 0
557 indent = ""
558 previous_line_tokens = set()
559 for i_line, contents in sorted(line_to_contents.items()):
560 while last_line < i_line - 1:
561 if show_lines:
562 stream.write("%s.\n" % (last_line + 1,))
563 else:
564 stream.write("\n")
565 last_line += 1
566
567 line_contents = []
568 dedents_found = 0
569 for part in contents:
570 if part is INDENT_MARKER:
571 if DEBUG:
572 print("found indent", i_line)
573 indent += " "
574 continue
575 if part is DEDENT_MARKER:
576 if DEBUG:
577 print("found dedent", i_line)
578 dedents_found += 1
579 continue
580 line_contents.append(part)
581
582 s = indent + _compose_line_contents(line_contents, previous_line_tokens)
583 if show_lines:
584 stream.write("%s. %s\n" % (i_line, s))
585 else:
586 stream.write("%s\n" % s)
587
588 if dedents_found:
589 indent = indent[: -(4 * dedents_found)]
590 last_line = i_line
591
592 return stream.getvalue()
