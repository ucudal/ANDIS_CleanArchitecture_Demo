37def write_headers(headers, write):
38 # "Since the Host field-value is critical information for handling a
39 # request, a user agent SHOULD generate Host as the first header field
40 # following the request-line." - RFC 7230
41 for name, value in headers:
42 if name == b"host":
43 write(bytesmod(b"%s: %s\r\n", (name, value)))
44 for name, value in headers:
45 if name != b"host":
46 write(bytesmod(b"%s: %s\r\n", (name, value)))
47 write(b"\r\n")

85# These are all careful not to do anything to 'data' except call len(data) and
86# write(data). This allows us to transparently pass-through funny objects,
87# like placeholder objects referring to files on disk that will be sent via
88# sendfile(2).
89#
90class ContentLengthWriter(BodyWriter):
91 def __init__(self, length):
92 self._length = length
93
94 def send_data(self, data, write):
95 self._length -= len(data)
96 if self._length < 0:
97 raise LocalProtocolError("Too much data for declared Content-Length")
98 write(data)
99
100 def send_eom(self, headers, write):
101 if self._length != 0:
102 raise LocalProtocolError("Too little data for declared Content-Length")
103 if headers:
104 raise LocalProtocolError("Content-Length and trailers don't mix")
