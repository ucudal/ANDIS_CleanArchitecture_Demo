78def _find_by_sys_executable() -> str | None:
79 def _norm(path: str) -> str | None:
80 _, exe = os.path.split(path.lower())
81 exe, _, _ = exe.partition('.exe')
82 if exe not in {'python', 'pythonw'} and find_executable(exe):
83 return exe
84 return None
85
86 # On linux, I see these common sys.executables:
87 #
88 # system `python`: /usr/bin/python -> python2.7
89 # system `python2`: /usr/bin/python2 -> python2.7
90 # virtualenv v: v/bin/python (will not return from this loop)
91 # virtualenv v -ppython2: v/bin/python -> python2
92 # virtualenv v -ppython2.7: v/bin/python -> python2.7
93 # virtualenv v -ppypy: v/bin/python -> v/bin/pypy
94 for path in (sys.executable, os.path.realpath(sys.executable)):
95 exe = _norm(path)
96 if exe:
97 return exe
98 return None

101@functools.lru_cache(maxsize=1)
102def get_default_version() -> str: # pragma: no cover (platform dependent)
103 # First attempt from `sys.executable` (or the realpath)
104 exe = _find_by_sys_executable()
105 if exe:
106 return exe
107
108 # Next try the `pythonX.X` executable
109 exe = f'python{sys.version_info[0]}.{sys.version_info[1]}'
110 if find_executable(exe):
111 return exe
112
113 if _find_by_py_launcher(exe):
114 return exe
115
116 # We tried!
117 return C.DEFAULT

120def _sys_executable_matches(version: str) -> bool:
121 if version == 'python':
122 return True
123 elif not version.startswith('python'):
124 return False
125
126 try:
127 info = tuple(int(p) for p in version[len('python'):].split('.'))
128 except ValueError:
129 return False
130
131 return sys.version_info[:len(info)] == info

134def norm_version(version: str) -> str | None:
135 if version == C.DEFAULT: # use virtualenv's default
136 return None
137 elif _sys_executable_matches(version): # virtualenv defaults to our exe
138 return None
139
140 if os.name == 'nt': # pragma: no cover (windows)
141 version_exec = _find_by_py_launcher(version)
142 if version_exec:
143 return version_exec
144
145 # Try looking up by name
146 version_exec = find_executable(version)
147 if version_exec and version_exec != version:
148 return version_exec
149
150 # Otherwise assume it is a path
151 return os.path.expanduser(version)

161def health_check(prefix: Prefix, version: str) -> str | None:
162 envdir = lang_base.environment_dir(prefix, ENVIRONMENT_DIR, version)
163 pyvenv_cfg = os.path.join(envdir, 'pyvenv.cfg')
164
165 # created with "old" virtualenv
166 if not os.path.exists(pyvenv_cfg):
167 return 'pyvenv.cfg does not exist (old virtualenv?)'
168
169 exe_name = win_exe('python')
170 py_exe = prefix.path(bin_dir(envdir), exe_name)
171 cfg = _read_pyvenv_cfg(pyvenv_cfg)
172
173 if 'version_info' not in cfg:
174 return "created virtualenv's pyvenv.cfg is missing `version_info`"
175
176 # always use uncached lookup here in case we replaced an unhealthy env
177 virtualenv_version = _version_info.__wrapped__(py_exe)
178 if virtualenv_version != cfg['version_info']:
179 return (
180 f'virtualenv python version did not match created version:\n'
181 f'- actual version: {virtualenv_version}\n'
182 f'- expected version: {cfg["version_info"]}\n'
183 )
184
185 # made with an older version of virtualenv? skip `base-executable` check
186 if 'base-executable' not in cfg:
187 return None
188
189 base_exe_version = _version_info(cfg['base-executable'])
190 if base_exe_version != cfg['version_info']:
191 return (
192 f'base executable python version does not match created version:\n'
193 f'- base-executable version: {base_exe_version}\n'
194 f'- expected version: {cfg["version_info"]}\n'
195 )
196 else:
197 return None
