30class BoxConstraints:
31 """Represents a box of potentially variable width and height.
32 Among other uses, this can be leveraged to produce a variably sized
33 box with a fixed aspect ratio.
34
35 If width/height are not defined yet, they can be set by assigning to the
36 :attr:`width` and :attr:`height` attributes.
37 """
38
39 _width: Optional[int]
40 _height: Optional[int]
41 _ar: Optional[Fraction]
42 _fully_specified: bool

44 def __init__(self, width=None, height=None, aspect_ratio: Fraction = None):
45 self._width = int(width) if width is not None else None
46 self._height = int(height) if height is not None else None
47
48 fully_specified = False
49
50 self._ar = None
51 if width is None and height is None and aspect_ratio is None:
52 return
53 elif width is not None and height is not None:
54 if aspect_ratio is not None:
55 raise BoxSpecificationError # overspecified
56 self._ar = Fraction(self._width, self._height)
57 fully_specified = True
58 elif aspect_ratio is not None:
59 self._ar = aspect_ratio
60 if height is not None:
61 self._width = height * aspect_ratio
62 elif width is not None:
63 self._height = width / aspect_ratio
64
65 self._fully_specified = fully_specified
66
67 def _recalculate(self):
68 if self._width is not None and self._height is not None:
69 self._ar = Fraction(self._width, self._height)
70 self._fully_specified = True
71 elif self._ar is not None:
72 if self._height is not None:
73 self._width = int(self._height * self._ar)
74 self._fully_specified = True
75 elif self._width is not None:
76 self._height = int(self._width / self._ar)
77 self._fully_specified = True

162class InnerScaling(enum.Enum):
163 """Class representing a scaling convention."""
164
165 NO_SCALING = enum.auto()
166 """Never scale content."""
167
168 STRETCH_FILL = enum.auto()
169 """Scale content to fill the entire container."""
170
171 STRETCH_TO_FIT = enum.auto()
172 """
173 Scale content while preserving aspect ratio until either the maximal
174 width or maximal height is reached.
175 """
176
177 SHRINK_TO_FIT = enum.auto()
178 """
179 Scale content down to fit in the container, while preserving the original
180 aspect ratio.
181 """
182
183 @classmethod
184 def from_config(cls, config_str: str) -> 'InnerScaling':
185 """
186 Convert from a configuration string.
187
188 :param config_str:
189 A string: 'none', 'stretch-fill', 'stretch-to-fit', 'shrink-to-fit'
190 :return:
191 An :class:`.InnerScaling` value.
192 :raise ConfigurationError: on unexpected string inputs.
193 """
194 try:
195 return {
196 'none': InnerScaling.NO_SCALING,
197 'stretch-fill': InnerScaling.STRETCH_FILL,
198 'stretch-to-fit': InnerScaling.STRETCH_TO_FIT,
199 'shrink-to-fit': InnerScaling.SHRINK_TO_FIT
200 }[config_str.lower()]
201 except KeyError:
202 raise ConfigurationError(
203 f"'{config_str}' is not a valid inner scaling setting; valid "
204 f"values are 'none', 'stretch-fill', 'stretch-to-fit', "
205 f"'shrink-to-fit'."
206 )

273 @property

277 def align(self, container_len: int, inner_len: int,
278 pre_margin, post_margin) -> int:
279
280 effective_max_len = Margins.effective(
281 'length', container_len, pre_margin, post_margin
282 )
283
284 if self == AxisAlignment.ALIGN_MAX:
285 # we want to start as far up the axis as possible.
286 # Ignoring margins, that would be at container_len - inner_len
287 # This computation makes sure that there's room for post_margin
288 # in the back.
289 return container_len - inner_len - post_margin
290 elif self == AxisAlignment.ALIGN_MIN:
291 return pre_margin
292 elif inner_len > effective_max_len:
293 logger.warning(
294 f"Content box width/height {inner_len} is too wide for "
295 f"container size {container_len} with margins "
296 f"({pre_margin}, {post_margin}); post_margin will be ignored"
297 )
298 return pre_margin
299 elif self == AxisAlignment.ALIGN_MID:
300 # we'll center the inner content *within* the margins
301 inner_offset = (effective_max_len - inner_len) // 2
302 return pre_margin + inner_offset
303
304
305# Class variables in enums are weird, so let's put this here
306_alignment_opposites = {
307 AxisAlignment.ALIGN_MID: AxisAlignment.ALIGN_MID,
308 AxisAlignment.ALIGN_MIN: AxisAlignment.ALIGN_MAX,
309 AxisAlignment.ALIGN_MAX: AxisAlignment.ALIGN_MIN
310}

393 """Internal helper method to compute effective margins."""

439class SimpleBoxLayoutRule(ConfigurableMixin):
440 """
441 Class describing alignment, scaling and margin rules for a box
442 positioned inside another box.
443 """

447 Horizontal alignment settings.

457 Container (inner) margins. Defaults to all zeroes.

465 @classmethod
466 def process_entries(cls, config_dict):
467 # in config processing, we default to MID for everything
468 x_align = config_dict.get('x_align', AxisAlignment.ALIGN_MID)
469 if isinstance(x_align, str):
470 x_align = AxisAlignment.from_x_align(x_align)
471 config_dict['x_align'] = x_align
472
473 y_align = config_dict.get('y_align', AxisAlignment.ALIGN_MID)
474 if isinstance(y_align, str):

476 config_dict['y_align'] = y_align

478 scaling = config_dict.get('inner_content_scaling', None)
479 if scaling is not None:

484 return SimpleBoxLayoutRule(

486 margins=new_margins,
487 inner_content_scaling=self.inner_content_scaling

490 def fit(self, container_box: BoxConstraints,
491 inner_nat_width: int, inner_nat_height: int) -> Positioning:
492 """
493 Position and possibly scale a box within a container, according
494 to this layout rule.
495
496 :param container_box:
497 :class:`.BoxConstraints` describing the container.
498 :param inner_nat_width:
499 The inner box's natural width.
500 :param inner_nat_height:
501 The inner box's natural height.
502 :return:
503 A :class:`.Positioning` describing the scaling & position of the
504 lower left corner of the inner box.
505 """
506
507 margins = self.margins
508 scaling = self.inner_content_scaling
509 x_scale = y_scale = 1
510 if scaling != InnerScaling.NO_SCALING and \
511 container_box.width_defined and container_box.height_defined:
512 eff_width = margins.effective_width(container_box.width)
513 eff_height = margins.effective_height(container_box.height)
514
515 x_scale = eff_width / inner_nat_width
516 y_scale = eff_height / inner_nat_height
517 if scaling == InnerScaling.STRETCH_TO_FIT:
518 x_scale = y_scale = min(x_scale, y_scale)
519 elif scaling == InnerScaling.SHRINK_TO_FIT:
520 # same as stretch to fit, with the additional stipulation
521 # that it can't scale up, only down.
522 x_scale = y_scale = min(x_scale, y_scale, 1)
523
524 x_pos = _aln_width(
525 self.x_align, container_box,
526 inner_nat_width * x_scale,
527 margins.left, margins.right
528 )
529 y_pos = _aln_height(
530 self.y_align, container_box,
531 inner_nat_height * y_scale,
532 margins.bottom, margins.top
533 )
534 return Positioning(
535 x_pos=x_pos, y_pos=y_pos, x_scale=x_scale, y_scale=y_scale
536 )
