75def get_instructions_to_add(stop_at_line, _pydev_stop_at_break=_pydev_stop_at_break, _pydev_needs_stop_at_break=_pydev_needs_stop_at_break):
76 """
77 This is the bytecode for something as:
78
79 if _pydev_needs_stop_at_break():
80 _pydev_stop_at_break()
81
82 but with some special handling for lines.
83 """
84 # Good reference to how things work regarding line numbers and jumps:
85 # https://github.com/python/cpython/blob/3.6/Objects/lnotab_notes.txt
86
87 # Usually use a stop line -1, but if that'd be 0, using line +1 is ok too.
88 spurious_line = stop_at_line - 1
89 if spurious_line <= 0:
90 spurious_line = stop_at_line + 1
91
92 label = Label()
93 return [
94 # -- if _pydev_needs_stop_at_break():
95 Instr("LOAD_CONST", _pydev_needs_stop_at_break, lineno=stop_at_line),
96 Instr("LOAD_CONST", stop_at_line, lineno=stop_at_line),
97 Instr("CALL_FUNCTION", 1, lineno=stop_at_line),
98 Instr("POP_JUMP_IF_FALSE", label, lineno=stop_at_line),
99 # -- _pydev_stop_at_break()
100 #
101 # Note that this has line numbers -1 so that when the NOP just below
102 # is executed we have a spurious line event.
103 Instr("LOAD_CONST", _pydev_stop_at_break, lineno=spurious_line),
104 Instr("LOAD_CONST", stop_at_line, lineno=spurious_line),
105 Instr("CALL_FUNCTION", 1, lineno=spurious_line),
106 Instr("POP_TOP", lineno=spurious_line),
107 # Reason for the NOP: Python will give us a 'line' trace event whenever we forward jump to
108 # the first instruction of a line, so, in the case where we haven't added a programmatic
109 # breakpoint (either because we didn't hit a breakpoint anymore or because it was already
110 # tracing), we don't want the spurious line event due to the line change, so, we make a jump
111 # to the instruction right after the NOP so that the spurious line event is NOT generated in
112 # this case (otherwise we'd have a line event even if the line didn't change).
113 Instr("NOP", lineno=stop_at_line),
114 label,
115 ]

153class _HelperBytecodeList(object):
154 """
155 A helper double-linked list to make the manipulation a bit easier (so that we don't need
156 to keep track of indices that change) and performant (because adding multiple items to
157 the middle of a regular list isn't ideal).
158 """
159
160 def __init__(self, lst=None):
161 self._head = None
162 self._tail = None
163 if lst:
164 node = self
165 for item in lst:
166 node = node.append(item)
167
168 def append(self, data):
169 if self._tail is None:
170 node = _Node(data)
171 self._head = self._tail = node
172 return node
173 else:
174 node = self._tail = self.tail.append(data)
175 return node
176
177 @property
178 def head(self):
179 node = self._head
180 # Manipulating the node directly may make it unsynchronized.
181 while node.prev:
182 self._head = node = node.prev
183 return node
184
185 @property
186 def tail(self):
187 node = self._tail
188 # Manipulating the node directly may make it unsynchronized.
189 while node.next:
190 self._tail = node = node.next
191 return node
192
193 def __iter__(self):
194 node = self.head
195
196 while node:
197 yield node.data
198 node = node.next

201_PREDICT_TABLE = {
202 "LIST_APPEND": ("JUMP_ABSOLUTE",),
203 "SET_ADD": ("JUMP_ABSOLUTE",),
204 "GET_ANEXT": ("LOAD_CONST",),
205 "GET_AWAITABLE": ("LOAD_CONST",),
206 "DICT_MERGE": ("CALL_FUNCTION_EX",),
207 "MAP_ADD": ("JUMP_ABSOLUTE",),
208 "COMPARE_OP": (
209 "POP_JUMP_IF_FALSE",
210 "POP_JUMP_IF_TRUE",
211 ),
212 "IS_OP": (
213 "POP_JUMP_IF_FALSE",
214 "POP_JUMP_IF_TRUE",
215 ),
216 "CONTAINS_OP": (
217 "POP_JUMP_IF_FALSE",
218 "POP_JUMP_IF_TRUE",
219 ),
220 # Note: there are some others with PREDICT on ceval, but they have more logic
221 # and it needs more experimentation to know how it behaves in the static generated
222 # code (and it's only an issue for us if there's actually a line change between
223 # those, so, we don't have to really handle all the cases, only the one where
224 # the line number actually changes from one instruction to the predicted one).
225}
226
227# 3.10 optimizations include copying code branches multiple times (for instance
228# if the body of a finally has a single assign statement it can copy the assign to the case
229# where an exception happens and doesn't happen for optimization purposes) and as such
230# we need to add the programmatic breakpoint multiple times.
231TRACK_MULTIPLE_BRANCHES = sys.version_info[:2] >= (3, 10)
232
233# When tracking multiple branches, we try to fix the bytecodes which would be PREDICTED in the
234# Python eval loop so that we don't have spurious line events that wouldn't usually be issued
235# in the tracing as they're ignored due to the eval prediction (even though they're in the bytecode).
236FIX_PREDICT = sys.version_info[:2] >= (3, 10)

267 # return False, code_to_modify
