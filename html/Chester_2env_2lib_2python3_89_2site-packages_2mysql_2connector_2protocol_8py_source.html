115 def make_auth(
116 self,
117 handshake: Optional[HandShakeType],
118 username: Optional[StrOrBytes] = None,
119 password: Optional[str] = None,
120 database: Optional[str] = None,
121 charset: int = 45,
122 client_flags: int = 0,
123 max_allowed_packet: int = 1073741824,
124 ssl_enabled: bool = False,
125 auth_plugin: Optional[str] = None,
126 conn_attrs: Optional[ConnAttrsType] = None,
127 auth_plugin_class: Optional[str] = None,
128 ) -> bytes:
129 """Make a MySQL Authentication packet"""
130 if handshake is None:
131 handshake = {}
132 try:
133 auth_data: Optional[bytes] = handshake["auth_data"]
134 auth_plugin = auth_plugin or handshake["auth_plugin"]
135 except (TypeError, KeyError) as err:
136 raise ProgrammingError(
137 f"Handshake misses authentication info ({err})"
138 ) from None
139
140 if not username:
141 username = b""
142 try:
143 username_bytes = username.encode("utf8") # type: ignore[union-attr]
144 except AttributeError:
145 # Username is already bytes
146 username_bytes = username
147
148 filler = "x" * 22
149 username_len = len(username_bytes)
150
151 packet = struct.pack(
152 f"<IIH{filler}{username_len}sx",
153 client_flags,
154 max_allowed_packet,
155 charset,
156 username_bytes,
157 )
158
159 packet += self._auth_response(
160 client_flags,
161 username,
162 password,
163 database,
164 auth_plugin_class,
165 auth_plugin,
166 auth_data,
167 ssl_enabled,
168 )
169
170 packet += self._connect_with_db(client_flags, database)
171
172 if client_flags & ClientFlag.PLUGIN_AUTH:
173 packet += auth_plugin.encode("utf8") + b"\x00"
174
175 if (client_flags & ClientFlag.CONNECT_ARGS) and conn_attrs is not None:
176 packet += self.make_conn_attrs(conn_attrs)
177
178 return packet

180 @staticmethod
181 def make_conn_attrs(conn_attrs: ConnAttrsType) -> bytes:
182 """Encode the connection attributes"""
183 for attr_name in conn_attrs:
184 if conn_attrs[attr_name] is None:
185 conn_attrs[attr_name] = ""
186 conn_attrs_len = (
187 sum(len(x) + len(conn_attrs[x]) for x in conn_attrs)
188 + len(conn_attrs.keys())
189 + len(conn_attrs.values())
190 )
191
192 conn_attrs_packet = struct.pack("<B", conn_attrs_len)
193 for attr_name in conn_attrs:
194 conn_attrs_packet += struct.pack("<B", len(attr_name))
195 conn_attrs_packet += attr_name.encode("utf8")
196 conn_attrs_packet += struct.pack("<B", len(conn_attrs[attr_name]))
197 conn_attrs_packet += conn_attrs[attr_name].encode("utf8") # type: ignore[union-attr]
198 return conn_attrs_packet

286 @staticmethod
287 def parse_handshake(packet: bytes) -> HandShakeType:
288 """Parse a MySQL Handshake-packet"""
289 res = {}
290 res["protocol"] = struct.unpack("<xxxxB", packet[0:5])[0]
291 if res["protocol"] != PROTOCOL_VERSION:
292 raise DatabaseError(
293 f"Protocol mismatch; server version = {res['protocol']}, "
294 f"client version = {PROTOCOL_VERSION}"
295 )
296 packet, res["server_version_original"] = utils.read_string(
297 packet[5:], end=b"\x00"
298 )
299
300 (
301 res["server_threadid"],
302 auth_data1,
303 capabilities1,
304 res["charset"],
305 res["server_status"],
306 capabilities2,
307 auth_data_length,
308 ) = struct.unpack("<I8sx2sBH2sBxxxxxxxxxx", packet[0:31])
309 res["server_version_original"] = res["server_version_original"].decode()
310
311 packet = packet[31:]
312
313 capabilities = utils.intread(capabilities1 + capabilities2)
314 auth_data2 = b""
315 if capabilities & ClientFlag.SECURE_CONNECTION:
316 size = min(13, auth_data_length - 8) if auth_data_length else 13
317 auth_data2 = packet[0:size]
318 packet = packet[size:]
319 if auth_data2[-1] == 0:
320 auth_data2 = auth_data2[:-1]
321
322 if capabilities & ClientFlag.PLUGIN_AUTH:
323 if b"\x00" not in packet and res["server_version_original"].startswith(
324 "5.5.8"
325 ):
326 # MySQL server 5.5.8 has a bug where end byte is not send
327 (packet, res["auth_plugin"]) = (b"", packet)
328 else:
329 (packet, res["auth_plugin"]) = utils.read_string(packet, end=b"\x00")
330 res["auth_plugin"] = res["auth_plugin"].decode("utf-8")
331 else:
332 res["auth_plugin"] = "mysql_native_password"
333
334 res["auth_data"] = auth_data1 + auth_data2
335 res["capabilities"] = capabilities
336 return res

412 def parse_eof(self, packet: bytes) -> EofPacketType:
413 """Parse a MySQL EOF-packet"""
414 if packet[4] == 0:
415 # EOF packet deprecation
416 return self.parse_ok(packet)
417
418 err_msg = "Failed parsing EOF packet."
419 res = {}
420 try:
421 unpacked = struct.unpack("<xxxBBHH", packet)
422 except struct.error as err:
423 raise InterfaceError(err_msg) from err
424
425 if not (unpacked[1] == 254 and len(packet) <= 9):
426 raise InterfaceError(err_msg)
427
428 res["warning_count"] = unpacked[2]
429 res["status_flag"] = unpacked[3]
430 return res

432 @staticmethod
433 def parse_statistics(packet: bytes, with_header: bool = True) -> StatsPacketType:
434 """Parse the statistics packet"""
435 errmsg = "Failed getting COM_STATISTICS information"
436 res: Dict[str, Union[int, Decimal]] = {}
437 # Information is separated by 2 spaces
438 pairs: List[bytes] = [b""]
439 lbl: StrOrBytes = b""
440 if with_header:
441 pairs = packet[4:].split(b"\x20\x20")
442 else:
443 pairs = packet.split(b"\x20\x20")
444 for pair in pairs:
445 try:
446 lbl, val = [v.strip() for v in pair.split(b":", 2)]
447 except ValueError as err:
448 raise InterfaceError(errmsg) from err
449
450 # It's either an integer or a decimal
451 lbl = lbl.decode("utf-8")
452 try:
453 res[lbl] = int(val)
454 except (KeyError, ValueError):
455 try:
456 res[lbl] = Decimal(val.decode("utf-8"))
457 except DecimalException as err:
458 raise InterfaceError(f"{errmsg} ({lbl}:{repr(val)})") from err
459 return res

461 def read_text_result(
462 self, sock: SocketType, version: Tuple[int, ...], count: int = 1
463 ) -> Tuple[List[Tuple[Optional[bytes], ...]], Optional[EofPacketType],]:
464 """Read MySQL text result
465
466 Reads all or given number of rows from the socket.
467
468 Returns a tuple with 2 elements: a list with all rows and
469 the EOF packet.
470 """
471 # Keep unused 'version' for API backward compatibility
472 _ = version
473 rows = []
474 eof = None
475 rowdata = None
476 i = 0
477 while True:
478 if eof or i == count:
479 break
480 packet = sock.recv()
481 if packet.startswith(b"\xff\xff\xff"):
482 datas = [packet[4:]]
483 packet = sock.recv()
484 while packet.startswith(b"\xff\xff\xff"):
485 datas.append(packet[4:])
486 packet = sock.recv()
487 datas.append(packet[4:])
488 rowdata = utils.read_lc_string_list(bytearray(b"").join(datas))
489 elif packet[4] == 254 and packet[0] < 7:
490 eof = self.parse_eof(packet)
491 rowdata = None
492 else:
493 eof = None
494 rowdata = utils.read_lc_string_list(packet[4:])
495 if eof is None and rowdata is not None:
496 rows.append(rowdata)
497 elif eof is None and rowdata is None:
498 raise get_exception(packet)
499 i += 1
500 return rows, eof

547 @staticmethod

552 """Parse a timestamp from a binary packet"""

570 day=packet[4],

573 second=packet[7],

579 @staticmethod
580 def _parse_binary_time(packet: bytes) -> Tuple[bytes, datetime.timedelta]:
581 """Parse a time value from a binary packet"""
582 length = packet[0]
583 if not length:
584 return (packet[1:], datetime.timedelta())
585 data = packet[1 : length + 1]
586 mcs = 0
587 if length > 8:
588 mcs = struct.unpack("<I", data[8:])[0]
589 days = struct.unpack("<I", data[1:5])[0]
590 if data[0] == 1:
591 days *= -1

595 microseconds=mcs,
596 minutes=data[6],
597 hours=data[5],
598 )
599

602 def _parse_binary_values(
603 self,
604 fields: List[DescriptionType],
605 packet: bytes,
606 charset: str = "utf-8",
607 ) -> Tuple[ParseValueFromBinaryResultPacketTypes, ...]:
608 """Parse values from a binary result packet"""
609 null_bitmap_length = (len(fields) + 7 + 2) // 8
610 null_bitmap = [int(i) for i in packet[0:null_bitmap_length]]
611 packet = packet[null_bitmap_length:]
612
613 values: List[Any] = []
614 value: Any = None
615 for pos, field in enumerate(fields):
616 if null_bitmap[int((pos + 2) / 8)] & (1 << (pos + 2) % 8):
617 values.append(None)
618 continue
619 if field[1] in (
620 FieldType.TINY,
621 FieldType.SHORT,
622 FieldType.INT24,
623 FieldType.LONG,
624 FieldType.LONGLONG,
625 ):
626 packet, value = self._parse_binary_integer(packet, field)
627 values.append(value)
628 elif field[1] in (FieldType.DOUBLE, FieldType.FLOAT):
629 packet, value = self._parse_binary_float(packet, field)
630 values.append(value)
631 elif field[1] in (FieldType.DECIMAL, FieldType.NEWDECIMAL):
632 packet, value = self._parse_binary_new_decimal(packet, charset)
633 values.append(value)
634 elif field[1] in (
635 FieldType.DATETIME,
636 FieldType.DATE,
637 FieldType.TIMESTAMP,
638 ):
639 (packet, value) = self._parse_binary_timestamp(packet, field[1])
640 values.append(value)
641 elif field[1] == FieldType.TIME:
642 (packet, value) = self._parse_binary_time(packet)
643 values.append(value)
644 else:
645 (packet, value) = utils.read_lc_string(packet)
646 try:
647 values.append(value.decode(charset))
648 except UnicodeDecodeError:
649 values.append(value)
650
651 return tuple(values)

653 def read_binary_result(
654 self,
655 sock: SocketType,
656 columns: List[DescriptionType],
657 count: int = 1,
658 charset: str = "utf-8",
659 ) -> Tuple[
660 List[Tuple[ParseValueFromBinaryResultPacketTypes, ...]],
661 Optional[EofPacketType],
662 ]:
663 """Read MySQL binary protocol result
664
665 Reads all or given number of binary resultset rows from the socket.
666 """
667 rows = []
668 eof = None
669 values = None
670 i = 0
671 while True:
672 if eof is not None:
673 break
674 if i == count:
675 break
676 packet = sock.recv()
677 if packet[4] == 254:
678 eof = self.parse_eof(packet)
679 values = None
680 elif packet[4] == 0:
681 eof = None
682 values = self._parse_binary_values(columns, packet[5:], charset)
683 if eof is None and values is not None:
684 rows.append(values)
685 elif eof is None and values is None:
686 raise get_exception(packet)
687 i += 1
688 return (rows, eof)

708 @staticmethod
709 def prepare_binary_integer(value: int) -> Tuple[bytes, int, int]:
710 """Prepare an integer for the MySQL binary protocol"""
711 field_type = None
712 flags = 0
713 if value < 0:
714 if value >= -128:
715 format_ = "<b"
716 field_type = FieldType.TINY
717 elif value >= -32768:
718 format_ = "<h"
719 field_type = FieldType.SHORT
720 elif value >= -2147483648:
721 format_ = "<i"
722 field_type = FieldType.LONG
723 else:
724 format_ = "<q"
725 field_type = FieldType.LONGLONG
726 else:
727 flags = 128
728 if value <= 255:
729 format_ = "<B"
730 field_type = FieldType.TINY
731 elif value <= 65535:
732 format_ = "<H"
733 field_type = FieldType.SHORT
734 elif value <= 4294967295:
735 format_ = "<I"
736 field_type = FieldType.LONG
737 else:
738 field_type = FieldType.LONGLONG
739 format_ = "<Q"
740 return (struct.pack(format_, value), field_type, flags)
