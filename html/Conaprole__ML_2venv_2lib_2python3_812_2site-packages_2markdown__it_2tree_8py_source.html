23 """A Markdown syntax tree node.
24
25 A class that can be used to construct a tree representation of a linear
26 `markdown-it-py` token stream.
27
28 Each node in the tree represents either:
29 - root of the Markdown document
30 - a single unnested `Token`
31 - a `Token` "_open" and "_close" token pair, and the tokens nested in

33 """

35 def __init__(
36 self, tokens: Sequence[Token] = (), *, create_root: bool = True
37 ) -> None:
38 """Initialize a `SyntaxTreeNode` from a token stream.
39
40 If `create_root` is True, create a root node for the document.
41 """
42 # Only nodes representing an unnested token have self.token
43 self.token: Token | None = None
44
45 # Only containers have nester tokens
46 self.nester_tokens: _NesterTokens | None = None
47
48 # Root node does not have self.parent
49 self._parent: Any = None
50
51 # Empty list unless a non-empty container, or unnested token that has
52 # children (i.e. inline or img)
53 self._children: list[Any] = []
54
55 if create_root:
56 self._set_children_from_tokens(tokens)
57 return

59 if not tokens:
60 raise ValueError(
61 "Can only create root from empty token sequence."
62 " Set `create_root=True`."
63 )
64 elif len(tokens) == 1:
65 inline_token = tokens[0]

67 raise ValueError(

73 else:

94 def recursive_collect_tokens(node: _NodeType, token_list: list[Token]) -> None:
95 if node.type == "root":
96 for child in node.children:
97 recursive_collect_tokens(child, token_list)
98 elif node.token:
99 token_list.append(node.token)
100 else:
101 assert node.nester_tokens

199 def _set_children_from_tokens(self, tokens: Sequence[Token]) -> None:
200 """Convert the token stream to a tree structure and set the resulting
201 nodes as children of `self`."""
202 reversed_tokens = list(reversed(tokens))
203 while reversed_tokens:
204 token = reversed_tokens.pop()
205
206 if not token.nesting:
207 self._add_child([token])
208 continue
209 if token.nesting != 1:
210 raise ValueError("Invalid token nesting")
211
212 nested_tokens = [token]
213 nesting = 1
214 while reversed_tokens and nesting:
215 token = reversed_tokens.pop()
216 nested_tokens.append(token)
217 nesting += token.nesting
218 if nesting:
219 raise ValueError(f"unclosed tokens starting {nested_tokens[0]}")
220
221 self._add_child(nested_tokens)

245 def walk(
246 self: _NodeType, *, include_self: bool = True
247 ) -> Generator[_NodeType, None, None]:
248 """Recursively yield all descendant nodes in the tree starting at self.
249
250 The order mimics the order of the underlying linear token
251 stream (i.e. depth first).
252 """
