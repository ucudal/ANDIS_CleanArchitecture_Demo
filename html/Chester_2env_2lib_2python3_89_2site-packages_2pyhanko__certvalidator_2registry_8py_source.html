112 Simple trustless certificate store.

122 def __init__(self):
123 self.certs = {}

127 def register(self, cert: x509.Certificate) -> bool:
128 """
129 Register a single certificate.
130
131 :param cert:
132 Certificate to add.
133 :return:
134 ``True`` if the certificate was added, ``False`` if it already
135 existed in this store.
136 """
137 if cert.issuer_serial in self.certs:
138 return False
139 self.certs[cert.issuer_serial] = cert
140 self._subject_map[cert.subject.hashable].append(cert)

143 else:
144 self._key_identifier_map[cert.public_key.sha1].append(cert)
145 return True

166class CertificateRegistry(SimpleCertificateStore):
167 """
168 Contains certificate lists used to build validation paths
169 """
170
171 # A dict with keys being asn1crypto.x509.Certificate.signature byte string.
172 # Each value is a bool - if the certificate is a CA cert.
173 _ca_lookup = None
174
175 def __init__(self,
176 trust_roots: Optional[Iterable[x509.Certificate]] = None,
177 extra_trust_roots: Optional[Iterable[x509.Certificate]] = None,
178 other_certs: Optional[Iterable[x509.Certificate]] = None,
179 *, cert_fetcher: CertificateFetcher = None):
180 """
181 :param trust_roots:
182 If the operating system's trust list should not be used, instead
183 pass a list of byte strings containing DER or PEM-encoded X.509
184 certificates, or asn1crypto.x509.Certificate objects. These
185 certificates will be used as the trust roots for the path being
186 built.
187
188 :param extra_trust_roots:
189 If the operating system's trust list should be used, but augmented
190 with one or more extra certificates. This should be a list of byte
191 strings containing DER or PEM-encoded X.509 certificates, or
192 asn1crypto.x509.Certificate objects.
193
194 :param other_certs:
195 A list of byte strings containing DER or PEM-encoded X.509
196 certificates, or a list of asn1crypto.x509.Certificate objects.
197 These other certs are usually provided by the service/item being
198 validated. In SSL, these would be intermediate chain certs.
199 """
200
201 super().__init__()
202
203 self._ca_lookup = set()
204
205 if other_certs is None:
206 other_certs = []
207 else:
208 other_certs = list(other_certs)
209
210 if trust_roots is None:
211 trust_roots = [e[0] for e in trust_list.get_list()]
212 else:
213 trust_roots = list(trust_roots)
214
215 if extra_trust_roots is not None:
216 trust_roots.extend(extra_trust_roots)
217
218 for trust_root in trust_roots:
219 self.register(trust_root)
220 self._ca_lookup.add(trust_root.signature)
221
222 for other_cert in other_certs:
223 self.register(other_cert)
224
225 self.fetcher = cert_fetcher

303 async def async_build_paths(
304 self, end_entity_cert: x509.Certificate):
305 """
306 Builds a list of ValidationPath objects from a certificate in the
307 operating system trust store to the end-entity certificate
308
309 :param end_entity_cert:
310 A byte string of a DER or PEM-encoded X.509 certificate, or an
311 instance of asn1crypto.x509.Certificate
312
313 :return:
314 A list of pyhanko_certvalidator.path.ValidationPath objects that

316 of the CA certs.

339 async def _walk_issuers(self, path, paths, failed_paths):
340 """
341 Recursively looks through the list of known certificates for the issuer
342 of the certificate specified, stopping once the certificate in question
343 is one contained within the CA certs list
344
345 :param path:
346 A ValidationPath object representing the current traversal of
347 possible paths
348
349 :param paths:
350 A list of completed ValidationPath objects. This is mutated as
351 results are found.
352
353 :param failed_paths:
354 A list of pyhanko_certvalidator.path.ValidationPath objects that failed due
355 to no matching issuer before reaching a certificate from the CA
356 certs list
357 """
358
359 if path.first.signature in self._ca_lookup:
360 paths.append(path)
361 return
362
363 new_branches = 0
364 for issuer in self._possible_issuers(path.first):
365 try:
366 await self._walk_issuers(
367 path.copy().prepend(issuer), paths, failed_paths
368 )
369 new_branches += 1
370 except DuplicateCertificateError:
371 pass
372
373 if not new_branches and self.fetcher is not None:
374 # attempt to download certs if there's nothing in the context
375 async for issuer in self.fetcher.fetch_cert_issuers(path.first):
376 # register the cert for future reference
377 self.add_other_cert(issuer)
378 try:
379 await self._walk_issuers(
380 path.copy().prepend(issuer), paths, failed_paths
381 )
382 new_branches += 1
383 except DuplicateCertificateError:
384 pass
385 if not new_branches:
386 failed_paths.append(path)

388 def _possible_issuers(self, cert):
389 """
390 Returns a generator that will list all possible issuers for the cert
391
392 :param cert:
393 An asn1crypto.x509.Certificate object to find the issuer of
394 """
395
396 issuer_hashable = cert.issuer.hashable
397 for issuer in self._subject_map[issuer_hashable]:
398 # Info from the authority key identifier extension can be used to
399 # eliminate possible options when multiple keys with the same
400 # subject exist, such as during a transition, or with cross-signing.
401 if cert.authority_key_identifier and issuer.key_identifier:
402 if cert.authority_key_identifier != issuer.key_identifier:
403 continue
404 elif cert.authority_issuer_serial:
405 if cert.authority_issuer_serial != issuer.issuer_serial:
406 continue
407
408 yield issuer
