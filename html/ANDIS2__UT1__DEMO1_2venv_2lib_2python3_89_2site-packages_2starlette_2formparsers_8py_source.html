83 async def parse(self) -> FormData:
84 # Callbacks dictionary.
85 callbacks: QuerystringCallbacks = {
86 "on_field_start": self.on_field_start,
87 "on_field_name": self.on_field_name,
88 "on_field_data": self.on_field_data,
89 "on_field_end": self.on_field_end,
90 "on_end": self.on_end,
91 }
92
93 # Create the parser.
94 parser = multipart.QuerystringParser(callbacks)
95 field_name = b""
96 field_value = b""
97
98 items: list[tuple[str, str | UploadFile]] = []
99
100 # Feed the parser with data from the request.
101 async for chunk in self.stream:
102 if chunk:
103 parser.write(chunk)
104 else:
105 parser.finalize()
106 messages = list(self.messages)
107 self.messages.clear()
108 for message_type, message_bytes in messages:
109 if message_type == FormMessage.FIELD_START:
110 field_name = b""
111 field_value = b""
112 elif message_type == FormMessage.FIELD_NAME:
113 field_name += message_bytes
114 elif message_type == FormMessage.FIELD_DATA:
115 field_value += message_bytes
116 elif message_type == FormMessage.FIELD_END:
117 name = unquote_plus(field_name.decode("latin-1"))
118 value = unquote_plus(field_value.decode("latin-1"))
119 items.append((name, value))
120
121 return FormData(items)

124class MultiPartParser:
125 spool_max_size = 1024 * 1024 # 1MB
126 """The maximum size of the spooled temporary file used to store file data."""
127 max_part_size = 1024 * 1024 # 1MB
128 """The maximum size of a part in the multipart request."""
129
130 def __init__(
131 self,
132 headers: Headers,
133 stream: typing.AsyncGenerator[bytes, None],
134 *,
135 max_files: int | float = 1000,
136 max_fields: int | float = 1000,
137 max_part_size: int = 1024 * 1024, # 1MB
138 ) -> None:
139 assert multipart is not None, "The `python-multipart` library must be installed to use form parsing."
140 self.headers = headers
141 self.stream = stream
142 self.max_files = max_files
143 self.max_fields = max_fields
144 self.items: list[tuple[str, str | UploadFile]] = []
145 self._current_files = 0
146 self._current_fields = 0
147 self._current_partial_header_name: bytes = b""
148 self._current_partial_header_value: bytes = b""
149 self._current_part = MultipartPart()
150 self._charset = ""
151 self._file_parts_to_write: list[tuple[MultipartPart, bytes]] = []
152 self._file_parts_to_finish: list[MultipartPart] = []
153 self._files_to_close_on_error: list[SpooledTemporaryFile[bytes]] = []
154 self.max_part_size = max_part_size
155
156 def on_part_begin(self) -> None:
157 self._current_part = MultipartPart()
158
159 def on_part_data(self, data: bytes, start: int, end: int) -> None:
160 message_bytes = data[start:end]
161 if self._current_part.file is None:
162 if len(self._current_part.data) + len(message_bytes) > self.max_part_size:
163 raise MultiPartException(f"Part exceeded maximum size of {int(self.max_part_size / 1024)}KB.")
164 self._current_part.data.extend(message_bytes)
165 else:
166 self._file_parts_to_write.append((self._current_part, message_bytes))
167
168 def on_part_end(self) -> None:
169 if self._current_part.file is None:
170 self.items.append(
171 (
172 self._current_part.field_name,
173 _user_safe_decode(self._current_part.data, self._charset),
174 )
175 )
176 else:
177 self._file_parts_to_finish.append(self._current_part)
178 # The file can be added to the items right now even though it's not
179 # finished yet, because it will be finished in the `parse()` method, before
180 # self.items is used in the return value.
181 self.items.append((self._current_part.field_name, self._current_part.file))
182
183 def on_header_field(self, data: bytes, start: int, end: int) -> None:
184 self._current_partial_header_name += data[start:end]
185
186 def on_header_value(self, data: bytes, start: int, end: int) -> None:
187 self._current_partial_header_value += data[start:end]
188
189 def on_header_end(self) -> None:
190 field = self._current_partial_header_name.lower()
191 if field == b"content-disposition":
192 self._current_part.content_disposition = self._current_partial_header_value
193 self._current_part.item_headers.append((field, self._current_partial_header_value))
194 self._current_partial_header_name = b""
195 self._current_partial_header_value = b""
196
197 def on_headers_finished(self) -> None:
198 disposition, options = parse_options_header(self._current_part.content_disposition)
199 try:
200 self._current_part.field_name = _user_safe_decode(options[b"name"], self._charset)
201 except KeyError:
202 raise MultiPartException('The Content-Disposition header field "name" must be provided.')
203 if b"filename" in options:
204 self._current_files += 1
205 if self._current_files > self.max_files:
206 raise MultiPartException(f"Too many files. Maximum number of files is {self.max_files}.")
207 filename = _user_safe_decode(options[b"filename"], self._charset)
208 tempfile = SpooledTemporaryFile(max_size=self.spool_max_size)
209 self._files_to_close_on_error.append(tempfile)
210 self._current_part.file = UploadFile(
211 file=tempfile, # type: ignore[arg-type]
212 size=0,
213 filename=filename,
214 headers=Headers(raw=self._current_part.item_headers),
215 )
216 else:
217 self._current_fields += 1
218 if self._current_fields > self.max_fields:
219 raise MultiPartException(f"Too many fields. Maximum number of fields is {self.max_fields}.")
220 self._current_part.file = None
221
222 def on_end(self) -> None:
223 pass

225 async def parse(self) -> FormData:
226 # Parse the Content-Type header to get the multipart boundary.
227 _, params = parse_options_header(self.headers["Content-Type"])
228 charset = params.get(b"charset", "utf-8")
229 if isinstance(charset, bytes):
230 charset = charset.decode("latin-1")
231 self._charset = charset
232 try:
233 boundary = params[b"boundary"]
234 except KeyError:
235 raise MultiPartException("Missing boundary in multipart.")
236
237 # Callbacks dictionary.
238 callbacks: MultipartCallbacks = {
239 "on_part_begin": self.on_part_begin,
240 "on_part_data": self.on_part_data,
241 "on_part_end": self.on_part_end,
242 "on_header_field": self.on_header_field,
243 "on_header_value": self.on_header_value,
244 "on_header_end": self.on_header_end,
245 "on_headers_finished": self.on_headers_finished,
246 "on_end": self.on_end,
247 }
248
249 # Create the parser.
250 parser = multipart.MultipartParser(boundary, callbacks)
251 try:
252 # Feed the parser with data from the request.
253 async for chunk in self.stream:
254 parser.write(chunk)
255 # Write file data, it needs to use await with the UploadFile methods
256 # that call the corresponding file methods *in a threadpool*,
257 # otherwise, if they were called directly in the callback methods above
258 # (regular, non-async functions), that would block the event loop in
259 # the main thread.
260 for part, data in self._file_parts_to_write:
261 assert part.file # for type checkers
262 await part.file.write(data)
263 for part in self._file_parts_to_finish:
264 assert part.file # for type checkers
265 await part.file.seek(0)
266 self._file_parts_to_write.clear()
267 self._file_parts_to_finish.clear()
268 except MultiPartException as exc:
269 # Close all the files if there was an error.
270 for file in self._files_to_close_on_error:
271 file.close()
272 raise exc
273
274 parser.finalize()
275 return FormData(self.items)
