450 class ForkSafeLock(object):
451 """
452 A lock which is fork-safe (when a fork is done, `pydevd_constants.after_fork()`
453 should be called to reset the locks in the new process to avoid deadlocks
454 from a lock which was locked during the fork).
455
456 Note:
457 Unlike `threading.Lock` this class is not completely atomic, so, doing:
458
459 lock = ForkSafeLock()
460 with lock:
461 ...
462
463 is different than using `threading.Lock` directly because the tracing may
464 find an additional function call on `__enter__` and on `__exit__`, so, it's
465 not recommended to use this in all places, only where the forking may be important
466 (so, for instance, the locks on PyDB should not be changed to this lock because
467 of that -- and those should all be collected in the new process because PyDB itself
468 should be completely cleared anyways).
469
470 It's possible to overcome this limitation by using `ForkSafeLock.acquire` and
471 `ForkSafeLock.release` instead of the context manager (as acquire/release are
472 bound to the original implementation, whereas __enter__/__exit__ is not due to Python
473 limitations).
474 """
475
476 def __init__(self, rlock=False):
477 self._rlock = rlock
478 self._init()
479 _fork_safe_locks.append(weakref.ref(self))
480
481 def __enter__(self):
482 return self._lock.__enter__()
483
484 def __exit__(self, exc_type, exc_val, exc_tb):
485 return self._lock.__exit__(exc_type, exc_val, exc_tb)
486
487 def _init(self):
488 if self._rlock:
489 self._lock = threading.RLock()
490 else:
491 self._lock = thread.allocate_lock()
492
493 self.acquire = self._lock.acquire
494 self.release = self._lock.release
495 _fork_safe_locks.append(weakref.ref(self))

498def after_fork():
499 """
500 Must be called after a fork operation (will reset the ForkSafeLock).
501 """
502 global _fork_safe_locks
503 locks = _fork_safe_locks[:]
504 _fork_safe_locks = []
505 for lock in locks:
506 lock = lock()
507 if lock is not None:
508 lock._init()
