12def _replace_new_line(match: re.Match[str]) -> str:
13 value = match.group(2)
14 value = re.sub(r"\n", r"\\n", value)
15 value = re.sub(r"\r", r"\\r", value)
16 value = re.sub(r"\t", r"\\t", value)
17 value = re.sub(r'(?<!\\)"', r"\"", value)
18
19 return match.group(1) + value + match.group(3)

45def parse_partial_json(s: str, *, strict: bool = False) -> Any:
46 """Parse a JSON string that may be missing closing braces.
47
48 Args:
49 s: The JSON string to parse.
50 strict: Whether to use strict parsing. Defaults to False.
51
52 Returns:
53 The parsed JSON object as a Python dictionary.
54 """
55 # Attempt to parse the string as-is.
56 try:
57 return json.loads(s, strict=strict)
58 except json.JSONDecodeError:
59 pass
60
61 # Initialize variables.
62 new_chars = []
63 stack = []
64 is_inside_string = False
65 escaped = False
66
67 # Process each character in the string one at a time.
68 for char in s:
69 new_char = char
70 if is_inside_string:
71 if char == '"' and not escaped:
72 is_inside_string = False
73 elif char == "\n" and not escaped:
74 new_char = (
75 "\\n" # Replace the newline character with the escape sequence.
76 )
77 elif char == "\\":
78 escaped = not escaped
79 else:
80 escaped = False
81 elif char == '"':
82 is_inside_string = True
83 escaped = False
84 elif char == "{":
85 stack.append("}")
86 elif char == "[":
87 stack.append("]")
88 elif char in {"}", "]"}:
89 if stack and stack[-1] == char:
90 stack.pop()
91 else:
92 # Mismatched closing character; the input is malformed.
93 return None
94
95 # Append the processed character to the new string.
96 new_chars.append(new_char)
97
98 # If we're still inside a string at the end of processing,
99 # we need to close the string.
100 if is_inside_string:
101 if escaped: # Remoe unterminated escape character
102 new_chars.pop()
103 new_chars.append('"')
104
105 # Reverse the stack to get the closing characters.
106 stack.reverse()
107
108 # Try to parse mods of string until we succeed or run out of characters.
109 while new_chars:
110 # Close any remaining open structures in the reverse
111 # order that they were opened.
112 # Attempt to parse the modified string as JSON.
113 try:
114 return json.loads("".join(new_chars + stack), strict=strict)
115 except json.JSONDecodeError:
116 # If we still can't parse the string as JSON,
117 # try removing the last character
118 new_chars.pop()
119
120 # If we got here, we ran out of characters to remove
121 # and still couldn't parse the string as JSON, so return the parse error
122 # for the original string.
123 return json.loads(s, strict=strict)

129def parse_json_markdown(
130 json_string: str, *, parser: Callable[[str], Any] = parse_partial_json
131) -> dict:
132 """Parse a JSON string from a Markdown string.
133
134 Args:
135 json_string: The Markdown string.
136 parser: The parser to use. Defaults to `parse_partial_json`.
137
138 Returns:
139 The parsed JSON object as a Python dictionary.
140 """
141 try:
142 return _parse_json(json_string, parser=parser)
143 except json.JSONDecodeError:

145 match = _json_markdown_re.search(json_string)
146

149 json_str = json_string if match is None else match.group(2)
