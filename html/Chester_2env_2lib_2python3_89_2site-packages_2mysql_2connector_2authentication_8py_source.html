48 """Return authentication class based on plugin name
49
50 This function returns the class for the authentication plugin plugin_name.
51 The returned class is a subclass of BaseAuthPlugin.
52
53 Args:
54 plugin_name (str): Authentication plugin name.
55 auth_plugin_class (str): Authentication plugin class name.
56
57 Raises:
58 NotSupportedError: When plugin_name is not supported.
59
60 Returns:
61 Subclass of `BaseAuthPlugin`.
62 """
63 package = DEFAULT_PLUGINS_PKG
64 if plugin_name:
65 try:
66 logger.info("package: %s", package)
67 logger.info("plugin_name: %s", plugin_name)
68 plugin_module = importlib.import_module(f".{plugin_name}", package)
69 if not auth_plugin_class or not hasattr(plugin_module, auth_plugin_class):
70 auth_plugin_class = plugin_module.AUTHENTICATION_PLUGIN_CLASS
71 logger.info("AUTHENTICATION_PLUGIN_CLASS: %s", auth_plugin_class)
72 return getattr(plugin_module, auth_plugin_class)
73 except ModuleNotFoundError as err:
74 logger.warning("Requested Module was not found: %s", err)
75 except ValueError as err:
76 raise ProgrammingError(f"Invalid module name: {err}") from err
77 raise NotSupportedError(f"Authentication plugin '{plugin_name}' is not supported")
