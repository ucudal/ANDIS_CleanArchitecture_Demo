7class CompilerFlags(IntFlag):
8 """Possible values of the co_flags attribute of Code object.
9
10 Note: We do not rely on inspect values here as some of them are missing and
11 furthermore would be version dependent.
12
13 """
14
15 OPTIMIZED = 0x00001 # noqa
16 NEWLOCALS = 0x00002 # noqa
17 VARARGS = 0x00004 # noqa
18 VARKEYWORDS = 0x00008 # noqa
19 NESTED = 0x00010 # noqa
20 GENERATOR = 0x00020 # noqa
21 NOFREE = 0x00040 # noqa
22 # New in Python 3.5
23 # Used for coroutines defined using async def ie native coroutine
24 COROUTINE = 0x00080 # noqa
25 # Used for coroutines defined as a generator and then decorated using
26 # types.coroutine
27 ITERABLE_COROUTINE = 0x00100 # noqa
28 # New in Python 3.6
29 # Generator defined in an async def function
30 ASYNC_GENERATOR = 0x00200 # noqa
31
32 # __future__ flags
33 # future flags changed in Python 3.9
34 if sys.version_info < (3, 9):
35 FUTURE_GENERATOR_STOP = 0x80000 # noqa
36 if sys.version_info > (3, 6):
37 FUTURE_ANNOTATIONS = 0x100000
38 else:
39 FUTURE_GENERATOR_STOP = 0x800000 # noqa
40 FUTURE_ANNOTATIONS = 0x1000000

43def infer_flags(bytecode, is_async=None):
44 """Infer the proper flags for a bytecode based on the instructions.
45
46 Because the bytecode does not have enough context to guess if a function
47 is asynchronous the algorithm tries to be conservative and will never turn
48 a previously async code into a sync one.
49
50 Parameters
51 ----------
52 bytecode : Bytecode | ConcreteBytecode | ControlFlowGraph
53 Bytecode for which to infer the proper flags
54 is_async : bool | None, optional
55 Force the code to be marked as asynchronous if True, prevent it from
56 being marked as asynchronous if False and simply infer the best
57 solution based on the opcode and the existing flag if None.
58
59 """
60 flags = CompilerFlags(0)
61 if not isinstance(
62 bytecode,
63 (_bytecode.Bytecode, _bytecode.ConcreteBytecode, _bytecode.ControlFlowGraph),
64 ):
65 msg = "Expected a Bytecode, ConcreteBytecode or ControlFlowGraph " "instance not %s"
66 raise ValueError(msg % bytecode)
67
68 instructions = bytecode.get_instructions() if isinstance(bytecode, _bytecode.ControlFlowGraph) else bytecode
69 instr_names = {i.name for i in instructions if not isinstance(i, (_bytecode.SetLineno, _bytecode.Label))}
70
71 # Identify optimized code
72 if not (instr_names & {"STORE_NAME", "LOAD_NAME", "DELETE_NAME"}):
73 flags |= CompilerFlags.OPTIMIZED
74
75 # Check for free variables
76 if not (
77 instr_names
78 & {
79 "LOAD_CLOSURE",
80 "LOAD_DEREF",
81 "STORE_DEREF",
82 "DELETE_DEREF",
83 "LOAD_CLASSDEREF",
84 }
85 ):
86 flags |= CompilerFlags.NOFREE
87
88 # Copy flags for which we cannot infer the right value
89 flags |= bytecode.flags & (CompilerFlags.NEWLOCALS | CompilerFlags.VARARGS | CompilerFlags.VARKEYWORDS | CompilerFlags.NESTED)
90

92 maybe_generator = instr_names & {"YIELD_VALUE", "YIELD_FROM"}
93
94 sure_async = instr_names & {
95 "GET_AWAITABLE",
96 "GET_AITER",
97 "GET_ANEXT",
98 "BEFORE_ASYNC_WITH",
99 "SETUP_ASYNC_WITH",
100 "END_ASYNC_FOR",
101 }
102
103 # If performing inference or forcing an async behavior, first inspect
104 # the flags since this is the only way to identify iterable coroutines
105 if is_async in (None, True):
106 if bytecode.flags & CompilerFlags.COROUTINE:
107 if sure_generator:
108 flags |= CompilerFlags.ASYNC_GENERATOR
109 else:
110 flags |= CompilerFlags.COROUTINE
111 elif bytecode.flags & CompilerFlags.ITERABLE_COROUTINE:
112 if sure_async:
113 msg = (
114 "The ITERABLE_COROUTINE flag is set but bytecode that"
115 "can only be used in async functions have been "
116 "detected. Please unset that flag before performing "
117 "inference."
118 )
119 raise ValueError(msg)
120 flags |= CompilerFlags.ITERABLE_COROUTINE
121 elif bytecode.flags & CompilerFlags.ASYNC_GENERATOR:
122 if not sure_generator:
123 flags |= CompilerFlags.COROUTINE
124 else:
125 flags |= CompilerFlags.ASYNC_GENERATOR
126
127 # If the code was not asynchronous before determine if it should now be
128 # asynchronous based on the opcode and the is_async argument.
129 else:
130 if sure_async:
131 # YIELD_FROM is not allowed in async generator
132 if sure_generator:
133 flags |= CompilerFlags.ASYNC_GENERATOR
134 else:
135 flags |= CompilerFlags.COROUTINE
136
137 elif maybe_generator:
138 if is_async:
139 if sure_generator:
140 flags |= CompilerFlags.ASYNC_GENERATOR
141 else:
142 flags |= CompilerFlags.COROUTINE
143 else:
144 flags |= CompilerFlags.GENERATOR
145
146 elif is_async:
147 flags |= CompilerFlags.COROUTINE
148
149 # If the code should not be asynchronous, check first it is possible and
150 # next set the GENERATOR flag if relevant
151 else:
152 if sure_async:
153 raise ValueError(
154 "The is_async argument is False but bytecodes " "that can only be used in async functions have " "been detected."
155 )

158 flags |= CompilerFlags.GENERATOR
