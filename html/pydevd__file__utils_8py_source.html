1r"""
2This module provides utilities to get the absolute filenames so that we can be sure that:
3 - The case of a file will match the actual file in the filesystem (otherwise breakpoints won't be hit).
4 - Providing means for the user to make path conversions when doing a remote debugging session in
5 one machine and debugging in another.
6
7To do that, the PATHS_FROM_ECLIPSE_TO_PYTHON constant must be filled with the appropriate paths.
8
9@note:
10 in this context, the server is where your python process is running
11 and the client is where eclipse is running.
12
13E.g.:
14 If the server (your python process) has the structure
15 /user/projects/my_project/src/package/module1.py
16
17 and the client has:
18 c:\my_project\src\package\module1.py
19
20 the PATHS_FROM_ECLIPSE_TO_PYTHON would have to be:
21 PATHS_FROM_ECLIPSE_TO_PYTHON = [(r'c:\my_project\src', r'/user/projects/my_project/src')]
22
23 alternatively, this can be set with an environment variable from the command line:
24 set PATHS_FROM_ECLIPSE_TO_PYTHON=[['c:\my_project\src','/user/projects/my_project/src']]
25
26@note: DEBUG_CLIENT_SERVER_TRANSLATION can be set to True to debug the result of those translations
27
28@note: the case of the paths is important! Note that this can be tricky to get right when one machine
29uses a case-independent filesystem and the other uses a case-dependent filesystem (if the system being
30debugged is case-independent, 'normcase()' should be used on the paths defined in PATHS_FROM_ECLIPSE_TO_PYTHON).
31
32@note: all the paths with breakpoints must be translated (otherwise they won't be found in the server)
33
34@note: to enable remote debugging in the target machine (pydev extensions in the eclipse installation)
35 import pydevd;pydevd.settrace(host, stdoutToServer, stderrToServer, port, suspend)
36
37 see parameter docs on pydevd.py
38
39@note: for doing a remote debugging session, all the pydevd_ files must be on the server accessible
40 through the PYTHONPATH (and the PATHS_FROM_ECLIPSE_TO_PYTHON only needs to be set on the target
41 machine for the paths that'll actually have breakpoints).
42"""

137def _resolve_listing(resolved, iter_parts_lowercase, cache=_listdir_cache):
138 while True: # Note: while True to make iterative and not recursive
139 try:
140 resolve_lowercase = next(iter_parts_lowercase) # must be lowercase already
141 except StopIteration:
142 return resolved
143
144 resolved_lower = resolved.lower()
145
146 resolved_joined = cache.get((resolved_lower, resolve_lowercase))
147 if resolved_joined is None:
148 dir_contents = cache.get(resolved_lower)
149 if dir_contents is None:
150 dir_contents = cache[resolved_lower] = os_listdir(resolved)
151
152 for filename in dir_contents:
153 if filename.lower() == resolve_lowercase:
154 resolved_joined = os.path.join(resolved, filename)
155 cache[(resolved_lower, resolve_lowercase)] = resolved_joined
156 break
157 else:
158 raise FileNotFoundError("Unable to find: %s in %s. Dir Contents: %s" % (resolve_lowercase, resolved, dir_contents))
159
160 resolved = resolved_joined

480def _apply_func_and_normalize_case(filename, func, isabs, normalize_case, os_path_exists=os_path_exists, join=join):
481 if filename.startswith("<"):
482 # Not really a file, rather a synthetic name like <string> or <ipython-...>;
483 # shouldn't be normalized.
484 return filename
485
486 r = func(filename)
487
488 if not isabs:
489 if not os_path_exists(r):
490 r = _get_relative_filename_abs_path(filename, func)
491
492 ind = r.find(".zip")
493 if ind == -1:
494 ind = r.find(".egg")
495 if ind != -1:
496 ind += 4
497 zip_path = r[:ind]
498 inner_path = r[ind:]
499 if inner_path.startswith("!"):
500 # Note (fabioz): although I can replicate this by creating a file ending as
501 # .zip! or .egg!, I don't really know what's the real-world case for this
502 # (still kept as it was added by @jetbrains, but it should probably be reviewed
503 # later on).
504 # Note 2: it goes hand-in-hand with 'exists'.
505 inner_path = inner_path[1:]
506 zip_path = zip_path + "!"
507
508 if inner_path.startswith("/") or inner_path.startswith("\\"):
509 inner_path = inner_path[1:]
510 if inner_path:
511 if normalize_case:
512 r = join(normcase(zip_path), inner_path)
513 else:
514 r = join(zip_path, inner_path)
515 return r
516
517 if normalize_case:
518 r = normcase(r)
519 return r

526def exists(filename):
527 if os_path_exists(filename):
528 return True
529
530 if not os.path.isabs(filename):
531 filename = _get_relative_filename_abs_path(filename, os.path.abspath)
532 if os_path_exists(filename):

534
535 ind = filename.find(".zip")
536 if ind == -1:
537 ind = filename.find(".egg")
538
539 if ind != -1:
540 ind += 4
541 zip_path = filename[:ind]
542 inner_path = filename[ind:]
543 if inner_path.startswith("!"):
544 # Note (fabioz): although I can replicate this by creating a file ending as
545 # .zip! or .egg!, I don't really know what's the real-world case for this
546 # (still kept as it was added by @jetbrains, but it should probably be reviewed
547 # later on).

549 inner_path = inner_path[1:]
550 zip_path = zip_path + "!"
551

553 if zip_file_obj is None:
554 return False
555 elif zip_file_obj is _NOT_FOUND_SENTINEL:
556 try:
557 import zipfile

712def setup_client_server_paths(paths):

727 python_sep = "\\" if IS_WINDOWS else "/"
728 eclipse_sep = "\\" if _ide_os == "WINDOWS" else "/"

739 # Apply normcase to the existing paths to follow the os preferences.
740
741 for i, (path0, path1) in enumerate(paths):
742 force_only_slash = path0.endswith(("/", "\\")) and path1.endswith(("/", "\\"))
743
744 if not force_only_slash:
745 path0 = _fix_path(path0, eclipse_sep, False)
746 path1 = _fix_path(path1, python_sep, False)
747 initial_paths.append((path0, path1))
748 paths_from_eclipse_to_python.append((_normcase_from_client(path0), normcase(path1)))

750 # Now, make a version with a slash in the end.
751 path0 = _fix_path(path0, eclipse_sep, True)
752 path1 = _fix_path(path1, python_sep, True)
753 initial_paths_with_end_sep.append((path0, path1))
754 paths_from_eclipse_to_python_with_end_sep.append((_normcase_from_client(path0), normcase(path1)))

756 if DEBUG_CLIENT_SERVER_TRANSLATION:
757 pydev_log.critical(
758 "pydev debugger: paths_from_eclipse_to_python %s",
759 ", ".join(
760 [
761 '"%s=>%s"' % (x[0], x[1])
762 for x in paths_from_eclipse_to_python
763 ]
764 ),
765 )
766 # Fix things so that we always match the versions with a slash in the end first.
767 initial_paths = initial_paths_with_end_sep + initial_paths
768 paths_from_eclipse_to_python = paths_from_eclipse_to_python_with_end_sep + paths_from_eclipse_to_python

770 if not paths_from_eclipse_to_python:

777 def _map_file_to_server(filename, cache=norm_filename_to_server_container):
778 # Eclipse will send the passed filename to be translated to the python process
779 # So, this would be 'NormFileFromEclipseToPython'
780 try:
781 return cache[filename]
782 except KeyError:
783 if eclipse_sep != python_sep:
784 # Make sure that the separators are what we expect from the IDE.
785 filename = filename.replace(python_sep, eclipse_sep)
786
787 # used to translate a path from the client to the debug server
788 translated = filename
789 translated_normalized = _normcase_from_client(filename)
790 for eclipse_prefix, server_prefix in paths_from_eclipse_to_python:
791 if translated_normalized.startswith(eclipse_prefix):
792 found_translation = True
793 if DEBUG_CLIENT_SERVER_TRANSLATION:
794 pydev_log.critical("pydev debugger: replacing to server: %s", filename)
795 translated = server_prefix + filename[len(eclipse_prefix) :]
796 if DEBUG_CLIENT_SERVER_TRANSLATION:
797 pydev_log.critical("pydev debugger: sent to server: %s - matched prefix: %s", translated, eclipse_prefix)
798 break
799 else:
800 found_translation = False
801
802 # Note that when going to the server, we do the replace first and only later do the norm file.
803 if eclipse_sep != python_sep:
804 translated = translated.replace(eclipse_sep, python_sep)
805
806 if found_translation:
807 # Note: we don't normalize it here, this must be done as a separate
808 # step by the caller.
809 translated = absolute_path(translated)
810 else:
811 if not os_path_exists(translated):
812 if not translated.startswith("<"):
813 # This is a configuration error, so, write it always so
814 # that the user can fix it.
815 error_once(
816 'pydev debugger: unable to find translation for: "%s" in [%s] (please revise your path mappings).\n',
817 filename,
818 ", ".join(['"%s"' % (x[0],) for x in paths_from_eclipse_to_python]),
819 )
820 else:
821 # It's possible that we had some round trip (say, we sent /usr/lib and received
822 # it back, so, having no translation is ok too).
823
824 # Note: we don't normalize it here, this must be done as a separate
825 # step by the caller.
826 translated = absolute_path(translated)
827
828 cache[filename] = translated
829 return translated

831 def _map_file_to_client(filename, cache=norm_filename_to_client_container):
832 # The result of this method will be passed to eclipse
833 # So, this would be 'NormFileFromPythonToEclipse'
834 try:
835 return cache[filename]
836 except KeyError:
837 abs_path = absolute_path(filename)
838 translated_proper_case = get_path_with_real_case(abs_path)
839 translated_normalized = normcase(abs_path)
840
841 path_mapping_applied = False
842
843 if translated_normalized.lower() != translated_proper_case.lower():
844 if DEBUG_CLIENT_SERVER_TRANSLATION:
845 pydev_log.critical(
846 "pydev debugger: translated_normalized changed path (from: %s to %s)", translated_proper_case, translated_normalized
847 )
848
849 for i, (eclipse_prefix, python_prefix) in enumerate(paths_from_eclipse_to_python):
850 if translated_normalized.startswith(python_prefix):
851 if DEBUG_CLIENT_SERVER_TRANSLATION:
852 pydev_log.critical("pydev debugger: replacing to client: %s", translated_normalized)
853
854 # Note: use the non-normalized version.
855 eclipse_prefix = initial_paths[i][0]
856 translated = eclipse_prefix + translated_proper_case[len(python_prefix) :]
857 if DEBUG_CLIENT_SERVER_TRANSLATION:
858 pydev_log.critical("pydev debugger: sent to client: %s - matched prefix: %s", translated, python_prefix)
859 path_mapping_applied = True
860 break
861 else:
862 if DEBUG_CLIENT_SERVER_TRANSLATION:
863 pydev_log.critical(
864 "pydev debugger: to client: unable to find matching prefix for: %s in %s",
865 translated_normalized,
866 [x[1] for x in paths_from_eclipse_to_python],
867 )
868 translated = translated_proper_case
869
870 if eclipse_sep != python_sep:
871 translated = translated.replace(python_sep, eclipse_sep)
872
873 translated = _path_to_expected_str(translated)
874
875 # The resulting path is not in the python process, so, we cannot do a normalize the path here,
876 # only at the beginning of this method.
877 cache[filename] = (translated, path_mapping_applied)
878
879 if translated not in _client_filename_in_utf8_to_source_reference:
880 if path_mapping_applied:
881 source_reference = 0
882 else:
883 source_reference = _next_source_reference()
884 pydev_log.debug("Created source reference: %s for untranslated path: %s", source_reference, filename)
885
886 _client_filename_in_utf8_to_source_reference[translated] = source_reference
887 _source_reference_to_server_filename[source_reference] = filename
888
889 return (translated, path_mapping_applied)
