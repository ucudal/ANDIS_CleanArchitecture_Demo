49@silence_warnings_decorator
50def getVariable(dbg, thread_id, frame_id, scope, locator):
51 """
52 returns the value of a variable
53
54 :scope: can be BY_ID, EXPRESSION, GLOBAL, LOCAL, FRAME
55
56 BY_ID means we'll traverse the list of all objects alive to get the object.
57
58 :locator: after reaching the proper scope, we have to get the attributes until we find
59 the proper location (i.e.: obj\tattr1\tattr2)
60
61 :note: when BY_ID is used, the frame_id is considered the id of the object to find and
62 not the frame (as we don't care about the frame in this case).
63 """
64 if scope == "BY_ID":
65 if thread_id != get_current_thread_id(threading.current_thread()):
66 raise VariableError("getVariable: must execute on same thread")
67
68 try:
69 import gc
70
71 objects = gc.get_objects()
72 except:
73 pass # Not all python variants have it.
74 else:
75 frame_id = int(frame_id)
76 for var in objects:
77 if id(var) == frame_id:
78 if locator is not None:
79 locator_parts = locator.split("\t")
80 for k in locator_parts:
81 _type, _type_name, resolver = get_type(var)
82 var = resolver.resolve(var, k)
83
84 return var
85
86 # If it didn't return previously, we coudn't find it by id (i.e.: already garbage collected).
87 sys.stderr.write("Unable to find object with id: %s\n" % (frame_id,))
88 return None
89
90 frame = dbg.find_frame(thread_id, frame_id)
91 if frame is None:
92 return {}
93
94 if locator is not None:
95 locator_parts = locator.split("\t")
96 else:
97 locator_parts = []
98
99 for attr in locator_parts:
100 attr.replace("@_@TAB_CHAR@_@", "\t")
101
102 if scope == "EXPRESSION":
103 for count in range(len(locator_parts)):
104 if count == 0:
105 # An Expression can be in any scope (globals/locals), therefore it needs to evaluated as an expression
106 var = evaluate_expression(dbg, frame, locator_parts[count], False)
107 else:
108 _type, _type_name, resolver = get_type(var)
109 var = resolver.resolve(var, locator_parts[count])
110 else:
111 if scope == "GLOBAL":
112 var = frame.f_globals
113 del locator_parts[0] # globals are special, and they get a single dummy unused attribute
114 else:
115 # in a frame access both locals and globals as Python does
116 var = {}
117 var.update(frame.f_globals)
118 var.update(frame.f_locals)
119
120 for k in locator_parts:
121 _type, _type_name, resolver = get_type(var)
122 var = resolver.resolve(var, k)
123
124 return var

187def custom_operation(dbg, thread_id, frame_id, scope, attrs, style, code_or_file, operation_fn_name):
188 """
189 We'll execute the code_or_file and then search in the namespace the operation_fn_name to execute with the given var.
190
191 code_or_file: either some code (i.e.: from pprint import pprint) or a file to be executed.
192 operation_fn_name: the name of the operation to execute after the exec (i.e.: pprint)
193 """
194 expressionValue = getVariable(dbg, thread_id, frame_id, scope, attrs)
195
196 try:
197 namespace = {"__name__": "<custom_operation>"}
198 if style == "EXECFILE":
199 namespace["__file__"] = code_or_file
200 execfile(code_or_file, namespace, namespace)
201 else: # style == EXEC
202 namespace["__file__"] = "<customOperationCode>"
203 Exec(code_or_file, namespace, namespace)
204
205 return str(namespace[operation_fn_name](expressionValue))
206 except:
207 pydev_log.exception()

210@lru_cache(3)
211def _expression_to_evaluate(expression):
212 keepends = True
213 lines = expression.splitlines(keepends)
214 # find first non-empty line
215 chars_to_strip = 0
216 for line in lines:
217 if line.strip(): # i.e.: check first non-empty line
218 for c in iter_chars(line):
219 if c.isspace():
220 chars_to_strip += 1
221 else:
222 break
223 break
224
225 if chars_to_strip:
226 # I.e.: check that the chars we'll remove are really only whitespaces.
227 proceed = True
228 new_lines = []
229 for line in lines:
230 if not proceed:
231 break
232 for c in iter_chars(line[:chars_to_strip]):
233 if not c.isspace():
234 proceed = False
235 break
236
237 new_lines.append(line[chars_to_strip:])
238
239 if proceed:
240 if isinstance(expression, bytes):
241 expression = b"".join(new_lines)
242 else:
243 expression = "".join(new_lines)
244
245 return expression

248def eval_in_context(expression, global_vars, local_vars, py_db=None):
249 result = None
250 try:
251 compiled = compile_as_eval(expression)
252 is_async = inspect.CO_COROUTINE & compiled.co_flags == inspect.CO_COROUTINE
253
254 if is_async:
255 if py_db is None:
256 py_db = get_global_debugger()
257 if py_db is None:
258 raise RuntimeError("Cannot evaluate async without py_db.")
259 t = _EvalAwaitInNewEventLoop(py_db, compiled, global_vars, local_vars)
260 t.start()
261 t.join()
262
263 if t.exc:
264 raise t.exc[1].with_traceback(t.exc[2])
265 else:
266 result = t.evaluated_value
267 else:
268 result = eval(compiled, global_vars, local_vars)
269 except (Exception, KeyboardInterrupt):
270 etype, result, tb = sys.exc_info()
271 result = ExceptionOnEvaluate(result, etype, tb)
272
273 # Ok, we have the initial error message, but let's see if we're dealing with a name mangling error...
274 try:
275 if ".__" in expression:
276 # Try to handle '__' name mangling (for simple cases such as self.__variable.__another_var).
277 split = expression.split(".")
278 entry = split[0]
279
280 if local_vars is None:
281 local_vars = global_vars
282 curr = local_vars[entry] # Note: we want the KeyError if it's not there.
283 for entry in split[1:]:
284 if entry.startswith("__") and not hasattr(curr, entry):
285 entry = "_%s%s" % (curr.__class__.__name__, entry)
286 curr = getattr(curr, entry)
287
288 result = curr
289 except:
290 pass
291 return result

311def _run_with_unblock_threads(original_func, py_db, curr_thread, frame, expression, is_exec):
312 on_timeout_unblock_threads = None
313 timeout_tracker = py_db.timeout_tracker # : :type timeout_tracker: TimeoutTracker
314
315 if py_db.multi_threads_single_notification:
316 unblock_threads_timeout = pydevd_constants.PYDEVD_UNBLOCK_THREADS_TIMEOUT

322 tid = get_current_thread_id(curr_thread)

325 on_timeout_unblock_threads.called = True

347def _evaluate_with_timeouts(original_func):
348 """
349 Provides a decorator that wraps the original evaluate to deal with slow evaluates.
350
351 If some evaluation is too slow, we may show a message, resume threads or interrupt them
352 as needed (based on the related configurations).
353 """
354
355 @functools.wraps(original_func)
356 def new_func(py_db, frame, expression, is_exec):
357 if py_db is None:
358 # Only for testing...
359 pydev_log.critical("_evaluate_with_timeouts called without py_db!")
360 return original_func(py_db, frame, expression, is_exec)
361 warn_evaluation_timeout = pydevd_constants.PYDEVD_WARN_EVALUATION_TIMEOUT
362 curr_thread = threading.current_thread()
363
364 def on_warn_evaluation_timeout():
365 py_db.writer.add_command(py_db.cmd_factory.make_evaluation_timeout_msg(py_db, expression, curr_thread))
366
367 timeout_tracker = py_db.timeout_tracker # : :type timeout_tracker: TimeoutTracker
368 with timeout_tracker.call_on_timeout(warn_evaluation_timeout, on_warn_evaluation_timeout):
369 return _run_with_unblock_threads(original_func, py_db, curr_thread, frame, expression, is_exec)
370
371 return new_func

442@_evaluate_with_timeouts
443def evaluate_expression(py_db, frame, expression, is_exec):
444 """
445 :param str expression:
446 The expression to be evaluated.
447
448 Note that if the expression is indented it's automatically dedented (based on the indentation
449 found on the first non-empty line).
450
451 i.e.: something as:
452
453 `
454 def method():
455 a = 1
456 `
457
458 becomes:
459
460 `
461 def method():
462 a = 1
463 `
464
465 Also, it's possible to evaluate calls with a top-level await (currently this is done by
466 creating a new event loop in a new thread and making the evaluate at that thread -- note
467 that this is still done synchronously so the evaluation has to finish before this
468 function returns).
469
470 :param is_exec: determines if we should do an exec or an eval.
471 There are some changes in this function depending on whether it's an exec or an eval.
472
473 When it's an exec (i.e.: is_exec==True):
474 This function returns None.
475 Any exception that happens during the evaluation is reraised.
476 If the expression could actually be evaluated, the variable is printed to the console if not None.
477
478 When it's an eval (i.e.: is_exec==False):
479 This function returns the result from the evaluation.
480 If some exception happens in this case, the exception is caught and a ExceptionOnEvaluate is returned.
481 Also, in this case we try to resolve name-mangling (i.e.: to be able to add a self.__my_var watch).
482
483 :param py_db:
484 The debugger. Only needed if some top-level await is detected (for creating a
485 PyDBDaemonThread).
486 """
487 if frame is None:
488 return
489
490 # This is very tricky. Some statements can change locals and use them in the same
491 # call (see https://github.com/microsoft/debugpy/issues/815), also, if locals and globals are
492 # passed separately, it's possible that one gets updated but apparently Python will still
493 # try to load from the other, so, what's done is that we merge all in a single dict and
494 # then go on and update the frame with the results afterwards.
495
496 # -- see tests in test_evaluate_expression.py
497
498 # This doesn't work because the variables aren't updated in the locals in case the
499 # evaluation tries to set a variable and use it in the same expression.
500 # updated_globals = frame.f_globals
501 # updated_locals = frame.f_locals
502
503 # This doesn't work because the variables aren't updated in the locals in case the
504 # evaluation tries to set a variable and use it in the same expression.
505 # updated_globals = {}
506 # updated_globals.update(frame.f_globals)
507 # updated_globals.update(frame.f_locals)
508 #
509 # updated_locals = frame.f_locals
510
511 # This doesn't work either in the case where the evaluation tries to set a variable and use
512 # it in the same expression (I really don't know why as it seems like this *should* work
513 # in theory but doesn't in practice).
514 # updated_globals = {}
515 # updated_globals.update(frame.f_globals)
516 #
517 # updated_locals = {}
518 # updated_globals.update(frame.f_locals)
519
520 # This is the only case that worked consistently to run the tests in test_evaluate_expression.py
521 # It's a bit unfortunate because although the exec works in this case, we have to manually
522 # put the updates in the frame locals afterwards.
523 updated_globals = {}
524 updated_globals.update(frame.f_globals)
525 updated_globals.update(frame.f_locals)
526 if "globals" not in updated_globals:
527 # If the user explicitly uses 'globals()' then we provide the
528 # frame globals (unless he has shadowed it already).
529 updated_globals["globals"] = lambda: frame.f_globals
530
531 initial_globals = updated_globals.copy()
532
533 updated_locals = None
534
535 try:
536 expression = expression.replace("@LINE@", "\n")
537
538 if is_exec:
539 try:
540 # Try to make it an eval (if it is an eval we can print it, otherwise we'll exec it and
541 # it will have whatever the user actually did)
542 compiled = compile_as_eval(expression)
543 except Exception:
544 compiled = None
545
546 if compiled is None:
547 try:
548 compiled = _compile_as_exec(expression)
549 is_async = inspect.CO_COROUTINE & compiled.co_flags == inspect.CO_COROUTINE
550 if is_async:
551 t = _EvalAwaitInNewEventLoop(py_db, compiled, updated_globals, updated_locals)
552 t.start()
553 t.join()
554
555 if t.exc:
556 raise t.exc[1].with_traceback(t.exc[2])
557 else:
558 Exec(compiled, updated_globals, updated_locals)
559 finally:
560 # Update the globals even if it errored as it may have partially worked.
561 update_globals_and_locals(updated_globals, initial_globals, frame)
562 else:
563 is_async = inspect.CO_COROUTINE & compiled.co_flags == inspect.CO_COROUTINE
564 if is_async:
565 t = _EvalAwaitInNewEventLoop(py_db, compiled, updated_globals, updated_locals)
566 t.start()
567 t.join()
568
569 if t.exc:
570 raise t.exc[1].with_traceback(t.exc[2])
571 else:
572 result = t.evaluated_value
573 else:
574 result = eval(compiled, updated_globals, updated_locals)
575 if result is not None: # Only print if it's not None (as python does)
576 sys.stdout.write("%s\n" % (result,))
577 return
578
579 else:
580 ret = eval_in_context(expression, updated_globals, updated_locals, py_db)
581 try:
582 is_exception_returned = ret.__class__ == ExceptionOnEvaluate
583 except:
584 pass
585 else:
586 if not is_exception_returned:
587 # i.e.: by using a walrus assignment (:=), expressions can change the locals,
588 # so, make sure that we save the locals back to the frame.
589 update_globals_and_locals(updated_globals, initial_globals, frame)
590 return ret
591 finally:
592 # Should not be kept alive if an exception happens and this frame is kept in the stack.
593 del updated_globals
594 del updated_locals
595 del initial_globals
596 del frame

599def change_attr_expression(frame, attr, expression, dbg, value=SENTINEL_VALUE, /, scope: Optional[ScopeRequest]=None):
600 """Changes some attribute in a given frame."""
601 if frame is None:
602 return
603
604 if scope is not None:
605 assert isinstance(scope, ScopeRequest)
606 scope = scope.scope
607
608 try:
609 expression = expression.replace("@LINE@", "\n")
610
611 if dbg.plugin and value is SENTINEL_VALUE:
612 result = dbg.plugin.change_variable(frame, attr, expression)
613 if result is not dbg.plugin.EMPTY_SENTINEL:
614 return result
615
616 if attr[:7] == "Globals" or scope == "globals":
617 attr = attr[8:] if attr.startswith("Globals") else attr
618 if attr in frame.f_globals:
619 if value is SENTINEL_VALUE:
620 value = eval(expression, frame.f_globals, frame.f_locals)
621 frame.f_globals[attr] = value
622 return frame.f_globals[attr]
623 else:
624 raise VariableError("Attribute %s not found in globals" % attr)
625 else:
626 if "." not in attr: # i.e.: if we have a '.', we're changing some attribute of a local var.
627 if pydevd_save_locals.is_save_locals_available():
628 if value is SENTINEL_VALUE:
629 value = eval(expression, frame.f_globals, frame.f_locals)
630 frame.f_locals[attr] = value
631 pydevd_save_locals.save_locals(frame)
632 return frame.f_locals[attr]
633
634 # i.e.: case with '.' or save locals not available (just exec the assignment in the frame).
635 if value is SENTINEL_VALUE:
636 value = eval(expression, frame.f_globals, frame.f_locals)
637 result = value
638 Exec("%s=%s" % (attr, expression), frame.f_globals, frame.f_locals)
639 return result
640
641 except Exception as e:
642 pydev_log.exception(e)

668def array_to_xml(array, roffset, coffset, rows, cols, format):
669 xml = ""
670 rows = min(rows, MAXIMUM_ARRAY_SIZE)
671 cols = min(cols, MAXIMUM_ARRAY_SIZE)
672
