39def cookie_parser(cookie_string: str) -> dict[str, str]:
40 """
41 This function parses a ``Cookie`` HTTP header into a dict of key/value pairs.
42
43 It attempts to mimic browser cookie parsing behavior: browsers and web servers
44 frequently disregard the spec (RFC 6265) when setting and reading cookies,
45 so we attempt to suit the common scenarios here.
46
47 This function has been adapted from Django 3.1.0.
48 Note: we are explicitly _NOT_ using `SimpleCookie.load` because it is based
49 on an outdated spec and will fail on lots of input we want to support
50 """
51 cookie_dict: dict[str, str] = {}
52 for chunk in cookie_string.split(";"):
53 if "=" in chunk:
54 key, val = chunk.split("=", 1)
55 else:
56 # Assume an empty name per
57 # https://bugzilla.mozilla.org/show_bug.cgi?id=169091
58 key, val = "", chunk
59 key, val = key.strip(), val.strip()
60 if key or val:
61 # unquote using Python's algorithm.
62 cookie_dict[key] = http_cookies._unquote(val)
63 return cookie_dict
