93def lxml_trace(data, html=True, **kwargs):
94 """Print out the lxml events that occur during parsing.
95
96 This lets you see how lxml parses a document when no Beautiful
97 Soup code is running. You can use this to determine whether
98 an lxml-specific problem is in Beautiful Soup's lxml tree builders
99 or in lxml itself.
100
101 :param data: Some markup.
102 :param html: If True, markup will be parsed with lxml's HTML parser.
103 if False, lxml's XML parser will be used.
104 """
105 from lxml import etree
106 recover = kwargs.pop('recover', True)
107 if isinstance(data, str):
108 data = data.encode("utf8")
109 reader = BytesIO(data)
110 for event, element in etree.iterparse(
111 reader, html=html, recover=recover, **kwargs
112 ):
113 print(("%s, %4s, %s" % (event, element.tag, element.text)))

178def rsentence(length=4):

180 return " ".join(rword(random.randint(4,9)) for i in range(length))

182def rdoc(num_elements=1000):
183 """Randomly generate an invalid HTML document."""
184 tag_names = ['p', 'div', 'span', 'i', 'b', 'script', 'table']
185 elements = []
186 for i in range(num_elements):
187 choice = random.randint(0,3)
188 if choice == 0:
189 # New tag.
190 tag_name = random.choice(tag_names)
191 elements.append("<%s>" % tag_name)
192 elif choice == 1:
193 elements.append(rsentence(random.randint(1,4)))
194 elif choice == 2:
195 # Close a tag.
196 tag_name = random.choice(tag_names)
197 elements.append("</%s>" % tag_name)
198 return "<html>" + "\n".join(elements) + "</html>"
