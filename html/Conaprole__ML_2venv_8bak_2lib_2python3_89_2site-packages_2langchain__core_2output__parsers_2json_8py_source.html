104 def get_format_instructions(self) -> str:
105 """Return the format instructions for the JSON output.
106
107 Returns:
108 The format instructions for the JSON output.
109 """
110 if self.pydantic_object is None:
111 return "Return a JSON object."
112 # Copy schema to avoid altering original Pydantic schema.
113 schema = dict(self._get_schema(self.pydantic_object).items())
114
115 # Remove extraneous fields.
116 reduced_schema = schema
117 if "title" in reduced_schema:
118 del reduced_schema["title"]
119 if "type" in reduced_schema:
120 del reduced_schema["type"]
121 # Ensure json in context is well-formed with double quotes.
122 schema_str = json.dumps(reduced_schema, ensure_ascii=False)
123 return JSON_FORMAT_INSTRUCTIONS.format(schema=schema_str)
