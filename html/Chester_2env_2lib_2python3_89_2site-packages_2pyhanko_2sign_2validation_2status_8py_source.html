145 @classmethod

147 cls, validator: CertificateValidator,
148 key_usage_settings: KeyUsageConstraints = None):
149 key_usage_settings = key_usage_settings or KeyUsageConstraints()
150 key_usage_settings = KeyUsageConstraints(
151 key_usage=(
152 cls.key_usage if key_usage_settings.key_usage is None
153 else key_usage_settings.key_usage
154 ),

156 cls.extd_key_usage if key_usage_settings.extd_key_usage is None
157 else key_usage_settings.extd_key_usage
158 )

164 # validate usage without going through pyhanko_certvalidator
165 key_usage_settings.validate(cert)
166 path = await validator.async_validate_usage(key_usage=set())
167 ades_status = None
168 except InvalidCertificateError as e:

170 logger.warning(e)

174 # revocation cases under AdES we're dealing with.
175 # (For now, assume the worst.)
176 ades_status = AdESFailure.REVOKED
177 except PathBuildingError as e:
178 logger.warning(e)

180 except PathValidationError as e:

183 # so we can actually set proper AdES values

185 # revinfo freshness...)
186 ades_status = AdESIndeterminate.CERTIFICATE_CHAIN_GENERAL_FAILURE

290 parse_error_fatal=False):

293 for result in results:

626@unique
627class SignatureCoverageLevel(OrderedEnum):
628 """
629 Indicate the extent to which a PDF signature (cryptographically) covers
630 a document. Note that this does *not* pass judgment on whether uncovered
631 updates are legitimate or not, but as a general rule, a legitimate signature
632 will satisfy at least :attr:`ENTIRE_REVISION`.
633 """
634
635 UNCLEAR = 0
636 """
637 The signature's coverage is unclear and/or disconnected.
638 In standard PDF signatures, this is usually a bad sign.
639 """
640
641 CONTIGUOUS_BLOCK_FROM_START = 1
642 """
643 The signature covers a contiguous block in the PDF file stretching from
644 the first byte of the file to the last byte in the indicated ``/ByteRange``.
645 In other words, the only interruption in the byte range is fully occupied
646 by the signature data itself.
647 """
648
649 ENTIRE_REVISION = 2
650 """
651 The signature covers the entire revision in which it occurs, but incremental
652 updates may have been added later. This is not necessarily evidence of
653 tampering. In particular, it is expected when a file contains multiple
654 signatures. Nonetheless, caution is required.
655 """
656
657 ENTIRE_FILE = 3
658 """
659 The entire file is covered by the signature.
660 """

668 """

671 """
672 Result of the difference analysis run on the file:
673
674 * If ``None``, no difference analysis was run.
675 * If the difference analysis was successful, this attribute will contain
676 a :class:`.DiffResult` object.

690 """
691
692 coverage = self.coverage
693 if self.diff_result is None:
694 if coverage == SignatureCoverageLevel.ENTIRE_REVISION:
695 # in this case, we can't know without the diff analysis result
696 return None
697 return (
698 ModificationLevel.NONE
699 if coverage == SignatureCoverageLevel.ENTIRE_FILE
700 else ModificationLevel.OTHER
701 )
702 elif isinstance(self.diff_result, DiffResult):
703 return self.diff_result.modification_level
704 else:
705 return ModificationLevel.OTHER
