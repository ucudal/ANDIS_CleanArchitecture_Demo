74 if not isinstance(code, types.CodeType):
75 path = code

81 def iterate():
82 # First, get all line starts for this code object. This does not include
83 # bodies of nested class and function definitions, as they have their
84 # own objects.
85 for _, lineno in dis.findlinestarts(code):
86 if lineno is not None:
87 yield lineno
88
89 # For nested class and function definitions, their respective code objects
90 # are constants referenced by this object.
91 for const in code.co_consts:
92 if isinstance(const, types.CodeType) and const.co_filename == code.co_filename:
93 for lineno in _get_code_lines(const):
94 yield lineno
95
96 return iterate()

1074 def list_children_and_stop_forking(initial_pid, stop=True):
1075 children_pids = []
1076 if stop:
1077 # Ask to stop forking (shouldn't be called for this process, only subprocesses).
1078 self._call(["kill", "-STOP", str(initial_pid)], stdout=subprocess.PIPE, stderr=subprocess.PIPE)
1079
1080 list_popen = self._popen(["pgrep", "-P", str(initial_pid)], stdout=subprocess.PIPE, stderr=subprocess.PIPE)
1081
1082 if list_popen is not None:
1083 stdout, _ = list_popen.communicate()
1084 for line in stdout.splitlines():
1085 line = line.decode("ascii").strip()
1086 if line:
1087 pid = str(line)
1088 if pid in dont_terminate_child_pids:
1089 continue
1090 children_pids.append(pid)
1091 # Recursively get children.
1092 children_pids.extend(list_children_and_stop_forking(pid))
1093 return children_pids
1094
1095 previously_found = set()
1096
1097 for _ in range(50): # Try this at most 50 times before giving up.
1098 children_pids = list_children_and_stop_forking(this_pid, stop=False)
1099 found_new = False
1100
1101 for pid in children_pids:
1102 if pid not in previously_found:
1103 found_new = True
1104 previously_found.add(pid)
1105 self._call(["kill", "-KILL", str(pid)], stdout=subprocess.PIPE, stderr=subprocess.PIPE)
1106
1107 if not found_new:
1108 break
