156 """Usage docs: https://docs.pydantic.dev/2.10/concepts/validators/#field-validators
157
158 A metadata class that indicates that a validation should be applied **instead** of the inner validation logic.
159
160 !!! note
161 Before v2.9, `PlainValidator` wasn't always compatible with JSON Schema generation for `mode='validation'`.
162 You can now use the `json_schema_input_type` argument to specify the input type of the function
163 to be used in the JSON schema when `mode='validation'` (the default). See the example below for more details.
164
165 Attributes:
166 func: The validator function.
167 json_schema_input_type: The input type of the function. This is only used to generate the appropriate
168 JSON Schema (in validation mode). If not provided, will default to `Any`.
169
170 Example:
171 ```python
172 from typing import Union
173
174 from typing_extensions import Annotated
175
176 from pydantic import BaseModel, PlainValidator
177
178 MyInt = Annotated[
179 int,
180 PlainValidator(
181 lambda v: int(v) + 1, json_schema_input_type=Union[str, int] # (1)!
182 ),
183 ]
184
185 class Model(BaseModel):
186 a: MyInt
187
188 print(Model(a='1').a)
189 #> 2
190
191 print(Model(a=1).a)
192 #> 2
193 ```
194
195 1. In this example, we've specified the `json_schema_input_type` as `Union[str, int]` which indicates to the JSON schema
196 generator that in validation mode, the input type for the `a` field can be either a `str` or an `int`.

202 def __get_pydantic_core_schema__(self, source_type: Any, handler: GetCoreSchemaHandler) -> core_schema.CoreSchema:
203 # Note that for some valid uses of PlainValidator, it is not possible to generate a core schema for the
204 # source_type, so calling `handler(source_type)` will error, which prevents us from generating a proper
205 # serialization schema. To work around this for use cases that will not involve serialization, we simply
206 # catch any PydanticSchemaGenerationError that may be raised while attempting to build the serialization schema
207 # and abort any attempts to handle special serialization.
208 from pydantic import PydanticSchemaGenerationError
209
210 try:
211 schema = handler(source_type)
212 # TODO if `schema['serialization']` is one of `'include-exclude-dict/sequence',
213 # schema validation will fail. That's why we use 'type ignore' comments below.
214 serialization = schema.get(
215 'serialization',
216 core_schema.wrap_serializer_function_ser_schema(
217 function=lambda v, h: h(v),
218 schema=schema,
219 return_schema=handler.generate_schema(source_type),
220 ),
221 )
222 except PydanticSchemaGenerationError:
223 serialization = None
224
225 input_schema = handler.generate_schema(self.json_schema_input_type)
226
227 info_arg = _inspect_validator(self.func, 'plain')
228 if info_arg:
229 func = cast(core_schema.WithInfoValidatorFunction, self.func)
230 return core_schema.with_info_plain_validator_function(
231 func,
232 field_name=handler.field_name,
233 serialization=serialization, # pyright: ignore[reportArgumentType]
234 json_schema_input_schema=input_schema,
235 )
236 else:
237 func = cast(core_schema.NoInfoValidatorFunction, self.func)
238 return core_schema.no_info_plain_validator_function(
239 func,
240 serialization=serialization, # pyright: ignore[reportArgumentType]
241 json_schema_input_schema=input_schema,
242 )

406def field_validator(
407 field: str,
408 /,
409 *fields: str,
410 mode: FieldValidatorModes = 'after',
411 check_fields: bool | None = None,
412 json_schema_input_type: Any = PydanticUndefined,
413) -> Callable[[Any], Any]:
414 """Usage docs: https://docs.pydantic.dev/2.10/concepts/validators/#field-validators
415
416 Decorate methods on the class indicating that they should be used to validate fields.
417
418 Example usage:
419 ```python
420 from typing import Any
421
422 from pydantic import (
423 BaseModel,
424 ValidationError,
425 field_validator,
426 )
427
428 class Model(BaseModel):
429 a: str
430
431 @field_validator('a')
432 @classmethod
433 def ensure_foobar(cls, v: Any):
434 if 'foobar' not in v:
435 raise ValueError('"foobar" not found in a')
436 return v
437
438 print(repr(Model(a='this is foobar good')))
439 #> Model(a='this is foobar good')
440
441 try:
442 Model(a='snap')
443 except ValidationError as exc_info:
444 print(exc_info)
445 '''
446 1 validation error for Model
447 a
448 Value error, "foobar" not found in a [type=value_error, input_value='snap', input_type=str]
449 '''
450 ```
451
452 For more in depth examples, see [Field Validators](../concepts/validators.md#field-validators).
453
454 Args:
455 field: The first field the `field_validator` should be called on; this is separate
456 from `fields` to ensure an error is raised if you don't pass at least one.
457 *fields: Additional field(s) the `field_validator` should be called on.
458 mode: Specifies whether to validate the fields before or after validation.
459 check_fields: Whether to check that the fields actually exist on the model.
460 json_schema_input_type: The input type of the function. This is only used to generate
461 the appropriate JSON Schema (in validation mode) and can only specified
462 when `mode` is either `'before'`, `'plain'` or `'wrap'`.
463
464 Returns:
465 A decorator that can be used to decorate a function to be used as a field_validator.
466
467 Raises:
468 PydanticUserError:
469 - If `@field_validator` is used bare (with no fields).
470 - If the args passed to `@field_validator` as fields are not strings.
471 - If `@field_validator` applied to instance methods.
472 """
473 if isinstance(field, FunctionType):
474 raise PydanticUserError(
475 '`@field_validator` should be used with fields and keyword arguments, not bare. '
476 "E.g. usage should be `@validator('<field_name>', ...)`",
477 code='validator-no-fields',
478 )
479
480 if mode not in ('before', 'plain', 'wrap') and json_schema_input_type is not PydanticUndefined:
481 raise PydanticUserError(
482 f"`json_schema_input_type` can't be used when mode is set to {mode!r}",
483 code='validator-input-type',
484 )
485
486 if json_schema_input_type is PydanticUndefined and mode == 'plain':
487 json_schema_input_type = Any
488
489 fields = field, *fields
490 if not all(isinstance(field, str) for field in fields):
491 raise PydanticUserError(
492 '`@field_validator` fields should be passed as separate string args. '
493 "E.g. usage should be `@validator('<field_name_1>', '<field_name_2>', ...)`",
494 code='validator-invalid-fields',
495 )
496
497 def dec(
498 f: Callable[..., Any] | staticmethod[Any, Any] | classmethod[Any, Any, Any],
499 ) -> _decorators.PydanticDescriptorProxy[Any]:
500 if _decorators.is_instance_method_from_sig(f):
501 raise PydanticUserError(
502 '`@field_validator` cannot be applied to instance methods', code='validator-instance-method'
503 )
504
505 # auto apply the @classmethod decorator
506 f = _decorators.ensure_classmethod_based_on_signature(f)
507
508 dec_info = _decorators.FieldValidatorDecoratorInfo(
509 fields=fields, mode=mode, check_fields=check_fields, json_schema_input_type=json_schema_input_type
510 )
511 return _decorators.PydanticDescriptorProxy(f, dec_info)
512
513 return dec

667def model_validator(
668 *,
669 mode: Literal['wrap', 'before', 'after'],
670) -> Any:
671 """Usage docs: https://docs.pydantic.dev/2.10/concepts/validators/#model-validators
672
673 Decorate model methods for validation purposes.
674
675 Example usage:
676 ```python
677 from typing_extensions import Self
678
679 from pydantic import BaseModel, ValidationError, model_validator
680
681 class Square(BaseModel):
682 width: float
683 height: float
684
685 @model_validator(mode='after')
686 def verify_square(self) -> Self:
687 if self.width != self.height:
688 raise ValueError('width and height do not match')
689 return self
690
691 s = Square(width=1, height=1)
692 print(repr(s))
693 #> Square(width=1.0, height=1.0)
694
695 try:
696 Square(width=1, height=2)
697 except ValidationError as e:
698 print(e)
699 '''
700 1 validation error for Square
701 Value error, width and height do not match [type=value_error, input_value={'width': 1, 'height': 2}, input_type=dict]
702 '''
703 ```
704
705 For more in depth examples, see [Model Validators](../concepts/validators.md#model-validators).
706
707 Args:
708 mode: A required string literal that specifies the validation mode.
709 It can be one of the following: 'wrap', 'before', or 'after'.
710
711 Returns:
712 A decorator that can be used to decorate a function to be used as a model validator.
713 """
714
715 def dec(f: Any) -> _decorators.PydanticDescriptorProxy[Any]:
716 # auto apply the @classmethod decorator
717 f = _decorators.ensure_classmethod_based_on_signature(f)
718 dec_info = _decorators.ModelValidatorDecoratorInfo(mode=mode)
719 return _decorators.PydanticDescriptorProxy(f, dec_info)
720
721 return dec
