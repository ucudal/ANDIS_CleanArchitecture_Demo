269 def open_handle(self, dwDesiredAccess=win32.THREAD_ALL_ACCESS):
270 """
271 Opens a new handle to the thread, closing the previous one.
272
273 The new handle is stored in the L{hThread} property.
274
275 @warn: Normally you should call L{get_handle} instead, since it's much
276 "smarter" and tries to reuse handles and merge access rights.
277
278 @type dwDesiredAccess: int
279 @param dwDesiredAccess: Desired access rights.
280 Defaults to L{win32.THREAD_ALL_ACCESS}.
281 See: U{http://msdn.microsoft.com/en-us/library/windows/desktop/ms686769(v=vs.85).aspx}
282
283 @raise WindowsError: It's not possible to open a handle to the thread
284 with the requested access rights. This tipically happens because
285 the target thread belongs to system process and the debugger is not
286 runnning with administrative rights.
287 """
288 hThread = win32.OpenThread(dwDesiredAccess, win32.FALSE, self.dwThreadId)
289
290 # In case hThread was set to an actual handle value instead of a Handle
291 # object. This shouldn't happen unless the user tinkered with it.
292 if not hasattr(self.hThread, "__del__"):
293 self.close_handle()
294
295 self.hThread = hThread
296
297 def close_handle(self):
298 """
299 Closes the handle to the thread.
300
301 @note: Normally you don't need to call this method. All handles
302 created by I{WinAppDbg} are automatically closed when the garbage
303 collector claims them.
304 """
305 try:
306 if hasattr(self.hThread, "close"):
307 self.hThread.close()
308 elif self.hThread not in (None, win32.INVALID_HANDLE_VALUE):
309 win32.CloseHandle(self.hThread)
310 finally:
311 self.hThread = None

313 def get_handle(self, dwDesiredAccess=win32.THREAD_ALL_ACCESS):
314 """
315 Returns a handle to the thread with I{at least} the access rights
316 requested.
317
318 @note:
319 If a handle was previously opened and has the required access
320 rights, it's reused. If not, a new handle is opened with the
321 combination of the old and new access rights.
322
323 @type dwDesiredAccess: int
324 @param dwDesiredAccess: Desired access rights.
325 See: U{http://msdn.microsoft.com/en-us/library/windows/desktop/ms686769(v=vs.85).aspx}
326
327 @rtype: ThreadHandle
328 @return: Handle to the thread.
329
330 @raise WindowsError: It's not possible to open a handle to the thread
331 with the requested access rights. This tipically happens because
332 the target thread belongs to system process and the debugger is not
333 runnning with administrative rights.
334 """
335 if self.hThread in (None, win32.INVALID_HANDLE_VALUE):
336 self.open_handle(dwDesiredAccess)
337 else:
338 dwAccess = self.hThread.dwAccess
339 if (dwAccess | dwDesiredAccess) != dwAccess:
340 self.open_handle(dwAccess | dwDesiredAccess)
341 return self.hThread

466 def get_context(self, ContextFlags=None, bSuspend=False):
467 """
468 Retrieves the execution context (i.e. the registers values) for this
469 thread.
470
471 @type ContextFlags: int
472 @param ContextFlags: Optional, specify which registers to retrieve.
473 Defaults to C{win32.CONTEXT_ALL} which retrieves all registes
474 for the current platform.
475
476 @type bSuspend: bool
477 @param bSuspend: C{True} to automatically suspend the thread before
478 getting its context, C{False} otherwise.
479
480 Defaults to C{False} because suspending the thread during some
481 debug events (like thread creation or destruction) may lead to
482 strange errors.
483
484 Note that WinAppDbg 1.4 used to suspend the thread automatically
485 always. This behavior was changed in version 1.5.
486
487 @rtype: dict( str S{->} int )
488 @return: Dictionary mapping register names to their values.
489
490 @see: L{set_context}
491 """
492
493 # Some words on the "strange errors" that lead to the bSuspend
494 # parameter. Peter Van Eeckhoutte and I were working on a fix
495 # for some bugs he found in the 1.5 betas when we stumbled upon
496 # what seemed to be a deadlock in the debug API that caused the
497 # GetThreadContext() call never to return. Since removing the
498 # call to SuspendThread() solved the problem, and a few Google
499 # searches showed a handful of problems related to these two
500 # APIs and Wow64 environments, I decided to break compatibility.
501 #
502 # Here are some pages about the weird behavior of SuspendThread:
503 # http://zachsaw.blogspot.com.es/2010/11/wow64-bug-getthreadcontext-may-return.html
504 # http://stackoverflow.com/questions/3444190/windows-suspendthread-doesnt-getthreadcontext-fails
505
506 # Get the thread handle.
507 dwAccess = win32.THREAD_GET_CONTEXT
508 if bSuspend:
509 dwAccess = dwAccess | win32.THREAD_SUSPEND_RESUME
510 hThread = self.get_handle(dwAccess)
511
512 # Suspend the thread if requested.
513 if bSuspend:
514 try:
515 self.suspend()
516 except WindowsError:
517 # Threads can't be suspended when the exit process event
518 # arrives, but you can still get the context.
519 bSuspend = False
520
521 # If an exception is raised, make sure the thread execution is resumed.
522 try:
523 if win32.bits == self.get_bits():
524 # 64 bit debugger attached to 64 bit process, or
525 # 32 bit debugger attached to 32 bit process.
526 ctx = win32.GetThreadContext(hThread, ContextFlags=ContextFlags)
527
528 else:
529 if self.is_wow64():
530 # 64 bit debugger attached to 32 bit process.
531 if ContextFlags is not None:
532 ContextFlags &= ~win32.ContextArchMask
533 ContextFlags |= win32.WOW64_CONTEXT_i386
534 ctx = win32.Wow64GetThreadContext(hThread, ContextFlags)
535
536 else:
537 # 32 bit debugger attached to 64 bit process.
538 # XXX only i386/AMD64 is supported in this particular case
539 if win32.arch not in (win32.ARCH_I386, win32.ARCH_AMD64):
540 raise NotImplementedError()
541 if ContextFlags is not None:
542 ContextFlags &= ~win32.ContextArchMask
543 ContextFlags |= win32.context_amd64.CONTEXT_AMD64
544 ctx = win32.context_amd64.GetThreadContext(hThread, ContextFlags=ContextFlags)
545
546 finally:
547 # Resume the thread if we suspended it.
548 if bSuspend:
549 self.resume()
550
551 # Return the context.
552 return ctx

554 def set_context(self, context, bSuspend=False):
555 """
556 Sets the values of the registers.
557
558 @see: L{get_context}
559
560 @type context: dict( str S{->} int )
561 @param context: Dictionary mapping register names to their values.
562
563 @type bSuspend: bool
564 @param bSuspend: C{True} to automatically suspend the thread before
565 setting its context, C{False} otherwise.
566
567 Defaults to C{False} because suspending the thread during some
568 debug events (like thread creation or destruction) may lead to
569 strange errors.
570
571 Note that WinAppDbg 1.4 used to suspend the thread automatically
572 always. This behavior was changed in version 1.5.
573 """
574
575 # Get the thread handle.
576 dwAccess = win32.THREAD_SET_CONTEXT
577 if bSuspend:
578 dwAccess = dwAccess | win32.THREAD_SUSPEND_RESUME
579 hThread = self.get_handle(dwAccess)
580
581 # Suspend the thread if requested.
582 if bSuspend:
583 self.suspend()
584 # No fix for the exit process event bug.
585 # Setting the context of a dead thread is pointless anyway.
586
587 # Set the thread context.
588 try:
589 if win32.bits == 64 and self.is_wow64():
590 win32.Wow64SetThreadContext(hThread, context)
591 else:
592 win32.SetThreadContext(hThread, context)
593
594 # Resume the thread if we suspended it.
595 finally:
596 if bSuspend:
597 self.resume()

940 def get_linear_address(self, segment, address):
941 """
942 Translates segment-relative addresses to linear addresses.
943
944 Linear addresses can be used to access a process memory,
945 calling L{Process.read} and L{Process.write}.
946
947 @type segment: str
948 @param segment: Segment register name.
949
950 @type address: int
951 @param address: Segment relative memory address.
952
953 @rtype: int
954 @return: Linear memory address.
955
956 @raise ValueError: Address is too large for selector.
957
958 @raise WindowsError:
959 The current architecture does not support selectors.
960 Selectors only exist in x86-based systems.
961 """
962 hThread = self.get_handle(win32.THREAD_QUERY_INFORMATION)
963 selector = self.get_register(segment)
964 ldt = win32.GetThreadSelectorEntry(hThread, selector)
965 BaseLow = ldt.BaseLow
966 BaseMid = ldt.HighWord.Bytes.BaseMid << 16
967 BaseHi = ldt.HighWord.Bytes.BaseHi << 24
968 Base = BaseLow | BaseMid | BaseHi
969 LimitLow = ldt.LimitLow
970 LimitHi = ldt.HighWord.Bits.LimitHi << 16
971 Limit = LimitLow | LimitHi
972 if address > Limit:
973 msg = "Address %s too large for segment %s (selector %d)"
974 msg = msg % (HexDump.address(address, self.get_bits()), segment, selector)
975 raise ValueError(msg)
976 return Base + address

1023 def get_seh_chain(self):
1024 """
1025 @rtype: list of tuple( int, int )
1026 @return: List of structured exception handlers.
1027 Each SEH is represented as a tuple of two addresses:
1028 - Address of this SEH block
1029 - Address of the SEH callback function

1031 where the first member is a pointer to the B{next} block instead.
1032
1033 @raise NotImplementedError:

1045 seh_chain.append((seh, None))

1079 def __get_stack_trace(self, depth=16, bUseLabels=True, bMakePretty=True):
1080 """
1081 Tries to get a stack trace for the current function using the debug
1082 helper API (dbghelp.dll).
1083
1084 @type depth: int
1085 @param depth: Maximum depth of stack trace.
1086
1087 @type bUseLabels: bool
1088 @param bUseLabels: C{True} to use labels, C{False} to use addresses.
1089
1090 @type bMakePretty: bool
1091 @param bMakePretty:
1092 C{True} for user readable labels,
1093 C{False} for labels that can be passed to L{Process.resolve_label}.
1094
1095 "Pretty" labels look better when producing output for the user to
1096 read, while pure labels are more useful programatically.
1097
1098 @rtype: tuple of tuple( int, int, str )
1099 @return: Stack trace of the thread as a tuple of
1100 ( return address, frame pointer address, module filename )
1101 when C{bUseLabels} is C{True}, or a tuple of
1102 ( return address, frame pointer label )
1103 when C{bUseLabels} is C{False}.
1104
1105 @raise WindowsError: Raises an exception on error.
1106 """
1107
1108 aProcess = self.get_process()
1109 arch = aProcess.get_arch()
1110 bits = aProcess.get_bits()
1111
1112 if arch == win32.ARCH_I386:
1113 MachineType = win32.IMAGE_FILE_MACHINE_I386
1114 elif arch == win32.ARCH_AMD64:
1115 MachineType = win32.IMAGE_FILE_MACHINE_AMD64
1116 elif arch == win32.ARCH_IA64:
1117 MachineType = win32.IMAGE_FILE_MACHINE_IA64
1118 else:
1119 msg = "Stack walking is not available for this architecture: %s"
1120 raise NotImplementedError(msg % arch)
1121
1122 hProcess = aProcess.get_handle(win32.PROCESS_VM_READ | win32.PROCESS_QUERY_INFORMATION)
1123 hThread = self.get_handle(win32.THREAD_GET_CONTEXT | win32.THREAD_QUERY_INFORMATION)
1124
1125 StackFrame = win32.STACKFRAME64()
1126 StackFrame.AddrPC = win32.ADDRESS64(self.get_pc())
1127 StackFrame.AddrFrame = win32.ADDRESS64(self.get_fp())
1128 StackFrame.AddrStack = win32.ADDRESS64(self.get_sp())
1129
1130 trace = list()
1131 while win32.StackWalk64(MachineType, hProcess, hThread, StackFrame):
1132 if depth <= 0:
1133 break
1134 fp = StackFrame.AddrFrame.Offset
1135 ra = aProcess.peek_pointer(fp + 4)
1136 if ra == 0:
1137 break
1138 lib = aProcess.get_module_at_address(ra)
1139 if lib is None:
1140 lib = ""
1141 else:
1142 if lib.fileName:
1143 lib = lib.fileName
1144 else:
1145 lib = "%s" % HexDump.address(lib.lpBaseOfDll, bits)
1146 if bUseLabels:
1147 label = aProcess.get_label_at_address(ra)
1148 if bMakePretty:
1149 label = "%s (%s)" % (HexDump.address(ra, bits), label)
1150 trace.append((fp, label))
1151 else:
1152 trace.append((fp, ra, lib))
1153 fp = aProcess.peek_pointer(fp)
1154 return tuple(trace)
