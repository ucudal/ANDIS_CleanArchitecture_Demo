33def _get_typing_objects_by_name_of(name: str) -> tuple[Any, ...]:
34 """Get the member named `name` from both `typing` and `typing-extensions` (if it exists)."""
35 result = tuple(getattr(module, name) for module in (typing, typing_extensions) if hasattr(module, name))
36 if not result:
37 raise ValueError(f'Neither `typing` nor `typing_extensions` has an object called {name!r}')
38 return result

244def is_classvar_annotation(tp: Any, /) -> bool:
245 """Return whether the provided argument represents a class variable annotation.
246
247 Although not explicitly stated by the typing specification, `ClassVar` can be used
248 inside `Annotated` and as such, this function checks for this specific scenario.
249
250 Because this function is used to detect class variables before evaluating forward references
251 (or because evaluation failed), we also implement a naive regex match implementation. This is
252 required because class variables are inspected before fields are collected, so we try to be
253 as accurate as possible.
254 """
255 if is_classvar(tp) or (anntp := annotated_type(tp)) is not None and is_classvar(anntp):
256 return True
257
258 str_ann: str | None = None
259 if isinstance(tp, typing.ForwardRef):
260 str_ann = tp.__forward_arg__
261 if isinstance(tp, str):
262 str_ann = tp
263
264 if str_ann is not None and _classvar_re.match(str_ann):
265 # stdlib dataclasses do something similar, although a bit more advanced
266 # (see `dataclass._is_type`).
267 return True
268
269 return False

418def parent_frame_namespace(*, parent_depth: int = 2, force: bool = False) -> dict[str, Any] | None:
419 """We allow use of items in parent namespace to get around the issue with `get_type_hints` only looking in the
420 global module namespace. See https://github.com/pydantic/pydantic/issues/2678#issuecomment-1008139014 -> Scope
421 and suggestion at the end of the next comment by @gvanrossum.
422
423 WARNING 1: it matters exactly where this is called. By default, this function will build a namespace from the
424 parent of where it is called.
425
426 WARNING 2: this only looks in the parent namespace, not other parents since (AFAIK) there's no way to collect a
427 dict of exactly what's in scope. Using `f_back` would work sometimes but would be very wrong and confusing in many
428 other cases. See https://discuss.python.org/t/is-there-a-way-to-access-parent-nested-namespaces/20659.
429
430 There are some cases where we want to force fetching the parent namespace, ex: during a `model_rebuild` call.
431 In this case, we want both the namespace of the class' module, if applicable, and the parent namespace of the
432 module where the rebuild is called.
433
434 In other cases, like during initial schema build, if a class is defined at the top module level, we don't need to
435 fetch that module's namespace, because the class' __module__ attribute can be used to access the parent namespace.
436 This is done in `_namespace_utils.get_module_ns_of`. Thus, there's no need to cache the parent frame namespace in this case.
437 """
438 frame = sys._getframe(parent_depth)
439
440 # note, we don't copy frame.f_locals here (or during the last return call), because we don't expect the namespace to be modified down the line
441 # if this becomes a problem, we could implement some sort of frozen mapping structure to enforce this
442 if force:
443 return frame.f_locals
444
445 # if either of the following conditions are true, the class is defined at the top module level
446 # to better understand why we need both of these checks, see
447 # https://github.com/pydantic/pydantic/pull/10113#discussion_r1714981531
448 if frame.f_back is None or frame.f_code.co_name == '<module>':
449 return None
450
451 return frame.f_locals

454def _type_convert(arg: Any) -> Any:
455 """Convert `None` to `NoneType` and strings to `ForwardRef` instances.
456
457 This is a backport of the private `typing._type_convert` function. When
458 evaluating a type, `ForwardRef._evaluate` ends up being called, and is
459 responsible for making this conversion. However, we still have to apply
460 it for the first argument passed to our type evaluation functions, similarly
461 to the `typing.get_type_hints` function.
462 """
463 if arg is None:
464 return NoneType
465 if isinstance(arg, str):
466 # Like `typing.get_type_hints`, assume the arg can be in any context,
467 # hence the proper `is_argument` and `is_class` args:
468 return _make_forward_ref(arg, is_argument=False, is_class=True)
469 return arg

472def get_model_type_hints(
473 obj: type[BaseModel],
474 *,
475 ns_resolver: NsResolver | None = None,
476) -> dict[str, tuple[Any, bool]]:
477 """Collect annotations from a Pydantic model class, including those from parent classes.
478
479 Args:
480 obj: The Pydantic model to inspect.
481 ns_resolver: A namespace resolver instance to use. Defaults to an empty instance.
482
483 Returns:
484 A dictionary mapping annotation names to a two-tuple: the first element is the evaluated
485 type or the original annotation if a `NameError` occurred, the second element is a boolean
486 indicating if whether the evaluation succeeded.
487 """
488 hints: dict[str, Any] | dict[str, tuple[Any, bool]] = {}
489 ns_resolver = ns_resolver or NsResolver()
490
491 for base in reversed(obj.__mro__):
492 ann: dict[str, Any] | None = base.__dict__.get('__annotations__')
493 if not ann or isinstance(ann, types.GetSetDescriptorType):
494 continue
495 with ns_resolver.push(base):
496 globalns, localns = ns_resolver.types_namespace
497 for name, value in ann.items():
498 if name.startswith('_'):
499 # For private attributes, we only need the annotation to detect the `ClassVar` special form.
500 # For this reason, we still try to evaluate it, but we also catch any possible exception (on
501 # top of the `NameError`s caught in `try_eval_type`) that could happen so that users are free
502 # to use any kind of forward annotation for private fields (e.g. circular imports, new typing
503 # syntax, etc).
504 try:
505 hints[name] = try_eval_type(value, globalns, localns)
506 except Exception:
507 hints[name] = (value, False)
508 else:
509 hints[name] = try_eval_type(value, globalns, localns)
510 return hints

593def eval_type_backport(
594 value: Any,
595 globalns: GlobalsNamespace | None = None,
596 localns: MappingNamespace | None = None,
597 type_params: tuple[Any, ...] | None = None,
598) -> Any:
599 """An enhanced version of `typing._eval_type` which will fall back to using the `eval_type_backport`
600 package if it's installed to let older Python versions use newer typing constructs.
601
602 Specifically, this transforms `X | Y` into `typing.Union[X, Y]` and `list[X]` into `typing.List[X]`
603 (as well as all the types made generic in PEP 585) if the original syntax is not supported in the
604 current Python version.
605
606 This function will also display a helpful error if the value passed fails to evaluate.
607 """
608 try:
609 return _eval_type_backport(value, globalns, localns, type_params)
610 except TypeError as e:
611 if 'Unable to evaluate type annotation' in str(e):
612 raise
613
614 # If it is a `TypeError` and value isn't a `ForwardRef`, it would have failed during annotation definition.
615 # Thus we assert here for type checking purposes:
616 assert isinstance(value, typing.ForwardRef)
617
618 message = f'Unable to evaluate type annotation {value.__forward_arg__!r}.'
619 if sys.version_info >= (3, 11):
620 e.add_note(message)
621 raise
622 else:
623 raise TypeError(message) from e

626def _eval_type_backport(
627 value: Any,
628 globalns: GlobalsNamespace | None = None,
629 localns: MappingNamespace | None = None,
630 type_params: tuple[Any, ...] | None = None,
631) -> Any:
632 try:
633 return _eval_type(value, globalns, localns, type_params)
634 except TypeError as e:
635 if not (isinstance(value, typing.ForwardRef) and is_backport_fixable_error(e)):
636 raise
637
638 try:
639 from eval_type_backport import eval_type_backport
640 except ImportError:
641 raise TypeError(
642 f'Unable to evaluate type annotation {value.__forward_arg__!r}. If you are making use '
643 'of the new typing syntax (unions using `|` since Python 3.10 or builtins subscripting '
644 'since Python 3.9), you should either replace the use of new syntax with the existing '
645 '`typing` constructs or install the `eval_type_backport` package.'
646 ) from e
647
648 return eval_type_backport(
649 value,
650 globalns,
651 localns, # pyright: ignore[reportArgumentType], waiting on a new `eval_type_backport` release.
652 try_default=False,
653 )

683def get_function_type_hints(
684 function: Callable[..., Any],
685 *,
686 include_keys: set[str] | None = None,
687 globalns: GlobalsNamespace | None = None,
688 localns: MappingNamespace | None = None,
689) -> dict[str, Any]:
690 """Return type hints for a function.
691
692 This is similar to the `typing.get_type_hints` function, with a few differences:
693 - Support `functools.partial` by using the underlying `func` attribute.
694 - If `function` happens to be a built-in type (e.g. `int`), assume it doesn't have annotations
695 but specify the `return` key as being the actual type.
696 - Do not wrap type annotation of a parameter with `Optional` if it has a default value of `None`
697 (related bug: https://github.com/python/cpython/issues/90353, only fixed in 3.11+).
698 """
699 try:
700 if isinstance(function, partial):
701 annotations = function.func.__annotations__
702 else:
703 annotations = function.__annotations__
704 except AttributeError:
705 type_hints = get_type_hints(function)
706 if isinstance(function, type):
707 # `type[...]` is a callable, which returns an instance of itself.
708 # At some point, we might even look into the return type of `__new__`
709 # if it returns something else.
710 type_hints.setdefault('return', function)
711 return type_hints
712
713 if globalns is None:
714 globalns = get_module_ns_of(function)
715 type_params: tuple[Any, ...] | None = None
716 if localns is None:
717 # If localns was specified, it is assumed to already contain type params. This is because
718 # Pydantic has more advanced logic to do so (see `_namespace_utils.ns_for_function`).
719 type_params = getattr(function, '__type_params__', ())
720
721 type_hints = {}
722 for name, value in annotations.items():
723 if include_keys is not None and name not in include_keys:
724 continue
725 if value is None:
726 value = NoneType
727 elif isinstance(value, str):
728 value = _make_forward_ref(value)
729
730 type_hints[name] = eval_type_backport(value, globalns, localns, type_params)
731
732 return type_hints

785
786 Return type hints for an object.
787
788 This is often the same as obj.__annotations__, but it handles
789 forward references encoded as string literals, adds Optional[t] if a
790 default value equal to None is set and recursively replaces all
791 'Annotated[T, ...]' with 'T' (unless 'include_extras=True').
792
793 The argument may be a module, class, method, or function. The annotations
794 are returned as a dictionary. For classes, annotations include also
795 inherited members.
796
797 TypeError is raised if the argument is not of a type that can contain
798 annotations, and an empty dictionary is returned if no annotations are
799 present.
800
801 BEWARE -- the behavior of globalns and localns is counterintuitive
802 (unless you are familiar with how eval() and exec() work). The
803 search order is locals first, then globals.
804
805 - If no dict arguments are passed, an attempt is made to use the
806 globals from obj (or the respective module's globals for classes),
807 and these are also used as the locals. If the object does not appear
808 to have globals, an empty dictionary is used. For classes, the search
809 order is globals first then locals.
810
811 - If one dict argument is passed, it is used for both globals and
812 locals.
813
814 - If two dict arguments are passed, they specify globals and
815 locals, respectively.
816 """
817 if getattr(obj, '__no_type_check__', None):
818 return {}

833 # get_type_hints only evaluated the globalns of
834 # a class. To maintain backwards compatibility, we reverse
835 # the globalns and localns order so that eval() looks into
836 # *base_globals* first rather than *base_locals*.
837 # This only affects ForwardRefs.
838 base_globals, base_locals = base_locals, base_globals
839 for name, value in ann.items():
840 if value is None:
841 value = type(None)
842 if isinstance(value, str):
843 value = _make_forward_ref(value, is_argument=False, is_class=True)
844
845 value = eval_type_backport(value, base_globals, base_locals)
846 hints[name] = value

851 }

871 if isinstance(obj, typing._allowed_types): # type: ignore

873 else:
874 raise TypeError(f'{obj!r} is not a module, class, method, ' 'or function.')
