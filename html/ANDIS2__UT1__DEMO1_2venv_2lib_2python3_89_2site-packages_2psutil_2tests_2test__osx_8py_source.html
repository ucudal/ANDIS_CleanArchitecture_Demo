32def sysctl(cmdline):
33 """Expects a sysctl command with an argument and parse the result
34 returning only the value of interest.
35 """
36 out = sh(cmdline)
37 result = out.split()[1]
38 try:
39 return int(result)
40 except ValueError:
41 return result

44def vm_stat(field):
45 """Wrapper around 'vm_stat' cmdline utility."""
46 out = sh('vm_stat')
47 for line in out.split('\n'):
48 if field in line:
49 break
50 else:
51 raise ValueError("line not found")
52 return int(re.search(r'\d+', line).group(0)) * getpagesize()

82 @retry_on_failure()
83 def test_disks(self):
84 # test psutil.disk_usage() and psutil.disk_partitions()
85 # against "df -a"
86 def df(path):
87 out = sh(f'df -k "{path}"').strip()
88 lines = out.split('\n')
89 lines.pop(0)
90 line = lines.pop(0)
91 dev, total, used, free = line.split()[:4]
92 if dev == 'none':
93 dev = ''
94 total = int(total) * 1024
95 used = int(used) * 1024
96 free = int(free) * 1024
97 return dev, total, used, free
98
99 for part in psutil.disk_partitions(all=False):
100 usage = psutil.disk_usage(part.mountpoint)
101 dev, total, used, free = df(part.mountpoint)
102 assert part.device == dev
103 assert usage.total == total
104 assert abs(usage.free - free) < TOLERANCE_DISK_USAGE
105 assert abs(usage.used - used) < TOLERANCE_DISK_USAGE
