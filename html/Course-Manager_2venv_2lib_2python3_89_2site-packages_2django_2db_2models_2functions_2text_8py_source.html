8class BytesToCharFieldConversionMixin:
9 """
10 Convert CharField results from bytes to str.
11
12 MySQL returns long data types (bytes) instead of chars when it can't
13 determine the length of the result string. For example:
14 LPAD(column1, CHAR_LENGTH(column2), ' ')
15 returns the LONGTEXT (bytes) instead of VARCHAR.
16 """
17 def convert_value(self, value, expression, connection):
18 if connection.features.db_functions_convert_bytes_to_str:
19 if self.output_field.get_internal_type() == 'CharField' and isinstance(value, bytes):
20 return value.decode()
21 return super().convert_value(value, expression, connection)

111class Concat(Func):
112 """
113 Concatenate text fields together. Backends that result in an entire
114 null expression when any arguments are null will wrap each argument in
115 coalesce functions to ensure a non-null result.
116 """
117 function = None
118 template = "%(expressions)s"
119
120 def __init__(self, *expressions, **extra):
121 if len(expressions) < 2:
122 raise ValueError('Concat must take at least two expressions')
123 paired = self._paired(expressions)
124 super().__init__(paired, **extra)
125
126 def _paired(self, expressions):
127 # wrap pairs of expressions in successive concat functions
128 # exp = [a, b, c, d]
129 # -> ConcatPair(a, ConcatPair(b, ConcatPair(c, d))))
130 if len(expressions) == 2:
131 return ConcatPair(*expressions)
132 return ConcatPair(expressions[0], self._paired(expressions[1:]))
