47def proc_info(pid):
48 tcase = PsutilTestCase()
49
50 def check_exception(exc, proc, name, ppid):
51 tcase.assertEqual(exc.pid, pid)
52 if exc.name is not None:
53 tcase.assertEqual(exc.name, name)
54 if isinstance(exc, psutil.ZombieProcess):
55 tcase.assertProcessZombie(proc)
56 if exc.ppid is not None:
57 tcase.assertGreaterEqual(exc.ppid, 0)
58 tcase.assertEqual(exc.ppid, ppid)
59 elif isinstance(exc, psutil.NoSuchProcess):
60 tcase.assertProcessGone(proc)
61 str(exc)
62 repr(exc)
63
64 def do_wait():
65 if pid != 0:
66 try:
67 proc.wait(0)
68 except psutil.Error as exc:
69 check_exception(exc, proc, name, ppid)
70
71 try:
72 proc = psutil.Process(pid)
73 except psutil.NoSuchProcess:
74 tcase.assertPidGone(pid)
75 return {}
76 try:

78 except psutil.NoSuchProcess:
79 tcase.assertProcessGone(proc)
80 else:
81 name, ppid = d['name'], d['ppid']
82 info = {'pid': proc.pid}
83 ns = process_namespace(proc)
84 # We don't use oneshot() because in order not to fool
85 # check_exception() in case of NSP.
86 for fun, fun_name in ns.iter(ns.getters, clear_cache=False):
87 try:
88 info[fun_name] = fun()
89 except psutil.Error as exc:
90 check_exception(exc, proc, name, ppid)
91 continue
92 do_wait()

460class TestPidsRange(PsutilTestCase):
461 """Given pid_exists() return value for a range of PIDs which may or
462 may not exist, make sure that psutil.Process() and psutil.pids()
463 agree with pid_exists(). This guarantees that the 3 APIs are all
464 consistent with each other. See:
465 https://github.com/giampaolo/psutil/issues/2359

467 XXX - Note about Windows: it turns out there are some "hidden" PIDs
468 which are not returned by psutil.pids() and are also not revealed
469 by taskmgr.exe and ProcessHacker, still they can be instantiated by
470 psutil.Process() and queried. One of such PIDs is "conhost.exe".
471 Running as_dict() for it reveals that some Process() APIs
472 erroneously raise NoSuchProcess, so we know we have problem there.
473 Let's ignore this for now, since it's quite a corner case (who even
474 imagined hidden PIDs existed on Windows?).

484 def is_linux_tid(pid):
485 try:
486 f = open(f"/proc/{pid}/status", "rb") # noqa: SIM115
487 except FileNotFoundError:
488 return False
489 else:
490 with f:
491 for line in f:
492 if line.startswith(b"Tgid:"):
493 tgid = int(line.split()[1])
