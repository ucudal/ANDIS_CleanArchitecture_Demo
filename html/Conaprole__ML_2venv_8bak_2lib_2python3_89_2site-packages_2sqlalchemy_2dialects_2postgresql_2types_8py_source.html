83class MONEY(sqltypes.TypeEngine[str]):
84 r"""Provide the PostgreSQL MONEY type.
85
86 Depending on driver, result rows using this type may return a
87 string value which includes currency symbols.
88
89 For this reason, it may be preferable to provide conversion to a
90 numerically-based currency datatype using :class:`_types.TypeDecorator`::
91
92 import re
93 import decimal
94 from sqlalchemy import Dialect
95 from sqlalchemy import TypeDecorator
96
97
98 class NumericMoney(TypeDecorator):
99 impl = MONEY
100
101 def process_result_value(self, value: Any, dialect: Dialect) -> None:
102 if value is not None:
103 # adjust this for the currency and numeric
104 m = re.match(r"\$([\d.]+)", value)
105 if m:
106 value = decimal.Decimal(m.group(1))
107 return value
108
109 Alternatively, the conversion may be applied as a CAST using
110 the :meth:`_types.TypeDecorator.column_expression` method as follows::
111
112 import decimal
113 from sqlalchemy import cast
114 from sqlalchemy import TypeDecorator
115
116
117 class NumericMoney(TypeDecorator):
118 impl = MONEY
119
120 def column_expression(self, column: Any):
121 return cast(column, Numeric())
122
123 .. versionadded:: 1.2
124
125 """ # noqa: E501
126
127 __visit_name__ = "MONEY"
