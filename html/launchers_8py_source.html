16class Launcher(components.Component):

21 def __init__(self, session, stream):
22 with session:
23 assert not session.launcher

34 @message_handler
35 def process_event(self, event):
36 self.pid = event("systemProcessId", int)
37 self.client.propagate_after_start(event)
38
39 @message_handler
40 def output_event(self, event):
41 self.client.propagate_after_start(event)
42
43 @message_handler
44 def exited_event(self, event):
45 self.exit_code = event("exitCode", int)
46 # We don't want to tell the client about this just yet, because it will then
47 # want to disconnect, and the launcher might still be waiting for keypress
48 # (if wait-on-exit was enabled). Instead, we'll report the event when we
49 # receive "terminated" from the launcher, right before it exits.
50
51 @message_handler
52 def terminated_event(self, event):
53 try:
54 self.client.channel.send_event("exited", {"exitCode": self.exit_code})
55 except Exception:
56 pass
57 self.channel.close()
58
59 def terminate_debuggee(self):
60 with self.session:
61 if self.exit_code is None:
62 try:
63 self.channel.request("terminate")
64 except Exception:
65 pass

68def spawn_debuggee(
69 session,
70 start_request,
71 python,
72 launcher_path,
73 adapter_host,
74 args,
75 shell_expand_args,
76 cwd,
77 console,
78 console_title,
79 sudo,
80):
81 global listener
82
83 # -E tells sudo to propagate environment variables to the target process - this
84 # is necessary for launcher to get DEBUGPY_LAUNCHER_PORT and DEBUGPY_LOG_DIR.
85 cmdline = ["sudo", "-E"] if sudo else []
86 cmdline += python
87 cmdline += [launcher_path]
88 env = {}
89
90 arguments = dict(start_request.arguments)
91 if not session.no_debug:
92 _, arguments["port"] = sockets.get_address(servers.listener)
93 arguments["adapterAccessToken"] = adapter.access_token
94
95 def on_launcher_connected(sock):
96 listener.close()
97 stream = messaging.JsonIOStream.from_socket(sock)
98 Launcher(session, stream)
99
100 try:
101 listener = sockets.serve(
102 "Launcher", on_launcher_connected, adapter_host, backlog=1
103 )
104 except Exception as exc:
105 raise start_request.cant_handle(
106 "{0} couldn't create listener socket for launcher: {1}", session, exc
107 )
108 sessions.report_sockets()
109
110 try:
111 launcher_host, launcher_port = sockets.get_address(listener)
112 localhost = sockets.get_default_localhost()
113 launcher_addr = (
114 launcher_port
115 if launcher_host == localhost
116 else f"{launcher_host}:{launcher_port}"
117 )
118 cmdline += [str(launcher_addr), "--"]
119 cmdline += args
120
121 if log.log_dir is not None:
122 env[str("DEBUGPY_LOG_DIR")] = log.log_dir
123 if log.stderr.levels != {"warning", "error"}:
124 env[str("DEBUGPY_LOG_STDERR")] = str(" ".join(log.stderr.levels))
125
126 if console == "internalConsole":
127 log.info("{0} spawning launcher: {1!r}", session, cmdline)
128 try:
129 # If we are talking to the client over stdio, sys.stdin and sys.stdout
130 # are redirected to avoid mangling the DAP message stream. Make sure
131 # the launcher also respects that.
132 subprocess.Popen(
133 cmdline,
134 cwd=cwd,
135 env=dict(list(os.environ.items()) + list(env.items())),
136 stdin=sys.stdin,
137 stdout=sys.stdout,
138 stderr=sys.stderr,
139 )
140 except Exception as exc:
141 raise start_request.cant_handle("Failed to spawn launcher: {0}", exc)
142 else:
143 log.info('{0} spawning launcher via "runInTerminal" request.', session)
144 session.client.capabilities.require("supportsRunInTerminalRequest")
145 kinds = {"integratedTerminal": "integrated", "externalTerminal": "external"}
146 request_args = {
147 "kind": kinds[console],
148 "title": console_title,
149 "args": cmdline,
150 "env": env,
151 }
152 if cwd is not None:
153 request_args["cwd"] = cwd
154 if shell_expand_args:
155 request_args["argsCanBeInterpretedByShell"] = True
156 try:
157 # It is unspecified whether this request receives a response immediately, or only
158 # after the spawned command has completed running, so do not block waiting for it.
159 session.client.channel.send_request("runInTerminal", request_args)
160 except messaging.MessageHandlingError as exc:
161 exc.propagate(start_request)
162
163 # If using sudo, it might prompt for password, and launcher won't start running
164 # until the user enters it, so don't apply timeout in that case.
165 if not session.wait_for(
166 lambda: session.launcher,
167 timeout=(None if sudo else common.PROCESS_SPAWN_TIMEOUT),
168 ):
169 raise start_request.cant_handle("Timed out waiting for launcher to connect")
170
171 try:
172 session.launcher.channel.request(start_request.command, arguments)
173 except messaging.MessageHandlingError as exc:
174 exc.propagate(start_request)
175
176 if not session.wait_for(
177 lambda: session.launcher.pid is not None,
178 timeout=common.PROCESS_SPAWN_TIMEOUT,
179 ):
180 raise start_request.cant_handle(
181 'Timed out waiting for "process" event from launcher'
182 )
183
184 if session.no_debug:
185 return
186
187 # Wait for the first incoming connection regardless of the PID - it won't
188 # necessarily match due to the use of stubs like py.exe or "conda run".
189 conn = servers.wait_for_connection(
190 session, lambda conn: True, timeout=common.PROCESS_SPAWN_TIMEOUT
191 )
192 if conn is None:
193 raise start_request.cant_handle("Timed out waiting for debuggee to spawn")
194 conn.attach_to_session(session)
195
196 finally:
197 listener.close()
198 listener = None
199 sessions.report_sockets()
