31class _DistributionFinder:
32 """Finder to locate distributions.
33
34 The main purpose of this class is to memoize found distributions' names, so
35 only one distribution is returned for each package name. At lot of pip code
36 assumes this (because it is setuptools's behavior), and not doing the same
37 can potentially cause a distribution in lower precedence path to override a
38 higher precedence one if the caller is not careful.
39
40 Eventually we probably want to make it possible to see lower precedence
41 installations as well. It's useful feature, after all.
42 """
43
44 FoundResult = Tuple[importlib.metadata.Distribution, Optional[BasePath]]

46 def __init__(self) -> None:
47 self._found_names: Set[NormalizedName] = set()

49 def _find_impl(self, location: str) -> Iterator[FoundResult]:
50 """Find distributions in a location."""
51 # Skip looking inside a wheel. Since a package inside a wheel is not
52 # always valid (due to .data directories etc.), its .dist-info entry
53 # should not be considered an installed distribution.
54 if _looks_like_wheel(location):
55 return
56 # To know exactly where we find a distribution, we have to feed in the
57 # paths one by one, instead of dumping the list to importlib.metadata.
58 for dist in importlib.metadata.distributions(path=[location]):
59 normalized_name = canonicalize_name(get_dist_name(dist))
60 if normalized_name in self._found_names:
61 continue
62 self._found_names.add(normalized_name)
63 info_location = get_info_location(dist)
64 yield dist, info_location
65
66 def find(self, location: str) -> Iterator[BaseDistribution]:
67 """Find distributions in a location.
68
69 The path can be either a directory, or a ZIP archive.
70 """
71 for dist, info_location in self._find_impl(location):
72 if info_location is None:
73 installed_location: Optional[BasePath] = None
74 else:
75 installed_location = info_location.parent
76 yield Distribution(dist, info_location, installed_location)

78 def find_linked(self, location: str) -> Iterator[BaseDistribution]:
79 """Read location in egg-link files and return distributions in there.
80
81 The path should be a directory; otherwise this returns nothing. This
82 follows how setuptools does this for compatibility. The first non-empty
83 line in the egg-link is read as a path (resolved against the egg-link's
84 containing directory if relative). Distributions found at that linked
85 location are returned.
86 """
87 path = pathlib.Path(location)
88 if not path.is_dir():
89 return
90 for child in path.iterdir():
91 if child.suffix != ".egg-link":
92 continue
93 with child.open() as f:
94 lines = (line.strip() for line in f)
95 target_rel = next((line for line in lines if line), "")
96 if not target_rel:
97 continue
98 target_location = str(path.joinpath(target_rel))
99 for dist, info_location in self._find_impl(target_location):
100 yield Distribution(dist, info_location, path)
