75# Workaround to support (cast_to: Type[ResponseT]) -> ResponseT
76# where ResponseT includes `None`. In order to support directly
77# passing `None`, overloads would have to be defined for every
78# method that uses `ResponseT` which would lead to an unacceptable
79# amount of code duplication and make it unreadable. See _base_client.py
80# for example usage.
81#
82# This unfortunately means that you will either have
83# to import this type and pass it explicitly:
84#
85# from openai import NoneType
86# client.get('/foo', cast_to=NoneType)
87#
88# or build it yourself:
89#
90# client.get('/foo', cast_to=type(None))

137class Omit:
138 """In certain situations you need to be able to represent a case where a default value has
139 to be explicitly removed and `None` is not an appropriate substitute, for example:
140
141 ```py
142 # as the default `Content-Type` header is `application/json` that will be sent
143 client.post("/upload/files", files={"file": b"my raw file content"})
144
145 # you can't explicitly override the header as it has to be dynamically generated
146 # to look something like: 'multipart/form-data; boundary=0d8382fcf5f8c3be01ca2e11002d2983'
147 client.post(..., headers={"Content-Type": "multipart/form-data"})
148
149 # instead you can remove the default `application/json` header by passing Omit
150 client.post(..., headers={"Content-Type": Omit()})
151 ```
152 """
153
154 def __bool__(self) -> Literal[False]:
155 return False
