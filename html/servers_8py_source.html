293 # boundary between the server and the adapter, and we cannot trust arbitrary

317 @message_handler
318 def continued_event(self, event):
319 # https://github.com/microsoft/ptvsd/issues/1530
320 #
321 # DAP specification says that a step request implies that only the thread on
322 # which that step occurred is resumed for the duration of the step. However,
323 # for VS compatibility, pydevd can operate in a mode that resumes all threads
324 # instead. This is set according to the value of "steppingResumesAllThreads"
325 # in "launch" or "attach" request, which defaults to true. If explicitly set
326 # to false, pydevd will only resume the thread that was stepping.
327 #
328 # To ensure that the client is aware that other threads are getting resumed in
329 # that mode, pydevd sends a "continued" event with "allThreadsResumed": true.
330 # when responding to a step request. This ensures correct behavior in VSCode
331 # and other DAP-conformant clients.
332 #
333 # On the other hand, VS does not follow the DAP specification in this regard.
334 # When it requests a step, it assumes that all threads will be resumed, and
335 # does not expect to see "continued" events explicitly reflecting that fact.
336 # If such events are sent regardless, VS behaves erratically. Thus, we have
337 # to suppress them specifically for VS.
338 if self.client.client_id not in ("visualstudio", "vsformac"):
339 self.client.propagate_after_start(event)
340
341 @message_handler

429 def wait_for_timeout():
