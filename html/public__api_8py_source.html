73@_api()
74def configure(__properties: dict[str, typing.Any] | None = None, **kwargs) -> None:
75 """Sets debug configuration properties that cannot be set in the
76 "attach" request, because they must be applied as early as possible
77 in the process being debugged.
78
79 For example, a "launch" configuration with subprocess debugging
80 disabled can be defined entirely in JSON::
81
82 {
83 "request": "launch",
84 "subProcess": false,
85 ...
86 }
87
88 But the same cannot be done with "attach", because "subProcess"
89 must be known at the point debugpy starts tracing execution. Thus,
90 it is not available in JSON, and must be omitted::
91
92 {
93 "request": "attach",
94 ...
95 }
96
97 and set from within the debugged process instead::
98
99 debugpy.configure(subProcess=False)
100 debugpy.listen(...)
101
102 Properties to set can be passed either as a single dict argument,
103 or as separate keyword arguments::
104
105 debugpy.configure({"subProcess": False})
106 """

109@_api()
110def listen(
111 __endpoint: Endpoint | int, *, in_process_debug_adapter: bool = False
112) -> Endpoint:
113 """Starts a debug adapter debugging this process, that listens for
114 incoming socket connections from clients on the specified address.
115
116 `__endpoint` must be either a (host, port) tuple as defined by the
117 standard `socket` module for the `AF_INET` address family, or a port
118 number. If only the port is specified, host is "127.0.0.1".
119
120 `in_process_debug_adapter`: by default a separate python process is
121 spawned and used to communicate with the client as the debug adapter.
122 By setting the value of `in_process_debug_adapter` to True a new
123 python process is not spawned. Note: the con of setting
124 `in_process_debug_adapter` to True is that subprocesses won't be
125 automatically debugged.
126
127 Returns the interface and the port on which the debug adapter is
128 actually listening, in the same format as `__endpoint`. This may be
129 different from address if port was 0 in the latter, in which case
130 the adapter will pick some unused ephemeral port to listen on.
131
132 This function does't wait for a client to connect to the debug
133 adapter that it starts. Use `wait_for_client` to block execution
134 until the client connects.
135 """

199@_api()
200def trace_this_thread(__should_trace: bool):
201 """Tells the debug adapter to enable or disable tracing on the
202 current thread.
203
204 When the thread is traced, the debug adapter can detect breakpoints
205 being hit, but execution is slower, especially in functions that
206 have any breakpoints set in them. Disabling tracing when breakpoints
207 are not anticipated to be hit can improve performance. It can also
208 be used to skip breakpoints on a particular thread.
209
210 Tracing is automatically disabled for all threads when there is no
211 client connected to the debug adapter.
212 """

215def get_cli_options() -> CliOptions | None:
216 """Returns the CLI options that were processed by debugpy.
217
218 These options are all the options after the CLI args and
219 environment variables that were processed on startup.
220
221 If the debugpy CLI entry point was not called in this process, the
222 returned value is None.
223 """
224 from debugpy.server import cli
225
226 options = cli.options
227 if options.mode is None or options.target_kind is None or options.address is None:
228 # The CLI entrypoint was not called so there are no options present.
229 return None
230
231 # We don't return the actual options object because we don't want callers
232 # to be able to mutate it. Instead we use a frozen dataclass as a snapshot
233 # with richer type annotations.
234 return CliOptions(
235 mode=options.mode,
236 target_kind=options.target_kind,
237 address=options.address,
238 log_to=options.log_to,
239 log_to_stderr=options.log_to_stderr,
240 target=options.target,
241 wait_for_client=options.wait_for_client,
242 adapter_access_token=options.adapter_access_token,
243 config=options.config,
244 parent_session_pid=options.parent_session_pid,
245 )
