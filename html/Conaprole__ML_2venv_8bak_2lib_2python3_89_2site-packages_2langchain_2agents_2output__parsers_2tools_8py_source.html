22def parse_ai_message_to_tool_action(
23 message: BaseMessage,
24) -> Union[List[AgentAction], AgentFinish]:
25 """Parse an AI message potentially containing tool_calls."""
26 if not isinstance(message, AIMessage):
27 raise TypeError(f"Expected an AI message got {type(message)}")
28
29 actions: List = []
30 if message.tool_calls:
31 tool_calls = message.tool_calls
32 else:
33 if not message.additional_kwargs.get("tool_calls"):
34 return AgentFinish(
35 return_values={"output": message.content}, log=str(message.content)
36 )
37 # Best-effort parsing
38 tool_calls = []
39 for tool_call in message.additional_kwargs["tool_calls"]:
40 function = tool_call["function"]
41 function_name = function["name"]
42 try:
43 args = json.loads(function["arguments"] or "{}")
44 tool_calls.append(
45 ToolCall(name=function_name, args=args, id=tool_call["id"])
46 )
47 except JSONDecodeError:
48 raise OutputParserException(
49 f"Could not parse tool input: {function} because "
50 f"the `arguments` is not valid JSON."
51 )
52 for tool_call in tool_calls:
53 # HACK HACK HACK:
54 # The code that encodes tool input into Open AI uses a special variable
55 # name called `__arg1` to handle old style tools that do not expose a
56 # schema and expect a single string argument as an input.
57 # We unpack the argument here if it exists.
58 # Open AI does not support passing in a JSON array as an argument.
59 function_name = tool_call["name"]
60 _tool_input = tool_call["args"]
61 if "__arg1" in _tool_input:
62 tool_input = _tool_input["__arg1"]
63 else:
64 tool_input = _tool_input
65
66 content_msg = f"responded: {message.content}\n" if message.content else "\n"
67 log = f"\nInvoking: `{function_name}` with `{tool_input}`\n{content_msg}\n"
68 actions.append(
69 ToolAgentAction(
70 tool=function_name,
71 tool_input=tool_input,
72 log=log,
73 message_log=[message],
74 tool_call_id=tool_call["id"],
75 )
76 )
77 return actions
