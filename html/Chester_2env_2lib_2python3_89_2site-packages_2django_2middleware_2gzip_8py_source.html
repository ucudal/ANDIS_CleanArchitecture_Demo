10class GZipMiddleware(MiddlewareMixin):
11 """
12 Compress content if the browser allows gzip compression.
13 Set the Vary header accordingly, so that caches will base their storage
14 on the Accept-Encoding header.
15 """
16 def process_response(self, request, response):
17 # It's not worth attempting to compress really short responses.
18 if not response.streaming and len(response.content) < 200:
19 return response
20
21 # Avoid gzipping if we've already got a content-encoding.
22 if response.has_header('Content-Encoding'):
23 return response
24
25 patch_vary_headers(response, ('Accept-Encoding',))
26
27 ae = request.META.get('HTTP_ACCEPT_ENCODING', '')
28 if not re_accepts_gzip.search(ae):
29 return response
30
31 if response.streaming:
32 # Delete the `Content-Length` header for streaming content, because
33 # we won't know the compressed size until we stream it.
34 response.streaming_content = compress_sequence(response.streaming_content)
35 del response['Content-Length']
36 else:
37 # Return the compressed content only if it's actually shorter.
38 compressed_content = compress_string(response.content)
39 if len(compressed_content) >= len(response.content):
40 return response
41 response.content = compressed_content
42 response['Content-Length'] = str(len(response.content))
43
44 # If there is a strong ETag, make it weak to fulfill the requirements
45 # of RFC 7232 section-2.1 while also allowing conditional request
46 # matches on ETags.
47 etag = response.get('ETag')
48 if etag and etag.startswith('"'):
49 response['ETag'] = 'W/' + etag
50 response['Content-Encoding'] = 'gzip'
51
52 return response
