361 def send_close(self, code: int | None = None, reason: str = "") -> None:
362 """
363 Send a `Close frame`_.
364
365 .. _Close frame:
366 https://datatracker.ietf.org/doc/html/rfc6455#section-5.5.1
367
368 Parameters:
369 code: close code.
370 reason: close reason.
371
372 Raises:
373 ProtocolError: If the code isn't valid or if a reason is provided
374 without a code.
375
376 """
377 # While RFC 6455 doesn't rule out sending more than one close Frame,
378 # websockets is conservative in what it sends and doesn't allow that.
379 if self._state is not OPEN:
380 raise InvalidState(f"connection is {self.state.name.lower()}")
381 if code is None:
382 if reason != "":
383 raise ProtocolError("cannot send a reason without a code")
384 close = Close(CloseCode.NO_STATUS_RCVD, "")
385 data = b""
386 else:
387 close = Close(code, reason)
388 data = close.serialize()
389 # 7.1.3. The WebSocket Closing Handshake is Started
390 self.send_frame(Frame(OP_CLOSE, data))
391 # Since the state is OPEN, no close frame was received yet.
392 # As a consequence, self.close_rcvd_then_sent remains None.
393 assert self.close_rcvd is None
394 self.close_sent = close
395 self.state = CLOSING

429 def fail(self, code: int, reason: str = "") -> None:
430 """
431 `Fail the WebSocket connection`_.
432
433 .. _Fail the WebSocket connection:
434 https://datatracker.ietf.org/doc/html/rfc6455#section-7.1.7
435
436 Parameters:
437 code: close code
438 reason: close reason
439
440 Raises:
441 ProtocolError: If the code isn't valid.
442 """
443 # 7.1.7. Fail the WebSocket Connection
444
445 # Send a close frame when the state is OPEN (a close frame was already
446 # sent if it's CLOSING), except when failing the connection because
447 # of an error reading from or writing to the network.
448 if self.state is OPEN:
449 if code != CloseCode.ABNORMAL_CLOSURE:
450 close = Close(code, reason)
451 data = close.serialize()
452 self.send_frame(Frame(OP_CLOSE, data))
453 self.close_sent = close
454 # If recv_messages() raised an exception upon receiving a close
455 # frame but before echoing it, then close_rcvd is not None even
456 # though the state is OPEN. This happens when the connection is
457 # closed while receiving a fragmented message.
458 if self.close_rcvd is not None:
459 self.close_rcvd_then_sent = True
460 self.state = CLOSING
461
462 # When failing the connection, a server closes the TCP connection
463 # without waiting for the client to complete the handshake, while a
464 # client waits for the server to close the TCP connection, possibly
465 # after sending a close frame that the client will ignore.
466 if self.side is SERVER and not self.eof_sent:
467 self.send_eof()
468
469 # 7.1.7. Fail the WebSocket Connection "An endpoint MUST NOT continue
470 # to attempt to process data(including a responding Close frame) from
471 # the remote endpoint after being instructed to _Fail the WebSocket
472 # Connection_."
473 self.parser = self.discard()
474 next(self.parser) # start coroutine

514 def close_expected(self) -> bool:
515 """
516 Tell if the TCP connection is expected to close soon.
517
518 Call this method immediately after any of the ``receive_*()``,
519 ``send_close()``, or :meth:`fail` methods.
520
521 If it returns :obj:`True`, schedule closing the TCP connection after a
522 short timeout if the other side hasn't already closed it.
523
524 Returns:
525 Whether the TCP connection is expected to close soon.
526
527 """
528 # During the opening handshake, when our state is CONNECTING, we expect
529 # a TCP close if and only if the hansdake fails. When it does, we start
530 # the TCP closing handshake by sending EOF with send_eof().
531
532 # Once the opening handshake completes successfully, we expect a TCP
533 # close if and only if we sent a close frame, meaning that our state
534 # progressed to CLOSING:
535
536 # * Normal closure: once we send a close frame, we expect a TCP close:
537 # server waits for client to complete the TCP closing handshake;
538 # client waits for server to initiate the TCP closing handshake.
539
540 # * Abnormal closure: we always send a close frame and the same logic
541 # applies, except on EOFError where we don't send a close frame
542 # because we already received the TCP close, so we don't expect it.
543
544 # If our state is CLOSED, we already received a TCP close so we don't
545 # expect it anymore.
546
547 # Micro-optimization: put the most common case first
548 if self.state is OPEN:
549 return False
550 if self.state is CLOSING:
551 return True
552 if self.state is CLOSED:
553 return False
554 assert self.state is CONNECTING
555 return self.eof_sent

559 def parse(self) -> Generator[None]:
560 """
561 Parse incoming data into frames.
562
563 :meth:`receive_data` and :meth:`receive_eof` run this generator
564 coroutine until it needs more data or reaches EOF.
565
566 :meth:`parse` never raises an exception. Instead, it sets the
567 :attr:`parser_exc` and yields control.
568
569 """
570 try:
571 while True:
572 if (yield from self.reader.at_eof()):
573 if self.debug:
574 self.logger.debug("< EOF")
575 # If the WebSocket connection is closed cleanly, with a
576 # closing handhshake, recv_frame() substitutes parse()
577 # with discard(). This branch is reached only when the
578 # connection isn't closed cleanly.
579 raise EOFError("unexpected end of stream")
580
581 if self.max_size is None:
582 max_size = None
583 elif self.cur_size is None:
584 max_size = self.max_size
585 else:
586 max_size = self.max_size - self.cur_size
587
588 # During a normal closure, execution ends here on the next
589 # iteration of the loop after receiving a close frame. At
590 # this point, recv_frame() replaced parse() by discard().
591 frame = yield from Frame.parse(
592 self.reader.read_exact,
593 mask=self.side is SERVER,
594 max_size=max_size,
595 extensions=self.extensions,
596 )
597
598 if self.debug:
599 self.logger.debug("< %s", frame)
600
601 self.recv_frame(frame)
602
603 except ProtocolError as exc:
604 self.fail(CloseCode.PROTOCOL_ERROR, str(exc))
605 self.parser_exc = exc
606
607 except EOFError as exc:
608 self.fail(CloseCode.ABNORMAL_CLOSURE, str(exc))
609 self.parser_exc = exc
610
611 except UnicodeDecodeError as exc:
612 self.fail(CloseCode.INVALID_DATA, f"{exc.reason} at position {exc.start}")
613 self.parser_exc = exc
614
615 except PayloadTooBig as exc:
616 exc.set_current_size(self.cur_size)
617 self.fail(CloseCode.MESSAGE_TOO_BIG, str(exc))
618 self.parser_exc = exc
619
620 except Exception as exc:
621 self.logger.error("parser failed", exc_info=True)
622 # Don't include exception details, which may be security-sensitive.
623 self.fail(CloseCode.INTERNAL_ERROR)
624 self.parser_exc = exc
625
626 # During an abnormal closure, execution ends here after catching an
627 # exception. At this point, fail() replaced parse() by discard().
628 yield
629 raise AssertionError("parse() shouldn't step after error")

631 def discard(self) -> Generator[None]:
632 """
633 Discard incoming data.
634
635 This coroutine replaces :meth:`parse`:
636
637 - after receiving a close frame, during a normal closure (1.4);
638 - after sending a close frame, during an abnormal closure (7.1.7).
639
640 """
641 # After the opening handshake completes, the server closes the TCP
642 # connection in the same circumstances where discard() replaces parse().
643 # The client closes it when it receives EOF from the server or times
644 # out. (The latter case cannot be handled in this Sans-I/O layer.)
645 assert (self.side is SERVER or self.state is CONNECTING) == (self.eof_sent)
646 while not (yield from self.reader.at_eof()):
647 self.reader.discard()
648 if self.debug:
649 self.logger.debug("< EOF")
650 # A server closes the TCP connection immediately, while a client
651 # waits for the server to close the TCP connection.
652 if self.side is CLIENT and self.state is not CONNECTING:
653 self.send_eof()
654 self.state = CLOSED
655 # If discard() completes normally, execution ends here.
656 yield
657 # Once the reader reaches EOF, its feed_data/eof() methods raise an
658 # error, so our receive_data/eof() methods don't step the generator.
659 raise AssertionError("discard() shouldn't step after EOF")
660
661 def recv_frame(self, frame: Frame) -> None:
662 """
663 Process an incoming frame.
664
665 """
666 if frame.opcode is OP_TEXT or frame.opcode is OP_BINARY:
667 if self.cur_size is not None:
668 raise ProtocolError("expected a continuation frame")
669 if not frame.fin:
670 self.cur_size = len(frame.data)
671
672 elif frame.opcode is OP_CONT:
673 if self.cur_size is None:
674 raise ProtocolError("unexpected continuation frame")
675 if frame.fin:
676 self.cur_size = None
677 else:
678 self.cur_size += len(frame.data)
679
680 elif frame.opcode is OP_PING:
681 # 5.5.2. Ping: "Upon receipt of a Ping frame, an endpoint MUST
682 # send a Pong frame in response"
683 pong_frame = Frame(OP_PONG, frame.data)
684 self.send_frame(pong_frame)
685
686 elif frame.opcode is OP_PONG:
687 # 5.5.3 Pong: "A response to an unsolicited Pong frame is not
688 # expected."
689 pass
690
691 elif frame.opcode is OP_CLOSE:
692 # 7.1.5. The WebSocket Connection Close Code
693 # 7.1.6. The WebSocket Connection Close Reason
694 self.close_rcvd = Close.parse(frame.data)
695 if self.state is CLOSING:
696 assert self.close_sent is not None
697 self.close_rcvd_then_sent = False
698
699 if self.cur_size is not None:
700 raise ProtocolError("incomplete fragmented message")
701
702 # 5.5.1 Close: "If an endpoint receives a Close frame and did
703 # not previously send a Close frame, the endpoint MUST send a
704 # Close frame in response. (When sending a Close frame in
705 # response, the endpoint typically echos the status code it
706 # received.)"
707
708 if self.state is OPEN:
709 # Echo the original data instead of re-serializing it with
710 # Close.serialize() because that fails when the close frame
711 # is empty and Close.parse() synthesizes a 1005 close code.
712 # The rest is identical to send_close().
713 self.send_frame(Frame(OP_CLOSE, frame.data))
714 self.close_sent = self.close_rcvd
715 self.close_rcvd_then_sent = True
716 self.state = CLOSING
717
718 # 7.1.2. Start the WebSocket Closing Handshake: "Once an
719 # endpoint has both sent and received a Close control frame,
720 # that endpoint SHOULD _Close the WebSocket Connection_"
721
722 # A server closes the TCP connection immediately, while a client
723 # waits for the server to close the TCP connection.
724 if self.side is SERVER:
725 self.send_eof()
726
727 # 1.4. Closing Handshake: "after receiving a control frame
728 # indicating the connection should be closed, a peer discards
729 # any further data received."
730 # RFC 6455 allows reading Ping and Pong frames after a Close frame.
731 # However, that doesn't seem useful; websockets doesn't support it.
732 self.parser = self.discard()
733 next(self.parser) # start coroutine
734
735 else:
736 # This can't happen because Frame.parse() validates opcodes.
737 raise AssertionError(f"unexpected opcode: {frame.opcode:02x}")
738
739 self.events.append(frame)
