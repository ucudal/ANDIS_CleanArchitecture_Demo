197def build_filter(
198 filter_name: str, ignore_paths_str: Optional[str]
199) -> Tuple[Union[None, DefaultFilter, Callable[[Change, str], bool]], str]:
200 ignore_paths: List[Path] = []
201 if ignore_paths_str:
202 ignore_paths = [Path(p).resolve() for p in ignore_paths_str.split(',')]
203
204 if filter_name == 'default':
205 return DefaultFilter(ignore_paths=ignore_paths), 'DefaultFilter'
206 elif filter_name == 'python':
207 return PythonFilter(ignore_paths=ignore_paths), 'PythonFilter'
208 elif filter_name == 'all':
209 if ignore_paths:
210 logger.warning('"--ignore-paths" argument ignored as "all" filter was selected')
211 return None, '(no filter)'
212
213 watch_filter_cls = import_exit(filter_name)
214 if isinstance(watch_filter_cls, type) and issubclass(watch_filter_cls, DefaultFilter):
215 return watch_filter_cls(ignore_paths=ignore_paths), watch_filter_cls.__name__
216
217 if ignore_paths:
218 logger.warning('"--ignore-paths" argument ignored as filter is not a subclass of DefaultFilter')
219
220 if isinstance(watch_filter_cls, type) and issubclass(watch_filter_cls, BaseFilter):
221 return watch_filter_cls(), watch_filter_cls.__name__
222 else:
223 watch_filter = cast(Callable[[Change, str], bool], watch_filter_cls)
224 return watch_filter, repr(watch_filter_cls)
