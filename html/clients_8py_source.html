115 def propagate_after_start(self, event):
116 # pydevd starts sending events as soon as we connect, but the client doesn't
117 # expect to see any until it receives the response to "launch" or "attach"
118 # request. If client is not ready yet, save the event instead of propagating
119 # it immediately.

122 log.debug("Propagation deferred.")

126 def _propagate_deferred_events(self):

131 log.info("All deferred events propagated to {0}.", self.client)

202 # Common code for "launch" and "attach" request handlers.
203 #
204 # See https://github.com/microsoft/vscode/issues/4902#issuecomment-368583522
205 # for the sequence of request and events necessary to orchestrate the start.
206 def _start_message_handler(f):
207 @components.Component.message_handler
208 def handle(self, request):
209 assert request.is_request("launch", "attach")
210 if self._initialize_request is None:
211 raise request.isnt_valid("Session is not initialized yet")
212 if self.launcher or self.server:
213 raise request.isnt_valid("Session is already started")
214
215 self.session.no_debug = request("noDebug", json.default(False))
216 if self.session.no_debug:
217 servers.dont_wait_for_first_connection()
218
219 self.session.debug_options = debug_options = set(
220 request("debugOptions", json.array(str))
221 )
222
223 f(self, request)
224 if request.response is not None:
225 return
226
227 if self.server:
228 self.server.initialize(self._initialize_request)
229 self._initialize_request = None
230
231 arguments = request.arguments
232 if self.launcher:
233 redirecting = arguments.get("console") == "internalConsole"
234 if "RedirectOutput" in debug_options:
235 # The launcher is doing output redirection, so we don't need the
236 # server to do it, as well.
237 arguments = dict(arguments)
238 arguments["debugOptions"] = list(
239 debug_options - {"RedirectOutput"}
240 )
241 redirecting = True
242
243 if arguments.get("redirectOutput"):
244 arguments = dict(arguments)
245 del arguments["redirectOutput"]
246 redirecting = True
247
248 arguments["isOutputRedirected"] = redirecting
249
250 # pydevd doesn't send "initialized", and responds to the start request
251 # immediately, without waiting for "configurationDone". If it changes
252 # to conform to the DAP spec, we'll need to defer waiting for response.
253 try:
254 self.server.channel.request(request.command, arguments)
255 except messaging.NoMoreMessages:
256 # Server closed connection before we could receive the response to
257 # "attach" or "launch" - this can happen when debuggee exits shortly
258 # after starting. It's not an error, but we can't do anything useful
259 # here at this point, either, so just bail out.
260 request.respond({})
261 self.session.finalize(
262 "{0} disconnected before responding to {1}".format(
263 self.server,
264 json.repr(request.command),
265 )
266 )
267 return
268 except messaging.MessageHandlingError as exc:
269 exc.propagate(request)
270
271 if self.session.no_debug:
272 self.start_request = request
273 self.has_started = True
274 request.respond({})
275 self._propagate_deferred_events()
276 return
277
278 # Let the client know that it can begin configuring the adapter.
279 self.channel.send_event("initialized")
280
281 self.start_request = request
282 return messaging.NO_RESPONSE # will respond on "configurationDone"
283
284 return handle

434 def attach_request(self, request):
435 if self.session.no_debug:
436 raise request.isnt_valid('"noDebug" is not supported for "attach"')
437
438 host = request("host", str, optional=True)
439 port = request("port", int, optional=True)
440 listen = request("listen", dict, optional=True)
441 connect = request("connect", dict, optional=True)
442 pid = request("processId", (int, str), optional=True)
443 sub_pid = request("subProcessId", int, optional=True)
444 on_terminate = request("onTerminate", bool, optional=True)
445
446 if on_terminate:
447 self._forward_terminate_request = on_terminate == "KeyboardInterrupt"
448
449 if host != () or port != ():
450 if listen != ():
451 raise request.isnt_valid(
452 '"listen" and "host"/"port" are mutually exclusive'
453 )
454 if connect != ():
455 raise request.isnt_valid(
456 '"connect" and "host"/"port" are mutually exclusive'
457 )
458 if listen != ():
459 if connect != ():
460 raise request.isnt_valid(
461 '"listen" and "connect" are mutually exclusive'
462 )
463 if pid != ():
464 raise request.isnt_valid(
465 '"listen" and "processId" are mutually exclusive'
466 )
467 if sub_pid != ():
468 raise request.isnt_valid(
469 '"listen" and "subProcessId" are mutually exclusive'
470 )
471 if pid != () and sub_pid != ():
472 raise request.isnt_valid(
473 '"processId" and "subProcessId" are mutually exclusive'
474 )
475
476 localhost = sockets.get_default_localhost()
477 if listen != ():
478 if servers.is_serving():
479 raise request.isnt_valid(
480 'Multiple concurrent "listen" sessions are not supported'
481 )
482 host = listen("host", localhost)
483 port = listen("port", int)
484 adapter.access_token = None
485 self.restart_requested = request("restart", False)
486 host, port = servers.serve(host, port)
487 else:
488 if not servers.is_serving():
489 servers.serve(localhost)
490 host, port = sockets.get_address(servers.listener)
491
492 # There are four distinct possibilities here.
493 #
494 # If "processId" is specified, this is attach-by-PID. We need to inject the
495 # debug server into the designated process, and then wait until it connects
496 # back to us. Since the injected server can crash, there must be a timeout.
497 #
498 # If "subProcessId" is specified, this is attach to a known subprocess, likely
499 # in response to a "debugpyAttach" event. If so, the debug server should be
500 # connected already, and thus the wait timeout is zero.
501 #
502 # If "listen" is specified, this is attach-by-socket with the server expected
503 # to connect to the adapter via debugpy.connect(). There is no PID known in
504 # advance, so just wait until the first server connection indefinitely, with
505 # no timeout.
506 #
507 # If "connect" is specified, this is attach-by-socket in which the server has
508 # spawned the adapter via debugpy.listen(). There is no PID known to the client
509 # in advance, but the server connection should be either be there already, or
510 # the server should be connecting shortly, so there must be a timeout.
511 #
512 # In the last two cases, if there's more than one server connection already,
513 # this is a multiprocess re-attach. The client doesn't know the PID, so we just
514 # connect it to the oldest server connection that we have - in most cases, it
515 # will be the one for the root debuggee process, but if it has exited already,
516 # it will be some subprocess.
517 if pid != ():
518 if not isinstance(pid, int):
519 try:
520 pid = int(pid)
521 except Exception:
522 raise request.isnt_valid('"processId" must be parseable as int')
523 debugpy_args = request("debugpyArgs", json.array(str))
524
525 def on_output(category, output):
526 self.channel.send_event(
527 "output",
528 {
529 "category": category,
530 "output": output,
531 },
532 )
533
534 try:
535 servers.inject(pid, debugpy_args, on_output)
536 except Exception as e:
537 log.swallow_exception()
538 self.session.finalize(
539 "Error when trying to attach to PID:\n%s" % (str(e),)
540 )
541 return
542
543 timeout = common.PROCESS_SPAWN_TIMEOUT
544 pred = lambda conn: conn.pid == pid
545 else:
546 if sub_pid == ():
547 pred = lambda conn: True
548 timeout = common.PROCESS_SPAWN_TIMEOUT if listen == () else None
549 else:
550 pred = lambda conn: conn.pid == sub_pid
551 timeout = 0
552
553 self.channel.send_event("debugpyWaitingForServer", {"host": host, "port": port})
554 conn = servers.wait_for_connection(self.session, pred, timeout)
555 if conn is None:
556 if sub_pid != ():
557 # If we can't find a matching subprocess, it's not always an error -
558 # it might have already exited, or didn't even get a chance to connect.
559 # To prevent the client from complaining, pretend that the "attach"
560 # request was successful, but that the session terminated immediately.
561 request.respond({})
562 self.session.finalize(
563 'No known subprocess with "subProcessId":{0}'.format(sub_pid)
564 )
565 return
566
567 raise request.cant_handle(
568 (
569 "Timed out waiting for debug server to connect."
570 if timeout
571 else "There is no debug server connected to this adapter."
572 ),
573 sub_pid,
574 )
575
576 try:
577 conn.attach_to_session(self.session)
578 except ValueError:
579 request.cant_handle("{0} is already being debugged.", conn)

581 @message_handler
582 def configurationDone_request(self, request):
583 if self.start_request is None or self.has_started:
584 request.cant_handle(
585 '"configurationDone" is only allowed during handling of a "launch" '
586 'or an "attach" request'
587 )
588
589 try:
590 self.has_started = True
591 try:
592 result = self.server.channel.delegate(request)
593 except messaging.NoMoreMessages:
594 # Server closed connection before we could receive the response to
595 # "configurationDone" - this can happen when debuggee exits shortly
596 # after starting. It's not an error, but we can't do anything useful
597 # here at this point, either, so just bail out.
598 request.respond({})
599 self.start_request.respond({})
600 self.session.finalize(
601 "{0} disconnected before responding to {1}".format(
602 self.server,
603 json.repr(request.command),
604 )
605 )
606 return
607 else:
608 request.respond(result)
609 except messaging.MessageHandlingError as exc:
610 self.start_request.cant_handle(str(exc))
611 finally:
612 if self.start_request.response is None:
613 self.start_request.respond({})
614 self._propagate_deferred_events()

617 # being debugged.
618 for conn in servers.connections():
619 if conn.server is None and conn.ppid == self.session.pid:

664 @message_handler
665 def terminate_request(self, request):
666 # If user specifically requests to terminate, it means that they don't want
667 # debug session auto-restart kicking in.
668 self.restart_requested = False
669
670 if self._forward_terminate_request:
671 # According to the spec, terminate should try to do a gracefull shutdown.
672 # We do this in the server by interrupting the main thread with a Ctrl+C.
673 # To force the kill a subsequent request would do a disconnect.
674 #
675 # We only do this if the onTerminate option is set though (the default
676 # is a hard-kill for the process and subprocesses).
677 return self.server.channel.delegate(request)
678
679 self.session.finalize('client requested "terminate"', terminate_debuggee=True)
680 return {}

732 def notify_of_subprocess(self, conn):
733 log.info("{1} is a subprocess of {0}.", self, conn)
734 with self.session:
735 if self.start_request is None or conn in self.known_subprocesses:
736 return
737 if "processId" in self.start_request.arguments:
738 log.warning(
739 "Not reporting subprocess for {0}, because the parent process "
740 'was attached to using "processId" rather than "port".',
741 self.session,
742 )
743 return
744
745 log.info("Notifying {0} about {1}.", self, conn)
746 body = dict(self.start_request.arguments)
