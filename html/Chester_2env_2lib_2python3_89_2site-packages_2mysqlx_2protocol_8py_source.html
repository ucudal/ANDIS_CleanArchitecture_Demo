95class Compressor:
96 """Implements compression/decompression using `zstd_stream`, `lz4_message`
97 and `deflate_stream` algorithms.
98
99 Args:
100 algorithm (str): Compression algorithm.
101
102 .. versionadded:: 8.0.21
103
104 """
105
106 def __init__(self, algorithm: str) -> None:
107 self._algorithm: str = algorithm
108 self._compressobj: Any = None
109 self._decompressobj: Any = None
110
111 if algorithm == "zstd_stream":
112 self._compressobj = zstd.ZstdCompressor()
113 self._decompressobj = zstd.ZstdDecompressor()
114 elif algorithm == "deflate_stream":
115 self._compressobj = zlib.compressobj()
116 self._decompressobj = zlib.decompressobj()
117
118 def compress(self, data: StrOrBytes) -> bytes:
119 """Compresses data and returns it.
120
121 Args:
122 data (str, bytes or buffer object): Data to be compressed.
123
124 Returns:
125 bytes: Compressed data.
126 """
127 if self._algorithm == "zstd_stream":
128 return self._compressobj.compress(data)
129 if self._algorithm == "lz4_message":
130 with lz4.frame.LZ4FrameCompressor() as compressor:
131 compressed = compressor.begin()
132 compressed += compressor.compress(data)
133 compressed += compressor.flush()
134 return compressed
135
136 # Using 'deflate_stream' algorithm
137 compressed = self._compressobj.compress(data)
138 compressed += self._compressobj.flush(zlib.Z_SYNC_FLUSH)
139 return compressed
140
141 def decompress(self, data: StrOrBytes) -> bytes:
142 """Decompresses a frame of data and returns it as a string of bytes.
143
144 Args:
145 data (str, bytes or buffer object): Data to be compressed.
146
147 Returns:
148 bytes: Decompresssed data.
149 """
150 if self._algorithm == "zstd_stream":
151 return self._decompressobj.decompress(data)
152 if self._algorithm == "lz4_message":
153 with lz4.frame.LZ4FrameDecompressor() as decompressor:
154 decompressed = decompressor.decompress(data)
155 return decompressed
156
157 # Using 'deflate' algorithm
158 decompressed = self._decompressobj.decompress(data)
159 decompressed += self._decompressobj.flush(zlib.Z_SYNC_FLUSH)
160 return decompressed

163class MessageReader:
164 """Implements a Message Reader.
165
166 Args:
167 socket_stream (mysqlx.connection.SocketStream): `SocketStream` object.
168
169 .. versionadded:: 8.0.21
170 """
171
172 def __init__(self, socket_stream: SocketType) -> None:
173 self._stream: SocketType = socket_stream
174 self._compressor: Optional[Compressor] = None
175 self._msg: MessageType = None
176 self._msg_queue: List[Message] = []
177
178 def _read_message(self) -> MessageType:
179 """Reads X Protocol messages from the stream and returns a
180 :class:`mysqlx.protobuf.Message` object.
181
182 Raises:
183 :class:`mysqlx.ProgrammingError`: If e connected server does not
184 have the MySQL X protocol plugin
185 enabled.
186
187 Returns:
188 mysqlx.protobuf.Message: MySQL X Protobuf Message.
189 """
190 if self._msg_queue:
191 return self._msg_queue.pop(0)
192
193 frame_size, frame_type = struct.unpack("<LB", self._stream.read(5))
194
195 if frame_type == 10:
196 raise ProgrammingError(
197 "The connected server does not have the "
198 "MySQL X protocol plugin enabled or "
199 "protocol mismatch"
200 )
201
202 frame_payload = self._stream.read(frame_size - 1)
203 if frame_type not in SERVER_MESSAGES:
204 raise ValueError(f"Unknown message type: {frame_type}")
205
206 # Do not parse empty notices, Message requires a type in payload
207 if frame_type == 11 and frame_payload == b"":
208 return self._read_message()
209
210 frame_msg = Message.from_server_message(frame_type, frame_payload)
211
212 if frame_type == 19: # Mysqlx.ServerMessages.Type.COMPRESSION
213 uncompressed_size = frame_msg["uncompressed_size"]
214 stream = BytesIO(self._compressor.decompress(frame_msg["payload"]))
215 bytes_processed = 0
216 while bytes_processed < uncompressed_size:
217 payload_size, msg_type = struct.unpack("<LB", stream.read(5))
218 payload = stream.read(payload_size - 1)
219 self._msg_queue.append(Message.from_server_message(msg_type, payload))
220 bytes_processed += payload_size + 4
221 return self._msg_queue.pop(0) if self._msg_queue else None
222
223 return frame_msg
224
225 def read_message(self) -> MessageType:
226 """Read message.
227
228 Returns:
229 mysqlx.protobuf.Message: MySQL X Protobuf Message.
230 """
231 if self._msg is not None:
232 msg = self._msg
233 self._msg = None
234 return msg
235 return self._read_message()
236
237 def push_message(self, msg: MessageType) -> None:
238 """Push message.
239
240 Args:
241 msg (mysqlx.protobuf.Message): MySQL X Protobuf Message.
242
243 Raises:
244 :class:`mysqlx.OperationalError`: If message push slot is full.
245 """
246 if self._msg is not None:
247 raise OperationalError("Message push slot is full")
248 self._msg = msg

263class MessageWriter:
264 """Implements a Message Writer.
265
266 Args:
267 socket_stream (mysqlx.connection.SocketStream): `SocketStream` object.
268
269 .. versionadded:: 8.0.21
270
271 """
272
273 def __init__(self, socket_stream: SocketType) -> None:
274 self._stream: SocketType = socket_stream
275 self._compressor: Optional[Compressor] = None
276
277 def write_message(self, msg_type: int, msg: MessageType) -> None:
278 """Write message.
279
280 Args:
281 msg_type (int): The message type.
282 msg (mysqlx.protobuf.Message): MySQL X Protobuf Message.
283 """
284 msg_size = msg.byte_size(msg)
285 if self._compressor and msg_size > _COMPRESSION_THRESHOLD:
286 msg_str = encode_to_bytes(msg.serialize_to_string())
287 header = struct.pack("<LB", msg_size + 1, msg_type)
288 compressed = self._compressor.compress(b"".join([header, msg_str]))
289
290 msg_first_fields = Message("Mysqlx.Connection.Compression")
291 msg_first_fields["client_messages"] = msg_type
292 msg_first_fields["uncompressed_size"] = msg_size + 5
293
294 msg_payload = Message("Mysqlx.Connection.Compression")
295 msg_payload["payload"] = compressed
296
297 output = b"".join(
298 [
299 encode_to_bytes(msg_first_fields.serialize_partial_to_string())[
300 :-2
301 ],
302 encode_to_bytes(msg_payload.serialize_partial_to_string()),
303 ]
304 )
305
306 msg_comp_id = mysqlxpb_enum("Mysqlx.ClientMessages.Type.COMPRESSION")
307 header = struct.pack("<LB", len(output) + 1, msg_comp_id)
308 self._stream.sendall(b"".join([header, output]))
309 else:
310 msg_str = encode_to_bytes(msg.serialize_to_string())
311 header = struct.pack("<LB", msg_size + 1, msg_type)
312 self._stream.sendall(b"".join([header, msg_str]))

362 def _create_any(self, arg: Any) -> Optional[MessageType]:
363 """Create any.
364
365 Args:
366 arg (object): Arbitrary object.
367
368 Returns:
369 mysqlx.protobuf.Message: MySQL X Protobuf Message.
370 """
371 if isinstance(arg, str):
372 value = Message("Mysqlx.Datatypes.Scalar.String", value=arg)
373 scalar = Message("Mysqlx.Datatypes.Scalar", type=8, v_string=value)
374 return Message("Mysqlx.Datatypes.Any", type=1, scalar=scalar)
375 if isinstance(arg, bool):
376 return Message(
377 "Mysqlx.Datatypes.Any", type=1, scalar=build_bool_scalar(arg)
378 )
379 if isinstance(arg, int):
380 if arg < 0:
381 return Message(
382 "Mysqlx.Datatypes.Any",
383 type=1,
384 scalar=build_int_scalar(arg),
385 )
386 return Message(
387 "Mysqlx.Datatypes.Any",
388 type=1,
389 scalar=build_unsigned_int_scalar(arg),
390 )
391 if isinstance(arg, tuple) and len(arg) == 2:
392 arg_key, arg_value = arg
393 obj_fld = Message(
394 "Mysqlx.Datatypes.Object.ObjectField",
395 key=arg_key,
396 value=self._create_any(arg_value),
397 )
398 obj = Message("Mysqlx.Datatypes.Object", fld=[obj_fld.get_message()])
399 return Message("Mysqlx.Datatypes.Any", type=2, obj=obj)
400 if isinstance(arg, dict) or (
401 isinstance(arg, (list, tuple)) and isinstance(arg[0], dict)
402 ):
403 array_values = []
404 for items in arg:
405 obj_flds = []
406 for key, value in items.items():
407 # Array can only handle Any types, Mysqlx.Datatypes.Any.obj
408 obj_fld = Message(
409 "Mysqlx.Datatypes.Object.ObjectField",
410 key=key,
411 value=self._create_any(value),
412 )
413 obj_flds.append(obj_fld.get_message())
414 msg_obj = Message("Mysqlx.Datatypes.Object", fld=obj_flds)
415 msg_any = Message("Mysqlx.Datatypes.Any", type=2, obj=msg_obj)
416 array_values.append(msg_any.get_message())
417
418 msg = Message("Mysqlx.Datatypes.Array")
419 msg["value"] = array_values
420 return Message("Mysqlx.Datatypes.Any", type=3, array=msg)
421 if isinstance(arg, list):
422 obj_flds = []
423 for key, value in arg:
424 obj_fld = Message(
425 "Mysqlx.Datatypes.Object.ObjectField",
426 key=key,
427 value=self._create_any(value),
428 )
429 obj_flds.append(obj_fld.get_message())
430 msg_obj = Message("Mysqlx.Datatypes.Object", fld=obj_flds)
431 msg_any = Message("Mysqlx.Datatypes.Any", type=2, obj=msg_obj)
432 return msg_any
433
434 return None

436 def _get_binding_args(
437 self, stmt: Union[FilterableStatement, SqlStatement], is_scalar: bool = True
438 ) -> Union[List[None], List[Union[ProtobufMessageType, ProtobufMessageCextType]]]:
439 """Returns the binding any/scalar.
440
441 Args:
442 stmt (Statement): A `Statement` based type object.
443 is_scalar (bool): `True` to return scalar values.
444
445 Raises:
446 :class:`mysqlx.ProgrammingError`: If unable to find placeholder for
447 parameter.
448
449 Returns:
450 list: A list of ``Any`` or ``Scalar`` objects.
451 """
452
453 def build_value(
454 value: Any,
455 ) -> Union[ProtobufMessageType, ProtobufMessageCextType]:
456 if is_scalar:
457 return build_scalar(value).get_message()
458 return self._create_any(value).get_message()
459
460 bindings = stmt.get_bindings()
461 binding_map = stmt.get_binding_map()
462
463 # If binding_map is None it's a SqlStatement object
464 if binding_map is None:
465 return [build_value(value) for value in bindings]
466
467 count = len(binding_map)
468 args: List[Any] = count * [None]
469 if count != len(bindings):
470 raise ProgrammingError(
471 "The number of bind parameters and placeholders do not match"
472 )
473 for name, value in bindings.items(): # type: ignore[union-attr]
474 if name not in binding_map:
475 raise ProgrammingError(
476 f"Unable to find placeholder for parameter: {name}"
477 )
478 pos = binding_map[name]
479 args[pos] = build_value(value)
480 return args

482 def _process_frame(self, msg: MessageType, result: ResultBaseType) -> None:
483 """Process frame.
484
485 Args:
486 msg (mysqlx.protobuf.Message): A MySQL X Protobuf Message.
487 result (Result): A `Result` based type object.
488 """
489 if msg["type"] == 1:
490 warn_msg = Message.from_message("Mysqlx.Notice.Warning", msg["payload"])
491 self._warnings.append(warn_msg.msg)
492 logger.warning(
493 "Protocol.process_frame Received Warning Notice code %s: %s",
494 warn_msg.code,
495 warn_msg.msg,
496 )
497 result.append_warning(warn_msg.level, warn_msg.code, warn_msg.msg)
498 elif msg["type"] == 2:
499 Message.from_message("Mysqlx.Notice.SessionVariableChanged", msg["payload"])
500 elif msg["type"] == 3:
501 sess_state_msg = Message.from_message(
502 "Mysqlx.Notice.SessionStateChanged", msg["payload"]
503 )
504 if sess_state_msg["param"] == mysqlxpb_enum(
505 "Mysqlx.Notice.SessionStateChanged.Parameter.GENERATED_DOCUMENT_IDS"
506 ):
507 result.set_generated_ids(
508 [
509 get_item_or_attr(
510 get_item_or_attr(value, "v_octets"), "value"
511 ).decode()
512 for value in sess_state_msg["value"]
513 ]
514 )
515 else: # Following results are unitary and not a list
516 sess_state_value = sess_state_msg["value"].pop()
517 if sess_state_msg["param"] == mysqlxpb_enum(
518 "Mysqlx.Notice.SessionStateChanged.Parameter.ROWS_AFFECTED"
519 ):
520 result.set_rows_affected(
521 get_item_or_attr(sess_state_value, "v_unsigned_int")
522 )
523 elif sess_state_msg["param"] == mysqlxpb_enum(
524 "Mysqlx.Notice.SessionStateChanged.Parameter.GENERATED_INSERT_ID"
525 ):
526 result.set_generated_insert_id(
527 get_item_or_attr(sess_state_value, "v_unsigned_int")
528 )

530 def _read_message(self, result: ResultBaseType) -> Optional[MessageType]:
531 """Read message.
532
533 Args:
534 result (Result): A `Result` based type object.
535 """
536 while True:
537 try:
538 msg = self._reader.read_message()
539 except RuntimeError as err:
540 warnings = repr(result.get_warnings())
541 if warnings:
542 raise RuntimeError(f"{err} reason: {warnings}") from err
543 if msg.type == "Mysqlx.Error":
544 raise OperationalError(msg["msg"], msg["code"])
545 if msg.type == "Mysqlx.Notice.Frame":
546 try:
547 self._process_frame(msg, result)
548 except (AttributeError, KeyError):
549 continue
550 elif msg.type == "Mysqlx.Sql.StmtExecuteOk":
551 return None
552 elif msg.type == "Mysqlx.Resultset.FetchDone":
553 result.set_closed(True)
554 elif msg.type == "Mysqlx.Resultset.FetchDoneMoreResultsets":
555 result.set_has_more_results(True)
556 elif msg.type == "Mysqlx.Resultset.Row":
557 result.set_has_data(True)
558 break
559 else:
560 break
561 return msg

597 def set_capabilities(self, **kwargs: Any) -> None:
598 """Set capabilities.
599
600 Args:
601 **kwargs: Arbitrary keyword arguments.
602
603 Returns:
604 mysqlx.protobuf.Message: MySQL X Protobuf Message.
605 """
606 if not kwargs:
607 return None
608 capabilities = Message("Mysqlx.Connection.Capabilities")
609 for key, value in kwargs.items():
610 capability = Message("Mysqlx.Connection.Capability")
611 capability["name"] = key
612 if isinstance(value, dict):
613 items = value
614 obj_flds = []
615 for item in items:
616 obj_fld = Message(
617 "Mysqlx.Datatypes.Object.ObjectField",
618 key=item,
619 value=self._create_any(items[item]),
620 )
621 obj_flds.append(obj_fld.get_message())
622 msg_obj = Message("Mysqlx.Datatypes.Object", fld=obj_flds)
623 msg_any = Message("Mysqlx.Datatypes.Any", type=2, obj=msg_obj)
624 capability["value"] = msg_any.get_message()
625 else:
626 capability["value"] = self._create_any(value)
627
628 capabilities["capabilities"].extend([capability.get_message()])
629 msg = Message("Mysqlx.Connection.CapabilitiesSet")
630 msg["capabilities"] = capabilities
631 self._writer.write_message(
632 mysqlxpb_enum("Mysqlx.ClientMessages.Type.CON_CAPABILITIES_SET"),
633 msg,
634 )
635
636 try:
637 return self.read_ok()
638 except InterfaceError as err:
639 # Skip capability "session_connect_attrs" error since
640 # is only available on version >= 8.0.16
641 if err.errno != 5002:
642 raise
643 return None

713 def send_prepare_prepare(
714 self,
715 msg_type: str,
716 msg: MessageType,
717 stmt: Union[
718 FindStatement,
719 DeleteStatement,
720 ModifyStatement,
721 ReadStatement,
722 RemoveStatement,
723 UpdateStatement,
724 ],
725 ) -> None:
726 """
727 Send prepare statement.
728
729 Args:
730 msg_type (str): Message ID string.
731 msg (mysqlx.protobuf.Message): MySQL X Protobuf Message.
732 stmt (Statement): A `Statement` based type object.
733
734 Raises:
735 :class:`mysqlx.NotSupportedError`: If prepared statements are not
736 supported.
737
738 .. versionadded:: 8.0.16
739 """
740 if stmt.has_limit and msg.type != "Mysqlx.Crud.Insert":
741 # Remove 'limit' from message by building a new one
742 if msg.type == "Mysqlx.Crud.Find":
743 _, msg = self.build_find(stmt) # type: ignore[arg-type]
744 elif msg.type == "Mysqlx.Crud.Update":
745 _, msg = self.build_update(stmt) # type: ignore[arg-type]
