278 def test_stack_effects(self):
279 # Verify all opcodes are handled and that "jump=None" really returns
280 # the max of the other cases.
281 from _pydevd_frame_eval.vendored.bytecode.concrete import ConcreteInstr
282
283 def check(instr):
284 jump = instr.stack_effect(jump=True)
285 no_jump = instr.stack_effect(jump=False)
286 max_effect = instr.stack_effect(jump=None)
287 self.assertEqual(instr.stack_effect(), max_effect)
288 self.assertEqual(max_effect, max(jump, no_jump))
289
290 if not instr.has_jump():
291 self.assertEqual(jump, no_jump)
292
293 for name, op in opcode.opmap.items():
294 with self.subTest(name):
295 # Use ConcreteInstr instead of Instr because it doesn't care
296 # what kind of argument it is constructed with.
297 if op < opcode.HAVE_ARGUMENT:
298 check(ConcreteInstr(name))
299 else:
300 for arg in range(256):
301 check(ConcreteInstr(name, arg))
302
303 # LOAD_CONST uses a concrete python object as its oparg, however, in
304 # dis.stack_effect(opcode.opmap['LOAD_CONST'], oparg),
305 # oparg should be the index of that python object in the constants.
306 #
307 # Fortunately, for an instruction whose oparg isn't equivalent to its
308 # form in binary files(pyc format), the stack effect is a
309 # constant which does not depend on its oparg.
310 #
311 # The second argument of dis.stack_effect cannot be
312 # more than 2**31 - 1. If stack effect of an instruction is
313 # independent of its oparg, we pass 0 as the second argument
314 # of dis.stack_effect.
315 # (As a result we can calculate stack_effect for
316 # any LOAD_CONST instructions, even for large integers)
317
318 for arg in 2**31, 2**32, 2**63, 2**64, -1:
319 self.assertEqual(Instr("LOAD_CONST", arg).stack_effect(), 1)
