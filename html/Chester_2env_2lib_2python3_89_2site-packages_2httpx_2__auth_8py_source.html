22 def auth_flow(self, request: Request) -> typing.Generator[Request, Response, None]:
23 """
24 Execute the authentication flow.
25
26 To dispatch a request, `yield` it:
27
28 ```
29 yield request
30 ```
31
32 The client will `.send()` the response back into the flow generator. You can
33 access it like so:
34
35 ```
36 response = yield request
37 ```
38
39 A `return` (or reaching the end of the generator) will result in the
40 client returning the last response obtained from the server.
41
42 You can dispatch as many requests as is necessary.

120 def _build_auth_header(
121 self, request: Request, challenge: "DigestAuthChallenge"
122 ) -> str:
123 hash_func = self.ALGORITHM_TO_HASH_FUNCTION[challenge.algorithm]
124
125 def digest(data: bytes) -> bytes:
126 return hash_func(data).hexdigest().encode()
127
128 A1 = b":".join((self.username, challenge.realm, self.password))
129
130 path = request.url.full_path.encode("utf-8")
131 A2 = b":".join((request.method.encode(), path))
132 # TODO: implement auth-int
133 HA2 = digest(A2)
134
135 nonce_count = 1 # TODO: implement nonce counting
136 nc_value = b"%08x" % nonce_count
137 cnonce = self._get_client_nonce(nonce_count, challenge.nonce)
138
139 HA1 = digest(A1)
140 if challenge.algorithm.lower().endswith("-sess"):
141 HA1 = digest(b":".join((HA1, challenge.nonce, cnonce)))
142
143 qop = self._resolve_qop(challenge.qop)
144 if qop is None:
145 digest_data = [HA1, challenge.nonce, HA2]
146 else:
147 digest_data = [challenge.nonce, nc_value, cnonce, qop, HA2]
148 key_digest = b":".join(digest_data)
149
150 format_args = {
151 "username": self.username,
152 "realm": challenge.realm,
153 "nonce": challenge.nonce,
154 "uri": path,
155 "response": digest(b":".join((HA1, key_digest))),
156 "algorithm": challenge.algorithm.encode(),
157 }
158 if challenge.opaque:
159 format_args["opaque"] = challenge.opaque
160 if qop:
161 format_args["qop"] = b"auth"
162 format_args["nc"] = nc_value
163 format_args["cnonce"] = cnonce
164
165 return "Digest " + self._get_header_value(format_args)

167 def _get_client_nonce(self, nonce_count: int, nonce: bytes) -> bytes:
168 s = str(nonce_count).encode()
169 s += nonce
170 s += time.ctime().encode()
171 s += os.urandom(8)
172
173 return hashlib.sha1(s).hexdigest()[:16].encode()

175 def _get_header_value(self, header_fields: typing.Dict[str, bytes]) -> str:
176 NON_QUOTED_FIELDS = ("algorithm", "qop", "nc")
177 QUOTED_TEMPLATE = '{}="{}"'
178 NON_QUOTED_TEMPLATE = "{}={}"
179
180 header_value = ""
181 for i, (field, value) in enumerate(header_fields.items()):
182 if i > 0:
183 header_value += ", "
184 template = (
185 QUOTED_TEMPLATE
186 if field not in NON_QUOTED_FIELDS
187 else NON_QUOTED_TEMPLATE
188 )
189 header_value += template.format(field, to_str(value))
190
191 return header_value

193 def _resolve_qop(self, qop: typing.Optional[bytes]) -> typing.Optional[bytes]:
194 if qop is None:
195 return None
196 qops = re.split(b", ?", qop)
197 if b"auth" in qops:
198 return b"auth"

200 if qops == [b"auth-int"]:

221 @classmethod
222 def from_header(cls, header: str) -> "DigestAuthChallenge":
223 """Returns a challenge from a Digest WWW-Authenticate header.
224 These take the form of:
225 `Digest realm="realm@host.com",qop="auth,auth-int",nonce="abc",opaque="xyz"`
226 """
227 scheme, _, fields = header.partition(" ")
228 if scheme.lower() != "digest":
229 raise ValueError("Header does not start with 'Digest'")
230
231 header_dict: typing.Dict[str, str] = {}
232 for field in parse_http_list(fields):
233 key, value = field.strip().split("=", 1)
234 header_dict[key] = unquote(value)
235
236 try:
237 return cls.from_header_dict(header_dict)
238 except KeyError as exc:
239 raise ValueError("Malformed Digest WWW-Authenticate header") from exc
