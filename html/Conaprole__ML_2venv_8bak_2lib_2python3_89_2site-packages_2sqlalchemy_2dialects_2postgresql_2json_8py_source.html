28class JSONPathType(sqltypes.JSON.JSONPathType):
29 def _processor(self, dialect, super_proc):
30 def process(value):
31 if isinstance(value, str):
32 # If it's already a string assume that it's in json path
33 # format. This allows using cast with json paths literals
34 return value
35 elif value:
36 # If it's already a string assume that it's in json path
37 # format. This allows using cast with json paths literals
38 value = "{%s}" % (", ".join(map(str, value)))
39 else:
40 value = "{}"
41 if super_proc:
42 value = super_proc(value)
43 return value
44
45 return process

48 return self._processor(dialect, self.string_bind_processor(dialect))

51 return self._processor(dialect, self.string_literal_processor(dialect))

72class JSON(sqltypes.JSON):
73 """Represent the PostgreSQL JSON type.
74
75 :class:`_postgresql.JSON` is used automatically whenever the base
76 :class:`_types.JSON` datatype is used against a PostgreSQL backend,
77 however base :class:`_types.JSON` datatype does not provide Python
78 accessors for PostgreSQL-specific comparison methods such as
79 :meth:`_postgresql.JSON.Comparator.astext`; additionally, to use
80 PostgreSQL ``JSONB``, the :class:`_postgresql.JSONB` datatype should
81 be used explicitly.
82
83 .. seealso::
84
85 :class:`_types.JSON` - main documentation for the generic
86 cross-platform JSON datatype.
87
88 The operators provided by the PostgreSQL version of :class:`_types.JSON`
89 include:
90
91 * Index operations (the ``->`` operator)::
92
93 data_table.c.data["some key"]
94
95 data_table.c.data[5]
96
97 * Index operations returning text
98 (the ``->>`` operator)::
99
100 data_table.c.data["some key"].astext == "some value"
101
102 Note that equivalent functionality is available via the
103 :attr:`.JSON.Comparator.as_string` accessor.
104
105 * Index operations with CAST
106 (equivalent to ``CAST(col ->> ['some key'] AS <type>)``)::
107
108 data_table.c.data["some key"].astext.cast(Integer) == 5
109
110 Note that equivalent functionality is available via the
111 :attr:`.JSON.Comparator.as_integer` and similar accessors.
112
113 * Path index operations (the ``#>`` operator)::
114
115 data_table.c.data[("key_1", "key_2", 5, ..., "key_n")]
116
117 * Path index operations returning text (the ``#>>`` operator)::
118
119 data_table.c.data[
120 ("key_1", "key_2", 5, ..., "key_n")
121 ].astext == "some value"
122
123 Index operations return an expression object whose type defaults to
124 :class:`_types.JSON` by default,
125 so that further JSON-oriented instructions
126 may be called upon the result type.
127
128 Custom serializers and deserializers are specified at the dialect level,
129 that is using :func:`_sa.create_engine`. The reason for this is that when
130 using psycopg2, the DBAPI only allows serializers at the per-cursor
131 or per-connection level. E.g.::
132
133 engine = create_engine(
134 "postgresql+psycopg2://scott:tiger@localhost/test",
135 json_serializer=my_serialize_fn,
136 json_deserializer=my_deserialize_fn,
137 )
138
139 When using the psycopg2 dialect, the json_deserializer is registered
140 against the database using ``psycopg2.extras.register_default_json``.
141
142 .. seealso::
143
144 :class:`_types.JSON` - Core level JSON type
145
146 :class:`_postgresql.JSONB`
147
148 """ # noqa
149
150 render_bind_cast = True
151 astext_type = sqltypes.Text()

174 super().__init__(none_as_null=none_as_null)

209class JSONB(JSON):
210 """Represent the PostgreSQL JSONB type.
211
212 The :class:`_postgresql.JSONB` type stores arbitrary JSONB format data,
213 e.g.::
214
215 data_table = Table(
216 "data_table",
217 metadata,
218 Column("id", Integer, primary_key=True),
219 Column("data", JSONB),
220 )
221
222 with engine.connect() as conn:
223 conn.execute(
224 data_table.insert(), data={"key1": "value1", "key2": "value2"}
225 )
226
227 The :class:`_postgresql.JSONB` type includes all operations provided by
228 :class:`_types.JSON`, including the same behaviors for indexing
229 operations.
230 It also adds additional operators specific to JSONB, including
231 :meth:`.JSONB.Comparator.has_key`, :meth:`.JSONB.Comparator.has_all`,
232 :meth:`.JSONB.Comparator.has_any`, :meth:`.JSONB.Comparator.contains`,
233 :meth:`.JSONB.Comparator.contained_by`,
234 :meth:`.JSONB.Comparator.delete_path`,
235 :meth:`.JSONB.Comparator.path_exists` and
236 :meth:`.JSONB.Comparator.path_match`.
237
238 Like the :class:`_types.JSON` type, the :class:`_postgresql.JSONB`
239 type does not detect
240 in-place changes when used with the ORM, unless the
241 :mod:`sqlalchemy.ext.mutable` extension is used.
242
243 Custom serializers and deserializers
244 are shared with the :class:`_types.JSON` class,
245 using the ``json_serializer``
246 and ``json_deserializer`` keyword arguments. These must be specified
247 at the dialect level using :func:`_sa.create_engine`. When using
248 psycopg2, the serializers are associated with the jsonb type using
249 ``psycopg2.extras.register_default_jsonb`` on a per-connection basis,
250 in the same way that ``psycopg2.extras.register_default_json`` is used
251 to register these handlers with the json type.
252
253 .. seealso::
254
255 :class:`_types.JSON`
256
257 """
258
259 __visit_name__ = "JSONB"

301 def delete_path(self, array):
302 """JSONB expression. Deletes field or array element specified in
303 the argument array (equivalent of the ``#-`` operator).
304
305 The input may be a list of strings that will be coerced to an
306 ``ARRAY`` or an instance of :meth:`_postgres.array`.
307
308 .. versionadded:: 2.0
309 """
310 if not isinstance(array, _pg_array):
311 array = _pg_array(array)
312 right_side = cast(array, ARRAY(sqltypes.TEXT))
313 return self.operate(DELETE_PATH, right_side, result_type=JSONB)
314
315 def path_exists(self, other):
316 """Boolean expression. Test for presence of item given by the
317 argument JSONPath expression (equivalent of the ``@?`` operator).
318
319 .. versionadded:: 2.0
320 """
321 return self.operate(
322 PATH_EXISTS, other, result_type=sqltypes.Boolean
323 )
324
325 def path_match(self, other):
326 """Boolean expression. Test if JSONPath predicate given by the
327 argument JSONPath expression matches
328 (equivalent of the ``@@`` operator).
329
330 Only the first item of the result is taken into account.
331
332 .. versionadded:: 2.0
333 """
334 return self.operate(
335 PATH_MATCH, other, result_type=sqltypes.Boolean
336 )
