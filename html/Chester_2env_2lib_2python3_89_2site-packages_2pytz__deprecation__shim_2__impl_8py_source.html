18def timezone(key, _cache={}):
19 """Builds an IANA database time zone shim.
20
21 This is the equivalent of ``pytz.timezone``.
22
23 :param key:
24 A valid key from the IANA time zone database.
25
26 :raises UnknownTimeZoneError:
27 If an unknown value is passed, this will raise an exception that can be
28 caught by :exc:`pytz_deprecation_shim.UnknownTimeZoneError` or
29 ``pytz.UnknownTimeZoneError``. Like
30 :exc:`zoneinfo.ZoneInfoNotFoundError`, both of those are subclasses of
31 :exc:`KeyError`.
32 """
33 instance = _cache.get(key, None)
34 if instance is None:
35 if len(key) == 3 and key.lower() == "utc":
36 instance = _cache.setdefault(key, UTC)
37 else:
38 try:
39 zone = _compat.get_timezone(key)
40 except KeyError:
41 raise get_exception(UnknownTimeZoneError, key)
42 instance = _cache.setdefault(key, wrap_zone(zone, key=key))
43
44 return instance

47def fixed_offset_timezone(offset, _cache={}):
48 """Builds a fixed offset time zone shim.
49
50 This is the equivalent of ``pytz.FixedOffset``. An alias is available as
51 ``pytz_deprecation_shim.FixedOffset`` as well.
52
53 :param offset:
54 A fixed offset from UTC, in minutes. This must be in the range ``-1439
55 <= offset <= 1439``.
56
57 :raises ValueError:
58 For offsets whose absolute value is greater than or equal to 24 hours.
59
60 :return:
61 A shim time zone.
62 """
63 if not (-1440 < offset < 1440):
64 raise ValueError("absolute offset is too large", offset)
65
66 instance = _cache.get(offset, None)
67 if instance is None:
68 if offset == 0:
69 instance = _cache.setdefault(offset, UTC)
70 else:
71 zone = _compat.get_fixed_offset_zone(offset)
72 instance = _cache.setdefault(offset, wrap_zone(zone, key=None))
73
74 return instance

102def wrap_zone(tz, key=KEY_SENTINEL, _cache={}):
103 """Wrap an existing time zone object in a shim class.
104
105 This is likely to be useful if you would like to work internally with
106 non-``pytz`` zones, but you expose an interface to callers relying on
107 ``pytz``'s interface. It may also be useful for passing non-``pytz`` zones
108 to libraries expecting to use ``pytz``'s interface.
109
110 :param tz:
111 A :pep:`495`-compatible time zone, such as those provided by
112 :mod:`dateutil.tz` or :mod:`zoneinfo`.
113
114 :param key:
115 The value for the IANA time zone key. This is optional for ``zoneinfo``
116 zones, but required for ``dateutil.tz`` zones.
117
118 :return:
119 A shim time zone.
120 """
121 if key is KEY_SENTINEL:
122 key = getattr(tz, "key", KEY_SENTINEL)
123
124 if key is KEY_SENTINEL:
125 raise TypeError(
126 "The `key` argument is required when wrapping zones that do not "
127 + "have a `key` attribute."
128 )
129
130 instance = _cache.get((id(tz), key), None)
131 if instance is None:
132 instance = _cache.setdefault((id(tz), key), _PytzShimTimezone(tz, key))
133
134 return instance

205 def localize(self, dt, is_dst=IS_DST_SENTINEL):
206 warnings.warn(
207 "The localize method is no longer necessary, as this "
208 + "time zone supports the fold attribute (PEP 495). "
209 + "For more details on migrating to a PEP 495-compliant "
210 + "implementation, see %s" % PYTZ_MIGRATION_GUIDE_URL,
211 PytzUsageWarning,
212 stacklevel=2,
213 )
214
215 if dt.tzinfo is not None:
216 raise ValueError("Not naive datetime (tzinfo is already set)")
217
218 dt_out = dt.replace(tzinfo=self)
219
220 if is_dst is IS_DST_SENTINEL:
221 return dt_out
222
223 dt_ambiguous = _compat.is_ambiguous(dt_out)
224 dt_imaginary = (
225 _compat.is_imaginary(dt_out) if not dt_ambiguous else False
226 )
227
228 if is_dst is None:
229 if dt_imaginary:
230 raise get_exception(
231 NonExistentTimeError, dt.replace(tzinfo=None)
232 )
233
234 if dt_ambiguous:
235 raise get_exception(AmbiguousTimeError, dt.replace(tzinfo=None))
236
237 elif dt_ambiguous or dt_imaginary:
238 # Start by normalizing the folds; dt_out may have fold=0 or fold=1,
239 # but we need to know the DST offset on both sides anyway, so we
240 # will get one datetime representing each side of the fold, then
241 # decide which one we're going to return.
242 if _compat.get_fold(dt_out):
243 dt_enfolded = dt_out
244 dt_out = _compat.enfold(dt_out, fold=0)
245 else:
246 dt_enfolded = _compat.enfold(dt_out, fold=1)
247
248 # Now we want to decide whether the fold=0 or fold=1 represents
249 # what pytz would return for `is_dst=True`
250 enfolded_dst = bool(dt_enfolded.dst())
251 if bool(dt_out.dst()) == enfolded_dst:
252 # If this is not a transition between standard time and
253 # daylight saving time, pytz will consider the larger offset
254 # the DST offset.
255 enfolded_dst = dt_enfolded.utcoffset() > dt_out.utcoffset()
256
257 # The default we've established is that dt_out is fold=0; swap it
258 # for the fold=1 datetime if is_dst == True and the enfolded side
259 # is DST or if is_dst == False and the enfolded side is *not* DST.
260 if is_dst == enfolded_dst:
261 dt_out = dt_enfolded
262
263 return dt_out
