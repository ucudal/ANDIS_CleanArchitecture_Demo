373 @classmethod
374 def get_file_version_info(cls, filename):
375 """
376 Get the program version from an executable file, if available.
377
378 @type filename: str
379 @param filename: Pathname to the executable file to query.
380
381 @rtype: tuple(str, str, bool, bool, str, str)
382 @return: Tuple with version information extracted from the executable
383 file metadata, containing the following:
384 - File version number (C{"major.minor"}).
385 - Product version number (C{"major.minor"}).
386 - C{True} for debug builds, C{False} for production builds.
387 - C{True} for legacy OS builds (DOS, OS/2, Win16),
388 C{False} for modern OS builds.
389 - Binary file type.
390 May be one of the following values:
391 - "application"
392 - "dynamic link library"
393 - "static link library"
394 - "font"
395 - "raster font"
396 - "TrueType font"
397 - "vector font"
398 - "driver"
399 - "communications driver"
400 - "display driver"
401 - "installable driver"
402 - "keyboard driver"
403 - "language driver"
404 - "legacy driver"
405 - "mouse driver"
406 - "network driver"
407 - "printer driver"
408 - "sound driver"
409 - "system driver"
410 - "versioned printer driver"
411 - Binary creation timestamp.
412 Any of the fields may be C{None} if not available.
413
414 @raise WindowsError: Raises an exception on error.
415 """
416
417 # Get the file version info structure.
418 pBlock = win32.GetFileVersionInfo(filename)
419 pBuffer, dwLen = win32.VerQueryValue(pBlock, "\\")
420 if dwLen != ctypes.sizeof(win32.VS_FIXEDFILEINFO):
421 raise ctypes.WinError(win32.ERROR_BAD_LENGTH)
422 pVersionInfo = ctypes.cast(pBuffer, ctypes.POINTER(win32.VS_FIXEDFILEINFO))
423 VersionInfo = pVersionInfo.contents
424 if VersionInfo.dwSignature != 0xFEEF04BD:
425 raise ctypes.WinError(win32.ERROR_BAD_ARGUMENTS)
426
427 # File and product versions.
428 FileVersion = "%d.%d" % (VersionInfo.dwFileVersionMS, VersionInfo.dwFileVersionLS)
429 ProductVersion = "%d.%d" % (VersionInfo.dwProductVersionMS, VersionInfo.dwProductVersionLS)
430
431 # Debug build?
432 if VersionInfo.dwFileFlagsMask & win32.VS_FF_DEBUG:
433 DebugBuild = (VersionInfo.dwFileFlags & win32.VS_FF_DEBUG) != 0
434 else:
435 DebugBuild = None
436
437 # Legacy OS build?
438 LegacyBuild = VersionInfo.dwFileOS != win32.VOS_NT_WINDOWS32
439
440 # File type.
441 FileType = cls.__binary_types.get(VersionInfo.dwFileType)
442 if VersionInfo.dwFileType == win32.VFT_DRV:
443 FileType = cls.__driver_types.get(VersionInfo.dwFileSubtype)
444 elif VersionInfo.dwFileType == win32.VFT_FONT:
445 FileType = cls.__font_types.get(VersionInfo.dwFileSubtype)
446
447 # Timestamp, ex: "Monday, July 7, 2013 (12:20:50.126)".
448 # FIXME: how do we know the time zone?
449 FileDate = (VersionInfo.dwFileDateMS << 32) + VersionInfo.dwFileDateLS
450 if FileDate:
451 CreationTime = win32.FileTimeToSystemTime(FileDate)
452 CreationTimestamp = "%s, %s %d, %d (%d:%d:%d.%d)" % (
453 cls.__days_of_the_week[CreationTime.wDayOfWeek],
454 cls.__months[CreationTime.wMonth],
455 CreationTime.wDay,
456 CreationTime.wYear,
457 CreationTime.wHour,
458 CreationTime.wMinute,
459 CreationTime.wSecond,
460 CreationTime.wMilliseconds,
461 )
462 else:
463 CreationTimestamp = None
464
465 # Return the file version info.
466 return (
467 FileVersion,
468 ProductVersion,
469 DebugBuild,
470 LegacyBuild,
471 FileType,
472 CreationTimestamp,
473 )

520 @classmethod
521 def load_dbghelp(cls, pathname=None):
522 """
523 Load the specified version of the C{dbghelp.dll} library.
524
525 This library is shipped with the Debugging Tools for Windows, and it's
526 required to load debug symbols.
527
528 Normally you don't need to call this method, as WinAppDbg already tries
529 to load the latest version automatically - but it may come in handy if
530 the Debugging Tools are installed in a non standard folder.
531
532 Example::
533 from winappdbg import Debug
534
535 def simple_debugger( argv ):
536
537 # Instance a Debug object, passing it the event handler callback
538 debug = Debug( my_event_handler )
539 try:
540
541 # Load a specific dbghelp.dll file
542 debug.system.load_dbghelp("C:\\Some folder\\dbghelp.dll")
543
544 # Start a new process for debugging
545 debug.execv( argv )
546
547 # Wait for the debugee to finish
548 debug.loop()
549
550 # Stop the debugger
551 finally:
552 debug.stop()
553
554 @see: U{http://msdn.microsoft.com/en-us/library/ms679294(VS.85).aspx}
555
556 @type pathname: str
557 @param pathname:
558 (Optional) Full pathname to the C{dbghelp.dll} library.
559 If not provided this method will try to autodetect it.
560
561 @rtype: ctypes.WinDLL
562 @return: Loaded instance of C{dbghelp.dll}.
563
564 @raise NotImplementedError: This feature was not implemented for the
565 current architecture.
566
567 @raise WindowsError: An error occured while processing this request.
568 """
569
570 # If an explicit pathname was not given, search for the library.
571 if not pathname:
572 # Under WOW64 we'll treat AMD64 as I386.
573 arch = win32.arch
574 if arch == win32.ARCH_AMD64 and win32.bits == 32:
575 arch = win32.ARCH_I386
576
577 # Check if the architecture is supported.
578 if not arch in cls.__dbghelp_locations:
579 msg = "Architecture %s is not currently supported."
580 raise NotImplementedError(msg % arch)
581
582 # Grab all versions of the library we can find.
583 found = []
584 for pathname in cls.__dbghelp_locations[arch]:
585 if path.isfile(pathname):
586 try:
587 f_ver, p_ver = cls.get_file_version_info(pathname)[:2]
588 except WindowsError:
589 msg = "Failed to parse file version metadata for: %s"
590 warnings.warn(msg % pathname)
591 if not f_ver:
592 f_ver = p_ver
593 elif p_ver and p_ver > f_ver:
594 f_ver = p_ver
595 found.append((f_ver, pathname))
596
597 # If we found any, use the newest version.
598 if found:
599 found.sort()
600 pathname = found.pop()[1]
601
602 # If we didn't find any, trust the default DLL search algorithm.
603 else:
604 pathname = "dbghelp.dll"
605
606 # Load the library.
607 dbghelp = ctypes.windll.LoadLibrary(pathname)
608
609 # Set it globally as the library to be used.
610 ctypes.windll.dbghelp = dbghelp
611
612 # Return the library.
613 return dbghelp

615 @staticmethod
616 def fix_symbol_store_path(symbol_store_path=None, remote=True, force=False):
617 """
618 Fix the symbol store path. Equivalent to the C{.symfix} command in
619 Microsoft WinDbg.
620
621 If the symbol store path environment variable hasn't been set, this
622 method will provide a default one.
623
624 @type symbol_store_path: str or None
625 @param symbol_store_path: (Optional) Symbol store path to set.
626
627 @type remote: bool
628 @param remote: (Optional) Defines the symbol store path to set when the
629 C{symbol_store_path} is C{None}.
630
631 If C{True} the default symbol store path is set to the Microsoft
632 symbol server. Debug symbols will be downloaded through HTTP.
633 This gives the best results but is also quite slow.
634
635 If C{False} the default symbol store path is set to the local
636 cache only. This prevents debug symbols from being downloaded and
637 is faster, but unless you've installed the debug symbols on this
638 machine or downloaded them in a previous debugging session, some
639 symbols may be missing.
640
641 If the C{symbol_store_path} argument is not C{None}, this argument
642 is ignored entirely.
643
644 @type force: bool
645 @param force: (Optional) If C{True} the new symbol store path is set
646 always. If C{False} the new symbol store path is only set if
647 missing.
648
649 This allows you to call this method preventively to ensure the
650 symbol server is always set up correctly when running your script,
651 but without messing up whatever configuration the user has.
652
653 Example::
654 from winappdbg import Debug, System
655
656 def simple_debugger( argv ):
657
658 # Instance a Debug object
659 debug = Debug( MyEventHandler() )
660 try:
661
662 # Make sure the remote symbol store is set
663 System.fix_symbol_store_path(remote = True,
664 force = False)
665
666 # Start a new process for debugging
667 debug.execv( argv )
668
669 # Wait for the debugee to finish
670 debug.loop()
671
672 # Stop the debugger
673 finally:
674 debug.stop()
675
676 @rtype: str or None
677 @return: The previously set symbol store path if any,
678 otherwise returns C{None}.
679 """
680 try:
681 if symbol_store_path is None:
682 local_path = "C:\\SYMBOLS"
683 if not path.isdir(local_path):
684 local_path = "C:\\Windows\\Symbols"
685 if not path.isdir(local_path):
686 local_path = path.abspath(".")
687 if remote:
688 symbol_store_path = "cache*;SRV*" + local_path + "*" "http://msdl.microsoft.com/download/symbols"
689 else:
690 symbol_store_path = "cache*;SRV*" + local_path
691 previous = os.environ.get("_NT_SYMBOL_PATH", None)
692 if not previous or force:
693 os.environ["_NT_SYMBOL_PATH"] = symbol_store_path
694 return previous
695 except Exception:
696 e = sys.exc_info()[1]
697 warnings.warn("Cannot fix symbol path, reason: %s" % str(e), RuntimeWarning)
