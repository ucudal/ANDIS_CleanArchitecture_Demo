58def fewtimes_if_linux():
59 """Decorator for those Linux functions which are implemented in pure
60 Python, and which we want to run faster.
61 """
62
63 def decorator(fun):
64 @functools.wraps(fun)
65 def wrapper(self, *args, **kwargs):
66 if LINUX:
67 before = self.__class__.times
68 try:
69 self.__class__.times = FEW_TIMES
70 return fun(self, *args, **kwargs)
71 finally:
72 self.__class__.times = before
73 else:
74 return fun(self, *args, **kwargs)
75
76 return wrapper
77
78 return decorator
