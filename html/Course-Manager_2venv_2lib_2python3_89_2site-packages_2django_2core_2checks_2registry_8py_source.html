28 def register(self, check=None, *tags, **kwargs):
29 """
30 Can be used as a function or a decorator. Register given function
31 `f` labeled with given `tags`. The function should receive **kwargs
32 and return list of Errors and Warnings.
33
34 Example::
35
36 registry = CheckRegistry()
37 @registry.register('mytag', 'anothertag')
38 def my_check(apps, **kwargs):
39 # ... perform checks and collect `errors` ...
40 return errors
41 # or
42 registry.register(my_check, 'mytag', 'anothertag')
43 """
44 def inner(check):
45 check.tags = tags
46 checks = self.deployment_checks if kwargs.get('deploy') else self.registered_checks
47 checks.add(check)
48 return check
49
50 if callable(check):
51 return inner(check)
52 else:
53 if check:
54 tags += (check,)
55 return inner

57 def run_checks(self, app_configs=None, tags=None, include_deployment_checks=False):
58 """
59 Run all registered checks and return list of Errors and Warnings.
60 """
61 errors = []
62 checks = self.get_checks(include_deployment_checks)
63
64 if tags is not None:
65 checks = [check for check in checks if not set(check.tags).isdisjoint(tags)]
66 else:
67 # By default, 'database'-tagged checks are not run as they do more
68 # than mere static code analysis.
69 checks = [check for check in checks if Tags.database not in check.tags]
70
71 for check in checks:
72 new_errors = check(app_configs=app_configs)
73 assert is_iterable(new_errors), (
74 "The function %r did not return a list. All functions registered "
75 "with the checks registry must return a list." % check)
76 errors.extend(new_errors)
77 return errors
