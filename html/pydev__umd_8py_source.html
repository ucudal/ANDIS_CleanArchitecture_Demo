40class UserModuleDeleter:
41 """
42 User Module Deleter (UMD) aims at deleting user modules
43 to force Python to deeply reload them during import
44
45 pathlist [list]: ignore list in terms of module path
46 namelist [list]: ignore list in terms of module name
47 """
48
49 def __init__(self, namelist=None, pathlist=None):
50 if namelist is None:
51 namelist = []
52 self.namelist = namelist
53 if pathlist is None:
54 pathlist = []
55 self.pathlist = pathlist
56 try:
57 # ignore all files in org.python.pydev/pysrc
58 import pydev_pysrc, inspect
59
60 self.pathlist.append(os.path.dirname(pydev_pysrc.__file__))
61 except:
62 pass
63 self.previous_modules = list(sys.modules.keys())

65 def is_module_ignored(self, modname, modpath):
66 for path in [sys.prefix] + self.pathlist:
67 if modpath.startswith(path):
68 return True
69 else:
70 return set(modname.split(".")) & set(self.namelist)

72 def run(self, verbose=False):
73 """
74 Del user modules to force Python to deeply reload them
75
76 Do not del modules which are considered as system modules, i.e.
77 modules installed in subdirectories of Python interpreter's binary
78 Do not del C modules
79 """
80 log = []
81 modules_copy = dict(sys.modules)
82 for modname, module in modules_copy.items():
83 if modname == "aaaaa":
84 print(modname, module)
85 print(self.previous_modules)
86 if modname not in self.previous_modules:
87 modpath = getattr(module, "__file__", None)
88 if modpath is None:
89 # *module* is a C module that is statically linked into the
90 # interpreter. There is no way to know its path, so we
91 # choose to ignore it.
92 continue
93 if not self.is_module_ignored(modname, modpath):
94 log.append(modname)
95 del sys.modules[modname]
96 if verbose and log:
97 print("\x1b[4;33m%s\x1b[24m%s\x1b[0m" % ("UMD has deleted", ": " + ", ".join(log)))
